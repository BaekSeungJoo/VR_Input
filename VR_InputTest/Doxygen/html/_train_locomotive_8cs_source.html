129 private IEnumerator StartStopTrain(bool startTrain)
130 {
131 float endSpeed = startTrain ? _initialSpeed : 0.0f;
132
133 var timePeriodForSpeedChange = 3.0f;
134 if (startTrain)
135 {
136 _smoke1.Play();
137 _isMoving = true;
138 var emissionModule1 = _smoke1.emission;
139 var mainModule = _smoke1.main;
140 emissionModule1.rateOverTimeMultiplier = _standardRateOverTimeMultiplier;
141 mainModule.maxParticles = _standardMaxParticles;
142 timePeriodForSpeedChange = PlayEngineSound(EngineSoundState.Start);
143 }
144 else
145 {
146 timePeriodForSpeedChange = PlayEngineSound(EngineSoundState.Stop);
147 }
148
149 // don't loop audio at first; only do when train continues movement below
150 _engineAudioSource.loop = false;
151
152 // make time period a tad shorter so that if it's not looping, we don't
153 // catch the beginning of the sound
154 timePeriodForSpeedChange = timePeriodForSpeedChange * 0.9f;
155 float startTime = Time.time;
156 float endTime = Time.time + timePeriodForSpeedChange;
157 float startSpeed = _currentSpeed;
158 while (Time.time < endTime)
159 {
160 float t = (Time.time - startTime) / timePeriodForSpeedChange;
161 _currentSpeed = startSpeed * (1.0f - t) + endSpeed * t;
162 UpdateSmokeEmissionBasedOnSpeed();
163 yield return null;
164 }
165
166 _currentSpeed = endSpeed;
167 _startStopTrainCr = null;
168 _isMoving = startTrain;
169 if (!_isMoving)
170 {
171 _smoke1.Stop();
172 }
173 else
174 {
175 _engineAudioSource.loop = true;
176 PlayEngineSound(EngineSoundState.AccelerateOrSetProperSpeed);
177 }
178 }

180 private float PlayEngineSound(EngineSoundState engineSoundState)
181 {
182 AudioClip audioClip = null;
183
184 if (engineSoundState == EngineSoundState.Start)
185 {
186 audioClip = _startUpSound;
187 }
188 else
189 {
190 AudioClip[] audioClips = engineSoundState == EngineSoundState.AccelerateOrSetProperSpeed
191 ? _accelerationSounds
192 : _decelerationSounds;
193 int numSounds = audioClips.Length;
194 int speedIndex = (int)Mathf.Round((_currentSpeed - MIN_SPEED) / _speedDiv);
195 audioClip = audioClips[Mathf.Clamp(speedIndex, 0, numSounds - 1)];
196 }
197
198 // if audio is already playing and we are playing the same track, don't interrupt it
199 if (_engineAudioSource.clip == audioClip && _engineAudioSource.isPlaying &&
200 engineSoundState == EngineSoundState.AccelerateOrSetProperSpeed)
201 {
202 return 0.0f;
203 }
204
205 _engineAudioSource.clip = audioClip;
206 _engineAudioSource.timeSamples = 0;
207 _engineAudioSource.Play();
208
209 return audioClip.length;
210 }

238 private void UpdateSmokeEmissionBasedOnSpeed()
239 {
240 var emissionModule = _smoke1.emission;
241 emissionModule.rateOverTimeMultiplier = GetCurrentSmokeEmission();
242 var mainModule = _smoke1.main;
243 mainModule.maxParticles = (int)Mathf.Lerp(_standardMaxParticles,
244 _standardMaxParticles * MAX_PARTICLES_MULTIPLIER,
245 _currentSpeed / (MAX_SPEED - MIN_SPEED));
246 }

248 private float GetCurrentSmokeEmission()
249 {
250 return Mathf.Lerp(_standardRateOverTimeMultiplier, _standardRateOverTimeMultiplier * SMOKE_SPEED_MULTIPLIER,
251 _currentSpeed / (MAX_SPEED - MIN_SPEED));
252 }
