24 [Header("Intent Settings")]

30 [FormerlySerializedAs("confidence")]
31 [Range(0, 1f), SerializeField] public float confidenceThreshold = .6f;
32
33 // Handle simple intent validation
34 protected override string OnValidateResponse(WitResponseNode response, bool isEarlyResponse)
35 {
36 // No response
37 if (response == null)
38 {
39 return "No response";
40 }
41 // Check against all intents
42 WitIntentData[] intents = response.GetIntents();
43 if (intents == null || intents.Length == 0)
44 {
45 return "No intents found";
46 }
47 // Find intent
48 WitIntentData found = null;
49 foreach (var intentData in intents)
50 {
51 if (string.Equals(intent, intentData.name, StringComparison.CurrentCultureIgnoreCase))
52 {
53 found = intentData;
54 break;
55 }
56 }
57 if (found == null)
58 {
59 return $"Missing required intent '{intent}'";
60 }
61 // Check confidence
62 if (found.confidence < confidenceThreshold)
63 {
64 return $"Required intent '{intent}' confidence too low: {found.confidence:0.000}\nRequired: {confidenceThreshold:0.000}";
65 }
66 return string.Empty;
67 }
68
69 protected override void OnEnable()
70 {
71 Manifest.WitResponseMatcherIntents.Add(intent);
72 base.OnEnable();
73 }
74
75 protected override void OnDisable()
76 {
77 Manifest.WitResponseMatcherIntents.Remove(intent);
78 base.OnDisable();
79 }
80 }
