207 // Get audio extension
208 public static string GetAudioExtension(AudioType audioType)
209 {
210 switch (audioType)
211 {
212 // PCM
213 case AudioType.UNKNOWN:
214 return "raw";
215 // OGG
216 case AudioType.OGGVORBIS:
217 return "ogg";
218 // MP3
219 case AudioType.MPEG:
220 return "mp3";
221 // WAV
222 case AudioType.WAV:
223 return "wav";
224 default:
225 VLog.W($"Attempting to process unsupported audio type: {audioType}");
226 return audioType.ToString().ToLower();
227 }
228 }
229 // Whether streamed audio is allowed by unity
230 public static bool CanStreamAudio(TTSWitAudioType witAudioType)
231 {
232 switch (witAudioType)
233 {
234 // Raw PCM: Supported by Wit.ai & custom unity implementation (DownloadHandlerRawPCM)
235 case TTSWitAudioType.PCM:
236 return true;
237 #if OGG_SUPPORT
238 // OGG: Supported by Unity (DownloadHandlerAudioClip) but not by Wit.ai
239 case TTSWitAudioType.OGG:
240 return true;
241 #endif
242 // MP3: Supported by Wit.ai but not by Unity (DownloadHandlerAudioClip)
243 case TTSWitAudioType.MPEG:
244 return false;
245 // WAV: does not support streaming
246 case TTSWitAudioType.WAV:
247 default:
248 return false;
249 }
250 }
