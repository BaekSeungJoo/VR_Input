29 public string GetSourceFilePathFromTypeName(string typeName, Manifest manifest, IAssemblyWalker assemblyWalker)
30 {
31 if (string.IsNullOrEmpty(typeName))
32 {
33 return null;
34 }
35
36 if (_typeToFilePath.ContainsKey(typeName))
37 {
38 return _typeToFilePath[typeName];
39 }
40
41 var type = GetTypeFromTypeName(typeName, manifest, assemblyWalker);
42
43 assemblyWalker.GetSourceCode(type, out var sourceCodeFile, out _);
44
45 return _typeToFilePath[typeName] = sourceCodeFile;
46 }

48 private Type GetTypeFromTypeName(string typeName, Manifest manifest, IAssemblyWalker assemblyWalker)
49 {
50 var entity = manifest.Entities.FirstOrDefault(manifestEntity => manifestEntity.Name.Equals(typeName));
51
52 if (entity == default)
53 {
54 VLog.D($"Could not find entity {typeName} in manifest");
55 return null;
56 }
57
58 var qualifiedTypeName = entity.GetQualifiedTypeName();
59
60 var allAssemblies = assemblyWalker.GetTargetAssemblies().ToList();
61
62 if (!allAssemblies.Any())
63 {
64 return null;
65 }
66
67 var assemblies = allAssemblies
68 .Where(a => a.FullName == entity.Assembly).ToList();
69
70 // If we can't find the assembly, the type may not be in the manifest yet. Try an exhaustive search.
71 if (assemblies.Count < 1)
72 {
73 assemblies = allAssemblies;
74 }
75
76 var assembly = assemblies.First();
77
78 var type = assembly.GetType(qualifiedTypeName);
79
80 // If we can't find a full qualified name, fallback to simple nam.e
81 return type ?? assembly.GetType(typeName);
82 }
83 }
