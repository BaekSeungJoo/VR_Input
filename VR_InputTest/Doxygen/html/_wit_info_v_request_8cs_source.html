33 // Get all apps & return the current app info
34 public bool RequestAppId(RequestCompleteDelegate<string> onComplete)
35 {
36 Dictionary<string, string> uriParameters = new Dictionary<string, string>();
37 uriParameters[WitEditorConstants.ENDPOINT_APPS_LIMIT] = 10000.ToString();
38 uriParameters[WitEditorConstants.ENDPOINT_APPS_OFFSET] = 0.ToString();
39 return RequestWitGet<WitResponseNode>(WitEditorConstants.ENDPOINT_APPS, uriParameters, (results, error) =>
40 {
41 if (string.IsNullOrEmpty(error) && results != null)
42 {
43 WitResponseArray nodes = results.AsArray;
44 if (nodes != null)
45 {
46 foreach (WitResponseNode node in nodes)
47 {
48 WitResponseClass child = node.AsObject;
49 if (child.HasChild(WitEditorConstants.ENDPOINT_APP_FOR_TOKEN) && child[WitEditorConstants.ENDPOINT_APP_FOR_TOKEN].AsBool && child.HasChild(WitEditorConstants.ENDPOINT_APP_ID))
50 {
51 onComplete?.Invoke(child[WitEditorConstants.ENDPOINT_APP_ID], null);
52 return;
53 }
54 }
55 }
56 error = "No app id found for token";
57 }
58 onComplete?.Invoke(null, error);
59 });
60 }

115 public bool RequestClientAppToken(string applicationId,
116 RequestCompleteDelegate<string> onComplete)
117 {
118 var jsonNode = new WitResponseClass()

120 { "refresh", "false" }

122 return RequestWitPost<WitResponseNode>($"{WitEditorConstants.ENDPOINT_APPS}/{applicationId}/{WitEditorConstants.ENDPOINT_CLIENTTOKENS}",
123 null, jsonNode.ToString(),
