255 protected virtual void UpdateController()
256 {
257 if (useProfileData)
258 {
259 if (InitialPose == null)
260 {
261 // Save the initial pose so it can be recovered if useProfileData
262 // is turned off later.
263 InitialPose = new OVRPose()
264 {
265 position = CameraRig.transform.localPosition,
266 orientation = CameraRig.transform.localRotation
267 };
268 }
269
270 var p = CameraRig.transform.localPosition;
271 if (OVRManager.instance.trackingOriginType == OVRManager.TrackingOrigin.EyeLevel)
272 {
273 p.y = OVRManager.profile.eyeHeight - (0.5f * Controller.height) + Controller.center.y;
274 }
275 else if (OVRManager.instance.trackingOriginType == OVRManager.TrackingOrigin.FloorLevel)
276 {
277 p.y = -(0.5f * Controller.height) + Controller.center.y;
278 }
279
280 CameraRig.transform.localPosition = p;
281 }
282 else if (InitialPose != null)
283 {
284 // Return to the initial pose if useProfileData was turned off at runtime
285 CameraRig.transform.localPosition = InitialPose.Value.position;
286 CameraRig.transform.localRotation = InitialPose.Value.orientation;
287 InitialPose = null;
288 }
289
290 CameraHeight = CameraRig.centerEyeAnchor.localPosition.y;
291
292 if (CameraUpdated != null)
293 {
294 CameraUpdated();
295 }
296
297 UpdateMovement();
298
299 Vector3 moveDirection = Vector3.zero;
300
301 float motorDamp = (1.0f + (Damping * SimulationRate * Time.deltaTime));
302
303 MoveThrottle.x /= motorDamp;
304 MoveThrottle.y = (MoveThrottle.y > 0.0f) ? (MoveThrottle.y / motorDamp) : MoveThrottle.y;
305 MoveThrottle.z /= motorDamp;
306
307 moveDirection += MoveThrottle * SimulationRate * Time.deltaTime;
308
309 // Gravity
310 if (Controller.isGrounded && FallSpeed <= 0)
311 FallSpeed = ((Physics.gravity.y * (GravityModifier * 0.002f)));
312 else
313 FallSpeed += ((Physics.gravity.y * (GravityModifier * 0.002f)) * SimulationRate * Time.deltaTime);
314
315 moveDirection.y += FallSpeed * SimulationRate * Time.deltaTime;
316
317
318 if (Controller.isGrounded && MoveThrottle.y <= transform.lossyScale.y * 0.001f)
319 {
320 // Offset correction for uneven ground
321 float bumpUpOffset = Mathf.Max(Controller.stepOffset,
322 new Vector3(moveDirection.x, 0, moveDirection.z).magnitude);
323 moveDirection -= bumpUpOffset * Vector3.up;
324 }
325
326 if (PreCharacterMove != null)
327 {
328 PreCharacterMove();
329 Teleported = false;
330 }
331
332 Vector3 predictedXZ = Vector3.Scale((Controller.transform.localPosition + moveDirection), new Vector3(1, 0, 1));
333
334 // Move contoller
335 Controller.Move(moveDirection);
336 Vector3 actualXZ = Vector3.Scale(Controller.transform.localPosition, new Vector3(1, 0, 1));
337
338 if (predictedXZ != actualXZ)
339 MoveThrottle += (actualXZ - predictedXZ) / (SimulationRate * Time.deltaTime);
340 }

343 public virtual void UpdateMovement()
344 {
345 //todo: enable for Unity Input System
346#if ENABLE_LEGACY_INPUT_MANAGER
347 if (HaltUpdateMovement)
348 return;
349
350 if (EnableLinearMovement)
351 {
352 bool moveForward = Input.GetKey(KeyCode.W) || Input.GetKey(KeyCode.UpArrow);
353 bool moveLeft = Input.GetKey(KeyCode.A) || Input.GetKey(KeyCode.LeftArrow);
354 bool moveRight = Input.GetKey(KeyCode.D) || Input.GetKey(KeyCode.RightArrow);
355 bool moveBack = Input.GetKey(KeyCode.S) || Input.GetKey(KeyCode.DownArrow);
356
357 bool dpad_move = false;
358
359 if (OVRInput.Get(OVRInput.Button.DpadUp))
360 {
361 moveForward = true;
362 dpad_move = true;
363 }
364
365 if (OVRInput.Get(OVRInput.Button.DpadDown))
366 {
367 moveBack = true;
368 dpad_move = true;
369 }
370
371 MoveScale = 1.0f;
372
373 if ((moveForward && moveLeft) || (moveForward && moveRight) ||
374 (moveBack && moveLeft) || (moveBack && moveRight))
375 MoveScale = 0.70710678f;
376
377 // No positional movement if we are in the air
378 if (!Controller.isGrounded)
379 MoveScale = 0.0f;
380
381 MoveScale *= SimulationRate * Time.deltaTime;
382
383 // Compute this for key movement
384 float moveInfluence = Acceleration * 0.1f * MoveScale * MoveScaleMultiplier;
385
386 // Run!
387 if (dpad_move || Input.GetKey(KeyCode.LeftShift) || Input.GetKey(KeyCode.RightShift))
388 moveInfluence *= 2.0f;
389
390 Quaternion ort = transform.rotation;
391 Vector3 ortEuler = ort.eulerAngles;
392 ortEuler.z = ortEuler.x = 0f;
393 ort = Quaternion.Euler(ortEuler);
394
395 if (moveForward)
396 MoveThrottle += ort * (transform.lossyScale.z * moveInfluence * Vector3.forward);
397 if (moveBack)
398 MoveThrottle += ort * (transform.lossyScale.z * moveInfluence * BackAndSideDampen * Vector3.back);
399 if (moveLeft)
400 MoveThrottle += ort * (transform.lossyScale.x * moveInfluence * BackAndSideDampen * Vector3.left);
401 if (moveRight)
402 MoveThrottle += ort * (transform.lossyScale.x * moveInfluence * BackAndSideDampen * Vector3.right);
403
404 moveInfluence = Acceleration * 0.1f * MoveScale * MoveScaleMultiplier;
405
406#if !UNITY_ANDROID // LeftTrigger not avail on Android game pad
407 moveInfluence *= 1.0f + OVRInput.Get(OVRInput.Axis1D.PrimaryIndexTrigger);
408#endif
409
410 Vector2 primaryAxis = OVRInput.Get(OVRInput.Axis2D.PrimaryThumbstick);
411
412 // If speed quantization is enabled, adjust the input to the number of fixed speed steps.
413 if (FixedSpeedSteps > 0)
414 {
415 primaryAxis.y = Mathf.Round(primaryAxis.y * FixedSpeedSteps) / FixedSpeedSteps;
416 primaryAxis.x = Mathf.Round(primaryAxis.x * FixedSpeedSteps) / FixedSpeedSteps;
417 }
418
419 if (primaryAxis.y > 0.0f)
420 MoveThrottle += ort * (primaryAxis.y * transform.lossyScale.z * moveInfluence * Vector3.forward);
421
422 if (primaryAxis.y < 0.0f)
423 MoveThrottle += ort * (Mathf.Abs(primaryAxis.y) * transform.lossyScale.z * moveInfluence *
424 BackAndSideDampen * Vector3.back);
425
426 if (primaryAxis.x < 0.0f)
427 MoveThrottle += ort * (Mathf.Abs(primaryAxis.x) * transform.lossyScale.x * moveInfluence *
428 BackAndSideDampen * Vector3.left);
429
430 if (primaryAxis.x > 0.0f)
431 MoveThrottle += ort * (primaryAxis.x * transform.lossyScale.x * moveInfluence * BackAndSideDampen *
432 Vector3.right);
433 }
434
435 if (EnableRotation)
436 {
437 Vector3 euler = RotateAroundGuardianCenter ? transform.rotation.eulerAngles : Vector3.zero;
438 float rotateInfluence = SimulationRate * Time.deltaTime * RotationAmount * RotationScaleMultiplier;
439
440 bool curHatLeft = OVRInput.Get(OVRInput.Button.PrimaryShoulder);
441
442 if (curHatLeft && !prevHatLeft)
443 euler.y -= RotationRatchet;
444
445 prevHatLeft = curHatLeft;
446
447 bool curHatRight = OVRInput.Get(OVRInput.Button.SecondaryShoulder);
448
449 if (curHatRight && !prevHatRight)
450 euler.y += RotationRatchet;
451
452 prevHatRight = curHatRight;
453
454 euler.y += buttonRotation;
455 buttonRotation = 0f;
456
457
458#if !UNITY_ANDROID || UNITY_EDITOR
459 if (!SkipMouseRotation)
460 euler.y += Input.GetAxis("Mouse X") * rotateInfluence * 3.25f;
461#endif
462
463 if (SnapRotation)
464 {
465 if (OVRInput.Get(OVRInput.Button.SecondaryThumbstickLeft) ||
466 (RotationEitherThumbstick && OVRInput.Get(OVRInput.Button.PrimaryThumbstickLeft)))
467 {
468 if (ReadyToSnapTurn)
469 {
470 euler.y -= RotationRatchet;
471 ReadyToSnapTurn = false;
472 }
473 }
474 else if (OVRInput.Get(OVRInput.Button.SecondaryThumbstickRight) ||
475 (RotationEitherThumbstick && OVRInput.Get(OVRInput.Button.PrimaryThumbstickRight)))
476 {
477 if (ReadyToSnapTurn)
478 {
479 euler.y += RotationRatchet;
480 ReadyToSnapTurn = false;
481 }
482 }
483 else
484 {
485 ReadyToSnapTurn = true;
486 }
487 }
488 else
489 {
490 Vector2 secondaryAxis = OVRInput.Get(OVRInput.Axis2D.SecondaryThumbstick);
491 if (RotationEitherThumbstick)
492 {
493 Vector2 altSecondaryAxis = OVRInput.Get(OVRInput.Axis2D.PrimaryThumbstick);
494 if (secondaryAxis.sqrMagnitude < altSecondaryAxis.sqrMagnitude)
495 {
496 secondaryAxis = altSecondaryAxis;
497 }
498 }
499
500 euler.y += secondaryAxis.x * rotateInfluence;
501 }
502
503 if (RotateAroundGuardianCenter)
504 {
505 transform.rotation = Quaternion.Euler(euler);
506 }
507 else
508 {
509 transform.RotateAround(CameraRig.centerEyeAnchor.position, Vector3.up, euler.y);
510 }
511 }
512#endif
513 }

519 public void UpdateTransform(OVRCameraRig rig)
520 {
521 Transform root = CameraRig.trackingSpace;
522 Transform centerEye = CameraRig.centerEyeAnchor;
523
524 if (HmdRotatesY && !Teleported)
525 {
526 Vector3 prevPos = root.position;
527 Quaternion prevRot = root.rotation;
528
529 transform.rotation = Quaternion.Euler(0.0f, centerEye.rotation.eulerAngles.y, 0.0f);
530
531 root.position = prevPos;
532 root.rotation = prevRot;
533 }
534
535 UpdateController();
536 if (TransformUpdated != null)
537 {
538 TransformUpdated(root);
539 }
