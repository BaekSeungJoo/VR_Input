112 public float visibilityStrength
113 {
114 get
115 {
116 // It's possible there are reasons to show the cursor - such as it hovering over some UI - and reasons to hide
117 // the cursor - such as another input method (e.g. mouse) being used. We take both of these in to account.
118
119
120 float strengthFromShowRequest;
121 if (hideByDefault)
122 {
123 // fade the cursor out with time
124 strengthFromShowRequest = Mathf.Clamp01(1 - (Time.time - lastShowRequestTime) / showTimeoutPeriod);
125 }
126 else
127 {
128 // keep it fully visible
129 strengthFromShowRequest = 1;
130 }
131
132 // Now consider factors requesting pointer to be hidden
133 float strengthFromHideRequest;
134
135 strengthFromHideRequest = (lastHideRequestTime + hideTimeoutPeriod > Time.time)
136 ? (dimOnHideRequest ? 0.1f : 0)
137 : 1;
138
139
140 // Hide requests take priority
141 return Mathf.Min(strengthFromShowRequest, strengthFromHideRequest);
142 }
143 }

172 void Update()
173 {
174 if (rayTransform == null && Camera.main != null)
175 rayTransform = Camera.main.transform;
176
177 // Move the gaze cursor to keep it in the middle of the view
178 transform.position = rayTransform.position + rayTransform.forward * depth;
179
180 // Should we show or hide the gaze cursor?
181 if (visibilityStrength == 0 && !hidden)
182 {
183 Hide();
184 }
185 else if (visibilityStrength > 0 && hidden)
186 {
187 Show();
188 }
189 }

196 public override void SetCursorStartDest(Vector3 _, Vector3 pos, Vector3 normal)
197 {
198 transform.position = pos;
199
200 if (!matchNormalOnPhysicsColliders) normal = rayTransform.forward;
201
202 // Set the rotation to match the normal of the surface it's on.
203 Quaternion newRot = transform.rotation;
204 newRot.SetLookRotation(normal, rayTransform.up);
205 transform.rotation = newRot;
206
207 // record depth so that distance doesn't pop when pointer leaves an object
208 depth = (rayTransform.position - pos).magnitude;
209
210 //set scale based on depth
211 currentScale = depth * depthScaleMultiplier;
212 transform.localScale = new Vector3(currentScale, currentScale, currentScale);
213
214 positionSetsThisFrame++;
215 RequestShow();
216 }

223 void LateUpdate()
224 {
225 // This happens after all Updates so we know that if positionSetsThisFrame is zero then nothing set the position this frame
226 if (positionSetsThisFrame == 0)
227 {
228 // No geometry intersections, so gazing into space. Make the cursor face directly at the camera
229 Quaternion newRot = transform.rotation;
230 newRot.SetLookRotation(rayTransform.forward, rayTransform.up);
231 transform.rotation = newRot;
232 }
233
234 Quaternion iconRotation = gazeIcon.rotation;
235 iconRotation.SetLookRotation(transform.rotation * new Vector3(0, 0, 1));
236 gazeIcon.rotation = iconRotation;
237
238 positionSetsThisFrame = 0;
