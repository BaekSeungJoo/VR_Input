73 void RefreshCrosshair()
74 {
75 if (m_crosshair)
76 {
77 if (isGrabbed) m_crosshair.SetState(GrabbableCrosshair.CrosshairState.Disabled);
78 else if (!InRange) m_crosshair.SetState(GrabbableCrosshair.CrosshairState.Disabled);
79 else
80 m_crosshair.SetState(Targeted
81 ? GrabbableCrosshair.CrosshairState.Targeted
82 : GrabbableCrosshair.CrosshairState.Enabled);
83 }
84
85 if (m_materialColorField != null)
86 {
87 m_renderer.GetPropertyBlock(m_mpb);
88 if (isGrabbed || !InRange)
89 m_mpb.SetColor(m_materialColorField, m_crosshairManager.OutlineColorOutOfRange);
90 else if (Targeted) m_mpb.SetColor(m_materialColorField, m_crosshairManager.OutlineColorHighlighted);
91 else m_mpb.SetColor(m_materialColorField, m_crosshairManager.OutlineColorInRange);
92 m_renderer.SetPropertyBlock(m_mpb);
93 }
