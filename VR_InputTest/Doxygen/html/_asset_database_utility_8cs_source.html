29 // Get all unity objects matching the name
30 public static T[] FindUnityAssets<T>(string filter, bool ignoreAdditional = false)
31 where T : UnityEngine.Object
32 {
33 List<T> results = new List<T>();
34 string[] guids = AssetDatabase.FindAssets(filter);
35 if (guids != null && guids.Length > 0)
36 {
37 foreach (var guid in guids)
38 {
39 string assetPath = AssetDatabase.GUIDToAssetPath(guid);
40 T asset = AssetDatabase.LoadAssetAtPath<T>(assetPath);
41 if (asset != null && !results.Contains(asset))
42 {
43 results.Add(asset);
44 if (ignoreAdditional)
45 {
46 break;
47 }
48 }
49 }
50 }
51 return results.ToArray();
52 }
53
54 // Gets the absolute string path of the asset(s) with the matching name
55 public static string[] FindUnityAssetPath(string filter, bool ignoreAdditional = false)
56 {
57 var relativePaths = AssetDatabase.FindAssets(filter);
58 char d = Path.DirectorySeparatorChar;
59
60 List<string> results = new List<string>();
61 foreach (var relPath in relativePaths)
62 {
63 results.Add($"{Application.dataPath}{d}..{d}{AssetDatabase.GUIDToAssetPath(relPath)}");
64 if (ignoreAdditional)
65 {
66 break;
67 }
68 }
69 return results.ToArray();
70 }
71 }
