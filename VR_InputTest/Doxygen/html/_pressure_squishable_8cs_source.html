56 public float ComputeUseStrength(float strength)
57 {
58 float squishAmount = Mathf.Lerp(1, 1 - _maxSquish, strength);
59 float stretchAmount = Mathf.Lerp(1, 1 + _maxStretch, strength);

61 // Perform a cheap axis squish and stretch effect
