150 private void CreateGhost()

156 Transform relativeTo = _handGrabPose.RelativeTo;
157 HandGhost ghostPrototype = _ghostVisualsProvider.GetHand(_handGrabPose.HandPose.Handedness);

161 Pose relativePose = _handGrabPose.RelativePose;
162 Pose pose = PoseUtils.GlobalPoseScaled(relativeTo, relativePose);

176 private void GhostFollowSurface()
177 {
178 if (_handGhost == null)
179 {
180 return;
181 }
182
183 Pose ghostTargetPose = _handGrabPose.RelativePose;
184
185 if (_handGrabPose.SnapSurface != null)
186 {
187 Vector3 mousePosition = Event.current.mousePosition;
188 Ray ray = HandleUtility.GUIPointToWorldRay(mousePosition);
189 if (_handGrabPose.SnapSurface.CalculateBestPoseAtSurface(ray, out Pose poseInSurface, _relativeTo))
190 {
191 ghostTargetPose = PoseUtils.DeltaScaled(_relativeTo, poseInSurface);
192 }
193 }
194
195 _handGhost.SetRootPose(ghostTargetPose, _relativeTo);
196 Handles.color = EditorConstants.PRIMARY_COLOR_DISABLED;
197 Handles.DrawSolidDisc(_handGhost.transform.position, _handGhost.transform.right, 0.01f);
198 }

200 private void GhostEditFingers()
201 {

209 private void DrawBonesRotator(List<HandJointMap> bones)
210 {
211 bool anyChanged = false;
212 for (int i = 0; i < FingersMetadata.HAND_JOINT_IDS.Length; i++)
213 {
214 bool changed = false;
215 HandJointId joint = FingersMetadata.HAND_JOINT_IDS[i];
216 HandFinger finger = FingersMetadata.JOINT_TO_FINGER[(int)joint];
217
218 if (_handGrabPose.HandPose.FingersFreedom[(int)finger] == JointFreedom.Free)
219 {
220 continue;
221 }
222
223 HandJointMap jointMap = bones.Find(b => b.id == joint);
224 if (jointMap == null)
225 {
226 continue;
227 }
228
229 Transform transform = jointMap.transform;
230 transform.localRotation = jointMap.RotationOffset * _handGrabPose.HandPose.JointRotations[i];
231
232 float scale = GIZMO_SCALE * _handGrabPose.transform.lossyScale.x;
233 Handles.color = EditorConstants.PRIMARY_COLOR;
234 Quaternion entryRotation = transform.rotation;
235 Quaternion rotation = Handles.Disc(entryRotation, transform.position,
236 transform.forward, scale, false, 0);
237 if (rotation != entryRotation)
238 {
239 changed = true;
240 }
241
242 if (FingersMetadata.HAND_JOINT_CAN_SPREAD[i])
243 {
244 Handles.color = EditorConstants.SECONDARY_COLOR;
245 Quaternion curlRotation = rotation;
246 rotation = Handles.Disc(curlRotation, transform.position,
247 transform.up, scale, false, 0);
248 if (rotation != curlRotation)
249 {
250 changed = true;
251 }
252 }
253
254 if (!changed)
255 {
256 continue;
257 }
258
259 transform.rotation = rotation;
260 Undo.RecordObject(_handGrabPose, "Bone Rotation");
261 _handGrabPose.HandPose.JointRotations[i] = jointMap.TrackedRotation;
262 anyChanged = true;
263 }
264
265 if (anyChanged)
266 {
267 EditorUtility.SetDirty(_handGrabPose);
268 }
269 }
270 #endregion
