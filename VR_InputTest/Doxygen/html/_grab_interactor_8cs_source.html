107 protected override GrabInteractable ComputeCandidate()
108 {
109 Vector3 position = Rigidbody.transform.position;
110 GrabInteractable closestInteractable = null;
111 GrabPoseScore bestScore = GrabPoseScore.Max;
112
113 var interactables = GrabInteractable.Registry.List(this);
114 foreach (GrabInteractable interactable in interactables)
115 {
116 Collider[] colliders = interactable.Colliders;
117 GrabPoseScore score = GrabPoseHelper.CollidersScore(position, interactable.Colliders, out Vector3 hit);
118 if (score.IsBetterThan(bestScore))
119 {
120 bestScore = score;
121 closestInteractable = interactable;
122 }
123 }
124
125 return closestInteractable;
126 }

230 protected override void DoSelectUpdate()
231 {
232 GrabInteractable interactable = _selectedInteractable;
233 if (interactable == null)
234 {
235 return;
236 }
237
238 _tween.UpdateTarget(_grabTarget.GetPose());
239 _tween.Tick();
240
241 _outsideReleaseDist = false;
242 if (interactable.ReleaseDistance > 0.0f)
243 {
244 float closestSqrDist = float.MaxValue;
245 Collider[] colliders = interactable.Colliders;
246 foreach (Collider collider in colliders)
247 {
248 float sqrDistanceFromCenter =
249 (collider.bounds.center - Rigidbody.transform.position).sqrMagnitude;
250 closestSqrDist = Mathf.Min(closestSqrDist, sqrDistanceFromCenter);
251 }
252
253 float sqrReleaseDistance = interactable.ReleaseDistance * interactable.ReleaseDistance;
254
255 if (closestSqrDist > sqrReleaseDistance)
256 {
257 _outsideReleaseDist = true;
258 }
259 }
