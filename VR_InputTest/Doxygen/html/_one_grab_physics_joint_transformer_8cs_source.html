102 public void BeginTransform()
103 {
104 Vector3 grabPointPosition = _grabbable.GrabPoints[0].position;
105 Quaternion grabPointRotation = _grabbable.GrabPoints[0].rotation;
106
107 _grabbingRigidbody = GetGrabRigidbody();
108 _grabbingRigidbody.transform.SetPositionAndRotation(grabPointPosition, grabPointRotation);

112 public void UpdateTransform()
113 {
114 Pose grabPoint = _grabbable.GrabPoints[0];
115 _targetPosition = grabPoint.position;
116 _targetRotation = grabPoint.rotation;

119 {
120 _grabbingRigidbody.transform.SetPositionAndRotation(_targetPosition, _targetRotation);
121 }
122 }

129 _grabbingRigidbody.MoveRotation(_targetRotation);

146 private Joint AddJoint(Rigidbody rigidbody)
147 {
148 RemoveCurrentJoint();
149
150 Joint joint;
151 if (_customJoint != null)
152 {
153 joint = CloneJoint(_customJoint, this.gameObject);
154 }
155 else
156 {
157 joint = CreateDefaultJoint();
158 }
159
160 joint.connectedBody = rigidbody;
161 joint.autoConfigureConnectedAnchor = false;
162 joint.anchor = joint.transform.InverseTransformPoint(rigidbody.transform.position);
163 joint.connectedAnchor = Vector3.zero;
164 return joint;
165 }

172 private void RemoveCurrentJoint()

185 private Rigidbody GetGrabRigidbody()
186 {
187 Rigidbody rigidbody = _cachedGrabbingRigidbodies.Find(rb => !rb.gameObject.activeSelf);
188 if (rigidbody == null)
189 {
190 rigidbody = CreateRigidBody();
191 _cachedGrabbingRigidbodies.Add(rigidbody);
192 }
193 rigidbody.gameObject.SetActive(true);
194 rigidbody.isKinematic = _isKinematicGrab;
195 return rigidbody;
