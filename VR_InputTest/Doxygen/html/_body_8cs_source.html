49 public bool GetJointPose(BodyJointId bodyJointId, out Pose pose)
50 {
51 pose = Pose.identity;
52 if (!IsTrackedDataValid || !SkeletonMapping.Joints.Contains(bodyJointId))
53 {
54 return false;

57 CheckJointPosesCacheUpdate();

101 private void InitializeJointPosesCache()
102 {
103 if (_jointPosesCache == null)
104 {
105 _jointPosesCache = new BodyJointsCache();
106 }
107 }
108
109 private void CheckJointPosesCacheUpdate()
110 {
111 if (_jointPosesCache != null
112 && CurrentDataVersion != _jointPosesCache.LocalDataVersion)
113 {
114 _jointPosesCache.Update(GetData(), CurrentDataVersion, _trackingSpace);
115 }
116 }
