422 public float GetDistanceToKeyboard(Vector3 point)
423 {
424 if (keyboardBoundingBox_ == null)
425 {
426 return Mathf.Infinity;
427 }
428
429 if (keyboardBoundingBox_.bounds.Contains(point))
430 {
431 return 0.0f;
432 }
433
434 var closestPointToKb = keyboardBoundingBox_.ClosestPointOnBounds(point);
435 var pointToKeyboard = closestPointToKb - point;
436 RaycastHit hitInfo;
437 bool didHit = keyboardBoundingBox_.Raycast(
438 new Ray(point, pointToKeyboard),
439 out hitInfo,
440 Mathf.Infinity);
441 return didHit ? hitInfo.distance : Mathf.Infinity;
442 }

470 private IEnumerator UpdateTrackingStateCoroutine()
471 {
472 for (; ; )
473 {
474 // On Link this is called before initialization.
475 //We don't want this on our normal flow because it breaks our tests.
476#if !UNITY_ANDROID && !UNITY_EDITOR
477 if (OVRPlugin.initialized)
478 {
479#endif
480 OVRKeyboard.TrackedKeyboardInfo keyboardInfo;
481 if (OVRKeyboard.GetSystemKeyboardInfo(KeyboardQueryFlags, out keyboardInfo))
482 {
483 bool systemKeyboardSwitched = false;
484 if (SystemKeyboardInfo.Identifier != keyboardInfo.Identifier ||
485 SystemKeyboardInfo.KeyboardFlags != keyboardInfo.KeyboardFlags)
486 {
487 Debug.Log(String.Format("New System keyboard info: [{0}] {1} (Flags {2}) ({3} {4})",
488 keyboardInfo.Identifier, keyboardInfo.Name,
489 keyboardInfo.KeyboardFlags,
490 (keyboardInfo.SupportedPresentationStyles &
491 OVRPlugin.TrackedKeyboardPresentationStyles.Opaque) != 0
492 ? "Supports Opaque"
493 : "",
494 (keyboardInfo.SupportedPresentationStyles &
495 OVRPlugin.TrackedKeyboardPresentationStyles.KeyLabel) != 0
496 ? "Supports Key Label"
497 : ""));
498 if (TrackingState == TrackedKeyboardState.NoTrackableKeyboard)
499 {
500 SetKeyboardState(TrackedKeyboardState.Offline);
501 }
502
503 SystemKeyboardInfo = keyboardInfo;
504 systemKeyboardSwitched = true;
505 }
506
507 bool keyboardExists = (keyboardInfo.KeyboardFlags & OVRPlugin.TrackedKeyboardFlags.Exists) != 0;
508 if ((keyboardExists && trackingEnabled) || showUntracked)
509 {
510 bool localKeyboard = (keyboardInfo.KeyboardFlags & OVRPlugin.TrackedKeyboardFlags.Local) != 0;
511 bool remoteKeyboard = (keyboardInfo.KeyboardFlags & OVRPlugin.TrackedKeyboardFlags.Remote) != 0;
512 bool connectedKeyboard =
513 (keyboardInfo.KeyboardFlags & OVRPlugin.TrackedKeyboardFlags.Connected) != 0;
514 bool shouldBeRunning = remoteKeyboard ||
515 (localKeyboard && (!connectionRequired || connectedKeyboard)) ||
516 showUntracked;
517
518 if (KeyboardTrackerIsRunning() && (systemKeyboardSwitched || !shouldBeRunning))
519 {
520 StopKeyboardTrackingInternal();
521 }
522
523 if (!KeyboardTrackerIsRunning() && shouldBeRunning)
524 {
525 yield return StartKeyboardTrackingCoroutine();
526 }
527 }
528 else
529 {
530 if (KeyboardTrackerIsRunning())
531 {
532 StopKeyboardTrackingInternal();
533 }
534
535 if (!keyboardExists)
536 {
537 SetKeyboardState(TrackedKeyboardState.NoTrackableKeyboard);
538 }
539 }
540 }
541 else
542 {
543 if (KeyboardTrackerIsRunning())
544 {
545 StopKeyboardTrackingInternal();
546 }
547
548 SetKeyboardState(TrackedKeyboardState.ErrorExtensionFailed);
549 }
550
551 SystemKeyboardInfo = keyboardInfo;
552#if !UNITY_ANDROID && !UNITY_EDITOR
553 }
554#endif
555 yield return new WaitForSeconds(.1f);
556 }
557 }

559 private IEnumerator StartKeyboardTrackingCoroutine()
560 {
561 if (KeyboardTrackerIsRunning())
562 {
563 Debug.Log("StartKeyboardTracking(): Keyboard already being tracked");
564 yield break;
565 }
566
567 Assert.IsTrue(
568 !KeyboardTrackerIsRunning()
569 && activeKeyboardMesh_ == null
570 && activeKeyboardRenderers_ == null
571 && updateKeyboardRoutine_ == null,
572 $"State: {TrackingState}, Mesh: {activeKeyboardMesh_}, Coroutine: {updateKeyboardRoutine_}");
573
574 InitializeKeyboardInfo();
575 RegisterPassthroughMeshToSDK();
576
577 Debug.Log("Calling StartKeyboardTracking with id " + SystemKeyboardInfo.Identifier);
578
579 if (!OVRPlugin.StartKeyboardTracking(SystemKeyboardInfo.Identifier))
580 {
581 if (!showUntracked)
582 {
583 Debug.LogWarning("OVRKeyboard.StartKeyboardTracking Failed");
584 SetKeyboardState(TrackedKeyboardState.Error);
585 yield break;
586 }
587 }
588
589 projectedPassthroughRoot.localScale = new Vector3
590 {
591 x = SystemKeyboardInfo.Dimensions.x * underlayScaleMultX_,
592 y = underlayScaleConstY_,
593 z = SystemKeyboardInfo.Dimensions.z * underlayScaleMultZ_
594 };
595
596 currentKeyboardPresentationStyles = SystemKeyboardInfo.SupportedPresentationStyles;
597 ActiveKeyboardInfo = SystemKeyboardInfo;
598 LoadKeyboardMesh();
599 updateKeyboardRoutine_ = StartCoroutine(UpdateKeyboardPose());
600 EWAPosition = null;
601 EWARotation = null;
602
603 TrackedKeyboardActiveChanged?.Invoke(new TrackedKeyboardSetActiveEvent(isEnabled: true));
604 SetKeyboardState(TrackedKeyboardState.StartedNotTracked);
605 }
606
607 private void StopKeyboardTrackingInternal()
608 {
609 if (!KeyboardTrackerIsRunning() || updateKeyboardRoutine_ == null)
610 {
611 SetKeyboardState(TrackedKeyboardState.Offline);
612 return;
613 }
614
615 projectedPassthroughOpaque_.hidden = true;
616 ProjectedPassthroughKeyLabel.hidden = true;
617
618 TrackedKeyboardActiveChanged?.Invoke(new TrackedKeyboardSetActiveEvent(isEnabled: false));
619
620 Debug.Log($"StopKeyboardTracking {ActiveKeyboardInfo.Name}");
621
622 StopCoroutine(updateKeyboardRoutine_);
623 updateKeyboardRoutine_ = null;
624
625 OVRKeyboard.StopKeyboardTracking(ActiveKeyboardInfo);
626 InitializeKeyboardInfo();
627
628 if (activeKeyboardMesh_ != null)
629 {
630 Destroy(activeKeyboardMesh_.gameObject);
631 activeKeyboardMesh_ = null;
632 activeKeyboardRenderers_ = null;
633 keyboardBoundingBox_ = null;
634 }
635
636 untrackedPosition_ = Vector3.zero;
637
638 SetKeyboardState(TrackedKeyboardState.Offline);
639 }

641 private IEnumerator UpdateKeyboardPose()
642 {
643 while (true)
644 {
645 transform.position = cameraRig_.trackingSpace.transform.position;
646 transform.rotation = cameraRig_.trackingSpace.transform.rotation;
647
648 var poseState = OVRKeyboard.GetKeyboardState();
649
650 // Emulate tracking when showUntracked is set
651 if ((!poseState.isPositionValid || !poseState.isPositionTracked) && showUntracked)
652 {
653 poseState.isPositionValid = true;
654 poseState.isPositionTracked = true;
655
656 if (untrackedPosition_ == Vector3.zero && Camera.main != null)
657 {
658 // Start keyboard in a nice position at waist level in front
659 Transform cameraTransform = Camera.main.transform;
660 Vector3 cameraDirectionHorizontal =
661 Vector3.ProjectOnPlane(cameraTransform.forward, Vector3.up).normalized;
662 untrackedPosition_ = cameraTransform.position +
663 cameraDirectionHorizontal * initialHorizontalDistanceKeyboard_ +
664 new Vector3(0.0f, -initialVerticalDistanceKeyboard_, 0.0f);
665 }
666
667 poseState.position = untrackedPosition_;
668 }
669
670 TrackedKeyboardState keyboardState = TrackedKeyboardState.StartedNotTracked;
671
672 if (poseState.isPositionValid)
673 {
674 if (poseState.isPositionTracked && activeKeyboardMesh_ != null)
675 {
676 float keyboardAngleFilter = UseHeuristicRollback ? 360f : 20f;
677 float ewaAlpha = UseHeuristicRollback ? 0f : 0.65f;
678
679 var worldRotation = transform.rotation * poseState.rotation;
680 var upRotated = worldRotation * Vector3.up;
681 CurrentKeyboardAngleFromUp = Vector3.Angle(upRotated, Vector3.up);
682
683 if (CurrentKeyboardAngleFromUp < keyboardAngleFilter)
684 {
685 if (!EWAPosition.HasValue)
686 {
687 EWAPosition = poseState.position;
688 }
689 else
690 {
691 EWAPosition = ewaAlpha * EWAPosition + (1f - ewaAlpha) * poseState.position;
692 }
693
694 if (!EWARotation.HasValue)
695 {
696 EWARotation = poseState.rotation;
697 }
698 else
699 {
700 EWARotation = Quaternion.Slerp(EWARotation.Value, poseState.rotation, 1f - ewaAlpha);
701 }
702
703 ActiveKeyboardTransform.localPosition = EWAPosition.Value;
704 ActiveKeyboardTransform.localRotation = EWARotation.Value;
705
706 projectedPassthroughRoot.localPosition = EWAPosition.Value + underlayOffset_ +
707 new Vector3(0f, HAND_HEIGHT_TUNING, 0f);
708 projectedPassthroughRoot.localRotation = EWARotation.Value;
709 }
710 else
711 {
712 skippedPoseCount_++;
713 }
714 }
715
716 keyboardState = poseState.isPositionTracked
717 ? TrackedKeyboardState.Valid
718 : TrackedKeyboardState.Stale;
719 }
720
721 SetKeyboardState(keyboardState);
722 UpdateSkippedPoseTimer();
723 yield return null;
724 }
725 }

727 private void UpdateSkippedPoseTimer()
728 {
729 sendFilteredPoseEventTimer_ += Time.deltaTime;
730 if (sendFilteredPoseEventTimer_ > FILTERED_POSE_TIMEOUT
731 && skippedPoseCount_ > 0)
732 {
733 // dispatcher_.Dispatch(new TrackedKeyboardSkippedPoseEvent(skippedPoseCount_));
734 skippedPoseCount_ = 0;
735 sendFilteredPoseEventTimer_ = 0f;
736 }

739 private void LoadKeyboardMesh()
740 {
741 Debug.Log("LoadKeyboardMesh");
742 activeKeyboardMesh_ = LoadRuntimeKeyboardMesh();
743 if (activeKeyboardMesh_ == null)
744 {
745 Debug.LogError("Failed to load keyboard mesh.");
746 SetKeyboardState(TrackedKeyboardState.Error);
747 return;
748 }
749
750 keyboardMeshNodes_ = new GameObject[activeKeyboardMesh_.transform.childCount];
751 for (int i = 0; i < activeKeyboardMesh_.transform.childCount; i++)
752 {
753 keyboardMeshNodes_[i] = activeKeyboardMesh_.transform.GetChild(i).gameObject;
754 }
755
756 keyboardBoundingBox_ = activeKeyboardMesh_.AddComponent<BoxCollider>();
757
758 keyboardBoundingBox_.center =
759 new Vector3(0.0f, ActiveKeyboardInfo.Dimensions.y / 2.0f, 0.0f);
760 keyboardBoundingBox_.size =
761 new Vector3(ActiveKeyboardInfo.Dimensions.x,
762 ActiveKeyboardInfo.Dimensions.y + boundingBoxAboveKeyboardY_,
763 ActiveKeyboardInfo.Dimensions.z);
764
765 activeKeyboardMeshRenderer_ = keyboardMeshNodes_[0].GetComponentInChildren<MeshRenderer>();
766 if (activeKeyboardMeshRenderer_ == null)
767 {
768 Debug.LogError("Failed to load activeKeyboardMeshRenderer_.");
769 SetKeyboardState(TrackedKeyboardState.Error);
770 return;
771 }
772
773 opaqueShader_ = activeKeyboardMeshRenderer_.material.shader;
774
775 passthroughQuad_ = GameObject.CreatePrimitive(PrimitiveType.Quad);
776 passthroughQuad_.transform.localPosition = new Vector3(0.0f, -0.01f, 0.0f);
777 passthroughQuad_.transform.parent = activeKeyboardMesh_.transform;
778 passthroughQuad_.transform.localRotation = Quaternion.Euler(90.0f, 0.0f, 0.0f);
779 float borderSize = ActiveKeyboardInfo.Dimensions.x * PassthroughBorderMultiplier;
780 passthroughQuad_.transform.localScale = new Vector3(
781 ActiveKeyboardInfo.Dimensions.x + borderSize,
782 ActiveKeyboardInfo.Dimensions.z + borderSize,
783 ActiveKeyboardInfo.Dimensions.y);
784
785 MeshRenderer meshRenderer = passthroughQuad_.GetComponent<MeshRenderer>();
786 meshRenderer.material.shader = PassthroughShader;
787
788 GameObject parent = new GameObject();
789 activeKeyboardMesh_.transform.parent = parent.transform;
790 activeKeyboardMesh_ = parent;
791
792 activeKeyboardRenderers_ = activeKeyboardMesh_.GetComponentsInChildren<MeshRenderer>();
793 activeKeyboardMesh_.transform.SetParent(ActiveKeyboardTransform, worldPositionStays: false);
794
795 ActiveKeyboardTransform.localRotation = Quaternion.identity;
796
797 // Make a copy of the current main texture (created by the OVRGLTFLoader) to apply our quality setting more freely
798 Texture readonlyTexture = activeKeyboardMeshRenderer_.material.mainTexture;
799 if (readonlyTexture != null)
800 {
801 dynamicQualityTexture_ = Texture2D.CreateExternalTexture(
802 readonlyTexture.width,
803 readonlyTexture.height,
804 TextureFormat.BC7,
805 mipChain: true, linear: true,
806 readonlyTexture.GetNativeTexturePtr());
807 }
808
809 UpdateTextureQuality();
810 UpdateKeyboardVisibility();
811 }

829 void UpdatePresentation(bool isVisible)
830 {
831 KeyboardPresentation presentationToUse = Presentation;
832 if (currentKeyboardPresentationStyles != 0)
833 {
834 if (Presentation == KeyboardPresentation.PreferOpaque &&
835 (currentKeyboardPresentationStyles & OVRPlugin.TrackedKeyboardPresentationStyles.Opaque) == 0)
836 {
837 if ((currentKeyboardPresentationStyles & OVRPlugin.TrackedKeyboardPresentationStyles.KeyLabel) != 0)
838 {
839 presentationToUse = KeyboardPresentation.PreferKeyLabels;
840 }
841 }
842 else if (Presentation == KeyboardPresentation.PreferKeyLabels &&
843 (currentKeyboardPresentationStyles & OVRPlugin.TrackedKeyboardPresentationStyles.KeyLabel) == 0)
844 {
845 if ((currentKeyboardPresentationStyles & OVRPlugin.TrackedKeyboardPresentationStyles.Opaque) != 0)
846 {
847 presentationToUse = KeyboardPresentation.PreferOpaque;
848 }
849 }
850 }
851
852 if (!isVisible)
853 {
854 projectedPassthroughOpaque_.hidden = true;
855 ProjectedPassthroughKeyLabel.hidden = true;
856 }
857 else if (presentationToUse == KeyboardPresentation.PreferOpaque)
858 {
859 activeKeyboardMeshRenderer_.material.shader = opaqueShader_;
860 passthroughQuad_.SetActive(false);
861 projectedPassthroughOpaque_.hidden = !GetKeyboardVisibility() || !HandsOverKeyboard;
862 ProjectedPassthroughKeyLabel.hidden = true;
863 for (int i = 1; i < keyboardMeshNodes_.Length; i++)
864 {
865 keyboardMeshNodes_[i].SetActive(true);
866 }
867 }
868 else
869 {
870 activeKeyboardMeshRenderer_.material.shader = KeyLabelModeShader;
871 passthroughQuad_.SetActive(true);
872 projectedPassthroughOpaque_.hidden = true;
873 ProjectedPassthroughKeyLabel.hidden = false; // Always shown
874 for (int i = 1; i < keyboardMeshNodes_.Length; i++)
875 {
876 keyboardMeshNodes_[i].SetActive(false);
877 }
878 }
879 }

881 private GameObject LoadRuntimeKeyboardMesh()
882 {
883 Debug.Log("LoadRuntimekeyboardMesh");
884 string[] modelPaths = OVRPlugin.GetRenderModelPaths();
885 if (modelPaths != null)
886 {
887 for (int i = 0; i < modelPaths.Length; i++)
888 {
889 if ((RemoteKeyboard && modelPaths[i].Equals("/model_fb/keyboard/remote")) ||
890 (!RemoteKeyboard && modelPaths[i].Equals("/model_fb/keyboard/local")))
