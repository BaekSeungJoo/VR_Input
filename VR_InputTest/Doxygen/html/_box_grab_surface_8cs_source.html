195 private void CalculateCorners(out Vector3 bottomLeft, out Vector3 bottomRight, out Vector3 topLeft, out Vector3 topRight,
196 Transform relativeTo)
197 {
198 Pose referencePose = GetReferencePose(relativeTo);
199 Vector3 size = GetSize(relativeTo);
200 float widthOffset = GetWidthOffset(relativeTo);
201 Vector3 rightRot = GetRotation(relativeTo) * Vector3.right;
202 bottomLeft = referencePose.position - rightRot * size.x * (1f - widthOffset);
203 bottomRight = referencePose.position + rightRot * size.x * (widthOffset);
204 Vector3 forwardOffset = GetRotation(relativeTo) * Vector3.forward * size.z;
205 topLeft = bottomLeft + forwardOffset;
206 topRight = bottomRight + forwardOffset;
207 }

209 private Vector3 ProjectOnSegment(Vector3 point, (Vector3, Vector3) segment)
210 {
211 Vector3 line = segment.Item2 - segment.Item1;
212 Vector3 projection = Vector3.Project(point - segment.Item1, line);
213 if (Vector3.Dot(projection, line) < 0f)
214 {
215 projection = segment.Item1;
216 }
217 else if (projection.magnitude > line.magnitude)
218 {
219 projection = segment.Item2;
220 }
221 else
222 {
223 projection += segment.Item1;
224 }
225 return projection;
226 }

228 public bool CalculateBestPoseAtSurface(Ray targetRay, out Pose bestPose, Transform relativeTo)
229 {
230 Pose recordedPose = GetReferencePose(relativeTo);
231 Plane plane = new Plane(GetRotation(relativeTo) * Vector3.up, this.transform.position);
232 plane.Raycast(targetRay, out float rayDistance);
233 Vector3 proximalPoint = targetRay.origin + targetRay.direction * rayDistance;
234
235 Vector3 surfacePoint = NearestPointInSurface(proximalPoint, relativeTo);
236 Pose desiredPose = new Pose(surfacePoint, recordedPose.rotation);
237 bestPose = MinimalTranslationPoseAtSurface(desiredPose, relativeTo);
238 return true;
239 }

247 private void NearestPointAndAngleInSurface(Vector3 targetPosition, out Vector3 surfacePoint, out float angle, Transform relativeTo)
248 {
249 Quaternion rotation = GetRotation(relativeTo);
250 Vector4 snappOffset = GetSnapOffset(relativeTo);
251 Vector3 rightDir = rotation * Vector3.right;
252 Vector3 forwardDir = rotation * Vector3.forward;
253 Vector3 bottomLeft, bottomRight, topLeft, topRight;
254 CalculateCorners(out bottomLeft, out bottomRight, out topLeft, out topRight, relativeTo);
255
256 Vector3 bottomP = ProjectOnSegment(targetPosition,
257 (bottomLeft + rightDir * snappOffset.y, bottomRight + rightDir * snappOffset.x));
258 Vector3 topP = ProjectOnSegment(targetPosition,
259 (topLeft - rightDir * snappOffset.x, topRight - rightDir * snappOffset.y));
260 Vector3 leftP = ProjectOnSegment(targetPosition,
261 (bottomLeft - forwardDir * snappOffset.z, topLeft - forwardDir * snappOffset.w));
262 Vector3 rightP = ProjectOnSegment(targetPosition,
263 (bottomRight + forwardDir * snappOffset.w, topRight + forwardDir * snappOffset.z));
264
265 float bottomDistance = (bottomP - targetPosition).sqrMagnitude;
266 float topDistance = (topP - targetPosition).sqrMagnitude;
267 float leftDistance = (leftP - targetPosition).sqrMagnitude;
268 float rightDistance = (rightP - targetPosition).sqrMagnitude;
269
270 float minDistance = Mathf.Min(bottomDistance,
271 Mathf.Min(topDistance,
272 Mathf.Min(leftDistance, rightDistance)));
273 if (bottomDistance == minDistance)
274 {
275 surfacePoint = bottomP;
276 angle = 0f;
277 return;
278 }
279 if (topDistance == minDistance)
280 {
281 surfacePoint = topP;
282 angle = 180f;
283 return;
284 }
285 if (leftDistance == minDistance)
286 {
287 surfacePoint = leftP;
288 angle = 90f;
289 return;
290 }
291 surfacePoint = rightP;
292 angle = -90f;
293 }

295 protected Pose MinimalRotationPoseAtSurface(in Pose userPose, Transform relativeTo)
296 {
297 Quaternion rotation = GetRotation(relativeTo);
298 Pose referencePose = GetReferencePose(relativeTo);
299 Vector4 snappOffset = GetSnapOffset(relativeTo);
300 Vector3 desiredPos = userPose.position;
301 Quaternion baseRot = referencePose.rotation;
302 Quaternion desiredRot = userPose.rotation;
303 Vector3 up = rotation * Vector3.up;
304
305 Quaternion bottomRot = baseRot;
306 Quaternion topRot = Quaternion.AngleAxis(180f, up) * baseRot;
307 Quaternion leftRot = Quaternion.AngleAxis(90f, up) * baseRot;
308 Quaternion rightRot = Quaternion.AngleAxis(-90f, up) * baseRot;
309
310 float bottomDot = RotationalScore(bottomRot, desiredRot);
311 float topDot = RotationalScore(topRot, desiredRot);
312 float leftDot = RotationalScore(leftRot, desiredRot);
313 float rightDot = RotationalScore(rightRot, desiredRot);
314
315 Vector3 rightDir = rotation * Vector3.right;
316 Vector3 forwardDir = rotation * Vector3.forward;
317 Vector3 bottomLeft, bottomRight, topLeft, topRight;
318 CalculateCorners(out bottomLeft, out bottomRight, out topLeft, out topRight, relativeTo);
319
320 float maxDot = Mathf.Max(bottomDot, Mathf.Max(topDot, Mathf.Max(leftDot, rightDot)));
321 if (bottomDot == maxDot)
322 {
323 Vector3 projBottom = ProjectOnSegment(desiredPos, (bottomLeft + rightDir * snappOffset.y, bottomRight + rightDir * snappOffset.x));
324 return new Pose(projBottom, bottomRot);
325 }
326 if (topDot == maxDot)
327 {
328 Vector3 projTop = ProjectOnSegment(desiredPos, (topLeft - rightDir * snappOffset.x, topRight - rightDir * snappOffset.y));
329 return new Pose(projTop, topRot);
330 }
331 if (leftDot == maxDot)
332 {
333 Vector3 projLeft = ProjectOnSegment(desiredPos, (bottomLeft - forwardDir * snappOffset.z, topLeft - forwardDir * snappOffset.w));
334 return new Pose(projLeft, leftRot);
335 }
336 Vector3 projRight = ProjectOnSegment(desiredPos, (bottomRight + forwardDir * snappOffset.w, topRight + forwardDir * snappOffset.z));
337 return new Pose(projRight, rightRot);
338 }

340 protected Pose MinimalTranslationPoseAtSurface(in Pose userPose, Transform relativeTo)
341 {
342 Pose referencePose = GetReferencePose(relativeTo);
343 Quaternion rotation = GetRotation(relativeTo);
344 Vector3 desiredPos = userPose.position;
345 Quaternion baseRot = referencePose.rotation;
346 NearestPointAndAngleInSurface(desiredPos, out Vector3 surfacePoint, out float surfaceAngle, relativeTo);
347 Quaternion surfaceRotation = Quaternion.AngleAxis(surfaceAngle, rotation * Vector3.up) * baseRot;
348 return new Pose(surfacePoint, surfaceRotation);
349 }
350
351 private static float RotationalScore(in Quaternion from, in Quaternion to)
352 {
353 float forwardDifference = Vector3.Dot(from * Vector3.forward, to * Vector3.forward) * 0.5f + 0.5f;
354 float upDifference = Vector3.Dot(from * Vector3.up, to * Vector3.up) * 0.5f + 0.5f;
355 return (forwardDifference * upDifference);
356 }
