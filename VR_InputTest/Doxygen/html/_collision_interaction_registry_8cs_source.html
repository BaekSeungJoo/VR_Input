32 public class CollisionInteractionRegistry<TInteractor, TInteractable> :
33 InteractableRegistry<TInteractor, TInteractable>
34 where TInteractor : Interactor<TInteractor, TInteractable>, IRigidbodyRef
35 where TInteractable : Interactable<TInteractor, TInteractable>, IRigidbodyRef
36 {
37 private Dictionary<Rigidbody, HashSet<TInteractable>> _rigidbodyCollisionMap;
38 private Dictionary<TInteractable, InteractableTriggerBroadcaster> _broadcasters;
39
40 public CollisionInteractionRegistry() : base()
41 {
42 _rigidbodyCollisionMap = new Dictionary<Rigidbody, HashSet<TInteractable>>();
43 _broadcasters = new Dictionary<TInteractable, InteractableTriggerBroadcaster>();
44 }
45
46 public override void Register(TInteractable interactable)
47 {
48 base.Register(interactable);
49
50 GameObject triggerGameObject = interactable.Rigidbody.gameObject;
51 InteractableTriggerBroadcaster broadcaster;
52 if (!_broadcasters.TryGetValue(interactable, out broadcaster))
53 {
54 broadcaster = triggerGameObject.AddComponent<InteractableTriggerBroadcaster>();
55 broadcaster.InjectAllInteractableTriggerBroadcaster(interactable);
56 _broadcasters.Add(interactable, broadcaster);
57 broadcaster.WhenTriggerEntered += HandleTriggerEntered;
58 broadcaster.WhenTriggerExited += HandleTriggerExited;
59 }
60 }
61
62 public override void Unregister(TInteractable interactable)
63 {
64 base.Unregister(interactable);
65
66 InteractableTriggerBroadcaster broadcaster;
67 if (_broadcasters.TryGetValue(interactable, out broadcaster))
68 {
69 _broadcasters.Remove(interactable);
70
71 if (broadcaster != null)
72 {
73 broadcaster.enabled = false;
74 broadcaster.WhenTriggerEntered -= HandleTriggerEntered;
75 broadcaster.WhenTriggerExited -= HandleTriggerExited;
76 Object.Destroy(broadcaster);
77 }
78 }
79 }
80
81 private void HandleTriggerEntered(IInteractable interactable, Rigidbody rigidbody)
82 {
83 TInteractable typedInteractable = interactable as TInteractable;
84 if (!_rigidbodyCollisionMap.ContainsKey(rigidbody))
85 {
86 _rigidbodyCollisionMap.Add(rigidbody, new HashSet<TInteractable>());
87 }
88
89 HashSet<TInteractable> interactables = _rigidbodyCollisionMap[rigidbody];
90 interactables.Add(typedInteractable);
91 }
92
93 private void HandleTriggerExited(IInteractable interactable, Rigidbody rigidbody)
94 {
95 TInteractable typedInteractable = interactable as TInteractable;
96 HashSet<TInteractable> interactables = _rigidbodyCollisionMap[rigidbody];
97 interactables.Remove(typedInteractable);
98
99 if (interactables.Count == 0)
100 {
101 _rigidbodyCollisionMap.Remove(rigidbody);
102 }
103 }
104
105 public override InteractableSet List(TInteractor interactor)
106 {
107 HashSet<TInteractable> colliding;
108 if (_rigidbodyCollisionMap.TryGetValue(interactor.Rigidbody, out colliding))
109 {
110 return List(interactor, colliding);
111 }
112 return _empty;
113 }
114
115 private static readonly InteractableSet _empty = new InteractableSet();
116 }
