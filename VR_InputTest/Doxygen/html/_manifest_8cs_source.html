80 public bool ResolveEntities()
81 {
82 bool allResolved = true;
83 foreach (var entity in Entities)
84 {
85 var typeName = string.IsNullOrEmpty(entity.Namespace) ? entity.ID : $"{entity.Namespace}.{entity.ID}";
86
87 var qualifiedTypeName = $"{typeName},{entity.Assembly}";
88 var type = Type.GetType(qualifiedTypeName);
89 if (type == null)
90 {
91 VLog.E($"Failed to resolve type: {qualifiedTypeName}");
92 allResolved = false;
93 }
94 CustomEntityTypes[entity.Name] = type;
95 }
96
97 return allResolved;

100 public Tuple<MethodInfo, Type> GetMethodInfo(IManifestMethod action)
101 {
102 var lastPeriod = action.ID.LastIndexOf('.');
103 if (lastPeriod <= 0)
104 {
105 VLog.E($"Invalid Action ID: {action.ID}");
106 return null;
107 }
108
109 var typeName = action.ID.Substring(0, lastPeriod);
110 var qualifiedTypeName = $"{typeName},{action.Assembly}";
111 var method = action.ID.Substring(lastPeriod + 1);
112
113 var targetType = Type.GetType(qualifiedTypeName);
114 if (targetType == null)
115 {
116 VLog.E($"Failed to resolve type: {qualifiedTypeName}");
117 return null;
118 }
119
120 var types = new Type[action.Parameters.Count];
121 for (var i = 0; i < action.Parameters.Count; i++)
122 {
123 var manifestParameter = action.Parameters[i];
124 var fullTypeName = $"{manifestParameter.QualifiedTypeName},{manifestParameter.TypeAssembly}";
125 types[i] = Type.GetType(fullTypeName);
126 if (types[i] == null)
127 {
128 VLog.E($"Failed to resolve type: {fullTypeName}");
129 }
130 }
131
132 var targetMethod = GetBestMethodMatch(targetType, method, types);
133 if (targetMethod == null)
134 {
135 VLog.E($"Failed to resolve method {typeName}.{method}.");
136 return null;
137 }
138
139 return Tuple.Create( targetMethod, targetType);
140 }

143 private bool ResolveAllActions()
144 {
145 var resolvedAll = true;
146 foreach (var action in this.Actions)
147 {
148 var methodInfo = GetMethodInfo(action);
149 if (methodInfo == null)
150 {
151 return false;
152 }
153 var targetMethod = methodInfo.Item1;
154 var targetType = methodInfo.Item2;
155 if (targetMethod == null)
156 {
157 VLog.E($"Invalid Action ID: {action.ID}");
158 resolvedAll = false;
159 continue;
160 }
161
162 var attributes = targetMethod.GetCustomAttributes(typeof(ConduitActionAttribute), false);
163 if (attributes.Length == 0)
164 {
165 VLog.E($"{targetMethod} - Did not have expected Conduit attribute");
166 resolvedAll = false;
167 continue;
168 }
169 var actionAttribute = attributes.First() as ConduitActionAttribute;
170
171 var invocationContext = new InvocationContext()
172 {
173 Type = targetType,
174 MethodInfo = targetMethod,
175 MinConfidence = actionAttribute.MinConfidence,
176 MaxConfidence = actionAttribute.MaxConfidence,
177 ValidatePartial = actionAttribute.ValidatePartial,
178 };
179
180 if (!_methodLookup.ContainsKey(action.Name))
181 {
182 _methodLookup.Add(action.Name, new List<InvocationContext>());
183 }
184
185 _methodLookup[action.Name].Add(invocationContext);
186 }
187
188 foreach (var invocationContext in _methodLookup.Values.Where(invocationContext =>
189 invocationContext.Count > 1))
190 {
191 // This is a slow operation. If there multiple overloads are common, we should optimize this
192 invocationContext.Sort((one, two) =>
193 two.MethodInfo.GetParameters().Length - one.MethodInfo.GetParameters().Length);
194 }
195
196 return resolvedAll;
197 }

199 private bool ResolveErrorHandlers()
200 {
201 if (this.ErrorHandlers == null)
202 {
203 return true;
204 }
205
206 var resolvedAll = true;
207 foreach (var action in this.ErrorHandlers)
208 {
209 var methodInfo = GetMethodInfo(action);
210 var targetMethod = methodInfo.Item1;
211 var targetType = methodInfo.Item2;
212 if (targetMethod == null)
213 {
214 VLog.E($"Invalid Action ID: {action.ID}");
215 resolvedAll = false;
216 continue;
217 }
218
219 var attributes = targetMethod.GetCustomAttributes(typeof(HandleEntityResolutionFailure), false);
220 if (attributes.Length == 0)
221 {
222 VLog.E($"{targetMethod} - Did not have expected Conduit attribute");
223 resolvedAll = false;
224 continue;
225 }
226 var actionAttribute = attributes.First() as HandleEntityResolutionFailure;
227 if (actionAttribute == null)
228 {
229 VLog.E("Found null attribute when one was expected");
230 continue;
231 }
232
233 var invocationContext = new InvocationContext()
234 {
235 Type = targetType,
236 MethodInfo = targetMethod,
237 CustomAttributeType = typeof(HandleEntityResolutionFailure)
238 };
239
240 if (!_methodLookup.ContainsKey(action.Name))
241 {
242 _methodLookup.Add(action.Name, new List<InvocationContext>());
243 }
244
245 _methodLookup[action.Name].Add(invocationContext);
246 }
247
248 foreach (var invocationContext in _methodLookup.Values.Where(invocationContext =>
249 invocationContext.Count > 1))
250 {
251 // This is a slow operation. If there multiple overloads are common, we should optimize this
252 invocationContext.Sort((one, two) =>
253 two.MethodInfo.GetParameters().Length - one.MethodInfo.GetParameters().Length);
254 }
255
256 return resolvedAll;
