118 public virtual void UpdateTeleportDestination(bool isValidDestination, Vector3? position, Quaternion? rotation,
119 Quaternion? landingRotation)
120 {
121 //Debug.Log("Teleport Destination: "+ position + " " + rotation);
122
123 IsValidDestination = isValidDestination;
124
125 LandingRotation = landingRotation.GetValueOrDefault();
126
127 // Show/hide the position indicator based on there being a valid position for it.
128 var go = PositionIndicator.gameObject;
129 var goActive = go.activeInHierarchy;
130 if (!position.HasValue)
131 {
132 if (goActive)
133 {
134 go.SetActive(false);
135 }
136
137 return;
138 }
139
140 if (!goActive)
141 {
142 go.SetActive(true);
143 }
144
145 transform.position = position.GetValueOrDefault();
146
147 // If there is no orientation indicator specified, rotate the entire indicator to match any provided orientation.
148 if (OrientationIndicator == null)
149 {
150 if (rotation.HasValue)
151 {
152 transform.rotation = rotation.GetValueOrDefault();
153 }
154
155 return;
156 }
157
158 // Rotate the orientation indicator to match any valid rotation and make it visible,
159 // otherwise hide it.
160 var orientationGO = OrientationIndicator.gameObject;
161 var ogoActive = orientationGO.activeInHierarchy;
162 if (!rotation.HasValue)
163 {
164 if (ogoActive)
165 {
166 orientationGO.SetActive(false);
167 }
168
169 return;
170 }
171
172 OrientationIndicator.rotation = rotation.GetValueOrDefault();
173 if (!ogoActive)
174 {
175 orientationGO.SetActive(true);
176 }
177 }
