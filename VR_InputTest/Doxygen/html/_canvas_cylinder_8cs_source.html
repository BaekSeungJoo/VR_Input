119 protected override void GenerateMesh(out List<Vector3> verts,
120 out List<int> tris,
121 out List<Vector2> uvs)
122 {
123 verts = new List<Vector3>();
124 tris = new List<int>();
125 uvs = new List<Vector2>();
126
127 Vector2 worldSize = GetWorldSize();
128 float scaledRadius = Radius * CylinderRelativeScale;
129
130 float xPos = worldSize.x * 0.5f;
131 float xNeg = -xPos;
132 float yPos = worldSize.y * 0.5f;
133 float yNeg = -yPos;
134
135 Vector2Int GetClampedResolution(float arcMax, float axisMax)
136 {
137 int horizontalResolution = Mathf.Max(2,
138 Mathf.RoundToInt(_meshGeneration.VerticesPerDegree *
139 Mathf.Rad2Deg * arcMax / scaledRadius));
140 int verticalResolution =
141 Mathf.Max(2, Mathf.RoundToInt(horizontalResolution * axisMax / arcMax));
142
143 horizontalResolution = Mathf.Clamp(horizontalResolution, 2,
144 _meshGeneration.MaxHorizontalResolution);
145 verticalResolution = Mathf.Clamp(verticalResolution, 2,
146 _meshGeneration.MaxVerticalResolution);
147
148 return new Vector2Int(horizontalResolution, verticalResolution);
149 }
150
151 Vector3 GetCurvedPoint(float u, float v)
152 {
153 float x = Mathf.Lerp(xNeg, xPos, u);
154 float y = Mathf.Lerp(yNeg, yPos, v);
155
156 float angle;
157 Vector3 point;
158
159 switch (_orientation)
160 {
161 default:
162 case CylinderOrientation.Vertical:
163 angle = x / scaledRadius;
164 point.x = Mathf.Sin(angle) * scaledRadius;
165 point.y = y;
166 point.z = Mathf.Cos(angle) * scaledRadius - scaledRadius;
167 break;
168 case CylinderOrientation.Horizontal:
169 angle = y / scaledRadius;
170 point.x = x;
171 point.y = Mathf.Sin(angle) * scaledRadius;
172 point.z = Mathf.Cos(angle) * scaledRadius - scaledRadius;
173 break;
174 }
175 return point;
176 }
177
178 Vector2Int resolution;
179 switch (_orientation)
180 {
181 default:
182 case CylinderOrientation.Vertical:
183 resolution = GetClampedResolution(xPos, yPos);
184 break;
185 case CylinderOrientation.Horizontal:
186 resolution = GetClampedResolution(yPos, xPos);
187 break;
188 }
189
190 for (int y = 0; y < resolution.y; y++)
191 {
192 for (int x = 0; x < resolution.x; x++)
193 {
194 float u = x / (resolution.x - 1.0f);
195 float v = y / (resolution.y - 1.0f);
196
197 verts.Add(GetCurvedPoint(u, v));
198 uvs.Add(new Vector2(u, v));
199 }
200 }
201
202 for (int y = 0; y < resolution.y - 1; y++)
203 {
204 for (int x = 0; x < resolution.x - 1; x++)
205 {
206 int v00 = x + y * resolution.x;
207 int v10 = v00 + 1;
208 int v01 = v00 + resolution.x;
209 int v11 = v00 + 1 + resolution.x;
210
211 tris.Add(v00);
212 tris.Add(v11);
213 tris.Add(v10);
214
215 tris.Add(v00);
216 tris.Add(v01);
217 tris.Add(v11);
218 }
219 }
220 }

222 private void UpdateMeshPosition()
223 {
224 Vector3 posInCylinder = _cylinder.transform.InverseTransformPoint(transform.position);
225
226 Vector3 localYOffset = new Vector3(0, posInCylinder.y, 0);
227 Vector3 localCancelY = posInCylinder - localYOffset;
228
229 // If canvas position is on cylinder center axis, project forward.
230 // Otherwise, project canvas onto cylinder wall from center axis.
231 Vector3 projection = Mathf.Approximately(localCancelY.sqrMagnitude, 0f) ?
232 Vector3.forward : localCancelY.normalized;
233
234 Vector3 localUp;
235 switch (_orientation)
236 {
237 default:
238 case CylinderOrientation.Vertical:
239 localUp = Vector3.up;
240 break;
241 case CylinderOrientation.Horizontal:
242 localUp = Vector3.right;
243 break;
244 }
245
246 transform.position = _cylinder.transform.TransformPoint((projection * _cylinder.Radius) + localYOffset);
247 transform.rotation = _cylinder.transform.rotation * Quaternion.LookRotation(projection, localUp);
248
249 if (_meshCollider != null &&
250 _meshCollider.transform != transform &&
251 !transform.IsChildOf(_meshCollider.transform))
252 {
253 _meshCollider.transform.position = transform.position;
254 _meshCollider.transform.rotation = transform.rotation;
255 _meshCollider.transform.localScale *= transform.lossyScale.x / _meshCollider.transform.lossyScale.x;
256 }
257 }

259 private Vector2 GetWorldSize()
260 {
261 Vector2Int resolution = _canvasRenderTexture.GetBaseResolutionToUse();
262 float width = _canvasRenderTexture.PixelsToUnits(Mathf.RoundToInt(resolution.x));
263 float height = _canvasRenderTexture.PixelsToUnits(Mathf.RoundToInt(resolution.y));
264 return new Vector2(width, height) / transform.lossyScale;
265 }

267 private void UpdateCurvedPlane()
268 {
269 // Get world size in cylinder space
270 Vector2 cylinderSize = GetWorldSize() / CylinderRelativeScale;
271
272 float arcSize, axisSize;
273 switch (_orientation)
274 {
275 default:
276 case CylinderOrientation.Vertical:
277 arcSize = cylinderSize.x;
278 axisSize = cylinderSize.y;
279 break;
280 case CylinderOrientation.Horizontal:
281 arcSize = cylinderSize.y;
282 axisSize = cylinderSize.x;
283 break;
284 }
285
286 Vector3 posInCylinder = Cylinder.transform.InverseTransformPoint(transform.position);
287 Rotation = Mathf.Atan2(posInCylinder.x, posInCylinder.z) * Mathf.Rad2Deg;
288 ArcDegrees = (arcSize * 0.5f / Radius) * 2f * Mathf.Rad2Deg;
289 Top = posInCylinder.y + (axisSize * 0.5f);
290 Bottom = posInCylinder.y - (axisSize * 0.5f);
291 }
