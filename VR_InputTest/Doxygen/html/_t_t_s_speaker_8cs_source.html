350 // Waits for all requests to complete
351 protected IEnumerator WaitForCompletion(List<TTSSpeakerRequestData> requestData)
352 {
353 // All done
354 int count = requestData?.Count ?? 0;
355 if (count == 0)
356 {
357 yield break;
358 }
359
360 // Current active requests
361 int activeRequests = 0;
362 UnityAction<TTSSpeaker, TTSClipData> onComplete = (speaker, clip) => activeRequests--;
363
364 // Add event delegates
365 for (int r = 0; r < count; r++)
366 {
367 TTSSpeakerRequestData request = requestData[r];
368 if (!IsClipRequestActive(request))
369 {
370 continue;
371 }
372 activeRequests++;
373 request.PlaybackEvents.OnComplete.AddListener(onComplete);
374 }
375
376 // Wait for active requests to be complete
377 yield return new WaitWhile(() => activeRequests > 0);
378
379 // Remove event delegates
380 for (int r = 0; r < count; r++)
381 {
382 TTSSpeakerRequestData request = requestData[r];
383 request.PlaybackEvents?.OnComplete.RemoveListener(onComplete);
384 }
385 }

394 public virtual List<string> GetFinalText(string textToSpeak)
395 {
396 // Get processors
397 RefreshProcessors();
398
399 // Get results
400 List<string> phrases = new List<string>();
401 phrases.Add(textToSpeak);
402
403 // Pre-processor
404 if (_textPreprocessors != null)
405 {
406 foreach (var preprocessor in _textPreprocessors)
407 {
408 preprocessor.OnPreprocessTTS(this, phrases);
409 }
410 }
411
412 // Add prepend & appended text to each item
413 for (int i = 0; i < phrases.Count; i++)
414 {
415 string phrase = phrases[i];
416 phrase = $"{PrependedText}{phrase}{AppendedText}".Trim();
417 phrases[i] = phrase;
418 }
419
420 // Post-processors
421 if (_textPostprocessors != null)
422 {
423 foreach (var postprocessor in _textPostprocessors)
424 {
425 postprocessor.OnPostprocessTTS(this, phrases);
426 }
427 }
428
429 // Return all text items
430 return phrases;
431 }

661 private List<TTSSpeakerRequestData> Speak(string textToSpeak, TTSDiskCacheSettings diskCacheSettings, TTSSpeakerClipEvents playbackEvents, bool addToQueue)
662 {
663 // Ensure voice settings exist
664 TTSVoiceSettings voiceSettings = VoiceSettings;
665 if (voiceSettings == null)
666 {
667 VLog.E($"No voice found with preset id: {presetVoiceID}");
668 return null;
669 }
670
671 // Get final text phrases to be spoken
672 List<string> phrases = GetFinalText(textToSpeak);
673 if (phrases == null || phrases.Count == 0)
674 {
675 VLog.W($"All phrases removed\nSource Phrase: {textToSpeak}");
676 return null;
677 }
678
679 // Cancel previous loading queue
680 if (!addToQueue)
681 {
682 _willHaveQueue = true;
683 StopLoading();
684 _willHaveQueue = false;
685 }
686
687 // Iterate voices
688 List<TTSSpeakerRequestData> results = new List<TTSSpeakerRequestData>();
689 foreach (var phrase in phrases)
690 {
691 TTSSpeakerRequestData requestData = HandleLoad(phrase, voiceSettings, diskCacheSettings, playbackEvents, addToQueue);
692 results.Add(requestData);
693
694 // Add additional to queue
695 if (!addToQueue)
696 {
697 addToQueue = true;
698 }
699 }
700 return results;
701 }

708 public virtual void Stop(string textToSpeak, bool allInstances = false)
709 {
710 // Found speaking clip
711 if (string.Equals(SpeakingClip?.textToSpeak, textToSpeak))
712 {
713 Stop(SpeakingClip, allInstances);
714 return;
715 }
716
717 // Find all clips that match & stop them
718 foreach (var clipData in QueuedClips)
719 {
720 if (string.Equals(clipData?.textToSpeak, textToSpeak))
721 {
722 Stop(clipData, allInstances);
723 if (!allInstances)
724 {
725 return;
726 }
727 }
728 }
729 }

736 public virtual void Stop(TTSClipData clipData, bool allInstances = false)
737 {
738 // Check if speaking
739 bool isSpeakingClip = SpeakingClip != null && clipData.Equals(SpeakingClip);
740
741 // Cancel queue
742 if (!isSpeakingClip || allInstances)
743 {
744 // Unload all instances
745 if (allInstances)
746 {
747 if (QueueContainsClip(clipData))
748 {
749 HandleUnload(clipData, string.Empty);
750 }
751 }
752 // Unload a single request
753 else
754 {
755 HandleUnload(GetQueuedRequest(clipData), string.Empty);
756 }
757 }
758
759 // Cancel playback
760 if (isSpeakingClip)
761 {
762 StopSpeaking();
763 }
764 }

1123 private TTSSpeakerRequestData HandleLoad(string textToSpeak, TTSVoiceSettings voiceSettings,
1124 TTSDiskCacheSettings diskCacheSettings, TTSSpeakerClipEvents playbackEvents,
1125 bool addToQueue)
1126 {
1127 // Generate request data
1128 TTSSpeakerRequestData requestData = new TTSSpeakerRequestData();
1129 requestData.StartTime = DateTime.Now;
1130 requestData.StopQueueOnLoad = !addToQueue;
1131 requestData.PlaybackEvents = playbackEvents ?? new TTSSpeakerClipEvents();
1132
1133 // Perform load request (Always waits a frame to ensure callbacks occur first)
1134 string clipId = TTSService.GetClipID(textToSpeak, voiceSettings);
