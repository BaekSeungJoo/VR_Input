50 private void InitializeArcPoints(Vector3[] linePoints)
51 {
52 if (_tubePoints == null
53 || _tubePoints.Length < linePoints.Length)
54 {
55 _tubePoints = new TubePoint[linePoints.Length];
56 }
57
58 float totalLength = 0f;
59 for (int i = 1; i < linePoints.Length; i++)
60 {
61 totalLength += (linePoints[i] - linePoints[i - 1]).magnitude;
62 }
63
64 for (int i = 0; i < linePoints.Length; i++)
65 {
66 Vector3 difference = i == 0 ? linePoints[i + 1] - linePoints[i]
67 : linePoints[i] - linePoints[i - 1];
68 _tubePoints[i].position = linePoints[i];
69 _tubePoints[i].rotation = Quaternion.LookRotation(difference);
70 _tubePoints[i].relativeLength = i == 0 ? 0f
71 : _tubePoints[i - 1].relativeLength + (difference.magnitude / totalLength);
72 }
73 }
