156 private bool UpdateTools(HashSet<InteractableTool> tools, bool resetCollisionData = false)
157 {
158 bool toolsEncounteredObjects = false;
159
160 foreach (InteractableTool currentInteractableTool in tools)
161 {
162 List<InteractableCollisionInfo> intersectingObjectsFound =
163 currentInteractableTool.GetNextIntersectingObjects();
164
165 if (intersectingObjectsFound.Count > 0 && !resetCollisionData)
166 {
167 if (!toolsEncounteredObjects)
168 {
169 toolsEncounteredObjects = intersectingObjectsFound.Count > 0;
170 }
171
172 // create map that indicates the furthest collider encountered per interactable element
173 currentInteractableTool.UpdateCurrentCollisionsBasedOnDepth();
174
175 if (currentInteractableTool.IsFarFieldTool)
176 {
177 var firstInteractable = currentInteractableTool.GetFirstCurrentCollisionInfo();
178 // if our tool is activated, make sure depth is set to "action"
179 if (currentInteractableTool.ToolInputState == ToolInputState.PrimaryInputUp)
180 {
181 firstInteractable.Value.InteractableCollider = firstInteractable.Key.ActionCollider;
182 firstInteractable.Value.CollisionDepth = InteractableCollisionDepth.Action;
183 }
184 else
185 {
186 firstInteractable.Value.InteractableCollider = firstInteractable.Key.ContactCollider;
187 firstInteractable.Value.CollisionDepth = InteractableCollisionDepth.Contact;
188 }
189
190 // far field tools only can focus elements -- pick first (for now)
191 currentInteractableTool.FocusOnInteractable(firstInteractable.Key,
192 firstInteractable.Value.InteractableCollider);
193 }
194 }
195 else
196 {
197 currentInteractableTool.DeFocus();
198 currentInteractableTool.ClearAllCurrentCollisionInfos();
199 }
200
201 currentInteractableTool.UpdateLatestCollisionData();
202 }
203
204 return toolsEncounteredObjects;
205 }
