100 public bool AddClip(TTSClipData clipData)
101 {
102 // Do not add null
103 if (clipData == null)
104 {
105 return false;
106 }
107 // Remove from order
108 bool wasAdded = true;
109 int clipIndex = _clipOrder.IndexOf(clipData.clipID);
110 if (clipIndex != -1)
111 {
112 wasAdded = false;
113 _clipOrder.RemoveAt(clipIndex);
114 }
115
116 // Add clip
117 _clips[clipData.clipID] = clipData;
118 // Add to end of order
119 _clipOrder.Add(clipData.clipID);
120
121 // Evict least recently used clips
122 while (IsCacheFull() && _clipOrder.Count > 0)
123 {
124 // Remove clip
125 RemoveClip(_clipOrder[0]);
126 }
127
128 // Call add delegate even if removed
129 if (wasAdded && _clips.Keys.Count > 0)
130 {
131 OnClipAdded?.Invoke(clipData);
132 }
133
134 // True if successfully added
135 return _clips.Keys.Count > 0;
136 }

142 public void RemoveClip(string clipID)
143 {
144 // Id not found
145 if (!_clips.ContainsKey(clipID))
146 {
147 return;
148 }
149
150 // Remove from dictionary
151 TTSClipData clipData = _clips[clipID];
152 _clips.Remove(clipID);
153
154 // Remove from order list
155 int clipIndex = _clipOrder.IndexOf(clipID);
156 _clipOrder.RemoveAt(clipIndex);
157
158 // Call remove delegate
159 OnClipRemoved?.Invoke(clipData);
160 }

165 protected bool IsCacheFull()

184 public int GetCacheDiskSize()
