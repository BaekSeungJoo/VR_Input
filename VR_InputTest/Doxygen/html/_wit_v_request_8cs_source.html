149 public static Uri GetWitUri(IWitRequestConfiguration configuration, string path, Dictionary<string, string> queryParams = null)
150 {
151 // Uri builder
152 UriBuilder uriBuilder = new UriBuilder();
153
154 // Append endpoint data
155 IWitRequestEndpointInfo endpoint = configuration.GetEndpointInfo();
156 uriBuilder.Scheme = endpoint.UriScheme;
157 uriBuilder.Host = endpoint.Authority;
158 uriBuilder.Port = endpoint.Port;
159
160 // Set path
161 uriBuilder.Path = path;
162
163 // Build query
164 string apiVersion = endpoint.WitApiVersion;
165 uriBuilder.Query = $"v={apiVersion}";
166 if (queryParams != null)
167 {
168 foreach (string key in queryParams.Keys)
169 {
170 uriBuilder.Query += $"&{key}={queryParams[key]}";
171 }
172 }
173
174 // Return custom uri
175 if (OnProvideCustomUri != null)
176 {
177 foreach (Func<UriBuilder, UriBuilder> del in OnProvideCustomUri.GetInvocationList())
178 {
179 uriBuilder = del(uriBuilder);
180 }
181 }
182
183 // Return uri
184 return uriBuilder.Uri;
185 }
189 public static Dictionary<string, string> GetWitHeaders(IWitRequestConfiguration configuration, string requestId, bool useServerToken)
190 {
191 // Get headers
192 Dictionary<string, string> headers = new Dictionary<string, string>();
193
194 // Set request id
195 headers[WitConstants.HEADER_REQUEST_ID] = string.IsNullOrEmpty(requestId) ? Guid.NewGuid().ToString() : requestId;
196 // Set User-Agent
197 headers[WitConstants.HEADER_USERAGENT] = GetUserAgentHeader(configuration);
198 // Set authorization
199 headers[WitConstants.HEADER_AUTH] = GetAuthorizationHeader(configuration, useServerToken);
200 // Allow overrides
201 if (OnProvideCustomHeaders != null)
202 {
203 // Allow overrides
204 foreach (Action<Dictionary<string, string>> del in OnProvideCustomHeaders.GetInvocationList())
205 {
206 del(headers);
207 }
208 }
209
210 // Return results
211 return headers;
212 }
216 private static string GetAuthorizationHeader(IWitRequestConfiguration configuration, bool useServerToken)
217 {
218 // Default to client access token
219 string token = configuration.GetClientAccessToken();
220 // Use server token
221 if (useServerToken)
222 {
223 #if UNITY_EDITOR
224 token = configuration.GetServerAccessToken();
225 #else
226 token = string.Empty;
227 #endif
228 }
229 // Trim token
230 if (!string.IsNullOrEmpty(token))
231 {
232 token = token.Trim();
233 }
234 // Use invalid token
235 else
236 {
237 token = "XXX";
238 }
239 // Return with bearer
240 return $"Bearer {token}";
241 }
242 // Build and return user agent header
243 private static string _operatingSystem;
244 private static string _deviceModel;
245 private static string _appIdentifier;
246 private static string _unityVersion;
247 private static string GetUserAgentHeader(IWitRequestConfiguration configuration)
248 {
249 // Generate user agent
250 StringBuilder userAgent = new StringBuilder();
251
252 // Append wit sdk version
253 userAgent.Append($"wit-unity-{WitConstants.SDK_VERSION}");
254
255 // Append operating system
256 if (_operatingSystem == null) _operatingSystem = UnityEngine.SystemInfo.operatingSystem;
257 userAgent.Append($",\"{_operatingSystem}\"");
258 // Append device model
259 if (_deviceModel == null) _deviceModel = UnityEngine.SystemInfo.deviceModel;
260 userAgent.Append($",\"{_deviceModel}\"");
261
262 // Append configuration log id
263 string logId = configuration.GetConfigurationId();
264 if (string.IsNullOrEmpty(logId))
265 {
266 logId = WitConstants.HEADER_USERAGENT_CONFID_MISSING;
267 }
268 userAgent.Append($",{logId}");
269
270 // Append app identifier
271 if (_appIdentifier == null) _appIdentifier = Application.identifier;
272 userAgent.Append($",{_appIdentifier}");
273
274 // Append editor identifier
275 #if UNITY_EDITOR
276 userAgent.Append(",Editor");
277 #else
278 userAgent.Append(",Runtime");
279 #endif
280
281 // Append unity version
282 if (_unityVersion == null) _unityVersion = Application.unityVersion;
283 userAgent.Append($",{_unityVersion}");
284
285 // Set custom user agent
286 if (OnProvideCustomUserAgent != null)
287 {
288 foreach (Action<StringBuilder> del in OnProvideCustomUserAgent.GetInvocationList())
289 {
290 del(userAgent);
291 }
292 }
293
294 // Return user agent
295 return userAgent.ToString();
296 }
