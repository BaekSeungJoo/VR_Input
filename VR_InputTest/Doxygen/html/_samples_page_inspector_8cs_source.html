38 private void OnEnable()
39 {
40 _samples.Clear();
41 _samplePage = (SamplesPage) target;
42 var samples = PageFinder.FindPages(typeof(Sample));
43
44 foreach (var so in samples)
45 {
46 var sample = (Sample)so;
47 if (sample.sampleSetId == _samplePage.SampleSetId)
48 {
49 _samples.Add(sample);
50 }
51 }
52
53 _samples.Sort((a, b) =>
54 {
55 int cmp = a.priority.CompareTo(b.priority);
56 if (cmp == 0) cmp = a.name.CompareTo(b.name);
57 return cmp;
58 });
59 }

61 public override void OnInspectorGUI()
62 {
63 _scrollOffset = GUILayout.BeginScrollView(_scrollOffset);
64 var windowWidth = (OverrideWidth > 0 ? OverrideWidth : EditorGUIUtility.currentViewWidth) - _minTileMargin;
65 var columns = (int) (windowWidth / (_tileSize + _minTileMargin));
66 int col = 0;
67 GUILayout.BeginHorizontal();
68 GUILayout.FlexibleSpace();
69 for (int i = 0; i < _samples.Count; i++)
70 {
71 if (col >= columns)
72 {
73 col = 0;
74 GUILayout.EndHorizontal();
75 GUILayout.BeginHorizontal();
76 GUILayout.FlexibleSpace();
77 }
78
79 var sample = _samples[i];
80 SampleEditor.DrawSample(sample, _tileSize);
81 GUILayout.FlexibleSpace();
82 col++;
83 }
84 GUILayout.EndHorizontal();
85
86#if VSDK_INTERNAL
87 GUILayout.Space(16);
88 edit = EditorGUILayout.Foldout(edit, "Edit");
89#endif
90 if (edit)
91 {
92 base.OnInspectorGUI();
93 }
94 GUILayout.EndScrollView();
95 }
