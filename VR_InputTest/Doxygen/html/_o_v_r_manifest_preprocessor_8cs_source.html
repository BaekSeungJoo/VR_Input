45 [MenuItem("Oculus/Tools/Create store-compatible AndroidManifest.xml", false, 100000)]
46 public static void GenerateManifestForSubmission()
47 {
48 var so = ScriptableObject.CreateInstance(typeof(OVRPluginInfo));
49 var script = MonoScript.FromScriptableObject(so);
50 string assetPath = AssetDatabase.GetAssetPath(script);
51 string editorDir = Directory.GetParent(assetPath).FullName;
52 string srcFile = editorDir + "/AndroidManifest.OVRSubmission.xml";
53
54 if (!File.Exists(srcFile))
55 {
56 Debug.LogError("Cannot find Android manifest template for submission." +
57 " Please delete the OVR folder and reimport the Oculus Utilities.");
58 return;
59 }
60
61 if (DoesAndroidManifestExist())
62 {
63 if (!EditorUtility.DisplayDialog("AndroidManifest.xml Already Exists!",
64 "Would you like to replace the existing manifest with a new one? All modifications will be lost.",
65 "Replace", "Cancel"))
66 {
67 return;
68 }
69 }
70
71 // IO methods use absolute paths
72 if (!Directory.Exists(ManifestFolderPathAbsolute))
73 Directory.CreateDirectory(ManifestFolderPathAbsolute);
74
75 PatchAndroidManifest(srcFile, BuildManifestFilePathAbsolute, false);
76
77 AssetDatabase.Refresh();
78 }

114 private static void AddOrRemoveTag(XmlDocument doc, string @namespace, string path, string elementName, string name,
115 bool required, bool modifyIfFound, params string[] attrs) // name, value pairs
116 {
117 var nodes = doc.SelectNodes(path + "/" + elementName);
118 XmlElement element = null;
119 foreach (XmlElement e in nodes)
120 {
121 if (name == null || name == e.GetAttribute("name", @namespace))
122 {
123 element = e;
124 break;
125 }
126 }
127
128 if (required)
129 {
130 if (element == null)
131 {
132 var parent = doc.SelectSingleNode(path);
133 element = doc.CreateElement(elementName);
134 element.SetAttribute("name", @namespace, name);
135 parent.AppendChild(element);
136 }
137
138 for (int i = 0; i < attrs.Length; i += 2)
139 {
140 if (modifyIfFound || string.IsNullOrEmpty(element.GetAttribute(attrs[i], @namespace)))
141 {
142 if (attrs[i + 1] != null)
143 {
144 element.SetAttribute(attrs[i], @namespace, attrs[i + 1]);
145 }
146 else
147 {
148 element.RemoveAttribute(attrs[i], @namespace);
149 }
150 }
151 }
152 }
153 else
154 {
155 if (element != null && modifyIfFound)
156 {
157 element.ParentNode.RemoveChild(element);
158 }
159 }
160 }

162 private static void AddReplaceValueTag(XmlDocument doc, string @namespace, string path, string elementName, string name)
163 {
164 XmlElement element = (XmlElement)doc.SelectSingleNode("/manifest");
165 if (element == null)
166 {
167 UnityEngine.Debug.LogError("Could not find manifest tag in android manifest.");
168 return;
169 }
170
171 string toolsNamespace = element.GetAttribute("xmlns:tools");
172 var nodes = doc.SelectNodes(path + "/" + elementName);
173 foreach (XmlElement e in nodes)
174 {
175 if (name == null || name == e.GetAttribute("name", @namespace))
176 {
177 element = e;
178 break;
179 }
180 }
181
182 element.SetAttribute("replace", toolsNamespace, "android:value");
183 }

185 public static void PatchAndroidManifest(string sourceFile, string destinationFile = null,
186 bool skipExistingAttributes = true, bool enableSecurity = false)
187 {
188 if (destinationFile == null)
189 {
190 destinationFile = sourceFile;
191 }
192
193 bool modifyIfFound = !skipExistingAttributes;
194
195 try
196 {
197 // Load android manfiest file
198 XmlDocument doc = new XmlDocument();
199 doc.Load(sourceFile);
200
201 string androidNamespaceURI;
202 XmlElement element = (XmlElement)doc.SelectSingleNode("/manifest");
203 if (element == null)
204 {
205 UnityEngine.Debug.LogError("Could not find manifest tag in android manifest.");
206 return;
207 }
208
209 // Get android namespace URI from the manifest
210 androidNamespaceURI = element.GetAttribute("xmlns:android");
211 if (string.IsNullOrEmpty(androidNamespaceURI))
212 {
213 UnityEngine.Debug.LogError("Could not find Android Namespace in manifest.");
214 return;
215 }
216
217 ApplyRequiredManfiestTags(doc, androidNamespaceURI, modifyIfFound, enableSecurity);
218 ApplyFeatureManfiestTags(doc, androidNamespaceURI, modifyIfFound);
219
220 // The following manifest entries are all handled through Oculus XR SDK Plugin
221#if !PRIORITIZE_OCULUS_XR_SETTINGS
222 ApplyOculusXRManifestTags(doc, androidNamespaceURI, modifyIfFound);
223 ApplyTargetDevicesManifestTags(doc, androidNamespaceURI, true /*modifyIfFound*/);
224#endif
225
226 doc.Save(destinationFile);
227 }
228 catch (System.Exception e)
229 {
230 UnityEngine.Debug.LogException(e);
231 }
232 }

234 private static void ApplyRequiredManfiestTags(XmlDocument doc, string androidNamespaceURI, bool modifyIfFound,
235 bool enableSecurity)
236 {
237 OVRProjectConfig projectConfig = OVRProjectConfig.GetProjectConfig();
238
239 AddOrRemoveTag(doc,
240 androidNamespaceURI,
241 "/manifest/application/activity/intent-filter",
242 "category",
243 "android.intent.category.LEANBACK_LAUNCHER",
244 required: false,
245 modifyIfFound: true); // always remove leanback launcher
246
247 // First add or remove headtracking flag if targeting Quest
248 AddOrRemoveTag(doc,
249 androidNamespaceURI,
250 "/manifest",
251 "uses-feature",
252 "android.hardware.vr.headtracking",
253 OVRDeviceSelector.isTargetDeviceQuestFamily,
254 true,
255 "version", "1",
256 "required", OVRProjectConfig.GetProjectConfig().allowOptional3DofHeadTracking ? "false" : "true");
257
258 // make sure android label and icon are set in the manifest
259 AddOrRemoveTag(doc,
260 androidNamespaceURI,
261 "/manifest",
262 "application",
263 null,
264 true,
265 modifyIfFound,
266 "label", "@string/app_name",
267 "icon", "@mipmap/app_icon",
268 // Disable allowBackup in manifest and add Android NSC XML file
269 "allowBackup", projectConfig.disableBackups ? "false" : "true",
270 "networkSecurityConfig", projectConfig.enableNSCConfig && enableSecurity ? "@xml/network_sec_config" : null
271 );
272 }

607 private static void ApplyTargetDevicesManifestTags(XmlDocument doc, string androidNamespaceURI, bool modifyIfFound)
608 {
609 // Add support devices manifest according to the target devices
610 if (OVRDeviceSelector.isTargetDeviceQuestFamily)
611 {
612 string targetDeviceValue = "";
613 if (OVRDeviceSelector.isTargetDeviceQuest)
614 {
615 if (string.IsNullOrEmpty(targetDeviceValue))
616 targetDeviceValue = "quest";
617 else
618 targetDeviceValue += "|quest";
619 }
620
621 if (OVRDeviceSelector.isTargetDeviceQuest2)
622 {
623 if (string.IsNullOrEmpty(targetDeviceValue))
624 targetDeviceValue = "quest2";
625 else
626 targetDeviceValue += "|quest2";
627 }
628
629 if (OVRDeviceSelector.isTargetDeviceQuestPro)
630 {
631 if (string.IsNullOrEmpty(targetDeviceValue))
632 targetDeviceValue = "questpro";
633 else
634 targetDeviceValue += "|questpro";
635 }
636 if (OVRDeviceSelector.isTargetDeviceQuest3)
637 {
638 if (string.IsNullOrEmpty(targetDeviceValue))
639 targetDeviceValue = "eureka";
640 else
641 targetDeviceValue += "|eureka";
642 }
643 if (string.IsNullOrEmpty(targetDeviceValue))
644 {
645 Debug.LogError("Empty target devices");
646 }
647
648 AddOrRemoveTag(doc,
649 androidNamespaceURI,
650 "/manifest/application",
651 "meta-data",
652 "com.oculus.supportedDevices",
653 true,
654 modifyIfFound,
655 "value", targetDeviceValue);
656
657#if XR_MGMT_4_4_0_OR_NEWER && USING_XR_SDK_OPENXR
658 // Fixes a manifest merge edge case where the supported devices tag collides with a cached version when using new XR Management manifest system
659 AddReplaceValueTag(doc,
660 androidNamespaceURI,
661 "/manifest/application",
662 "meta-data",
663 "com.oculus.supportedDevices");
664#endif
665 }

668 private static string ColorSpaceToManifestTag(OVRManager.ColorSpace colorSpace)
669 {
670 switch (colorSpace)
671 {
672 case OVRManager.ColorSpace.Unmanaged:
673 return "!Unmanaged";
674 case OVRManager.ColorSpace.Rec_2020:
675 return "Rec.2020";
676 case OVRManager.ColorSpace.Rec_709:
677 return "Rec.709";
678 case OVRManager.ColorSpace.Rift_CV1:
679 return "!RiftCV1";
680 case OVRManager.ColorSpace.Rift_S:
681 return "!RiftS";
682 case OVRManager.ColorSpace.Quest:
683 return "!Quest";
684 case OVRManager.ColorSpace.P3:
685 return "P3";
686 case OVRManager.ColorSpace.Adobe_RGB:
687 return "Adobe";
688 default:
689 return "";
690 }
691 }
