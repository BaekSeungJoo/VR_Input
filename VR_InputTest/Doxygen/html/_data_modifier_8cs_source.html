23namespace Oculus.Interaction.Input

25 public abstract class
26 DataModifier<TData> : DataSource<TData>
27 where TData : class, ICopyFrom<TData>, new()
28 {
29 [Header("Data Modifier")]
30 [SerializeField, Interface(nameof(_modifyDataFromSource))]
31 protected UnityEngine.Object _iModifyDataFromSourceMono;
32 private IDataSource<TData> _modifyDataFromSource;
33
34 [SerializeField]
35 [Tooltip("If this is false, then this modifier will simply pass through " +
36 "data without performing any modification. This saves on memory " +
37 "and computation")]
38 private bool _applyModifier = true;
39
40 private static TData InvalidAsset { get; } = new TData();
41 private TData _thisDataAsset;
42 private TData _currentDataAsset = InvalidAsset;
43
44 protected override TData DataAsset => _currentDataAsset;
45
46 public virtual IDataSource<TData> ModifyDataFromSource => _modifyDataFromSource == null
47 ? (_modifyDataFromSource = _iModifyDataFromSourceMono as IDataSource<TData>)
48 : _modifyDataFromSource;
49
50 public override int CurrentDataVersion
51 {
52 get
53 {
54 return _applyModifier
55 ? base.CurrentDataVersion
56 : ModifyDataFromSource.CurrentDataVersion;
57 }
58 }
59
60 public void ResetSources(IDataSource<TData> modifyDataFromSource, IDataSource updateAfter, UpdateModeFlags updateMode)
61 {
62 ResetUpdateAfter(updateAfter, updateMode);
63 _modifyDataFromSource = modifyDataFromSource;
64 _currentDataAsset = InvalidAsset;
65 }
66
67 protected override void UpdateData()
68 {
69 if (_applyModifier)
70 {
71 if (_thisDataAsset == null)
72 {
73 _thisDataAsset = new TData();
74 }
75
76 _thisDataAsset.CopyFrom(ModifyDataFromSource.GetData());
77 _currentDataAsset = _thisDataAsset;
78 Apply(_currentDataAsset);
79 }
80 else
81 {
82 _currentDataAsset = ModifyDataFromSource.GetData();
83 }
84 }
85
86 protected abstract void Apply(TData data);
87
88 protected override void Start()
89 {
90 this.BeginStart(ref _started, ()=>base.Start());
91 this.AssertField(ModifyDataFromSource, nameof(ModifyDataFromSource));
92 this.EndStart(ref _started);
93 }
94

97 {
98 base.InjectAllDataSource(updateMode, updateAfter);
99 InjectModifyDataFromSource(modifyDataFromSource);

101 }
102
103 public void InjectModifyDataFromSource(IDataSource<TData> modifyDataFromSource)

106 }
107

110 _applyModifier = applyModifier;
111 }
112 #endregion
113 }
