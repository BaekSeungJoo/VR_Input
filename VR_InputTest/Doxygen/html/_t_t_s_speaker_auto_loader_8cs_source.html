66 // Load all phrase clips
67 public virtual void LoadClips()
68 {
69 // Done
70 if (_clips != null)
71 {
72 VLog.W("Cannot autoload clips twice.");
73 return;
74 }
75
76 // Set phrase list
77 _phrases = GetAllPhrases().ToArray();
78
79 // Load all clips
80 List<TTSClipData> list = new List<TTSClipData>();
81 foreach (var phrase in _phrases)
82 {
83 _clipsLoading++;
84 TTSClipData clip = TTSService.Instance.Load(phrase, Speaker.presetVoiceID, null, OnClipReady);
85 list.Add(clip);
86 }
87 _clips = list.ToArray();
88 }
89 // Return all phrases
90 public virtual List<string> GetAllPhrases()
91 {
92 // Ensure speaker exists
93 SetupSpeaker();
94
95 // Get all phrases unformatted
96 List<string> unformattedPhrases = new List<string>();
97 // Add phrases split from phrase file
98 AddUniquePhrases(unformattedPhrases, PhraseFile?.text.Split('\n'));
99 // Add phrases serialized in phrase array
100 AddUniquePhrases(unformattedPhrases, Phrases);
101
102 // Iterate old phrases
103 List<string> phrases = new List<string>();
104 for (int i = 0; i < unformattedPhrases.Count; i++)
105 {
106 // Format phrases
107 List<string> newPhrases = Speaker.GetFinalText(unformattedPhrases[i]);
108 // Add to final list
109 if (newPhrases != null && newPhrases.Count > 0)
110 {
111 phrases.AddRange(newPhrases);
112 }
113 }
114
115 // Return array
116 return phrases;
117 }
118 // Add unique, non-null phrases
119 private void AddUniquePhrases(List<string> list, string[] newPhrases)
120 {
121 if (newPhrases != null)
122 {
123 foreach (var phrase in newPhrases)
124 {
125 if (!string.IsNullOrEmpty(phrase) && !list.Contains(phrase))
126 {
127 list.Add(phrase);
128 }
129 }
130 }
131 }

156 protected virtual void UnloadClips()
157 {
158 if (_clips == null)

164 TTSService.Instance?.Unload(clip);
165 }
166 _clips = null;
167 _phrases = null;
168 }
