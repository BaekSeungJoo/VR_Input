52 // Stop click
53 private void StopClick() => _speaker.Stop();
54 // Pause click
55 private void PauseClick()
56 {
57 if (_speaker.IsPaused)
58 {
59 _speaker.Resume();
60 }
61 else
62 {
63 _speaker.Pause();
64 }
65 }
66 // Speak phrase click
67 private void SpeakClick()
68 {
69 // Speak phrase
70 string phrase = FormatText(_input.text);
71 bool queued = _queueButton != null && _queueButton.isOn;
72 bool async = _asyncToggle != null && _asyncToggle.isOn;
73
74 // Speak async
75 if (async)
76 {
77 StartCoroutine(SpeakAsync(phrase, queued));
78 }
79 // Speak queued
80 else if (queued)
81 {
82 _speaker.SpeakQueued(phrase);
83 }
84 // Speak
85 else
86 {
87 _speaker.Speak(phrase);
88 }
89
90 // Queue additional phrases
91 if (_queuedText != null && _queuedText.Length > 0 && queued)
92 {
93 foreach (var text in _queuedText)
94 {
95 _speaker.SpeakQueued(FormatText(text));
96 }
97 }
98 }
99 // Speak async
100 private IEnumerator SpeakAsync(string phrase, bool queued)
101 {
102 // Queue
103 if (queued)
104 {
105 yield return _speaker.SpeakQueuedAsync(new string[] { phrase });
106 }
107 // Default
108 else
109 {
110 yield return _speaker.SpeakAsync(phrase);
111 }
112
113 // Play complete clip
114 if (_asyncClip != null)
115 {
116 _speaker.AudioSource.PlayOneShot(_asyncClip);
117 }
118 }
119 // Format text with current datetime
120 private string FormatText(string text)
121 {
122 string result = text;
123 if (result.Contains(_dateId))
124 {
125 DateTime now = DateTime.Now;
126 string dateString = $"{now.ToLongDateString()} at {now.ToLongTimeString()}";
127 result = text.Replace(_dateId, dateString);
128 }
129 return result;
130 }
131 // Remove delegates
132 private void OnDisable()
133 {
134 _stopButton.onClick.RemoveListener(StopClick);
135 _speakButton.onClick.RemoveListener(SpeakClick);
136 }
137
138 // Preset text fields
139 private void Update()
140 {
141 // On preset voice id update
142 if (!string.Equals(_voice, _speaker.presetVoiceID))
143 {
144 _voice = _speaker.presetVoiceID;
145 _input.placeholder.GetComponent<Text>().text = $"Write something to say in {_voice}'s voice";
146 }
147 // On state changes
148 if (_loading != _speaker.IsLoading)
149 {
150 RefreshStopButton();
151 }
152 if (_speaking != _speaker.IsSpeaking)
153 {
154 RefreshStopButton();
155 }
156 if (_paused != _speaker.IsPaused)
157 {
158 RefreshPauseButton();
159 }
160 }
161 // Refresh interactable based on states
162 private void RefreshStopButton()
163 {
164 _loading = _speaker.IsLoading;
165 _speaking = _speaker.IsSpeaking;
166 _stopButton.interactable = _loading || _speaking;
167 }
168 // Refresh text based on pause state
169 private void RefreshPauseButton()
170 {
171 _paused = _speaker.IsPaused;
172 _pauseButton.GetComponentInChildren<Text>().text = _paused ? "Resume" : "Pause";
173 }
174 }
