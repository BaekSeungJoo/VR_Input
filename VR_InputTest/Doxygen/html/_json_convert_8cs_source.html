128 {
129 // Could not parse
130 if (jsonToken == null)
131 {
132 return false;
133 }
134 // Use default if no customs are added
135 if (customConverters == null)
136 {
137 customConverters = DefaultConverters;
138 }
139
140 // Auto cast
141 Type iType = typeof(IN_TYPE);
142 if (iType == typeof(WitResponseNode))
143 {
144 object result = jsonToken;

147 }

184 private static object DeserializeToken(Type toType, object oldValue, WitResponseNode jsonToken, StringBuilder log, JsonConverter[] customConverters)
185 {
186 // Iterate custom converters
187 if (customConverters != null)
188 {
189 foreach (var converter in customConverters)
190 {
191 if (converter.CanRead && converter.CanConvert(toType))
192 {
193 return converter.ReadJson(jsonToken, toType, oldValue);
194 }
195 }
196 }
197 // Return default
198 if (toType == typeof(string))
199 {
200 return jsonToken.Value;
201 }
202 // Enum parse
203 if (toType.IsEnum)
204 {
205 string enumStr = jsonToken.Value;
206 foreach (var enumVal in Enum.GetValues(toType))
207 {
208 foreach (JsonPropertyAttribute renameAttribute in toType.GetMember(enumVal.ToString())[0].GetCustomAttributes(typeof(JsonPropertyAttribute), false))
209 {
210 if (!string.IsNullOrEmpty(renameAttribute.PropertyName) && string.Equals(jsonToken.Value, renameAttribute.PropertyName, StringComparison.CurrentCultureIgnoreCase))
211 {
212 enumStr = enumVal.ToString();
213 break;
214 }
215 }
216 }
217 // Call try parse
218 return DeserializeEnum(toType, EnsureExists(toType, oldValue), enumStr, log);
219 }
220 // Deserialize dictionary
221 if (toType.GetInterfaces().Contains(typeof(IDictionary)))
222 {
223 return DeserializeDictionary(toType, EnsureExists(toType, oldValue), jsonToken.AsObject, log, customConverters);
224 }
225 // Deserialize List
226 if (toType.GetInterfaces().Contains(typeof(IEnumerable)))
227 {
228 // Element type
229 Type elementType = toType.GetElementType();
230 if (elementType == null)
231 {
232 // Try arguments
233 Type[] genericArguments = toType.GetGenericArguments();
234 if (genericArguments != null && genericArguments.Length > 0)
235 {
236 elementType = genericArguments[0];
237 }
238 }
239
240 if (elementType != null)
241 {
242 // Make array
243 object newArray = newArray = typeof(JsonConvert)
244 .GetMethod("DeserializeArray", BindingFlags.Public | BindingFlags.Static)
245 .MakeGenericMethod(new Type[] { elementType })
246 .Invoke(null, new object[] { oldValue, jsonToken, log, customConverters });
247
248 // Return array
249 if (toType.IsArray)
250 {
251 return newArray;
252 }
253 // Convert to list
254 if (toType.GetInterfaces().Contains(typeof(IList)))
255 {
256 return Activator.CreateInstance(toType, new object[] { newArray });
257 }
258 }
259 }
260 // Deserialize class
261 if (toType.IsClass)
262 {
263 return DeserializeClass(toType, oldValue, jsonToken.AsObject, log, customConverters);
264 }
265 // Deserialize struct
266 if (toType.IsValueType && !toType.IsPrimitive)
267 {
268 object oldStruct = Activator.CreateInstance(toType);
269 object newStruct = DeserializeClass(toType, oldStruct, jsonToken.AsObject, log, customConverters);
270 return newStruct;
271 }
272
273 try
274 {
275 // Convert to basic values
276 return Convert.ChangeType(jsonToken.Value, toType);
277 }
278 catch (Exception e)
279 {
280 // Could not cast
281 log.AppendLine($"\nJson Deserializer failed to cast '{jsonToken.Value}' to type '{toType}'\n{e}");
282 return oldValue;
283 }
284 }

287 private static MethodInfo _enumParseMethod;
288 private static object DeserializeEnum(Type toType, object oldValue, string enumString, StringBuilder log)
289 {
290 // Find enum parse method
291 if (_enumParseMethod == null)
292 {
293 _enumParseMethod = typeof(Enum).GetMethods().ToList().Find(method =>
294 method.IsGenericMethod && method.GetParameters().Length == 3 &&
295 string.Equals(method.Name, "TryParse"));
296 }
297
298 // Attempt to parse (Enum.TryParse<TEnum>(enumString, false, out oldValue))
299 var parseMethod = _enumParseMethod.MakeGenericMethod(new[] {toType});
300 object[] parseParams = new object[] {enumString, false, Activator.CreateInstance(toType)};
301
302 // Invoke
303 if ((bool)parseMethod.Invoke(null, parseParams))
304 {
305 // Return the parsed enum
306 return parseParams[2];
307 }
308
309 // Failed
310 log.AppendLine($"\nJson Deserializer Failed to cast '{enumString}' to enum type '{toType}'");
311 return oldValue;
312 }

324 {
325 // Failed
326 if (jsonToken == null)
327 {
328 return (ITEM_TYPE[])oldArray;
329 }

331 // Generate array
332 WitResponseArray jsonArray = jsonToken.AsArray;
333 ITEM_TYPE[] newArray = new ITEM_TYPE[jsonArray.Count];
334
335 // Deserialize array elements
336 Type elementType = typeof(ITEM_TYPE);
337 for (int i = 0; i < jsonArray.Count; i++)
338 {

341 newArray[i] = newItem;
342 }
343
344 // Return array
345 return newArray;

354 private static object DeserializeClass(Type toType, object oldObject, WitResponseClass jsonClass, StringBuilder log, JsonConverter[] customConverters)
355 {
356 // Failed
357 if (jsonClass == null)
358 {
359 return oldObject;
360 }
361
362 // Use old value
363 object newObject = oldObject;
364 // Generate new if needed
365 if (newObject == null)
366 {
367 newObject = Activator.CreateInstance(toType);
368 }
369
370 // Add renames
371 Dictionary<string, FieldInfo> fieldRenames = new Dictionary<string, FieldInfo>();
372 foreach (var field in toType.GetFields().Where(field => field.IsDefined(typeof(JsonPropertyAttribute), false)))
373 {
374 foreach (var renameAttribute in field.GetCustomAttributes<JsonPropertyAttribute>())
375 {
376 if (!string.IsNullOrEmpty(renameAttribute.PropertyName))
377 {
378 fieldRenames[renameAttribute.PropertyName] = field;
379 }
380 }
381 }
382 Dictionary<string, PropertyInfo> propRenames = new Dictionary<string, PropertyInfo>();
383 foreach (var prop in toType.GetProperties().Where(prop => prop.IsDefined(typeof(JsonPropertyAttribute), false)))
384 {
385 foreach (var renameAttribute in prop.GetCustomAttributes<JsonPropertyAttribute>())
386 {
387 if (!string.IsNullOrEmpty(renameAttribute.PropertyName))
388 {
389 propRenames[renameAttribute.PropertyName] = prop;
390 }
391 }
392 }
393
394 // Iterate each child node
395 foreach (var childTokenName in jsonClass.ChildNodeNames)
396 {
397 // Check field
398 FieldInfo field = toType.GetField(childTokenName, BIND_FLAGS);
399 if (fieldRenames.ContainsKey(childTokenName))
400 {
401 field = fieldRenames[childTokenName];
402 }
403 if (field != null)
404 {
405 // Get old value
406 object oldValue = field.GetValue(newObject);
407
408 // Deserialize new value
409 object newValue = DeserializeToken(field.FieldType, oldValue, jsonClass[childTokenName], log, customConverters);
410
411 // Apply new value
412 field.SetValue(newObject, newValue);
413 continue;
414 }
415
416 // Check property
417 PropertyInfo property = toType.GetProperty(childTokenName, BIND_FLAGS);
418 if (propRenames.ContainsKey(childTokenName))
419 {
420 property = propRenames[childTokenName];
421 }
422 if (property != null && property.GetSetMethod() != null)
423 {
424 // Get old value
425 object oldValue = null;
426 if (property.GetGetMethod() != null)
427 {
428 oldValue = property.GetValue(newObject);
429 }
430 oldValue = EnsureExists(property.PropertyType, oldValue);
431
432 // Deserialize new value
433 object newValue = DeserializeToken(property.PropertyType, oldValue, jsonClass[childTokenName], log, customConverters);
434
435 // Apply new value
436 property.SetValue(newObject, newValue);
437 continue;
438 }
439
440 // Not found
441 log.AppendLine($"\t{toType.FullName} does not have a public '{childTokenName}' field or property.");
442 }
443
444 // Use deserializer if applicable
445 if (toType.GetInterfaces().Contains(typeof(IJsonDeserializer)))
446 {
447 IJsonDeserializer deserializer = newObject as IJsonDeserializer;
448 if (!deserializer.DeserializeObject(jsonClass))
449 {
450 log.AppendLine($"\tIJsonDeserializer '{toType}' failed");
451 }
452 }
453
454 // Success
455 return newObject;
456 }

466 private static object DeserializeDictionary(Type toType, object oldObject, WitResponseClass jsonClass, StringBuilder log, JsonConverter[] customConverters)
467 {
468 // Ensure types are correct
469 Type[] dictGenericTypes = toType.GetGenericArguments();
470 if (dictGenericTypes == null || dictGenericTypes.Length != 2)
471 {
472 return oldObject;
473 }
474
475 // Generate dictionary
476 IDictionary newDictionary = oldObject as IDictionary;
477
478 // Get types
479 Type keyType = dictGenericTypes[0];
480 Type valType = dictGenericTypes[1];
481
482 // Iterate children
483 foreach (var childName in jsonClass.ChildNodeNames)
484 {
485 // Cast key if possible
486 object childKey = Convert.ChangeType(childName, keyType);
487
488 // Cast value if possible
489 object newChildValue = DeserializeToken(valType, null, jsonClass[childName], log, customConverters);
490
491 // Apply
492 newDictionary[childKey] = newChildValue;
493 }
494
495 // Return dictionary
496 return newDictionary;
497 }

549 // Convert data to node
550 private static WitResponseNode SerializeToken(Type inType, object inObject, StringBuilder log, JsonConverter[] customConverters)
551 {
552 // Use object type instead if possible
553 if (inObject != null && inType == typeof(object))
554 {
555 inType = inObject.GetType();
556 }
557
558 // Iterate custom converters
559 if (customConverters != null)
560 {
561 foreach (var converter in customConverters)
562 {
563 if (converter.CanWrite && converter.CanConvert(inType))
564 {
565 return converter.WriteJson(inObject);
566 }
567 }
568 }
569
570 // Null
571 if (inObject == null)
572 {
573 return null;
574 }
575 // Most likely error in this class
576 if (inType == null)
577 {
578 throw new ArgumentException("In Type cannot be null");
579 }
580
581 // Serialize to string
582 if (inType == typeof(string))
583 {
584 return new WitResponseData((string)inObject);
585 }
586 // Convert to bool
587 if (inType == typeof(bool))
588 {
589 return new WitResponseData((bool)inObject);
590 }
591 // Convert to int
592 else if (inType == typeof(int))
593 {
594 return new WitResponseData((int)inObject);
595 }
596 // Convert to float
597 else if (inType == typeof(float))
598 {
599 return new WitResponseData((float)inObject);
600 }
601 // Convert to double
602 else if (inType == typeof(double))
603 {
604 return new WitResponseData((double)inObject);
605 }
606 // Convert to enum
607 else if (inType.IsEnum)
608 {
609 return new WitResponseData(inObject.ToString());
610 }
611 // Serialize a dictionary into a node
612 else if (inType.GetInterfaces().Contains(typeof(IDictionary)))
613 {
614 IDictionary oldDictionary = (IDictionary) inObject;
615 WitResponseClass newDictionary = new WitResponseClass();
616 Type valType = inType.GetGenericArguments()[1];
617 foreach (var key in oldDictionary.Keys)
618 {
619 object newObj = oldDictionary[key];
620 if (newObj == null)
621 {
622 if (valType == typeof(string))
623 {
624 newObj = string.Empty;
625 }
626 else
627 {
628 newObj = Activator.CreateInstance(valType);
629 }
630 }
631 newDictionary.Add(key.ToString(), SerializeToken(valType, newObj, log, customConverters));
632 }
633 return newDictionary;
634 }
635 // Serialize enumerable into array
636 else if (inType.GetInterfaces().Contains(typeof(IEnumerable)))
637 {
638 // Get enum
639 WitResponseArray newArray = new WitResponseArray();
640 IEnumerator oldEnumerable = ((IEnumerable) inObject).GetEnumerator();
641
642 // Array[]
643 Type elementType = inType.GetElementType();
644
645 // Try generic argument (List<>)
646 if (elementType == null)
647 {
648 Type[] genericArguments = inType.GetGenericArguments();
649 if (genericArguments != null && genericArguments.Length > 0)
650 {
651 elementType = genericArguments[0];
652 }
653 }
654
655 // Serialize each
656 while (oldEnumerable.MoveNext())
657 {
658 object newObj = EnsureExists(elementType, oldEnumerable.Current);
659 newArray.Add(string.Empty, SerializeToken(elementType, newObj, log, customConverters));
660 }
661 return newArray;
662 }
663 // Serialize a class or a struct into a node
664 else if (inType.IsClass || (inType.IsValueType && !inType.IsPrimitive))
665 {
666 WitResponseClass newClass = new WitResponseClass();
667 foreach (var field in inType.GetFields(BIND_FLAGS))
668 {
669 JsonPropertyAttribute[] fieldAttributes = (JsonPropertyAttribute[])field.GetCustomAttributes(typeof(JsonPropertyAttribute));
670 SerializeProperty(newClass, field.FieldType, field.Name, field.GetValue(inObject),
671 fieldAttributes, log, customConverters);
672 }
673 foreach (var property in inType.GetProperties(BIND_FLAGS))
674 {
675 MethodInfo getter = property.GetGetMethod();
676 if (getter != null && getter.GetParameters().Length == 0)
677 {
678 JsonPropertyAttribute[] propertyAttributes = (JsonPropertyAttribute[])property.GetCustomAttributes(typeof(JsonPropertyAttribute));
679 SerializeProperty(newClass, property.PropertyType, property.Name, property.GetValue(inObject),
680 propertyAttributes, log, customConverters);
681 }
682 }
683 if (inType.GetInterfaces().Contains(typeof(IJsonSerializer)))
684 {
685 IJsonSerializer serializer = inObject as IJsonSerializer;
686 if (!serializer.SerializeObject(newClass))
687 {
688 log.AppendLine($"\tIJsonSerializer '{inType}' failed");
689 }
690 }
691 return newClass;
692 }
693
694 // Warn & incode to string
695 log.AppendLine($"\tJson Serializer cannot serialize: {inType}");
696 return inObject == null ? null : new WitResponseData(inObject.ToString());
697 }
