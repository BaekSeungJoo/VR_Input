254 void CameraAndRenderTargetSetup()
255 {
256 float overlayWidth = cameraRenderOverlay.transform.localScale.x;
257 float overlayHeight = cameraRenderOverlay.transform.localScale.y;
258 float overlayRadius = cameraRenderOverlay.transform.localScale.z;
259
260#if UNITY_ANDROID
261 // Gear VR display panel resolution
262 float hmdPanelResWidth = 2560;
263 float hmdPanelResHeight = 1440;
264#else
265 // Rift display panel resolution
266 float hmdPanelResWidth = 2160;
267 float hmdPanelResHeight = 1200;
268#endif
269
270 float singleEyeScreenPhysicalResX = hmdPanelResWidth * 0.5f;
271 float singleEyeScreenPhysicalResY = hmdPanelResHeight;
272
273 // Calculate RT Height
274 // screenSizeYInWorld : how much world unity the full screen can cover at overlayQuad's location vertically
275 // pixelDensityY: pixels / world unit ( meter )
276
277 float halfFovY = mainCamera.GetComponent<Camera>().fieldOfView / 2;
278 float screenSizeYInWorld = 2 * overlayRadius * Mathf.Tan(Mathf.Deg2Rad * halfFovY);
279 float pixelDensityYPerWorldUnit = singleEyeScreenPhysicalResY / screenSizeYInWorld;
280 float renderTargetHeight = pixelDensityYPerWorldUnit * overlayWidth;
281
282 // Calculate RT width
283 float renderTargetWidth = 0.0f;
284
285 // screenSizeXInWorld : how much world unity the full screen can cover at overlayQuad's location horizontally
286 // pixelDensityY: pixels / world unit ( meter )
287
288 float screenSizeXInWorld = screenSizeYInWorld * mainCamera.GetComponent<Camera>().aspect;
289 float pixelDensityXPerWorldUnit = singleEyeScreenPhysicalResX / screenSizeXInWorld;
290 renderTargetWidth = pixelDensityXPerWorldUnit * overlayWidth;
291
292 // Compute the orthographic size for the camera
293 float orthographicSize = overlayHeight / 2.0f;
294 float orthoCameraAspect = overlayWidth / overlayHeight;
295 uiCamera.GetComponent<Camera>().orthographicSize = orthographicSize;
296 uiCamera.GetComponent<Camera>().aspect = orthoCameraAspect;
297
298 if (uiCamera.GetComponent<Camera>().targetTexture != null)
299 uiCamera.GetComponent<Camera>().targetTexture.Release();
300
301 RenderTexture overlayRT = new RenderTexture(
302 (int)renderTargetWidth * 2,
303 (int)renderTargetHeight * 2,
304 0,
305 RenderTextureFormat.ARGB32,
306 RenderTextureReadWrite.sRGB);
307 Debug.Log("Created RT of resolution w: " + renderTargetWidth + " and h: " + renderTargetHeight);
308
309 overlayRT.hideFlags = HideFlags.DontSave;
310 overlayRT.useMipMap = true;
311 overlayRT.filterMode = FilterMode.Trilinear;
312 overlayRT.anisoLevel = 4;
313#if UNITY_5_5_OR_NEWER
314 overlayRT.autoGenerateMips = true;
315#else
316 overlayRT.generateMips = true;
317#endif
318 uiCamera.GetComponent<Camera>().targetTexture = overlayRT;
319
320 cameraRenderOverlay.textures[0] = overlayRT;
321 }

327 void SimulateLevelLoad()
328 {
329 int numToPrint = 0;
330 for (int p = 0; p < numLoopsTrigger; p++)
331 {
332 numToPrint++;
333 }
334
335 Debug.Log("Finished " + numToPrint + " Loops");
336 Vector3 playerPos = mainCamera.transform.position;
337 playerPos.y = 0.5f;
338 // Generate a bunch of blocks, "blocking" the mainthread ;)
339 for (int j = 0; j < numLevels; j++)
340 {
341 for (var i = 0; i < numObjectsPerLevel; i++)
342 {
343 var angle = i * Mathf.PI * 2 / numObjectsPerLevel;
344 float stagger = (i % 2 == 0) ? 1.5f : 1.0f;
345 var pos = new Vector3(Mathf.Cos(angle), 0, Mathf.Sin(angle)) * cubeSpawnRadius * stagger;
346 pos.y = j * heightBetweenItems;
347 var newInst = Instantiate(prefabForLevelLoadSim, pos + playerPos, Quaternion.identity);
348 var newObjTransform = newInst.transform;
349 newObjTransform.LookAt(playerPos);
350 Vector3 newAngle = newObjTransform.rotation.eulerAngles;
351 newAngle.x = 0.0f;
352 newObjTransform.rotation = Quaternion.Euler(newAngle);
353 spawnedCubes.Add(newInst);
354 }
355 }
356 }
