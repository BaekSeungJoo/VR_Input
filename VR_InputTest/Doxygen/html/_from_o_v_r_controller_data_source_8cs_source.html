233 protected override void UpdateData()
234 {
235 _controllerDataAsset.Config = Config;
236 var worldToTrackingSpace = TrackingToWorldTransformer.Transform.worldToLocalMatrix;
237 Transform ovrController = _ovrControllerAnchor;
238
239 _controllerDataAsset.IsDataValid = true;
240 _controllerDataAsset.IsConnected =
241 (OVRInput.GetConnectedControllers() & _ovrController) > 0;
242 if (!_controllerDataAsset.IsConnected)
243 {
244 // revert state fields to their defaults
245 _controllerDataAsset.IsTracked = default;
246 _controllerDataAsset.ButtonUsageMask = default;
247 _controllerDataAsset.RootPoseOrigin = default;
248 return;
249 }
250
251 _controllerDataAsset.IsTracked = true;
252
253 // Update button usages
254 _controllerDataAsset.ButtonUsageMask = ControllerButtonUsage.None;
255 OVRInput.Controller controllerMask = _ovrController;
256 foreach (UsageMapping mapping in ControllerUsageMappings)
257 {
258 bool usageActive;
259 if (mapping.IsTouch)
260 {
261 usageActive = OVRInput.Get(mapping.Touch, controllerMask);
262 }
263 else
264 {
265 this.AssertIsTrue(mapping.IsButton,
266 $"Element in {AssertUtils.Nicify(nameof(ControllerUsageMappings))} has {AssertUtils.Nicify(nameof(mapping.IsButton))} set to false.");
267
268 usageActive = OVRInput.Get(mapping.Button, controllerMask);
269 }
270
271 if (usageActive)
272 {
273 _controllerDataAsset.ButtonUsageMask |= mapping.Usage;
274 }
275 }
276
277 // Update poses
278
279 // Root pose, in tracking space.
280 _controllerDataAsset.RootPose = new Pose(
281 OVRInput.GetLocalControllerPosition(_ovrController),
282 OVRInput.GetLocalControllerRotation(_ovrController));
283 _controllerDataAsset.RootPoseOrigin = PoseOrigin.RawTrackedPose;
284
285 // Convert controller pointer pose from local to tracking space.
286 Matrix4x4 controllerModelToTracking = Matrix4x4.TRS(
287 _controllerDataAsset.RootPose.position, _controllerDataAsset.RootPose.rotation,
288 Vector3.one);
289 _controllerDataAsset.PointerPose =
290 new Pose(controllerModelToTracking.MultiplyPoint3x4(_pointerPoseSelector.LocalPointerPose.position),
291 _controllerDataAsset.RootPose.rotation * _pointerPoseSelector.LocalPointerPose.rotation);
292
293 _controllerDataAsset.PointerPoseOrigin = PoseOrigin.RawTrackedPose;
294 }
