69 protected virtual void Awake()
70 {
71 FingerFeatureStateProvider = _fingerFeatureStateProvider as IFingerFeatureStateProvider;
72 this.AssertField(_shapeRecognizerActiveState, nameof(_shapeRecognizerActiveState));
73 this.AssertField(_target, nameof(_target));
74 this.AssertField(_fingerFeatureDebugVisualPrefab, nameof(_fingerFeatureDebugVisualPrefab));
75 this.AssertField(_targetText, nameof(_targetText));
76 _material = _target.material;
77
78 _material.color = _lastActiveValue ? _activeColor : _normalColor;
79
80 if (_fingerFeatureParent == null)
81 {
82 _fingerFeatureParent = transform;
83 }
84 }
85
86 protected virtual void Start()
87 {
88 this.AssertField(FingerFeatureStateProvider, nameof(FingerFeatureStateProvider));
89
90 Vector3 fingerOffset = Vector3.zero;
91
92 var statesByFinger = AllFeatureStates()
93 .GroupBy(s => s.Item1)
94 .Select(group => new
95 {
96 HandFinger = group.Key,
97 FingerFeatures = group.SelectMany(item => item.Item2)
98 });
99 foreach (var g in statesByFinger)
100 {
101 Vector3 fingerDebugFeatureTotalDisp = fingerOffset;
102 foreach (var config in g.FingerFeatures)
103 {
104 var fingerFeatureDebugVisInst = Instantiate(_fingerFeatureDebugVisualPrefab, _fingerFeatureParent);
105 var debugVisComp = fingerFeatureDebugVisInst.GetComponent<FingerFeatureDebugVisual>();
106
107 debugVisComp.Initialize(g.HandFinger, config, FingerFeatureStateProvider);
108 var debugVisTransform = debugVisComp.transform;
109 debugVisTransform.localScale = _fingerFeatureDebugLocalScale;
110 debugVisTransform.localRotation = Quaternion.identity;
111 debugVisTransform.localPosition = fingerDebugFeatureTotalDisp;
112
113 fingerDebugFeatureTotalDisp += _fingerFeatureSpacingVec;
114 }
115
116 fingerOffset += _fingerSpacingVec;
117 }
118
119 string shapeNames = "";
120 foreach (ShapeRecognizer shapeRecognizer in _shapeRecognizerActiveState.Shapes)
121 {
122 shapeNames += shapeRecognizer.ShapeName;
123 }
124
125 _targetText.text = $"{_shapeRecognizerActiveState.Handedness} Hand: {shapeNames} ";
126 }
127
128 private IEnumerable<ValueTuple<HandFinger, IReadOnlyList<ShapeRecognizer.FingerFeatureConfig>>> AllFeatureStates()
129 {
130 foreach (ShapeRecognizer shapeRecognizer in _shapeRecognizerActiveState.Shapes)
131 {
132 foreach (var handFingerConfigs in shapeRecognizer.GetFingerFeatureConfigs())
133 {
134 yield return handFingerConfigs;
135 }
136 }
137 }
138
139 protected virtual void OnDestroy()

141 Destroy(_material);
142 }
143
144 protected virtual void Update()
145 {
146 bool isActive = _shapeRecognizerActiveState.Active;
147 if (_lastActiveValue != isActive)
148 {
149 _material.color = isActive ? _activeColor : _normalColor;
150 _lastActiveValue = isActive;
151 }
152 }
153 }
