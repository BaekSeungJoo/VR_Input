33 public override void OnInspectorGUI()
34 {
35 float padding = 55;
36 var markdownPage = ((MarkdownPage)target);
37 if (!markdownPage)
38 {
39 base.OnInspectorGUI();
40 return;
41 }
42
43 var markdownFile = markdownPage.MarkdownFile;
44 if (!markdownFile)
45 {
46 base.OnInspectorGUI();
47 return;
48 }
49
50 var text = markdownFile.text;
51
52 if (_linkStyle == null)
53 {
54 _linkStyle = new GUIStyle(GUI.skin.label)
55 {
56 richText = true,
57 wordWrap = true,
58 alignment = TextAnchor.MiddleLeft
59 };
60 }
61
62 if (_normalTextStyle == null)
63 {
64 _normalTextStyle = new GUIStyle(GUI.skin.label)
65 {
66 wordWrap = true,
67 richText = true,
68 alignment = TextAnchor.MiddleLeft
69 };
70 }
71
72 Event currentEvent = Event.current;
73 Regex urlRegex = new Regex(@"(https?://[^\s]+)");
74 Regex imageRegex = new Regex(@"!\[(.*?)\]\‍((.*?)\‍)");
75 Regex splitRegex = new Regex(@"(!\[.*?\]\‍(.*?\‍))|(https?://[^\s]+)");
76 string[] parts = splitRegex.Split(text);
77
78 scrollView = GUILayout.BeginScrollView(scrollView);
79 var windowWidth = (OverrideWidth > 0 ? OverrideWidth : EditorGUIUtility.currentViewWidth) - padding;
80 GUILayout.BeginVertical(GUILayout.Width(windowWidth));
81 foreach (string part in parts)
82 {
83 if (imageRegex.IsMatch(part))
84 {
85 Match imageMatch = imageRegex.Match(part);
86
87 if (imageMatch.Success)
88 {
89 string imagePath = imageMatch.Groups[2].Value;
90
91 if (!_cachedImages.ContainsKey(imagePath))
92 {
93 if (urlRegex.IsMatch(imagePath))
94 {
95 LoadImageFromUrl(imagePath);
96 }
97 else
98 {
99 var path = AssetDatabase.GetAssetPath(markdownPage);
100 var dir = Path.GetDirectoryName(path);
101 Texture2D image = AssetDatabase.LoadAssetAtPath<Texture2D>(dir + "/" + imagePath);
102 if (!image)
103 {
104 // Get the path of target markdown file
105 string markdownPath = AssetDatabase.GetAssetPath(markdownFile);
106 // Get the directory of the markdown file
107 string markdownDir = System.IO.Path.GetDirectoryName(markdownPath);
108 image = AssetDatabase.LoadAssetAtPath<Texture2D>(Path.Combine(markdownDir,
109 imagePath));
110 if (!image) image = _badImageTex;
111 }
112
113 _cachedImages[imagePath] = image;
114 }
115 }
116
117 if (_cachedImages.TryGetValue(imagePath, out Texture2D img) && img && img != _badImageTex)
118 {
119 float aspectRatio = 1;
120 float width = img.width;
121 float height = img.height;
122 if (img.width > windowWidth - padding)
123 {
124 width = windowWidth - padding;
125 aspectRatio = img.width / (float) img.height;
126 height = width / aspectRatio;
127 }
128
129 if (null == _imageLabelStyle)
130 {
131 _imageLabelStyle = new GUIStyle(GUI.skin.label)
132 {
133 alignment = TextAnchor.MiddleCenter,
134 imagePosition = ImagePosition.ImageAbove
135 };
136 }
137
138 GUIContent content = new GUIContent(img);
139 Rect imageLabelRect = GUILayoutUtility.GetRect(content, _imageLabelStyle,
140 GUILayout.Height(height), GUILayout.Width(width));
141
142 if (GUI.Button(imageLabelRect, content, _imageLabelStyle))
143 {
144 ImageViewer.ShowWindow(img, Path.GetFileNameWithoutExtension(imagePath));
145 }
146 }
147 }
148 }
149 else if (urlRegex.IsMatch(part))
150 {
151 EditorGUILayout.BeginHorizontal();
152
153 GUILayout.Space(EditorGUI.indentLevel * 15);
154 GUILayout.Label("<color=blue>" + part + "</color>", _linkStyle, GUILayout.MaxWidth(windowWidth));
155
156 Rect linkRect = GUILayoutUtility.GetLastRect();
157 if (currentEvent.type == EventType.MouseDown && linkRect.Contains(currentEvent.mousePosition))
158 {
159 Application.OpenURL(part);
160 }
161
162 EditorGUILayout.EndHorizontal();
163 }
164 else
165 {
166 EditorGUILayout.LabelField(ParseMarkdown(part), _normalTextStyle, GUILayout.MaxWidth(windowWidth));
167 }
168 }
169
170 GUILayout.EndVertical();
171 GUILayout.EndScrollView();
172 }

174 public static string ParseMarkdown(string markdown)
175 {
176 // Headers
177 markdown = Regex.Replace(markdown, @"^######\s(.*?)$", "<size=14><b>$1</b></size>", RegexOptions.Multiline);
178 markdown = Regex.Replace(markdown, @"^#####\s(.*?)$", "<size=16><b>$1</b></size>", RegexOptions.Multiline);
179 markdown = Regex.Replace(markdown, @"^####\s(.*?)$", "<size=18><b>$1</b></size>", RegexOptions.Multiline);
180 markdown = Regex.Replace(markdown, @"^###\s(.*?)$", "<size=20><b>$1</b></size>", RegexOptions.Multiline);
181 markdown = Regex.Replace(markdown, @"^##\s(.*?)$", "<size=22><b>$1</b></size>", RegexOptions.Multiline);
182 markdown = Regex.Replace(markdown, @"^#\s(.*?)$", "<size=24><b>$1</b></size>", RegexOptions.Multiline);
183
184 // Bold
185 markdown = Regex.Replace(markdown, @"\*\*(.*?)\*\*", "<b>$1</b>", RegexOptions.Multiline);
186
187 // Italic
188 markdown = Regex.Replace(markdown, @"\*(.*?)\*", "<i>$1</i>", RegexOptions.Multiline);
189
190 // Code blocks
191 markdown = Regex.Replace(markdown, @"(?s)```(.*?)```", m =>
192 {
193 var codeLines = m.Groups[1].Value.Trim().Split('\n');
194 string result = string.Empty;
195 foreach (var line in codeLines)
196 {
197 result += $" <color=#a1b56c>{line}</color>\n";
198 }
199
200 return result;
201 }, RegexOptions.Multiline);
202
203 // Raw Urls
204 markdown = Regex.Replace(markdown, @"(https?:\/\/[^\s""'<>]+)",
205 "<link><color=#a1b56c><u>$1</u></color></link>", RegexOptions.Multiline);
206
207 // Unordered lists
208 markdown = Regex.Replace(markdown, @"^\s*\*\s(.*?)$", "• $1", RegexOptions.Multiline);
209
210 // Ordered lists
211 markdown = Regex.Replace(markdown, @"^(\d+)\.\s(.*?)$", "$1. $2", RegexOptions.Multiline);
212
213 return markdown;
214 }

216 private void LoadImageFromUrl(string url)
217 {
218 _cachedImages[url] = _badImageTex;
219 UnityWebRequest request = UnityWebRequestTexture.GetTexture(url);
220 request.SendWebRequest().completed += operation =>
221 {
222 if (request.responseCode == 200)
223 {
224 Texture2D texture = DownloadHandlerTexture.GetContent(request);
225 _cachedImages[url] = texture;
226 Repaint();
227 }
228 else
229 {
230 Debug.LogError($"Failed to load image from URL [Error {request.responseCode}]: {url}");
231 }
232 };
233 }
