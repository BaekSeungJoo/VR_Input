281 private Vector2 GetRawMoveVector()
282 {
283 Vector2 move = Vector2.zero;
284 move.x = Input.GetAxisRaw(m_HorizontalAxis);
285 move.y = Input.GetAxisRaw(m_VerticalAxis);
286
287 if (Input.GetButtonDown(m_HorizontalAxis))
288 {
289 if (move.x < 0)
290 move.x = -1f;
291 if (move.x > 0)
292 move.x = 1f;
293 }
294
295 if (Input.GetButtonDown(m_VerticalAxis))
296 {
297 if (move.y < 0)
298 move.y = -1f;
299 if (move.y > 0)
300 move.y = 1f;
301 }
302
303 return move;
304 }

309 private bool SendMoveEventToSelectedObject()
310 {
311 float time = Time.unscaledTime;
312
313 if (!AllowMoveEventProcessing(time))
314 return false;
315
316 Vector2 movement = GetRawMoveVector();
317 // Debug.Log(m_ProcessingEvent.rawType + " axis:" + m_AllowAxisEvents + " value:" + "(" + x + "," + y + ")");
318 var axisEventData = GetAxisEventData(movement.x, movement.y, 0.6f);
319 if (!Mathf.Approximately(axisEventData.moveVector.x, 0f)
320 || !Mathf.Approximately(axisEventData.moveVector.y, 0f))
321 {
322 ExecuteEvents.Execute(eventSystem.currentSelectedGameObject, axisEventData, ExecuteEvents.moveHandler);
323 }
324
325 m_NextAction = time + 1f / m_InputActionsPerSecond;
326 return axisEventData.used;
327 }

343 private void ProcessMousePress(MouseButtonEventData data)
344 {
345 var pointerEvent = data.buttonData;
346 var currentOverGo = pointerEvent.pointerCurrentRaycast.gameObject;
347
348 // PointerDown notification
349 if (data.PressedThisFrame())
350 {
351 pointerEvent.eligibleForClick = true;
352 pointerEvent.delta = Vector2.zero;
353 pointerEvent.dragging = false;
354 pointerEvent.useDragThreshold = true;
355 pointerEvent.pressPosition = pointerEvent.position;
356 if (pointerEvent.IsVRPointer())
357 {
358#if ENABLE_LEGACY_INPUT_MANAGER
359 pointerEvent.SetSwipeStart(Input.mousePosition);
360#endif
361 }
362
363 pointerEvent.pointerPressRaycast = pointerEvent.pointerCurrentRaycast;
364
365 DeselectIfSelectionChanged(currentOverGo, pointerEvent);
366
367 // search for the control that will receive the press
368 // if we can't find a press handler set the press
369 // handler to be what would receive a click.
370 var newPressed =
371 ExecuteEvents.ExecuteHierarchy(currentOverGo, pointerEvent, ExecuteEvents.pointerDownHandler);
372
373 // didnt find a press handler... search for a click handler
374 if (newPressed == null)
375 newPressed = ExecuteEvents.GetEventHandler<IPointerClickHandler>(currentOverGo);
376
377 // Debug.Log("Pressed: " + newPressed);
378
379 float time = Time.unscaledTime;
380
381 if (newPressed == pointerEvent.lastPress)
382 {
383 var diffTime = time - pointerEvent.clickTime;
384 if (diffTime < 0.3f)
385 ++pointerEvent.clickCount;
386 else
387 pointerEvent.clickCount = 1;
388
389 pointerEvent.clickTime = time;
390 }
391 else
392 {
393 pointerEvent.clickCount = 1;
394 }
395
396 pointerEvent.pointerPress = newPressed;
397 pointerEvent.rawPointerPress = currentOverGo;
398
399 pointerEvent.clickTime = time;
400
401 // Save the drag handler as well
402 pointerEvent.pointerDrag = ExecuteEvents.GetEventHandler<IDragHandler>(currentOverGo);
403
404 if (pointerEvent.pointerDrag != null)
405 ExecuteEvents.Execute(pointerEvent.pointerDrag, pointerEvent,
406 ExecuteEvents.initializePotentialDrag);
407 }
408
409 // PointerUp notification
410 if (data.ReleasedThisFrame())
411 {
412 // Debug.Log("Executing pressup on: " + pointer.pointerPress);
413 ExecuteEvents.Execute(pointerEvent.pointerPress, pointerEvent, ExecuteEvents.pointerUpHandler);
414
415 // Debug.Log("KeyCode: " + pointer.eventData.keyCode);
416
417 // see if we mouse up on the same element that we clicked on...
418 var pointerUpHandler = ExecuteEvents.GetEventHandler<IPointerClickHandler>(currentOverGo);
419
420 // PointerClick and Drop events
421 if (pointerEvent.pointerPress == pointerUpHandler && pointerEvent.eligibleForClick)
422 {
423 ExecuteEvents.Execute(pointerEvent.pointerPress, pointerEvent, ExecuteEvents.pointerClickHandler);
424 }
425 else if (pointerEvent.pointerDrag != null)
426 {
427 ExecuteEvents.ExecuteHierarchy(currentOverGo, pointerEvent, ExecuteEvents.dropHandler);
428 }
429
430 pointerEvent.eligibleForClick = false;
431 pointerEvent.pointerPress = null;
432 pointerEvent.rawPointerPress = null;
433
434 if (pointerEvent.pointerDrag != null && pointerEvent.dragging)
435 ExecuteEvents.Execute(pointerEvent.pointerDrag, pointerEvent, ExecuteEvents.endDragHandler);
436
437 pointerEvent.dragging = false;
438 pointerEvent.pointerDrag = null;
439
440 // redo pointer enter / exit to refresh state
441 // so that if we moused over somethign that ignored it before
442 // due to having pressed on something else
443 // it now gets it.
444 if (currentOverGo != pointerEvent.pointerEnter)
445 {
446 HandlePointerExitAndEnter(pointerEvent, null);
447 HandlePointerExitAndEnter(pointerEvent, currentOverGo);
448 }
449 }
450 }

461 private void ProcessMouseEvent(MouseState mouseData)
462 {
463 var pressed = mouseData.AnyPressesThisFrame();
464 var released = mouseData.AnyReleasesThisFrame();
465
466 var leftButtonData = mouseData.GetButtonState(PointerEventData.InputButton.Left).eventData;
467
468 if (!UseMouse(pressed, released, leftButtonData.buttonData))
469 return;
470
471 // Process the first mouse button fully
472 ProcessMousePress(leftButtonData);
473 ProcessMove(leftButtonData.buttonData);
474 ProcessDrag(leftButtonData.buttonData);
475
476 // Now process right / middle clicks
477 ProcessMousePress(mouseData.GetButtonState(PointerEventData.InputButton.Right).eventData);
478 ProcessDrag(mouseData.GetButtonState(PointerEventData.InputButton.Right).eventData.buttonData);
479 ProcessMousePress(mouseData.GetButtonState(PointerEventData.InputButton.Middle).eventData);
480 ProcessDrag(mouseData.GetButtonState(PointerEventData.InputButton.Middle).eventData.buttonData);
481
482 if (!Mathf.Approximately(leftButtonData.buttonData.scrollDelta.sqrMagnitude, 0.0f))
483 {
484 var scrollHandler = ExecuteEvents.GetEventHandler<IScrollHandler>(
485 leftButtonData.buttonData.pointerCurrentRaycast.gameObject);
486 ExecuteEvents.ExecuteHierarchy(scrollHandler, leftButtonData.buttonData, ExecuteEvents.scrollHandler);
487 }
488 }

612 static Vector3 GetRectTransformNormal(RectTransform rectTransform)
613 {
614 Vector3[] corners = new Vector3[4];
615 rectTransform.GetWorldCorners(corners);
616 Vector3 BottomEdge = corners[3] - corners[0];
617 Vector3 LeftEdge = corners[1] - corners[0];
618 rectTransform.GetWorldCorners(corners);
619 return Vector3.Cross(BottomEdge, LeftEdge).normalized;
620 }

625 // The following 2 functions are equivalent to PointerInputModule.GetMousePointerEventData but are customized to
626 // get data for ray pointers and canvas mouse pointers.
627
632 virtual protected MouseState GetGazePointerData()
633 {
634 // Get the OVRRayPointerEventData reference
635 OVRPointerEventData leftData;
636 GetPointerData(kMouseLeftId, out leftData, true);
637 leftData.Reset();
638
639 //Now set the world space ray. This ray is what the user uses to point at UI elements
640 leftData.worldSpaceRay = new Ray(rayTransform.position, rayTransform.forward);
641 leftData.scrollDelta = GetExtraScrollDelta();
642
643 //Populate some default values
644 leftData.button = PointerEventData.InputButton.Left;
645 leftData.useDragThreshold = true;
646 // Perform raycast to find intersections with world
647 eventSystem.RaycastAll(leftData, m_RaycastResultCache);
648 var raycast = FindFirstRaycast(m_RaycastResultCache);
649 leftData.pointerCurrentRaycast = raycast;
650 m_RaycastResultCache.Clear();
651
652 m_Cursor.SetCursorRay(rayTransform);
653
654 OVRRaycaster ovrRaycaster = raycast.module as OVRRaycaster;
655 // We're only interested in intersections from OVRRaycasters
656 if (ovrRaycaster)
657 {
658 // The Unity UI system expects event data to have a screen position
659 // so even though this raycast came from a world space ray we must get a screen
660 // space position for the camera attached to this raycaster for compatability
661 leftData.position = ovrRaycaster.GetScreenPosition(raycast);
662
663 // Find the world position and normal the Graphic the ray intersected
664 RectTransform graphicRect = raycast.gameObject.GetComponent<RectTransform>();
665 if (graphicRect != null)
666 {
667 // Set are gaze indicator with this world position and normal
668 Vector3 worldPos = raycast.worldPosition;
669 Vector3 normal = GetRectTransformNormal(graphicRect);
670 m_Cursor.SetCursorStartDest(rayTransform.position, worldPos, normal);
671 }
672 }
673
674 // Now process physical raycast intersections
675 OVRPhysicsRaycaster physicsRaycaster = raycast.module as OVRPhysicsRaycaster;
676 if (physicsRaycaster)
677 {
678 Vector3 position = raycast.worldPosition;
679
680 if (performSphereCastForGazepointer)
681 {
682 // Here we cast a sphere into the scene rather than a ray. This gives a more accurate depth
683 // for positioning a circular gaze pointer
684 List<RaycastResult> results = new List<RaycastResult>();
685 physicsRaycaster.Spherecast(leftData, results, m_SpherecastRadius);
686 if (results.Count > 0 && results[0].distance < raycast.distance)
687 {
688 position = results[0].worldPosition;
689 }
690 }
691
692 leftData.position = physicsRaycaster.GetScreenPos(raycast.worldPosition);
693
694 m_Cursor.SetCursorStartDest(rayTransform.position, position, raycast.worldNormal);
695 }
696
697 // Stick default data values in right and middle slots for compatability
698
699 // copy the apropriate data into right and middle slots
700 OVRPointerEventData rightData;
701 GetPointerData(kMouseRightId, out rightData, true);
702 CopyFromTo(leftData, rightData);
703 rightData.button = PointerEventData.InputButton.Right;
704
705 OVRPointerEventData middleData;
706 GetPointerData(kMouseMiddleId, out middleData, true);
707 CopyFromTo(leftData, middleData);
708 middleData.button = PointerEventData.InputButton.Middle;
709
710
711 m_MouseState.SetButtonState(PointerEventData.InputButton.Left,
712 GetGazeButtonState(), leftData);
713 m_MouseState.SetButtonState(PointerEventData.InputButton.Right,
714 PointerEventData.FramePressState.NotChanged, rightData);
715 m_MouseState.SetButtonState(PointerEventData.InputButton.Middle,
716 PointerEventData.FramePressState.NotChanged, middleData);
717 return m_MouseState;
718 }

724 protected MouseState GetCanvasPointerData()
725 {
726 // Get the OVRRayPointerEventData reference
727 PointerEventData leftData;
728 GetPointerData(kMouseLeftId, out leftData, true);
729 leftData.Reset();
730
731 // Setup default values here. Set position to zero because we don't actually know the pointer
732 // positions. Each canvas knows the position of its canvas pointer.
733 leftData.position = Vector2.zero;
734#if ENABLE_LEGACY_INPUT_MANAGER
735 leftData.scrollDelta = Input.mouseScrollDelta;
736#endif
737 leftData.button = PointerEventData.InputButton.Left;
738
739 if (activeGraphicRaycaster)
740 {
741 // Let the active raycaster find intersections on its canvas
742 activeGraphicRaycaster.RaycastPointer(leftData, m_RaycastResultCache);
743 var raycast = FindFirstRaycast(m_RaycastResultCache);
744 leftData.pointerCurrentRaycast = raycast;
745 m_RaycastResultCache.Clear();
746
747 OVRRaycaster ovrRaycaster = raycast.module as OVRRaycaster;
748 if (ovrRaycaster) // raycast may not actually contain a result
749 {
750 // The Unity UI system expects event data to have a screen position
751 // so even though this raycast came from a world space ray we must get a screen
752 // space position for the camera attached to this raycaster for compatability
753 Vector2 position = ovrRaycaster.GetScreenPosition(raycast);
754
755 leftData.delta = position - leftData.position;
756 leftData.position = position;
757 }
758 }
759
760 // copy the apropriate data into right and middle slots
761 PointerEventData rightData;
762 GetPointerData(kMouseRightId, out rightData, true);
763 CopyFromTo(leftData, rightData);
764 rightData.button = PointerEventData.InputButton.Right;
765
766 PointerEventData middleData;
767 GetPointerData(kMouseMiddleId, out middleData, true);
768 CopyFromTo(leftData, middleData);
769 middleData.button = PointerEventData.InputButton.Middle;
770
771#if ENABLE_LEGACY_INPUT_MANAGER
772 m_MouseState.SetButtonState(PointerEventData.InputButton.Left, StateForMouseButton(0), leftData);
773 m_MouseState.SetButtonState(PointerEventData.InputButton.Right, StateForMouseButton(1), rightData);
774 m_MouseState.SetButtonState(PointerEventData.InputButton.Middle, StateForMouseButton(2), middleData);
775#endif
776 return m_MouseState;
777 }

786 private bool ShouldStartDrag(PointerEventData pointerEvent)
787 {
788 if (!pointerEvent.useDragThreshold)
789 return true;
790
791 if (!pointerEvent.IsVRPointer())
792 {
793 // Same as original behaviour for canvas based pointers
794 return (pointerEvent.pressPosition - pointerEvent.position).sqrMagnitude >=
795 eventSystem.pixelDragThreshold * eventSystem.pixelDragThreshold;
796 }
797 else
798 {
799#if UNITY_ANDROID && !UNITY_EDITOR // On android allow swiping to start drag
800 if (useSwipeScroll && ((Vector3)pointerEvent.GetSwipeStart() - Input.mousePosition).magnitude > swipeDragThreshold)
801 {
802 return true;
803 }
804#endif
805 // When it's not a screen space pointer we have to look at the angle it moved rather than the pixels distance
806 // For gaze based pointing screen-space distance moved will always be near 0
807 Vector3 cameraPos = pointerEvent.pressEventCamera.transform.position;
808 Vector3 pressDir = (pointerEvent.pointerPressRaycast.worldPosition - cameraPos).normalized;
809 Vector3 currentDir = (pointerEvent.pointerCurrentRaycast.worldPosition - cameraPos).normalized;
810 return Vector3.Dot(pressDir, currentDir) < Mathf.Cos(Mathf.Deg2Rad * (angleDragThreshold));
811 }
812 }

833 protected Vector2 SwipeAdjustedPosition(Vector2 originalPosition, PointerEventData pointerEvent)
834 {
835#if UNITY_ANDROID && !UNITY_EDITOR
836 // On android we use the touchpad position (accessed through Input.mousePosition) to modify
837 // the effective cursor position for events related to dragging. This allows the user to
838 // use the touchpad to drag draggable UI elements
839 if (useSwipeScroll)
840 {
841 Vector2 delta = (Vector2)Input.mousePosition - pointerEvent.GetSwipeStart();
842 if (InvertSwipeXAxis)
843 delta.x *= -1;
844 return originalPosition + delta * swipeDragScale;
845 }
846#endif
847 return originalPosition;

857 protected override void ProcessDrag(PointerEventData pointerEvent)
858 {
859 Vector2 originalPosition = pointerEvent.position;
860 bool moving = IsPointerMoving(pointerEvent);
861 if (moving && pointerEvent.pointerDrag != null
862 && !pointerEvent.dragging
863 && ShouldStartDrag(pointerEvent))
864 {
865 if (pointerEvent.IsVRPointer())
866 {
867 //adjust the position used based on swiping action. Allowing the user to
868 //drag items by swiping on the touchpad
869 pointerEvent.position = SwipeAdjustedPosition(originalPosition, pointerEvent);
870 }
871
872 ExecuteEvents.Execute(pointerEvent.pointerDrag, pointerEvent, ExecuteEvents.beginDragHandler);
873 pointerEvent.dragging = true;
874 }
875
876 // Drag notification
877 if (pointerEvent.dragging && moving && pointerEvent.pointerDrag != null)
878 {
879 if (pointerEvent.IsVRPointer())
880 {
881 pointerEvent.position = SwipeAdjustedPosition(originalPosition, pointerEvent);
882 }
883
884 // Before doing drag we should cancel any pointer down state
885 // And clear selection!
886 if (pointerEvent.pointerPress != pointerEvent.pointerDrag)
887 {
888 ExecuteEvents.Execute(pointerEvent.pointerPress, pointerEvent, ExecuteEvents.pointerUpHandler);
889
890 pointerEvent.eligibleForClick = false;
891 pointerEvent.pointerPress = null;
892 pointerEvent.rawPointerPress = null;
893 }
894
895 ExecuteEvents.Execute(pointerEvent.pointerDrag, pointerEvent, ExecuteEvents.dragHandler);
896 }
