21using UnityEngine;
22using UnityEngine.Assertions;
23using UnityEngine.UI;
24
25namespace Oculus.Interaction.Samples
26{
27 public class CarouselView : MonoBehaviour
28 {
29 [SerializeField]
30 private RectTransform _viewport;
31
32 [SerializeField]
33 private RectTransform _content;
34
35 [SerializeField]
36 private AnimationCurve _easeCurve = AnimationCurve.EaseInOut(0, 0, 1, 1);
37
38 [SerializeField, Optional]
39 private GameObject _emptyCarouselVisuals;
40
41 public int CurrentChildIndex => _currentChildIndex;
42
43 public RectTransform ContentArea => _content;
44
45 private int _currentChildIndex = 0;
46 private float _scrollVal = 0;
47
48 protected virtual void Start()
49 {
50 this.AssertField(_viewport, nameof(_viewport));
51 this.AssertField(_content, nameof(_content));
52 }
53
54 public void ScrollRight()
55 {
56 if (_content.childCount <= 1)
57 {
58 return;
59 }
60 else if (_currentChildIndex > 0)
61 {
62 RectTransform currentChild = GetCurrentChild();
63 _content.GetChild(0).SetAsLastSibling();
64 LayoutRebuilder.ForceRebuildLayoutImmediate(_content);
65 ScrollToChild(currentChild, 1);
66 }
67 else
68 {
69 _currentChildIndex++;
70 }
71 _scrollVal = Time.time;
72 }
73
74 public void ScrollLeft()
75 {
76 if (_content.childCount <= 1)
77 {
78 return;
79 }
80 else if (_currentChildIndex < _content.childCount - 1)
81 {
82 RectTransform currentChild = GetCurrentChild();
83 _content.GetChild(_content.childCount - 1).SetAsFirstSibling();
84 LayoutRebuilder.ForceRebuildLayoutImmediate(_content);
85 ScrollToChild(currentChild, 1);
86 }
87 else
88 {
89 _currentChildIndex--;
90 }
91 _scrollVal = Time.time;
92 }
93
94 private RectTransform GetCurrentChild()
95 {
96 return _content.GetChild(_currentChildIndex) as RectTransform;
97 }
98
99 private void ScrollToChild(RectTransform child, float amount01)
100 {
101 if (child == null)
102 {
103 return;
104 }
105
106 amount01 = Mathf.Clamp01(amount01);
107
108 Vector3 viewportCenter = _viewport.TransformPoint(_viewport.rect.center);
109 Vector3 imageCenter = child.TransformPoint(child.rect.center);
110 Vector3 offset = imageCenter - viewportCenter;
111
112 if (offset.sqrMagnitude > float.Epsilon)
113 {
114 Vector3 targetPosition = _content.position - offset;
115 float lerp = Mathf.Clamp01(_easeCurve.Evaluate(amount01));
116 _content.position = Vector3.Lerp(_content.position, targetPosition, lerp);
117 }
118 }
119
120 protected virtual void Update()
121 {
122 _currentChildIndex = Mathf.Clamp(
123 _currentChildIndex, 0, _content.childCount - 1);
124
125 bool hasImages = _content.childCount > 0;
126 if (hasImages)
127 {
128 RectTransform currentImage = GetCurrentChild();
129 ScrollToChild(currentImage, Time.time - _scrollVal);
130 }
131
132 if (_emptyCarouselVisuals != null)
133 {
134 _emptyCarouselVisuals.SetActive(!hasImages);
135 }
136 }
137 }
138}
