26 // A top level component that provides hand pose data, pinch states, and more.
27 // Rather than sourcing data directly from the runtime layer, provides one
28 // level of abstraction so that the aforementioned data can be injected
29 // from other sources.
30 public class Hand : DataModifier<HandDataAsset>, IHand
31 {
32 public Handedness Handedness => GetData().Config.Handedness;
33
34 public ITrackingToWorldTransformer TrackingToWorldTransformer =>
35 GetData().Config.TrackingToWorldTransformer;
36
37 public HandSkeleton HandSkeleton => GetData().Config.HandSkeleton;
38
39 private HandJointCache _jointPosesCache;
40
41 public event Action WhenHandUpdated = delegate { };
42
43 public bool IsConnected => GetData().IsDataValidAndConnected;
44 public bool IsHighConfidence => GetData().IsHighConfidence;
45 public bool IsDominantHand => GetData().IsDominantHand;
46
47 public float Scale => (TrackingToWorldTransformer != null
48 ? TrackingToWorldTransformer.Transform.lossyScale.x
49 : 1) * GetData().HandScale;
50
51 private static readonly Vector3 PALM_LOCAL_OFFSET = new Vector3(0.08f, -0.01f, 0.0f);
52
53 protected override void Apply(HandDataAsset data)
54 {
55 // Default implementation does nothing, to allow instantiation of this modifier directly
56 }
57
58 public override void MarkInputDataRequiresUpdate()
59 {
60 base.MarkInputDataRequiresUpdate();
61
62 if (Started)
63 {
64 InitializeJointPosesCache();
65 WhenHandUpdated.Invoke();
66 }
67 }
68
69 private void InitializeJointPosesCache()
70 {
71 if (_jointPosesCache == null && GetData().IsDataValidAndConnected)
72 {
73 _jointPosesCache = new HandJointCache(HandSkeleton);
74 }
75 }
76
77 private void CheckJointPosesCacheUpdate()
78 {
79 if (_jointPosesCache != null
80 && CurrentDataVersion != _jointPosesCache.LocalDataVersion)
81 {
82 _jointPosesCache.Update(GetData(), CurrentDataVersion);
83 }
84 }

86 #region IHandState implementation
87
88 public bool GetFingerIsPinching(HandFinger finger)
89 {
90 HandDataAsset currentData = GetData();
91 return currentData.IsConnected && currentData.IsFingerPinching[(int)finger];
92 }
93
94 public bool GetIndexFingerIsPinching()

97 }
98
99 public bool IsPointerPoseValid => IsPoseOriginAllowed(GetData().PointerPoseOrigin);
100
101 public bool GetPointerPose(out Pose pose)
102 {
103 HandDataAsset currentData = GetData();
104 return ValidatePose(currentData.PointerPose, currentData.PointerPoseOrigin,
105 out pose);
106 }
107
108 public bool GetJointPose(HandJointId handJointId, out Pose pose)
109 {
110 pose = Pose.identity;
111
112 if (!IsTrackedDataValid
113 || _jointPosesCache == null
114 || !GetRootPose(out Pose rootPose))
115 {
116 return false;
117 }
118 CheckJointPosesCacheUpdate();
119 pose = _jointPosesCache.WorldJointPose(handJointId, rootPose, Scale);
120 return true;
121 }
122
123 public bool GetJointPoseLocal(HandJointId handJointId, out Pose pose)
124 {
125 pose = Pose.identity;
126 if (!GetJointPosesLocal(out ReadOnlyHandJointPoses localJointPoses))
127 {
128 return false;
129 }
130
131 pose = localJointPoses[(int)handJointId];
132 return true;
133 }
134
135 public bool GetJointPosesLocal(out ReadOnlyHandJointPoses localJointPoses)
136 {
137 if (!IsTrackedDataValid || _jointPosesCache == null)
138 {
139 localJointPoses = ReadOnlyHandJointPoses.Empty;
140 return false;
141 }
142 CheckJointPosesCacheUpdate();
143 return _jointPosesCache.GetAllLocalPoses(out localJointPoses);
144 }
145
146 public bool GetJointPoseFromWrist(HandJointId handJointId, out Pose pose)
147 {
148 pose = Pose.identity;
149 if (!GetJointPosesFromWrist(out ReadOnlyHandJointPoses jointPosesFromWrist))
150 {
151 return false;
152 }
153
154 pose = jointPosesFromWrist[(int)handJointId];
155 return true;
156 }
157
158 public bool GetJointPosesFromWrist(out ReadOnlyHandJointPoses jointPosesFromWrist)
159 {
160 if (!IsTrackedDataValid || _jointPosesCache == null)
161 {
162 jointPosesFromWrist = ReadOnlyHandJointPoses.Empty;
163 return false;
164 }
165 CheckJointPosesCacheUpdate();
166 return _jointPosesCache.GetAllPosesFromWrist(out jointPosesFromWrist);
167 }
168
169 public bool GetPalmPoseLocal(out Pose pose)
170 {
171 Quaternion rotationQuat = Quaternion.identity;
172 Vector3 offset = PALM_LOCAL_OFFSET;
173 if (Handedness == Handedness.Left)
174 {
175 offset = -offset;
176 }
177 pose = new Pose(offset * Scale, rotationQuat);
178 return true;
179 }
180
181 public bool GetFingerIsHighConfidence(HandFinger finger)
182 {
183 return GetData().IsFingerHighConfidence[(int)finger];
184 }
185
186 public float GetFingerPinchStrength(HandFinger finger)
187 {
188 return GetData().FingerPinchStrength[(int)finger];
189 }
190
191 public bool IsTrackedDataValid => IsPoseOriginAllowed(GetData().RootPoseOrigin);
192
193 public bool GetRootPose(out Pose pose)
194 {
195 HandDataAsset currentData = GetData();
196 return ValidatePose(currentData.Root, currentData.RootPoseOrigin, out pose);
197 }

199 #endregion

202 private bool ValidatePose(in Pose sourcePose, PoseOrigin sourcePoseOrigin, out Pose pose)
203 {
204 if (IsPoseOriginDisallowed(sourcePoseOrigin))
205 {
206 pose = Pose.identity;
207 return false;
208 }
209
210 pose = TrackingToWorldTransformer != null
211 ? TrackingToWorldTransformer.ToWorldPose(sourcePose)
212 : sourcePose;
213
214 return true;
215 }
216
217 private bool IsPoseOriginAllowed(PoseOrigin poseOrigin)
218 {
219 return poseOrigin != PoseOrigin.None;
220 }
221
222 private bool IsPoseOriginDisallowed(PoseOrigin poseOrigin)
223 {
224 return poseOrigin == PoseOrigin.None;
225 }

233 }

235 #endregion
