33 public class FeatureConfigList
34 {
35 private readonly SerializedProperty _root;
36 private uint _flags;
37 private readonly IReadOnlyDictionary<int, FeatureDescription> _featureDescriptions;
38
39 public FeatureConfigList(SerializedProperty root,
40 IReadOnlyDictionary<int, FeatureDescription> featureDescriptions)
41 {
42 Assert.IsNotNull(root);
43
44 _root = root;
45 _flags = GetFlags(_root);
46 _featureDescriptions = featureDescriptions;
47 }
48
49 private uint GetFlags(SerializedProperty root)
50 {
51 if (!root.isArray)
52 {
53 return 0U;
54 }
55
56 uint flags = 0U;
57 for (int i = 0; i < root.arraySize; i++)
58 {
59 var elemProp = root.GetArrayElementAtIndex(i);
60 var featureProp = elemProp.FindPropertyRelative("_feature");
61 flags |= 1U << featureProp.enumValueIndex;
62 }
63
64 return flags;
65 }
66
67 public uint Flags
68 {
69 get => _flags;
70 set
71 {
72 uint flagsToCreate = value;
73 for (int i = 0; i < _root.arraySize;)
74 {
75 var elemProp = _root.GetArrayElementAtIndex(i);
76 var featureProp = elemProp.FindPropertyRelative("_feature");
77 uint propFlags = (1U << featureProp.enumValueIndex);
78 if ((flagsToCreate & propFlags) == 0U)
79 {
80 // Feature is in list, but not flags... delete list entry.
81 _root.DeleteArrayElementAtIndex(i);
82 }
83 else
84 {
85 // Feature is in list, and in flags; remove from list of things we need to create.
86 flagsToCreate &= ~propFlags;
87 i++;
88 }
89 }
90
91 // Create missing elements.
92 foreach (var feature in _featureDescriptions.Keys)
93 {
94 uint flags = 1U << feature;
95 if ((flagsToCreate & flags) == 0U)
96 {
97 continue;
98 }
99
100 var lastIndex = _root.arraySize;
101 _root.InsertArrayElementAtIndex(lastIndex);
102 var model = new FeatureConfig(_root.GetArrayElementAtIndex(lastIndex));
103 model.Feature = feature;
104 model.FeatureState = _featureDescriptions[feature].FeatureStates[0].Id;
105 }
106
107 _flags = value;
108 }
109 }
110
111 public IEnumerable<FeatureConfig> ConfigModels
112 {
113 get
114 {
115 List<FeatureConfig> models = new List<FeatureConfig>(_root.arraySize);
116 for (int i = 0; i < _root.arraySize; i++)
117 {
118 models.Add(new FeatureConfig(_root.GetArrayElementAtIndex(i)));
119 }
120
121 return models;
122 }
123 }

177 public abstract class FeatureListPropertyDrawer : PropertyDrawer
178 {
179 public float ControlLineHeight => EditorGUIUtility.singleLineHeight + EditorGUIUtility.standardVerticalSpacing; // = 18 + 2
180 public float BottomMargin => EditorGUIUtility.standardVerticalSpacing * 2;
181 private const float IndentSize = 16;
182
183 public override float GetPropertyHeight(SerializedProperty property, GUIContent label)
184 {
185 var height = base.GetPropertyHeight(property, label); // includes height of first line
186
187 var model = CreateModel(property);
188 if (property.isExpanded)
189 {
190 var controlCount = model.ConfigModels.Count();
191 height += controlCount * ControlLineHeight;
192 height += BottomMargin;
193 }
194
195 return height;
196 }
197
198 public override bool CanCacheInspectorGUI(SerializedProperty property)
199 {
200 return true;
201 }
202
203 public override void OnGUI(Rect drawerPos, SerializedProperty property, GUIContent label)
204 {
205 EditorGUI.BeginProperty(drawerPos, label, property);
206
207 var oldIndent = EditorGUI.indentLevel;
208 EditorGUI.indentLevel = 0;
209
210 var labelPos = new Rect(drawerPos)
211 {
212 width = EditorGUIUtility.labelWidth - drawerPos.x,
213 height = EditorGUIUtility.singleLineHeight
214 };
215
216 var model = CreateModel(property);
217 property.isExpanded = EditorGUI.Foldout(labelPos, property.isExpanded, label, true);
218 if (property.isExpanded)
219 {
220 RenderExpanded(drawerPos, model);
221 }
222 else
223 {
224 RenderCollapsed(drawerPos, model);
225 }
226
227 EditorGUI.indentLevel = oldIndent;
228 EditorGUI.EndProperty();
229 }

231 private void RenderExpanded(Rect drawerPos, FeatureConfigList model) {
232 Rect controlPos = new Rect(drawerPos.x, drawerPos.y, drawerPos.width,
233 EditorGUIUtility.singleLineHeight);
234
235 var flagsPos = Indent(controlPos, EditorGUIUtility.labelWidth);
236 var newFlags = EnumToFlags(EditorGUI.EnumFlagsField(flagsPos, FlagsToEnum(model.Flags)));
237 if (newFlags != model.Flags)
238 {
239 model.Flags = newFlags;
240 }
241
242 controlPos = Indent(controlPos, IndentSize);
243 foreach (var configModel in model.ConfigModels)
244 {
245 controlPos.y += ControlLineHeight;
246
247 // Render the label
248 float indent = 0f;
249 var labelPos = Indent(controlPos, indent);
250 labelPos.width = EditorGUIUtility.labelWidth - IndentSize;
251 var featureName = FeatureToString(configModel.Feature);
252 featureName = ObjectNames.NicifyVariableName(featureName);
253
254 EditorGUI.PrefixLabel(labelPos, new GUIContent(featureName));
255
256 // Render the mode dropdown
257 indent += labelPos.width;
258 var modePos = Indent(controlPos, indent);
259 var allowedModes = GetAllowedModes(configModel);
260 if (allowedModes.Length > 1)
261 {
262 modePos.width = 70;
263 configModel.Mode =
264 (FeatureStateActiveMode)EditorGUI.EnumPopup(modePos, configModel.Mode);
265 }
266 else if (allowedModes.Length == 1)
267 {
268 configModel.Mode = allowedModes[0];
269 modePos.width = 15;
270 EditorGUI.SelectableLabel(modePos,
271 ObjectNames.NicifyVariableName(allowedModes[0] + ": "));
272 }
273 else
274 {
275 modePos.width = -2;
276 }
277
278 // Render the state dropdown
279 indent += modePos.width + 2;
280 var statePos = Indent(controlPos, indent);
281 var featureStates = GetStatesForFeature(configModel.Feature);
282 string[] options = featureStates.Select(fs => fs.Name).ToArray();
283 int selectedIndex = Array.FindIndex(featureStates, fs => fs.Id == configModel.FeatureState);
284 int newSelectedIndex = EditorGUI.Popup(statePos, selectedIndex, options);
285 if (newSelectedIndex != selectedIndex) {
286 configModel.FeatureState = featureStates[newSelectedIndex].Id;
287 }
288 }
289 }

291 private void RenderCollapsed(Rect drawerPos, FeatureConfigList model)
292 {
293 Rect controlPos = drawerPos;
294 controlPos.height = EditorGUIUtility.singleLineHeight;
295
296 var valuePos = Indent(controlPos, EditorGUIUtility.labelWidth + 2);
297 valuePos.width = drawerPos.width - valuePos.x;
298 var flagsEnum = FlagsToEnum(model.Flags);
299 var values = Enum.GetValues(flagsEnum.GetType())
300 .Cast<Enum>()
301 .Where(e => flagsEnum.HasFlag(e))
302 .Select(e => e.ToString());
303 EditorGUI.SelectableLabel(valuePos, String.Join(", ", values));
304 }

315 protected virtual FeatureStateActiveMode[] GetAllowedModes(FeatureConfig model)
316 {
317 var statesForFeature = GetStatesForFeature(model.Feature);
318 if (statesForFeature.Length > 2)
