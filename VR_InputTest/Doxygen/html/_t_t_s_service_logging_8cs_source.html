156 #region LOG HELPERS
157 // Generate request data & apply initial annotations
158 private void LogStart(TTSClipData clipData)
159 {
160 TTSServiceRequestLog requestData = GetRequestData(clipData);
161 requestData.startTime = DateTime.Now;
162 requestData.annotations = new Dictionary<string, string>();
163 LogTimestamp(requestData, TTS_START_TIME_ANNOTATION);
164 LogAnnotate(requestData, TTS_FILETYPE_ANNOTATION, AudioStreamHandler.GetDecodeType(clipData.audioType).ToString());
165 LogAnnotate(requestData, TTS_FILESTREAM_ANNOTATION, clipData.queryStream.ToString(CultureInfo.InvariantCulture));
166 _requests[clipData.queryRequestId] = requestData;
167 }
168 // Get request data
169 private TTSServiceRequestLog GetRequestData(TTSClipData clipData)
170 {
171 if (_requests.ContainsKey(clipData.queryRequestId))
172 {
173 return _requests[clipData.queryRequestId];
174 }
175 return new TTSServiceRequestLog();
176 }
177 // Append the elapsed ms
178 private void LogTimestamp(TTSClipData clipData, string key)
179 {
180 LogTimestamp(GetRequestData(clipData), key);
181 }
182 // Append the elapsed ms
183 private void LogTimestamp(TTSServiceRequestLog requestData, string key)
184 {
185 LogAnnotate(requestData, key, DateTimeUtility.ElapsedMilliseconds.ToString());
186 }
187 // Append if possible
188 private void LogAnnotate(TTSServiceRequestLog requestData, string key, string value)
189 {
190 if (requestData.annotations != null)
191 {
192 requestData.annotations[key] = value;
193 }
194 }
195 // Append if possible
196 private void LogComplete(TTSClipData clipData, string error = null)
197 {
198 // Get data & ensure annotations exist
199 TTSServiceRequestLog requestData = GetRequestData(clipData);
200 if (requestData.annotations == null)
201 {
202 return;
203 }
204
205 // Add error & completion time
206 if (!string.IsNullOrEmpty(error))
207 {
208 LogAnnotate(requestData, TTS_ERROR_ANNOTATION, error);
209 }
210 LogTimestamp(requestData, TTS_FINISH_TIME_ANNOTATION);
211
212 // Send full log
213 if (_voiceSDKLoggerImpl != null)
214 {
215 _voiceSDKLoggerImpl.LogInteractionStart(clipData.queryRequestId, WitConstants.ENDPOINT_TTS);
216 foreach (var key in requestData.annotations.Keys)
217 {
218 _voiceSDKLoggerImpl.LogAnnotation(key, requestData.annotations[key]);
219 }
220 if (string.IsNullOrEmpty(error))
221 {
222 _voiceSDKLoggerImpl.LogInteractionEndSuccess();
223 }
224 else
225 {
226 _voiceSDKLoggerImpl.LogInteractionEndFailure(error);
227 }
228 }
229
230 // Remove cache
231 _requests.Remove(clipData.queryRequestId);
232 }
233 #endregion
