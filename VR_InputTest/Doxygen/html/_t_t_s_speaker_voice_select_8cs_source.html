43 // Fix voice selection if changed elsewhere
44 private void Update()

52 // Refresh dropdown using voice settings
53 private void RefreshDropdown()
54 {
55 if (!Speaker)
56 {
57 VLog.W("No speaker found");
58 return;
59 }
60 if (!Speaker.TTSService)
61 {
62 VLog.W("No speaker service found");
63 return;
64 }
65 if (!_dropdown)
66 {
67 VLog.W("No dropdown found");
68 return;
69 }
70
71 // Get all voice names & load dropdown
72 string[] voiceNames = Speaker.TTSService.GetAllPresetVoiceSettings()
73 .Select((voiceSetting) => voiceSetting.SettingsId).ToArray();
74 _dropdown.LoadDropdown(voiceNames);
75
76 // Get selected voice &
77 _dropdown.SelectOption(Speaker.presetVoiceID);
78 }
79
80 // Apply voice on option select
81 private void OnOptionSelected(string newOption)
82 {
83 if (!Speaker)
84 {
85 VLog.W("No speaker found");
86 return;
87 }
88 Speaker.presetVoiceID = newOption;
89 }
90 }
