106 public Vector3? ComputeWorldPosition(Camera camera)
107 {
108 if (camera == null) throw new ArgumentNullException(nameof(camera));
109 if (!Position.HasValue) return null;
110
111 var headPose = OVRPose.identity;
112 if (!OVRNodeStateProperties.GetNodeStatePropertyVector3(UnityEngine.XR.XRNode.Head,
113 NodeStatePropertyType.Position, OVRPlugin.Node.Head, OVRPlugin.Step.Render, out headPose.position))
114 return null;
115
116 if (!OVRNodeStateProperties.GetNodeStatePropertyQuaternion(UnityEngine.XR.XRNode.Head,
117 NodeStatePropertyType.Orientation, OVRPlugin.Node.Head, OVRPlugin.Step.Render,
118 out headPose.orientation))
119 return null;
120
121 headPose = headPose.Inverse();
122
123 var headTrackingPosition = headPose.position + headPose.orientation * Position.Value;
124 return camera.transform.localToWorldMatrix.MultiplyPoint(headTrackingPosition);
125 }

139 public Quaternion? ComputeWorldRotation(Camera camera)
140 {
141 if (camera == null) throw new ArgumentNullException(nameof(camera));
142 if (!Rotation.HasValue) return null;
143
144 if (!OVRNodeStateProperties.GetNodeStatePropertyQuaternion(UnityEngine.XR.XRNode.Head,
145 NodeStatePropertyType.Orientation, OVRPlugin.Node.Head, OVRPlugin.Step.Render,
146 out var headPoseRotation))
147 return null;
148
149 headPoseRotation = Quaternion.Inverse(headPoseRotation);
150
151 var headTrackingOrientation = headPoseRotation * Rotation.Value;
152 return camera.transform.rotation * headTrackingOrientation;
153 }
