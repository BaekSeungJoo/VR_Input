257 // Safely start microphone
258 public void SafeStartMicrophone()
259 {
260 // Can't start if inactive
261 if (!gameObject.activeInHierarchy)
262 {
263 return;
264 }
265 // Look for devices
266 if (Devices == null || Devices.Count == 0)
267 {
268 // Check for devices
269 RefreshMicDevices();
270 // None found
271 if (Devices == null || Devices.Count == 0)
272 {
273 return;
274 }
275 }
276
277 // Ignore if already setup & recording
278 string micID = CurrentDeviceName;
279 if (!string.IsNullOrEmpty(micID) && AudioClip != null && string.Equals(micID, AudioClip.name) && MicrophoneIsRecording(micID))
280 {
281 return;
282 }
283
284 // Set device
285 ChangeDevice(CurrentDeviceIndex < 0 ? 0 : CurrentDeviceIndex);
286 }

291 public void RefreshMicDevices()

362 public void StartRecording(int sampleLen = 10)
363 {
364 // Still unavailable, exit
365 if (!IsInputAvailable)
366 {
367 return;
368 }
369
370 // Stop recording if doing so
371 StopRecording();
372
373 IsRecording = true;
374
375 SampleDurationMS = sampleLen;
376
377 if (AudioClip)
378 {
379 Sample = new float[AudioClipSampleRate / MS_TO_SECONDS * SampleDurationMS * AudioClip.channels];
380
381 StartCoroutine(ReadRawAudio());
382
383#if !UNITY_WEBGL || UNITY_EDITOR
384 // Make sure we seek before we start reading data
385 MicrophoneGetPosition(CurrentDeviceName);
386 VLog.D("Started recording with " + CurrentDeviceName);
387#endif
388 if (OnStartRecording != null)
389 OnStartRecording.Invoke();
390 }
391 else
392 {
393 OnStartRecordingFailed.Invoke();
394 }
395 }
396
400 public void StopRecording()
401 {
402 if (!IsRecording) return;
403
404 IsRecording = false;
405
406 StopCoroutine(ReadRawAudio());
407
408 VLog.D("Stopped recording with " + CurrentDeviceName);
409 if (OnStopRecording != null)
410 OnStopRecording.Invoke();
411 }
412
413 IEnumerator ReadRawAudio()
414 {
415 int loops = 0;
416 int readAbsPos = MicrophoneGetPosition(CurrentDeviceName);
417 int prevPos = readAbsPos;
418 float[] temp = new float[Sample.Length];
419
420 while (AudioClip != null && MicrophoneIsRecording(CurrentDeviceName) && IsRecording)
421 {
422 bool isNewDataAvailable = true;
423
424 while (isNewDataAvailable && AudioClip)
425 {
426 int currPos = MicrophoneGetPosition(CurrentDeviceName);
427 if (currPos < prevPos)
428 loops++;
429 prevPos = currPos;
430
431 var currAbsPos = loops * AudioClip.samples + currPos;
432 var nextReadAbsPos = readAbsPos + temp.Length;
433 float levelMax = 0;
434
435 if (nextReadAbsPos < currAbsPos)
436 {
437 AudioClip.GetData(temp, readAbsPos % AudioClip.samples);
438
439 for (int i = 0; i < temp.Length; i++)
440 {
441 float wavePeak = temp[i] * temp[i];
442 if (levelMax < wavePeak)
443 {
444 levelMax = wavePeak;
445 }
446 }
447
448 Sample = temp;
449 m_SampleCount++;
450 OnSampleReady?.Invoke(m_SampleCount, Sample, levelMax);
451
452 readAbsPos = nextReadAbsPos;
453 isNewDataAvailable = true;
454 }
455 else
456 isNewDataAvailable = false;
457 }
458
459 yield return null;
460 }
