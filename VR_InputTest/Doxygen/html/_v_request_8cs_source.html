111 public virtual bool Request(UnityWebRequest unityRequest, RequestCompleteDelegate<UnityWebRequest> onComplete)
112 {
113 // Already setup
114 if (_request != null)
115 {
116 onComplete?.Invoke(unityRequest, "Request is already being performed");
117 return false;
118 }
119
120 // Setup
121 _request = unityRequest;
122 _onComplete = onComplete;
123 IsPerforming = false;
124 IsComplete = false;
125 UploadProgress = 0f;
126 DownloadProgress = 0f;
127
128 // Add all headers
129 Dictionary<string, string> headers = GetHeaders();
130 if (headers != null)
131 {
132 foreach (var key in headers.Keys)
133 {
134 _request.SetRequestHeader(key, headers[key]);
135 }
136 }
137
138 // Use request's timeout value
139 _request.timeout = Timeout;
140
141 // Dispose handlers automatically
142 _request.disposeUploadHandlerOnDispose = true;
143 _request.disposeDownloadHandlerOnDispose = true;
144
145 // Begin
146 _coroutine = CoroutineUtility.StartCoroutine(PerformUpdate());
147
148 // Success
149 return true;
150 }
154 public virtual string CleanUrl(string url)
155 {
156 // Get url
157 string result = url;
158 // Add file:// if needed
159 if (!Regex.IsMatch(result, "(http:|https:|file:|jar:).*"))
160 {
161 result = $"file://{result}";
162 }
163 // Return url
164 return result;
165 }
166 // Override for custom headers
167 protected virtual Dictionary<string, string> GetHeaders() => null;
168 // Perform update
169 protected virtual IEnumerator PerformUpdate()
170 {
171 // Continue while request exists & is not complete
172 while (!IsRequestComplete())
173 {
174 // Wait
175 yield return null;
176
177 // Waiting to begin
178 if (!IsPerforming)
179 {
180 // Can start
181 if (MaxConcurrentRequests <= 0 || _requestCount < MaxConcurrentRequests)
182 {
183 _requestCount++;
184 Begin();
185 }
186 }
187 // Update progresses
188 else
189 {
190 // Set upload progress
191 float newProgress = _request.uploadProgress;
192 if (!UploadProgress.Equals(newProgress))
193 {
194 UploadProgress = newProgress;
195 }
196
197 // Set download progress
198 newProgress = _request.downloadProgress;
199 if (!DownloadProgress.Equals(newProgress))
200 {
201 DownloadProgress = newProgress;
202 _onDownloadProgress?.Invoke(DownloadProgress);
203 }
204
205 // First response received
206 if (_onFirstResponse != null && _request.downloadedBytes > 0)
207 {
208 _onFirstResponse.Invoke();
209 _onFirstResponse = null;
210 }
211
212 // Stream is ready
213 if (_onComplete != null && _request.downloadHandler is IVRequestStreamable streamHandler)
214 {
215 if (streamHandler.IsStreamReady)
216 {
217 _onComplete.Invoke(_request, string.Empty);
218 _onComplete = null;
219 }
220 }
221 }
222 }
223 // Complete
224 Complete();
225 }
226 // Begin request
227 protected virtual void Begin()
228 {
229 IsPerforming = true;
230 UploadProgress = 0f;
231 DownloadProgress = 0f;
232 _onDownloadProgress?.Invoke(DownloadProgress);
233 _request.SendWebRequest();
234 }
235 // Check for whether request is complete
236 protected virtual bool IsRequestComplete()
237 {
238 // No request
239 if (_request == null)
240 {
241 return true;
242 }
243 // Request still in progress
244 if (!_request.isDone)
245 {
246 return false;
247 }
248 // No error & download handler
249 if (string.IsNullOrEmpty(_request.error) && _request.downloadHandler != null)
250 {
251 // Stream is still finishing
252 if (_request.downloadHandler is IVRequestStreamable streamHandler)
253 {
254 if (!streamHandler.IsStreamComplete)
255 {
256 return false;
257 }
258 }
259 // Download handler not complete
260 else if (!_request.downloadHandler.isDone)
261 {
262 return false;
263 }
264 }
265 // Complete
266 return true;
267 }
268 // Request complete
269 protected virtual void Complete()
270 {
271 // Perform callback
272 if (IsPerforming && IsRequestComplete())
273 {
274 DownloadProgress = 1f;
275 ResponseCode = (int)_request.responseCode;
276 _onDownloadProgress?.Invoke(DownloadProgress);
277 _onComplete?.Invoke(_request, GetSpecificRequestError(_request));
278 }
279
280 // Unload
281 Unload();
282 }
283 // Abort request
284 public virtual void Cancel()
285 {
286 // Cancel
287 if (_onComplete != null && _request != null)
288 {
289 DownloadProgress = 1f;
290 _onDownloadProgress?.Invoke(DownloadProgress);
291 _onComplete?.Invoke(_request, WitConstants.CANCEL_ERROR);
292 }
293
294 // Unload
295 Unload();
296 }
297 // Request destroy
298 protected virtual void Unload()
299 {
300 // Cancel coroutine
301 if (_coroutine != null)
302 {
303 _coroutine.CoroutineCancel();
304 _coroutine = null;
305 }
306
307 // Complete
308 if (IsPerforming)
309 {
310 IsPerforming = false;
311 _requestCount--;
312 }
313
314 // Remove delegates
315 _onDownloadProgress = null;
316 _onFirstResponse = null;
317 _onComplete = null;
318
319 // Dispose
320 if (_request != null)
321 {
322 // Additional cleanup
323 if (_request.downloadHandler is AudioStreamHandler audioStreamer)
324 {
325 audioStreamer.CleanUp();
326 }
327 // Dispose handlers
328 _request.uploadHandler?.Dispose();
329 _request.downloadHandler?.Dispose();
330 // Dispose request
331 _request.Dispose();
332 _request = null;
333 }
334
335 // Officially complete
336 IsComplete = true;
337 }
338 // Returns more specific request error
339 public static string GetSpecificRequestError(UnityWebRequest request)
340 {
341 // Get error & return if empty
342 string error = request.error;
343 string result = error;
344 if (string.IsNullOrEmpty(result))
345 {
346 return result;
347 }
348
349 // Ignore without download handler
350 if (request.downloadHandler == null)
351 {
352 return result;
353 }
354
355 // Ignore without downloaded json
356 string downloadedJson = string.Empty;
357 try
358 {
359 byte[] downloadedBytes = request.downloadHandler.data;
360 if (downloadedBytes != null)
361 {
362 downloadedJson = Encoding.UTF8.GetString(downloadedBytes);
363 }
364 }
365 catch (Exception e)
366 {
367 VLog.W($"VRequest failed to parse downloaded text\n{e}");
368 }
369 if (string.IsNullOrEmpty(downloadedJson))
370 {
371 return result;
372 }
373
374 // Set error to json
375 result = $"{error}\nServer Response: {downloadedJson}";
376
377 // Decode
378 WitResponseNode downloadedNode = WitResponseNode.Parse(downloadedJson);
379 if (downloadedNode == null)
380 {
381 return result;
382 }
383
384 // Check for error
385 WitResponseClass downloadedClass = downloadedNode.AsObject;
386 if (!downloadedClass.HasChild(WitConstants.ENDPOINT_ERROR_PARAM))
387 {
388 return result;
389 }
390
391 // Get final result
392 return $"{request.error}\nServer Response Message: {downloadedClass[WitConstants.ENDPOINT_ERROR_PARAM].Value}";
393 }

471 public bool RequestFileDownload(string downloadPath, UnityWebRequest unityRequest,
472 RequestCompleteDelegate<bool> onComplete)
473 {
474 // Get temporary path for download
475 string tempDownloadPath = downloadPath + ".tmp";
476 try
477 {
478 // Delete temporary file if it already exists
479 if (File.Exists(tempDownloadPath))
480 {
481 File.Delete(tempDownloadPath);
482 }
483 }
484 catch (Exception e)
485 {
486 // Failed to delete file
487 string error = $"Deleting Download File Failed\nPath: {tempDownloadPath}\n\n{e}";
488 VLog.W(error);
489 onComplete?.Invoke(false, error);
490 return false;
491 }
492
493 // Add file download handler
494 DownloadHandlerFile fileHandler = new DownloadHandlerFile(tempDownloadPath, true);
495 unityRequest.downloadHandler = fileHandler;
496 unityRequest.disposeDownloadHandlerOnDispose = true;
497
498 // Perform request
499 return Request(unityRequest, (response, error) =>
500 {
501 try
502 {
503 // Handle existing temp file
504 if (File.Exists(tempDownloadPath))
505 {
506 // For error, remove
507 if (!string.IsNullOrEmpty(error))
508 {
509 File.Delete(tempDownloadPath);
510 }
511 // For success, move to final path
512 else
513 {
514 // File already at download path, delete it
515 if (File.Exists(downloadPath))
516 {
517 File.Delete(downloadPath);
518 }
519
520 // Move to final path
521 File.Move(tempDownloadPath, downloadPath);
522 }
523 }
524 }
525 catch (Exception e)
526 {
527 VLog.W($"Moving Download File Failed\nFrom: {tempDownloadPath}\nTo: {downloadPath}\n\n{e}");
528 }
529
530 // Complete
531 onComplete?.Invoke(string.IsNullOrEmpty(error), error);
532 });
533 }

540 public bool RequestFileExists(string checkPath,
541 RequestCompleteDelegate<bool> onComplete)
542 {
543 // WebGL & web files, perform a header lookup
544 if (checkPath.StartsWith("http"))
545 {
546 Uri uri = new Uri(CleanUrl(checkPath));
547 return RequestFileHeaders(uri, (headers, error) =>
548 {
549 onComplete?.Invoke(headers != null, error);
550 });
551 }
552
553#if FAKE_JAR_LOAD
554 // Android editor: simulate jar handling
555 if (Application.isPlaying && checkPath.StartsWith(Application.streamingAssetsPath))
556#else
557 // Jar file
558 if (checkPath.StartsWith("jar"))
559#endif
560 {
561 Uri uri = new Uri(CleanUrl(checkPath));
562 _onDownloadProgress = (progress) =>
563 {
564 // Stop as early as possible
565 if (progress > 0f && progress < 1f)
566 {
567 var localHandle = onComplete;
568 onComplete = null;
569 Cancel();
570 localHandle?.Invoke(true, String.Empty);
571 VLog.D("Async Check File Exists Success");
572 }
573 };
574 return RequestFile(uri, (response, error) =>
575 {
576 // If getting here, most likely failed but double check anyway
577 onComplete?.Invoke(string.IsNullOrEmpty(error), error);
578 });
579 }
580
581 // Can simply use File.IO otherwise
582 bool found = File.Exists(checkPath);
583 onComplete?.Invoke(found, string.Empty);
584 return true;
585 }

626 public bool RequestJson<TData>(UnityWebRequest unityRequest,
627 RequestCompleteDelegate<TData> onComplete)
628 {
629 // Set request header for json
630 unityRequest.SetRequestHeader("Content-Type", "application/json");
631
632 // Perform text request
633 return RequestText(unityRequest, (text, error) =>
634 {
635 // Request error
636 if (!string.IsNullOrEmpty(error) && string.IsNullOrEmpty(text))
637 {
638 onComplete?.Invoke(default(TData), error);
639 return;
640 }
641
642 // Deserialize
643 JsonConvert.DeserializeObjectAsync<TData>(text, (result, deserializeSuccess) =>
644 {
645 // Return parsed error
646 if (!string.IsNullOrEmpty(error))
647 {
648 onComplete?.Invoke(result, error);
649 }
650 // Parse failed
651 else if (!deserializeSuccess)
652 {
653 onComplete?.Invoke(result, $"Failed to parse json\n{text}");
654 }
655 // Success
656 else
657 {
658 onComplete?.Invoke(result, string.Empty);
659 }
660 });
661 });

749 public bool RequestAudioStream(IAudioClipStream clipStream,
750 UnityWebRequest unityRequest,
751 RequestCompleteDelegate<IAudioClipStream> onClipStreamReady,
752 AudioType audioType, bool audioStream)
753 {
754 // Audio streaming
755#if UNITY_WEBGL
756 if (audioStream && audioType != AudioType.OGGVORBIS)
757#else
758 if (audioStream && !AudioStreamHandler.CanDecodeType(audioType))
759#endif
760 {
761 VLog.W($"Audio streaming not currently supported by for {(audioType == AudioType.UNKNOWN ? "PCM" : audioType.ToString())}");
762 audioStream = false;
763 }
764
765 // Add audio download handler
766 if (unityRequest.downloadHandler == null)
767 {
768 if (AudioStreamHandler.CanDecodeType(audioType))
769 {
770 // Cannot download without clip stream info
771 if (clipStream == null)
772 {
773 onClipStreamReady?.Invoke(null, "No clip stream provided");
774 return false;
775 }
776 // Use custom audio stream handler
777 if (audioStream)
778 {
779 unityRequest.downloadHandler = new AudioStreamHandler(clipStream, audioType);
780 }
781 // Use buffer stream handler
782 else
783 {
784 unityRequest.downloadHandler = new DownloadHandlerBuffer();
785 }
786 }
787 else
788 {
789 unityRequest.downloadHandler = new DownloadHandlerAudioClip(unityRequest.uri, audioType);
790 }
791 }
792
793 // Set stream settings if applicable
794 if (unityRequest.downloadHandler is DownloadHandlerAudioClip audioDownloader)
795 {
796 audioDownloader.streamAudio = audioStream;
797 }
798
799 // Perform default request operation
800 return Request(unityRequest, (response, error) => OnRequestAudioReady(clipStream, audioType, response, error, onClipStreamReady));
801 }
802 // Called on audio ready to be decoded
803 private void OnRequestAudioReady(IAudioClipStream clipStream, AudioType audioType, UnityWebRequest request, string error,
804 RequestCompleteDelegate<IAudioClipStream> onClipStreamReady)
805 {
806 // Check error
807 if (!string.IsNullOrEmpty(error))
808 {
809 onClipStreamReady?.Invoke(null, error);
810 return;
811 }
812
813 // Get clip
814 try
815 {
816 // Custom Raw PCM streaming
817 if (request.downloadHandler is AudioStreamHandler downloadHandlerRaw)
818 {
819 clipStream = downloadHandlerRaw.ClipStream;
820 }
821 // Unity audio clip stream with existing clip
822 else if (request.downloadHandler is DownloadHandlerBuffer rawDownloader)
823 {
824 byte[] data = rawDownloader.data;
825 float[] samples = AudioStreamHandler.DecodeAudio(data,
826 AudioStreamHandler.GetDecodeType(audioType));
827 clipStream.SetTotalSamples(samples.Length);
828 clipStream.AddSamples(samples);
829 }
830 // Unity audio clip stream with existing clip
831 else if (request.downloadHandler is DownloadHandlerAudioClip audioDownloader)
832 {
833 clipStream?.Unload();
834 AudioClip clip = audioDownloader.audioClip;
835 clipStream = new UnityAudioClipStream(clip);
836 clipStream.UpdateState();
837 }
838 // Failed to decode audio clip
839 else if (request.downloadHandler != null)
840 {
841 onClipStreamReady?.Invoke(null, $"Invalid download handler: {request.downloadHandler.GetType()}");
842 return;
843 }
844 // Failed to decode audio clip
845 else
846 {
847 onClipStreamReady?.Invoke(null, $"Missing download handler");
848 return;
849 }
850 }
851 catch (Exception e)
852 {
853 // Failed to decode audio clip
854 onClipStreamReady?.Invoke(null, $"Failed to decode audio clip\n{e}");
855 return;
856 }
857
858 // Invalid clip
859 if (clipStream != null && clipStream.TotalSamples == 0)
860 {
861 clipStream.Unload();
862 onClipStreamReady?.Invoke(null, "Clip has no samples");
863 return;
864 }
865
866 // Clip is still missing
867 if (clipStream == null)
868 {
869 onClipStreamReady?.Invoke(null, "Failed to decode audio clip stream");
870 return;
871 }
872
873 // Success
874 onClipStreamReady?.Invoke(clipStream, string.Empty);
875 }
