155 private bool FindScaledHandPose(List<HandGrabPose> _handGrabPoses, float handScale, ref HandPose handPose)
156 {
157 if (_handGrabPoses.Count == 1 && _handGrabPoses[0].HandPose != null)
158 {
159 handPose.CopyFrom(_handGrabPoses[0].HandPose);
160 return true;
161 }
162 else if (_handGrabPoses.Count > 1)
163 {
164 float relativeHandScale = handScale / this.transform.lossyScale.x;
165 GrabPoseFinder.FindInterpolationRange(relativeHandScale, _handGrabPoses,
166 out HandGrabPose under, out HandGrabPose over, out float t);
167 if (under.HandPose != null && over.HandPose != null)
168 {
169 HandPose.Lerp(under.HandPose, over.HandPose, t, ref handPose);
170 return true;
171 }
172 else if (under.HandPose != null)
173 {
174 handPose.CopyFrom(under.HandPose);
175 return true;
176 }
177 else if (over.HandPose != null)
178 {
179 handPose.CopyFrom(over.HandPose);
180 return true;
181 }
182
183 return false;
184 }
185
186 return false;
187 }
