155 private IEnumerator StampRoutine(int stampCount, float randomness, float spread, float strength)
156 {
157 StartStamping();
158 Pose originalPose = _nozzle.GetPose();
159 for (int i = 0; i < stampCount; i++)
160 {
161 yield return WAIT_TIME;
162 Pose randomPose = originalPose;
163 randomPose.rotation =
164 randomPose.rotation *
165 Quaternion.Euler(
166 Random.Range(-randomness, randomness),
167 Random.Range(-randomness, randomness),
168 0f);
169
170 Stamp(randomPose, _maxDistance, spread, strength);
171 }
172 StartDrying();
173 }

175 private void StartStamping()

180 private void StartDrying()

186 private void Stamp(Pose pose, float maxDistance, float angle, float strength)
187 {
188 _sprayStampMaterial.SetMatrix(STAMP_MATRIX_PROPERTY, CreateStampMatrix(pose, angle));
189 _sprayStampMaterial.SetFloat(STAMP_MULTIPLIER_PROPERTY, strength);
190
191 float radius = Mathf.Tan(Mathf.Deg2Rad * angle / 2) * maxDistance;
192 Vector3 startPoint = pose.position + pose.forward * radius;
193 Vector3 endPoint = pose.position + pose.forward * maxDistance;
194 int hitCount = Physics.OverlapCapsuleNonAlloc(startPoint, endPoint, radius, NonAlloc._overlapResults, _raycastLayerMask.value, QueryTriggerInteraction.Ignore);
195 HashSet<Transform> roots = NonAlloc.GetRootsFromOverlapResults(hitCount);
196
197 foreach (Transform rootObject in roots)
198 {
199 RenderSplash(rootObject);
200 }
201
202 roots.Clear();
203 }
204
208 private void RenderSplash(Transform rootObject)
209 {
210 List<MeshFilter> meshFilters = NonAlloc.GetMeshFiltersInChildren(rootObject);
211 for (int i = 0; i < meshFilters.Count; i++)
212 {
213 int id = meshFilters[i].GetInstanceID();
214 if (!NonAlloc._blits.ContainsKey(id)) { NonAlloc._blits[id] = CreateMeshBlit(meshFilters[i]); }
215 NonAlloc._blits[id].Blit();
216 }
217 }
218
222 private MeshBlit CreateMeshBlit(MeshFilter meshFilter)
223 {
224 MeshBlit newBlit = meshFilter.gameObject.AddComponent<MeshBlit>();
225 newBlit.material = _sprayStampMaterial;
226 newBlit.renderTexture = new RenderTexture(512, 512, 0, RenderTextureFormat.RHalf);
227 newBlit.BlitsPerSecond = 30;
228
229 if (meshFilter.TryGetComponent(out Renderer renderer))
230 {
231 renderer.GetPropertyBlock(NonAlloc.PropertyBlock);
232
233 NonAlloc.PropertyBlock.SetTexture(WET_MAP_PROPERTY, newBlit.renderTexture);
234 if (_waterBumpOverride)
235 {
236 NonAlloc.PropertyBlock.SetTexture(WET_BUMPMAP_PROPERTY, _waterBumpOverride);
237 }
238
239 renderer.SetPropertyBlock(NonAlloc.PropertyBlock);
240 }
241
242 return newBlit;
243 }

245 private Matrix4x4 CreateStampMatrix(Pose pose, float angle)
246 {
247 Matrix4x4 viewMatrix = Matrix4x4.TRS(pose.position, pose.rotation, Vector3.one).inverse;
248 viewMatrix.m20 *= -1f;
249 viewMatrix.m21 *= -1f;
250 viewMatrix.m22 *= -1f;
251 viewMatrix.m23 *= -1f;
252 return GL.GetGPUProjectionMatrix(Matrix4x4.Perspective(angle, 1, 0, _maxDistance), true) * viewMatrix;
253 }

274 public float ComputeUseStrength(float strength)
275 {
276 float delta = Time.realtimeSinceStartup - _lastUseTime;
277 _lastUseTime = Time.realtimeSinceStartup;
278 if (strength > _dampedUseStrength)
279 {
280 _dampedUseStrength = Mathf.Lerp(_dampedUseStrength, strength, _triggerSpeed * delta);
281 }
282 else
283 {
284 _dampedUseStrength = strength;
285 }
286 float progress = _strengthCurve.Evaluate(_dampedUseStrength);
287 UpdateTriggerProgress(progress);
288 return progress;
289 }
290
291 private void UpdateTriggerProgress(float progress)
292 {
293 UpdateTriggerRotation(progress);
294
295 if (progress >= _fireThresold && !_wasFired)
296 {
297 _wasFired = true;
298 SprayWater();
299 }
300 else if (progress <= _releaseThresold)
301 {
302 _wasFired = false;
303 }
304 }

326 public static HashSet<Transform> GetRootsFromOverlapResults(int hitCount)
327 {
328 _roots.Clear();

340 static Transform GetRoot(Collider hit)
341 {
342 return hit.attachedRigidbody ? hit.attachedRigidbody.transform :
343 hit.transform.parent ? hit.transform.parent :
344 hit.transform;

347 public static void CleanUpDestroyedBlits()
348 {
349 if (!_blits.ContainsValue(null)) { return; }
350
351 foreach (int key in new List<int>(_blits.Keys))
352 {
353 if (_blits[key] == null) _blits.Remove(key);
