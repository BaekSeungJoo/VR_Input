98 public bool InvokeAction(IParameterProvider parameterProvider, string actionId, bool relaxed,
99 float confidence = 1f, bool partial = false)
100 {
101 if (!Manifest.ContainsAction(actionId))
102 {
103 var hasBeenHandledWithoutConduit = Manifest.WitResponseMatcherIntents.Contains(actionId);
104 if (!_ignoredActionIds.Contains(actionId) && !hasBeenHandledWithoutConduit)
105 {
106 _ignoredActionIds.Add(actionId);
107 InvokeError(actionId, new Exception($"Conduit did not find intent '{actionId}' in manifest."));
108 VLog.W($"Conduit did not find intent '{actionId}' in manifest.");
109 }
110 return false;
111 }
112
113 parameterProvider.PopulateRoles(_parameterToRoleMap);
114
115 var filter =
116 new InvocationContextFilter(parameterProvider, Manifest.GetInvocationContexts(actionId), relaxed);
117
118 var invocationContexts = filter.ResolveInvocationContexts(actionId, confidence, partial);
119 if (invocationContexts.Count < 1)
120 {
121 // Only log if this is non-partial and inverse does not contain any matches either
122 if (!partial && filter.ResolveInvocationContexts(actionId, confidence, true).Count < 1)
123 {
124 VLog.W(
125 $"Failed to resolve {(partial ? "partial" : "final")} method for {actionId} with supplied context");
126 InvokeError(actionId, new Exception($"Failed to resolve {(partial ? "partial" : "final")} method for {actionId} with supplied context")
127 );
128 }
129
130 return false;
131 }
132
133 var allSucceeded = true;
134 foreach (var invocationContext in invocationContexts)
135 {
136 try
137 {
138 if (!InvokeMethod(invocationContext, parameterProvider, relaxed))
139 {
140 allSucceeded = false;
141 }
142 }
143 catch (Exception e)
144 {
145 VLog.W($"Failed to invoke {invocationContext.MethodInfo.Name}. {e}");
146 allSucceeded = false;
147 InvokeError( invocationContext.MethodInfo.Name, e);
148 }
149 }
150
151 return allSucceeded;
152 }

154 public bool InvokeError(string actionId, Exception exception)
155 {
156 var contexts = Manifest.GetErrorHandlerContexts();
157 foreach (var context in contexts)
158 {
159 var parameterProvider = new ParameterProvider();
160 parameterProvider.AddParameter("intent", actionId);
161 parameterProvider.AddParameter("error", exception);
162
163 InvokeMethod(context, parameterProvider, true);
164 }
165
166 return true;

177 private bool InvokeMethod(InvocationContext invocationContext, IParameterProvider parameterProvider,
178 bool relaxed)
179 {
180 var method = invocationContext.MethodInfo;
181 var formalParametersInfo = method.GetParameters();
182 var parameterObjects = new object[formalParametersInfo.Length];
183 for (var i = 0; i < formalParametersInfo.Length; i++)
184 {
185 var log = new StringBuilder();
186 parameterObjects[i] = parameterProvider.GetParameterValue(formalParametersInfo[i],
187 invocationContext.ParameterMap, relaxed);
188
189 if (parameterObjects[i] == null)
190 {
191 InvokeError( invocationContext.MethodInfo.Name, new Exception(
192 $"Failed to find method param while invoking\nType: {invocationContext.Type.FullName}\nMethod: {invocationContext.MethodInfo.Name}\nParameter Issues\n{log}"
193 ));
194
195 VLog.W(
196 $"Failed to find method param while invoking\nType: {invocationContext.Type.FullName}\nMethod: {invocationContext.MethodInfo.Name}\nParameter Issues\n{log}");
197 return false;
198 }
199 }
200
201 if (method.IsStatic)
202 {
203 try
204 {
205 method.Invoke(null, parameterObjects.ToArray());
206 }
207 catch (Exception e)
208 {
209 VLog.W($"Failed to invoke static method {method.Name}. {e}");
210 InvokeError( invocationContext.MethodInfo.Name, e);
211 return false;
212 }
213
214 return true;
215 }
216 else
217 {
218 var allSucceeded = true;
219 foreach (var obj in _instanceResolver.GetObjectsOfType(invocationContext.Type))
220 {
221 try
222 {
223 method.Invoke(obj, parameterObjects.ToArray());
224 }
225 catch (Exception e)
226 {
227 VLog.W($"Failed to invoke method {method.Name}. {e} on {obj}");
228 allSucceeded = false;
229 InvokeError( invocationContext.MethodInfo.Name,e);
230 continue;
231 }
232 }
233
234 return allSucceeded;
235 }

282 public List<InvocationContext> ResolveInvocationContexts(string actionId, float confidence, bool partial)
283 {
284 // We may have multiple overloads, find the correct match.
285 return _actionContexts != null ? _actionContexts.Where(context => CompatibleInvocationContext(context, confidence, partial))
286 .ToList() : new List<InvocationContext>();
287 }
288
298 private bool CompatibleInvocationContext(InvocationContext invocationContext, float confidence, bool partial)
299 {
300 var parameterMap = new Dictionary<string, string>();
301 var parameters = invocationContext.MethodInfo.GetParameters();
302 if (invocationContext.ValidatePartial != partial)
303 {
304 return false;
305 }
306
307 if (invocationContext.MinConfidence > confidence || confidence > invocationContext.MaxConfidence)
308 {
309 //todo: throw error for out of confidence
310 return false;
311 }
312
313 var exactMatches = new HashSet<string>();
314
315 var log = new StringBuilder();
316 var allParametersMatched = true;
317 foreach (var parameter in parameters)
318 {
319 if (_parameterProvider.ContainsParameter(parameter, log))
320 {
321 exactMatches.Add(parameter.Name);
322 continue;
323 }
324
325 VLog.D(!_relaxed
326 ? $"Could not find value for parameter: {parameter.Name}"
327 : $"Could not find exact value for parameter: {parameter.Name}. Will attempt resolving by type.");
328
329 allParametersMatched = false;
330 }
331
332 if (allParametersMatched)
333 {
334 return true;
335 }
336
337 if (!_relaxed)
338 {
339 VLog.D(
340 $"Failed to resolve parameters. \nType: {invocationContext.Type.FullName}\nMethod: {invocationContext.MethodInfo.Name}\n{log}");
341 return false;
342 }
343
344 var actualTypes = new HashSet<Type>();
345 foreach (var parameter in parameters)
346 {
347 if (exactMatches.Contains(parameter.Name))
348 {
349 continue;
350 }
351
352 if (actualTypes.Contains(parameter.ParameterType))
353 {
354 VLog.D(
355 $"Failed to resolve parameters by type. More than one value of type {parameter.ParameterType} were provided.");
356 return false;
357 }
358
359 actualTypes.Add(parameter.ParameterType);
360
361 var actualParameterNames = _parameterProvider.GetParameterNamesOfType(parameter.ParameterType)
362 .Where(parameterName => !exactMatches.Contains(parameterName)).ToList();
363
364 if (actualParameterNames.Count != 1)
365 {
366 VLog.D($"Failed to find compatible value for {parameter.Name}");
367
368 return false;
369 }
370
371 parameterMap[parameter.Name] = actualParameterNames[0];
372 }
373
374 invocationContext.ParameterMap = parameterMap;
375 return true;
376 }
377 }
