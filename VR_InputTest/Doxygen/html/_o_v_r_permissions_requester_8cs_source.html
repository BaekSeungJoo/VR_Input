73 public static string GetPermissionId(Permission permission)
74 {
75 return permission switch
76 {
77 Permission.FaceTracking => FaceTrackingPermission,
78 Permission.BodyTracking => BodyTrackingPermission,
79 Permission.EyeTracking => EyeTrackingPermission,
80 Permission.Scene => ScenePermission,
81 _ => throw new ArgumentOutOfRangeException(nameof(permission), permission, null)
82 };
83 }
84
85 private static bool IsPermissionSupportedByPlatform(Permission permission)
86 {
87 return permission switch
88 {
89 Permission.FaceTracking => OVRPlugin.faceTrackingSupported,
90 Permission.BodyTracking => OVRPlugin.bodyTrackingSupported,
91 Permission.EyeTracking => OVRPlugin.eyeTrackingSupported,
92 // Scene is a no-op on unsupported platforms, but the request can always be made
93 Permission.Scene => true,
94 _ => throw new ArgumentOutOfRangeException(nameof(permission), permission, null)
95 };
96 }

107 public static bool IsPermissionGranted(Permission permission)
108 {
109#if UNITY_ANDROID && !UNITY_EDITOR
110 return UnityEngine.Android.Permission.HasUserAuthorizedPermission(GetPermissionId(permission));
111#else
112 return true;
113#endif
114 }

122#if UNITY_ANDROID && !UNITY_EDITOR
123 RequestPermissions(permissions);
124#endif // UNITY_ANDROID && !UNITY_EDITOR
125 }

127 private static void RequestPermissions(IEnumerable<Permission> permissions)
128 {
129 var permissionIdsToRequest = new List<string>();
130
131 foreach (var permission in permissions)
132 {
133 if (ShouldRequestPermission(permission))
134 {
135 permissionIdsToRequest.Add(GetPermissionId(permission));
136 }
137 }
138
139 if (permissionIdsToRequest.Count > 0)
140 {
141 UnityEngine.Android.Permission.RequestUserPermissions(permissionIdsToRequest.ToArray(),
142 BuildPermissionCallbacks());
143 }
144 }
145
146 private static bool ShouldRequestPermission(Permission permission)
147 {
148 if (!IsPermissionSupportedByPlatform(permission))
149 {
150 Debug.LogWarning(
151 $"[[{nameof(OVRPermissionsRequester)}] Permission {permission} is not supported by the platform and can't be requested.");
152 return false;
153 }
154
155 return !IsPermissionGranted(permission);
156 }
157
158 private static PermissionCallbacks BuildPermissionCallbacks()
159 {
160 var permissionCallbacks = new PermissionCallbacks();
161 permissionCallbacks.PermissionDenied += permissionId =>
162 {
163 Debug.LogWarning($"[{nameof(OVRPermissionsRequester)}] Permission {permissionId} was denied.");
164 };
165 permissionCallbacks.PermissionDeniedAndDontAskAgain += permissionId =>
166 {
167 Debug.LogWarning(
168 $"[{nameof(OVRPermissionsRequester)}] Permission {permissionId} was denied and blocked from being requested again.");
169 };
170 permissionCallbacks.PermissionGranted += permissionId =>
171 {
172 Debug.Log($"[{nameof(OVRPermissionsRequester)}] Permission {permissionId} was granted.");
173 PermissionGranted?.Invoke(permissionId);
174 };
175 return permissionCallbacks;
176 }
