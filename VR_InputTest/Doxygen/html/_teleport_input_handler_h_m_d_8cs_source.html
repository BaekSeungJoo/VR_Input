47 public override LocomotionTeleport.TeleportIntentions GetIntention()
48 {
49 if (!isActiveAndEnabled)
50 {
51 return global::LocomotionTeleport.TeleportIntentions.None;
52 }
53
54 if (LocomotionTeleport.CurrentIntention == global::LocomotionTeleport.TeleportIntentions.Aim)
55 {
56 // If the user has actually pressed the teleport button, enter the preteleport state.
57 if (OVRInput.GetDown(TeleportButton))
58 {
59 return FastTeleport
60 ? global::LocomotionTeleport.TeleportIntentions.Teleport
61 : global::LocomotionTeleport.TeleportIntentions.PreTeleport;
62 }
63 }
64
65 // If the user is already in the preteleport state, the intention will be to either remain in this state or switch to Teleport
66 if (LocomotionTeleport.CurrentIntention == global::LocomotionTeleport.TeleportIntentions.PreTeleport)
67 {
68 // If they released the button, switch to Teleport.
69 if (OVRInput.GetUp(TeleportButton))
70 {
71 // Button released, enter the teleport state.
72 return global::LocomotionTeleport.TeleportIntentions.Teleport;
73 }
74
75 // Button still down, remain in PreTeleport so they can orient the destination if an orientation handler supports it.
76 return global::LocomotionTeleport.TeleportIntentions.PreTeleport;
77 }
78
79
80 // Update the aim intention based on the button press state of the AimButton.
81 if (OVRInput.Get(AimButton))
82 {
83 return global::LocomotionTeleport.TeleportIntentions.Aim;
84 }
85
86 // If the aim button is the same as the teleport button, then there is no way to abort the teleport.
87 // Since the process of triggering the teleport is complete, returning the Teleport intention is necessary otherwise
88 // the input handler would remain active.
89 if (AimButton == TeleportButton)
90 {
91 return global::LocomotionTeleport.TeleportIntentions.Teleport;
92 }
93
94 return LocomotionTeleport.TeleportIntentions.None;
95 }
