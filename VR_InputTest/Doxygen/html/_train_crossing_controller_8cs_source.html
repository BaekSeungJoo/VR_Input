95 private void ActivateTrainCrossing()
96 {
97 int maxSoundIndex = _crossingSounds.Length - 1;
98 var audioClip = _crossingSounds[(int)(Random.value * maxSoundIndex)];
99 _audioSource.clip = audioClip;
100 _audioSource.timeSamples = 0;
101 _audioSource.Play();
102 if (_xingAnimationCr != null)
103 {
104 StopCoroutine(_xingAnimationCr);
105 }
106
107 _xingAnimationCr = StartCoroutine(AnimateCrossing(audioClip.length * 0.75f));

110 private IEnumerator AnimateCrossing(float animationLength)
111 {
112 ToggleLightObjects(true);
113
114 float animationEndTime = Time.time + animationLength;
115
116 float lightBlinkDuration = animationLength * 0.1f;
117 float lightBlinkStartTime = Time.time;
118 float lightBlinkEndTime = Time.time + lightBlinkDuration;
119 Material lightToBlinkOn = _lightsSide1Mat;
120 Material lightToBlinkOff = _lightsSide2Mat;
121 Color onColor = new Color(1.0f, 1.0f, 1.0f, 1.0f);
122 Color offColor = new Color(1.0f, 1.0f, 1.0f, 0.0f);
123
124 while (Time.time < animationEndTime)
125 {
126 float t = (Time.time - lightBlinkStartTime) / lightBlinkDuration;
127 lightToBlinkOn.SetColor(_colorId, Color.Lerp(offColor, onColor, t));
128 lightToBlinkOff.SetColor(_colorId, Color.Lerp(onColor, offColor, t));
129
130 // switch which lights blink on and off when time runs out
131 if (Time.time > lightBlinkEndTime)
132 {
133 Material temp = lightToBlinkOn;
134 lightToBlinkOn = lightToBlinkOff;
135 lightToBlinkOff = temp;
136 lightBlinkStartTime = Time.time;
137 lightBlinkEndTime = Time.time + lightBlinkDuration;
138 }
139
140 yield return null;
141 }
142
143 ToggleLightObjects(false);
144 }
