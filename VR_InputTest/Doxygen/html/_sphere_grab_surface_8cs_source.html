133 public bool CalculateBestPoseAtSurface(Ray targetRay, out Pose bestPose, Transform relativeTo)
134 {
135 Vector3 centre = GetCentre(relativeTo);
136 Vector3 projection = Vector3.Project(centre - targetRay.origin, targetRay.direction);
137 Vector3 nearestCentre = targetRay.origin + projection;
138 float radius = GetRadius(relativeTo);
139 float distanceToSurface = Mathf.Max(Vector3.Distance(centre, nearestCentre) - radius);
140 if (distanceToSurface < radius)
141 {
142 float adjustedDistance = Mathf.Sqrt(radius * radius - distanceToSurface * distanceToSurface);
143 nearestCentre -= targetRay.direction * adjustedDistance;
144 }
145
146 Pose recordedPose = GetReferencePose(relativeTo);
147 Vector3 surfacePoint = NearestPointInSurface(nearestCentre, relativeTo);
148 Pose desiredPose = new Pose(surfacePoint, recordedPose.rotation);
149 bestPose = MinimalTranslationPoseAtSurface(desiredPose, relativeTo);
150 return true;
151 }

176 protected Vector3 NearestPointInSurface(Vector3 targetPosition, Transform relativeTo)
177 {
178 Vector3 centre = GetCentre(relativeTo);
179 Vector3 direction = (targetPosition - centre).normalized;
180 float radius = GetRadius(relativeTo);
181 return centre + direction * radius;
182 }

184 protected Pose MinimalRotationPoseAtSurface(in Pose userPose, Transform relativeTo)
185 {
186 Vector3 centre = GetCentre(relativeTo);
187 Pose referencePose = GetReferencePose(relativeTo);
188 float radius = GetRadius(relativeTo);
189 Quaternion rotCorrection = userPose.rotation * Quaternion.Inverse(referencePose.rotation);
190 Vector3 correctedDir = rotCorrection * GetDirection(relativeTo);
191 Vector3 surfacePoint = NearestPointInSurface(centre + correctedDir * radius, relativeTo);
192 Quaternion surfaceRotation = RotationAtPoint(surfacePoint, referencePose.rotation, userPose.rotation, relativeTo);
193
194 return new Pose(surfacePoint, surfaceRotation);
195 }
196
197 protected Pose MinimalTranslationPoseAtSurface(in Pose userPose, Transform relativeTo)
198 {
199 Pose referencePose = GetReferencePose(relativeTo);
200 Vector3 desiredPos = userPose.position;
201 Quaternion baseRot = referencePose.rotation;
202 Vector3 surfacePoint = NearestPointInSurface(desiredPos, relativeTo);
203 Quaternion surfaceRotation = RotationAtPoint(surfacePoint, baseRot, userPose.rotation, relativeTo);
204 return new Pose(surfacePoint, surfaceRotation);
205 }
206
207 protected Quaternion RotationAtPoint(Vector3 surfacePoint,
208 Quaternion baseRot, Quaternion desiredRotation,
209 Transform relativeTo)
210 {
211 Vector3 desiredDirection = (surfacePoint - GetCentre(relativeTo)).normalized;
212 Quaternion targetRotation = Quaternion.FromToRotation(GetDirection(relativeTo), desiredDirection) * baseRot;
213 Vector3 targetProjected = Vector3.ProjectOnPlane(targetRotation * Vector3.forward, desiredDirection).normalized;
214 Vector3 desiredProjected = Vector3.ProjectOnPlane(desiredRotation * Vector3.forward, desiredDirection).normalized;
215 Quaternion rotCorrection = Quaternion.FromToRotation(targetProjected, desiredProjected);
216 return rotCorrection * targetRotation;
217 }
