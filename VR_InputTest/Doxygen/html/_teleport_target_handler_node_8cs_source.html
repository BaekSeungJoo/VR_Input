38 protected override bool ConsiderTeleport(Vector3 start, ref Vector3 end)
39 {
40 // If the ray hits the world, consider it valid and update the aimRay to the end point.
41 if (!LocomotionTeleport.AimCollisionTest(start, end, AimCollisionLayerMask | TeleportLayerMask,
42 out AimData.TargetHitInfo))
43 {
44 return false;
45 }
46
47 TeleportPoint tp = AimData.TargetHitInfo.collider.gameObject.GetComponent<TeleportPoint>();
48 if (tp == null)
49 {
50 return false;
51 }
52
53 // The targeting test discovered a valid teleport node. Now test to make sure there is line of sight to the
54 // actual destination. Since the teleport destination is expected to be right on the ground, use the LOSOffset
55 // to bump the collision check up off the ground a bit.
56 var dest = tp.destTransform.position;
57 var offsetEnd = new Vector3(dest.x, dest.y + LOSOffset, dest.z);
58 if (LocomotionTeleport.AimCollisionTest(start, offsetEnd, AimCollisionLayerMask & ~TeleportLayerMask,
59 out AimData.TargetHitInfo))
60 {
61 return false;
62 }
63
64 end = dest;
65 return true;
66 }
