16 [CustomPropertyDrawer(typeof(StringReference<>), true)]
17 public class StringReferenceDrawer : PropertyDrawer
18 {
19 private const int buttonWidth = 20;
20
21 public override void OnGUI(Rect position, SerializedProperty property, GUIContent label)
22 {
23 SerializedProperty stringValueProperty = property.FindPropertyRelative("stringValue");
24 SerializedProperty stringObjectProperty = property.FindPropertyRelative("stringObject");
25
26 EditorGUI.BeginProperty(position, label, property);
27
28 // Draw the label
29 position = EditorGUI.PrefixLabel(position, GUIUtility.GetControlID(FocusType.Passive), label);
30
31 // Draw the text field
32 Rect textFieldRect = new Rect(position);
33 textFieldRect.width -= buttonWidth;
34 Rect objFieldRect = new Rect(position.x + textFieldRect.width, position.y, buttonWidth,
35 EditorGUIUtility.singleLineHeight);
36
37 EditorGUI.PropertyField(stringObjectProperty.objectReferenceValue == null ? objFieldRect : position,
38 stringObjectProperty, GUIContent.none);
39 if (stringObjectProperty.objectReferenceValue == null)
40 {
41 stringValueProperty.stringValue = EditorGUI.TextField(textFieldRect, stringValueProperty.stringValue);
42 }
43 else
44 {
45 stringValueProperty.stringValue = ((IStringReference)stringObjectProperty.objectReferenceValue).Value;
46 }
47
48 Type targetType = fieldInfo.FieldType.BaseType.GenericTypeArguments[0];
49 // Handle drag and drop
50 if (Event.current.type == EventType.DragUpdated && textFieldRect.Contains(Event.current.mousePosition))
51 {
52 var validType = false;
53 foreach (var draggedObject in DragAndDrop.objectReferences)
54 {
55 if (draggedObject.GetType() == targetType)
56 {
57 validType = true;
58 break;
59 }
60 }
61
62 if (validType)
63 {
64 DragAndDrop.visualMode = DragAndDropVisualMode.Copy;
65 }
66 Event.current.Use();
67 }
68 else if (Event.current.type == EventType.DragPerform && textFieldRect.Contains(Event.current.mousePosition))
69 {
70 DragAndDrop.AcceptDrag();
71
72 foreach (var draggedObject in DragAndDrop.objectReferences)
73 {
74 if (draggedObject.GetType() == targetType)
75 {
76 stringObjectProperty.objectReferenceValue = draggedObject;
77 property.serializedObject.ApplyModifiedProperties();
78 break;
79 }
80 }
81
82 Event.current.Use();
83 }
84
85 EditorGUI.EndProperty();
86 }
87 }
88}
