65 // Start is called before the first frame update
66 void Start()
67 {
68 _canvas = GetComponent<Canvas>();
69
70 _rectTransform = _canvas.GetComponent<RectTransform>();
71
72 float rectWidth = _rectTransform.rect.width;
73 float rectHeight = _rectTransform.rect.height;
74
75 float aspectX = rectWidth >= rectHeight ? 1 : rectWidth / rectHeight;
76 float aspectY = rectHeight >= rectWidth ? 1 : rectHeight / rectWidth;
77
78 // if we are scaling the viewport we don't need to add a border
79 int pixelBorder = ScaleViewport ? 0 : 8;
80 int innerWidth = Mathf.CeilToInt(aspectX * (MaxTextureSize - pixelBorder * 2));
81 int innerHeight = Mathf.CeilToInt(aspectY * (MaxTextureSize - pixelBorder * 2));
82 int width = innerWidth + pixelBorder * 2;
83 int height = innerHeight + pixelBorder * 2;
84
85 float paddedWidth = rectWidth * (width / (float)innerWidth);
86 float paddedHeight = rectHeight * (height / (float)innerHeight);
87
88 float insetRectWidth = innerWidth / (float)width;
89 float insetRectHeight = innerHeight / (float)height;
90
91 // ever so slightly shrink our opaque mesh to avoid black borders
92 Vector2 opaqueTrim = Opacity == DrawMode.Opaque
93 ? new Vector2(0.005f / _rectTransform.lossyScale.x, 0.005f / _rectTransform.lossyScale.y)
94 : Vector2.zero;
95
96 _renderTexture = new RenderTexture(width, height, 0, RenderTextureFormat.ARGB32, RenderTextureReadWrite.Linear);
97 // if we can't scale the viewport, generate mipmaps instead
98 _renderTexture.useMipMap = !ScaleViewport;
99
100 GameObject overlayCamera = new GameObject(name + " Overlay Camera")
101 {
102#if !DEBUG_OVERLAY_CANVAS
103 hideFlags = HideFlags.HideInHierarchy | HideFlags.NotEditable
104#endif
105 };
106 overlayCamera.transform.SetParent(transform, false);
107
108 _camera = overlayCamera.AddComponent<Camera>();
109 _camera.stereoTargetEye = StereoTargetEyeMask.None;
110 _camera.transform.position = transform.position - transform.forward;
111 _camera.orthographic = true;
112 _camera.enabled = false;
113 _camera.targetTexture = _renderTexture;
114 _camera.cullingMask = 1 << gameObject.layer;
115 _camera.clearFlags = CameraClearFlags.SolidColor;
116 _camera.backgroundColor = Color.clear;
117 _camera.orthographicSize = 0.5f * paddedHeight * _rectTransform.localScale.y;
118 _camera.nearClipPlane = 0.99f;
119 _camera.farClipPlane = 1.01f;
120
121 _quad = new Mesh()
122 {
123 name = name + " Overlay Quad",
124 hideFlags = HideFlags.HideAndDontSave
125 };
126
127 _quad.vertices = new Vector3[]
128 { new Vector3(-0.5f, -0.5f), new Vector3(-0.5f, 0.5f), new Vector3(0.5f, 0.5f), new Vector3(0.5f, -0.5f) };
129 _quad.uv = new Vector2[] { new Vector2(0, 0), new Vector2(0, 1), new Vector2(1, 1), new Vector2(1, 0) };
130 _quad.triangles = new int[] { 0, 1, 2, 2, 3, 0 };
131 _quad.bounds = new Bounds(Vector3.zero, Vector3.one);
132 _quad.UploadMeshData(true);
133
134 switch (Opacity)
135 {
136 case DrawMode.Opaque:
137 _defaultMat = new Material(_opaqueShader);
138 break;
139 case DrawMode.OpaqueWithClip:
140 _defaultMat = new Material(_opaqueShader);
141 _defaultMat.EnableKeyword("WITH_CLIP");
142 break;
143 case DrawMode.TransparentDefaultAlpha:
144 _defaultMat = new Material(_transparentShader);
145 _defaultMat.EnableKeyword("ALPHA_SQUARED");
146 break;
147 case DrawMode.TransparentCorrectAlpha:
148 _defaultMat = new Material(_transparentShader);
149 break;
150 }
151
152 _defaultMat.mainTexture = _renderTexture;
153 _defaultMat.color = Color.black;
154 _defaultMat.mainTextureOffset = new Vector2(0.5f - 0.5f * insetRectWidth, 0.5f - 0.5f * insetRectHeight);
155 _defaultMat.mainTextureScale = new Vector2(insetRectWidth, insetRectHeight);
156
157 GameObject meshRenderer = new GameObject(name + " MeshRenderer")
158 {
159#if !DEBUG_OVERLAY_CANVAS
160 hideFlags = HideFlags.HideInHierarchy | HideFlags.NotEditable
161#endif
162 };
163
164 meshRenderer.transform.SetParent(transform, false);
165 meshRenderer.AddComponent<MeshFilter>().sharedMesh = _quad;
166 _meshRenderer = meshRenderer.AddComponent<MeshRenderer>();
167 _meshRenderer.sharedMaterial = _defaultMat;
168 meshRenderer.layer = Layer;
169 meshRenderer.transform.localScale = new Vector3(rectWidth - opaqueTrim.x, rectHeight - opaqueTrim.y, 1);
170
171 GameObject overlay = new GameObject(name + " Overlay")
172 {
173#if !DEBUG_OVERLAY_CANVAS
174 hideFlags = HideFlags.HideInHierarchy | HideFlags.NotEditable
175#endif
176 };
177 overlay.transform.SetParent(transform, false);
178 _overlay = overlay.AddComponent<OVROverlay>();
179 _overlay.isDynamic = true;
180 _overlay.noDepthBufferTesting = true;
181 _overlay.isAlphaPremultiplied = !Application.isMobilePlatform;
182 _overlay.textures[0] = _renderTexture;
183 _overlay.currentOverlayType = OVROverlay.OverlayType.Underlay;
184 _overlay.transform.localScale = new Vector3(paddedWidth, paddedHeight, 1);
185 _overlay.useExpensiveSuperSample = Expensive;
186 }

223 private static readonly Plane[] _FrustumPlanes = new Plane[6];
224
225 protected virtual bool ShouldRender()
226 {
227 if (DrawRate > 1)
228 {
229 if (Time.frameCount % DrawRate != DrawFrameOffset % DrawRate)
230 {
231 return false;
232 }
233 }
234
235 if (Camera.main != null)
236 {
237 // Perform Frustum culling
238 for (int i = 0; i < 2; i++)
239 {
240 var eye = (Camera.StereoscopicEye)i;
241 var mat = Camera.main.GetStereoProjectionMatrix(eye) * Camera.main.GetStereoViewMatrix(eye);
242 GeometryUtility.CalculateFrustumPlanes(mat, _FrustumPlanes);
243 if (GeometryUtility.TestPlanesAABB(_FrustumPlanes, _meshRenderer.bounds))
244 {
245 return true;
246 }
247 }
248
249 return false;
250 }
251
252 return true;
253 }
254
255 private void Update()
256 {
257 if (ShouldRender())
258 {
259 if (ScaleViewport)
260 {
261 if (Camera.main != null)
262 {
263 float d = (Camera.main.transform.position - transform.position).magnitude;
264
265 float size = PixelsPerUnit * Mathf.Max(_rectTransform.rect.width * transform.lossyScale.x,
266 _rectTransform.rect.height * transform.lossyScale.y) / d;
267
268 // quantize to even pixel sizes
269 const float quantize = 8;
270 float pixelHeight = Mathf.Ceil(size / quantize * _renderTexture.height) * quantize;
271
272 // clamp between or min size and our max size
273 pixelHeight = Mathf.Clamp(pixelHeight, MinTextureSize, _renderTexture.height);
274
275 float innerPixelHeight = pixelHeight - 2;
276
277 _camera.orthographicSize = 0.5f * _rectTransform.rect.height * _rectTransform.localScale.y *
278 pixelHeight / innerPixelHeight;
279
280 float aspect = (_rectTransform.rect.width / _rectTransform.rect.height);
281
282 float innerPixelWidth = innerPixelHeight * aspect;
283 float pixelWidth = Mathf.Ceil((innerPixelWidth + 2) * 0.5f) * 2;
284
285 float sizeX = pixelWidth / _renderTexture.width;
286 float sizeY = pixelHeight / _renderTexture.height;
287
288 // trim a half pixel off each size if this is opaque (transparent should fade)
289 float inset = Opacity == DrawMode.Opaque ? 1.001f : 0;
290
291 float innerSizeX = (innerPixelWidth - inset) / _renderTexture.width;
292 float innerSizeY = (innerPixelHeight - inset) / _renderTexture.height;
293
294 // scale the camera rect
295 _camera.rect = new Rect((1 - sizeX) / 2, (1 - sizeY) / 2, sizeX, sizeY);
296
297 Rect src = new Rect(0.5f - (0.5f * innerSizeX), 0.5f - (0.5f * innerSizeY), innerSizeX, innerSizeY);
298
299 _defaultMat.mainTextureOffset = src.min;
300 _defaultMat.mainTextureScale = src.size;
301
302 // update the overlay to use this same size
303 _overlay.overrideTextureRectMatrix = true;
304 src.y = 1 - src.height - src.y;
305 Rect dst = new Rect(0, 0, 1, 1);
306 _overlay.SetSrcDestRects(src, src, dst, dst);
307 }
308 }
309
310 _camera.Render();
311 }
312 }

315 {
316 get { return _overlay && _overlay.enabled; }
317 set
318 {
319 if (_overlay)
320 {
321 _overlay.enabled = value;
322 _defaultMat.color = value ? Color.black : Color.white;
323 }
324 }
