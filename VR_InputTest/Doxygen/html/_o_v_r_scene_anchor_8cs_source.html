88 public void Initialize(OVRAnchor anchor)
89 {
90 var space = (OVRSpace)anchor.Handle;
91 var uuid = anchor.Uuid;
92
93 if (Space.Valid)
94 throw new InvalidOperationException($"[{uuid}] {nameof(OVRSceneAnchor)} has already been initialized.");
95
96 if (!space.Valid)
97 throw new ArgumentException($"[{uuid}] {nameof(space)} must be valid.", nameof(space));
98
99 Space = space;
100 Uuid = uuid;
101 Anchor = anchor;
102
103 ClearPoseCache();
104
105 SceneAnchors[this.Uuid] = this;
106 SceneAnchorsList.Add(this);
107
108 AnchorReferenceCountDictionary.TryGetValue(Space, out var referenceCount);
109 AnchorReferenceCountDictionary[Space] = referenceCount + 1;
110
111 // Generally, we want to set the transform as soon as possible, but there is a valid use case where we want to
112 // disable this component as soon as its added to override the transform.
113 if (enabled)
114 {
115 var updateTransformSucceeded = TryUpdateTransform(false);
116
117 // This should work; so add some development-only logs so we know if something is wrong here.
118 if (updateTransformSucceeded)
119 {
120 IsTracked = true;
121 OVRSceneManager.Development.Log(nameof(OVRSceneAnchor),
122 $"[{uuid}] Initial transform set.", gameObject);
123 }
124 else
125 {
126 OVRSceneManager.Development.LogWarning(nameof(OVRSceneAnchor),
127 $"[{uuid}] {nameof(OVRPlugin.TryLocateSpace)} failed. The entity may have the wrong initial transform.",
128 gameObject);
129 }
130 }

132 SyncComponent<OVRSemanticClassification>(OVRPlugin.SpaceComponentType.SemanticLabels);
133 SyncComponent<OVRSceneVolume>(OVRPlugin.SpaceComponentType.Bounded3D);

165 internal bool TryUpdateTransform(bool useCache)
166 {
167 if (!Space.Valid || !enabled) return false;
168
169 if (!useCache || _pose == null)
170 {
171 var tryLocateSpace = OVRPlugin.TryLocateSpace(Space, OVRPlugin.GetTrackingOriginType(), out var pose,
172 out var locationFlags);
173 if (!tryLocateSpace || !locationFlags.IsOrientationValid() || !locationFlags.IsPositionValid())
174 {
175 return false;
176 }
177
178 _pose = pose;
179 }
180
181 // NOTE: This transformation performs the following steps:
182 // 1. Flip Z to convert from OpenXR's right-handed to Unity's left-handed coordinate system.
183 // OpenXR Unity
184 // | y y | / z
185 // | | /
186 // +----> x +----> x
187 // /
188 // z/ (normal)
189 //
190 // 2. (1) means that Z now points in the opposite direction from OpenXR. However, the design is such that a
191 // plane's normal should coincide with +Z, so we rotate 180 degrees around the +Y axis to make Z now point
192 // in the intended direction.
193 // OpenXR Unity
194 // | y y |
195 // | |
196 // +----> x <----+
197 // / /
198 // z/ z / (normal)
199 //
200 // 3. Convert from tracking space to world space.
201 var worldSpacePose = new OVRPose
202 {
203 position = _pose.Value.Position.FromFlippedZVector3f(),
204 orientation = _pose.Value.Orientation.FromFlippedZQuatf() * RotateY180
205 }.ToWorldSpacePose(Camera.main);
206 transform.SetPositionAndRotation(worldSpacePose.position, worldSpacePose.orientation);
207 return true;
208 }

210 private void OnDestroy()
211 {
212 SceneAnchors.Remove(this.Uuid);
213 SceneAnchorsList.Remove(this);
214
215 if (!Space.Valid)
216 {
217 return;
218 }
219
220 if (!AnchorReferenceCountDictionary.TryGetValue(Space, out var referenceCount))
221 {
222 OVRSceneManager.Development.LogError(nameof(OVRSceneAnchor),
223 $"[Anchor {Space.Handle}] has not been found, can't find it for deletion",
224 gameObject);
225 return;
226 }
227
228 if (referenceCount == 1)
229 {
230 // last reference to this anchor, delete it
231 if (Space.Valid)
232 {
233 OVRPlugin.DestroySpace(Space);
234 }
235
236 // remove instead of decrement to not waste memory
237 AnchorReferenceCountDictionary.Remove(Space);
238 }
239 else
240 {
241 AnchorReferenceCountDictionary[Space] = referenceCount - 1;
242 }
243 }
