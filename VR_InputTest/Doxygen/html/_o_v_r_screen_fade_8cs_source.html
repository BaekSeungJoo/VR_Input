208 IEnumerator Fade(float startAlpha, float endAlpha)
209 {
210 float elapsedTime = 0.0f;
211 while (elapsedTime < fadeTime)
212 {
213 elapsedTime += Time.deltaTime;
214 animatedFadeAlpha = Mathf.Lerp(startAlpha, endAlpha, Mathf.Clamp01(elapsedTime / fadeTime));
215 SetMaterialAlpha();
216 yield return new WaitForEndOfFrame();
217 }
218
219 animatedFadeAlpha = endAlpha;
220 SetMaterialAlpha();
221 }
222
227 private void SetMaterialAlpha()
228 {
229 Color color = fadeColor;
230 color.a = currentAlpha;
231 isFading = color.a > 0;
232 if (fadeMaterial != null)
233 {
234 fadeMaterial.color = color;
235 fadeMaterial.renderQueue = renderQueue;
236 fadeRenderer.material = fadeMaterial;
237 fadeRenderer.enabled = isFading;
238 }
