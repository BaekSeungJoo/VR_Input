26namespace Oculus.Voice.Dictation.Bindings.Android
27{
28 public class DictationConfigurationBinding
29 {
30 private readonly WitDictationRuntimeConfiguration _runtimeConfiguration;
31 private readonly DictationConfiguration _dictationConfiguration;
32 private readonly int MAX_PLATFORM_SUPPORTED_RECORDING_TIME_SECONDS = 300;
33
34 public DictationConfigurationBinding(WitDictationRuntimeConfiguration runtimeConfiguration)
35 {
36 if (null == runtimeConfiguration)
37 {
38 // No config defined, use the default configuration.
39 VLog.W("No dictation config has been defined. Using the default configuration.");
40 _dictationConfiguration = new DictationConfiguration();
41 }
42 else
43 {
44 _dictationConfiguration = runtimeConfiguration.dictationConfiguration;
45 _runtimeConfiguration = runtimeConfiguration;
46 }
47 }
48
49 public AndroidJavaObject ToJavaObject()
50 {
51 AndroidJavaObject jo = new AndroidJavaObject("com.oculus.assistant.api.voicesdk.dictation.PlatformDictationConfiguration");
52 jo.Set("multiPhrase", _dictationConfiguration.multiPhrase);
53 jo.Set("scenario", _dictationConfiguration.scenario);
54 jo.Set("inputType", _dictationConfiguration.inputType);
55 if (_runtimeConfiguration != null)
56 {
57 int maxRecordingTime = (int) _runtimeConfiguration.maxRecordingTime;
58 if (maxRecordingTime < 0)
59 {
60 maxRecordingTime = MAX_PLATFORM_SUPPORTED_RECORDING_TIME_SECONDS;
61 }
62 jo.Set("interactionTimeoutSeconds", maxRecordingTime);
63 }
64
65 return jo;
66 }
67 }
