37 public static TTSPreloadSettings CreatePreloadSettings(string savePath)
38 {
39 // Ignore if empty
40 if (string.IsNullOrEmpty(savePath))
41 {
42 return null;
43 }
44
45 // Get asset path
46 string assetPath = savePath.Replace("\\", "/");
47 if (!assetPath.StartsWith(Application.dataPath))
48 {
49 VLog.E(
50 $"TTS Preload Utility - Cannot Create Setting Outside of Assets Directory\nPath: {assetPath}");
51 return null;
52 }
53 assetPath = assetPath.Replace(Application.dataPath, "Assets");
54
55 // Generate & save
56 TTSPreloadSettings settings = ScriptableObject.CreateInstance<TTSPreloadSettings>();
57 AssetDatabase.CreateAsset(settings, assetPath);
58 AssetDatabase.SaveAssets();
59
60 // Reload & return
61 return AssetDatabase.LoadAssetAtPath<TTSPreloadSettings>(assetPath);
62 }

73 TTSPreloadSettings settings = AssetDatabase.LoadAssetAtPath<TTSPreloadSettings>(path);
74 results.Add(settings);
75 }
76 return results.ToArray();

131 private static void IteratePhrases(TTSService service, TTSPreloadData preloadData, TTSPreloadIterateDelegate onIterate, Action<float> onProgress, Action<string> onComplete)
132 {
133 // Skip if check fails
134 if (!CheckIterateData(service, preloadData, onIterate, onProgress, onComplete))
135 {
136 return;
137 }
138
139 // Unload previous coroutine performer
140 if (_performer != null)
141 {
142 _performer.gameObject.DestroySafely();
143 _performer = null;
144 }
145
146 // Run new coroutine
147 _performer = CoroutineUtility.StartCoroutine(PerformIteratePhrases(service, preloadData, onIterate, onProgress, onComplete));

149 // Perform iterate
150 private static IEnumerator PerformIteratePhrases(TTSService service, TTSPreloadData preloadData, TTSPreloadIterateDelegate onIterate, Action<float> onProgress, Action<string> onComplete)
151 {
152 // Get cache settings
153 TTSDiskCacheSettings cacheSettings = new TTSDiskCacheSettings()
154 {
155 DiskCacheLocation = TTSDiskCacheLocation.Preload
156 };
157
158 // Get total phrases
159 int phraseTotal = 0;
160 foreach (var voice in preloadData.voices)
161 {
162 if (voice.phrases == null)
163 {
164 continue;
165 }
166 foreach (var phrase in voice.phrases)
167 {
168 phraseTotal++;
169 }
170 }
171
172 // Begin
173 onProgress?.Invoke(0f);
174
175 // Iterate
176 int phraseCount = 0;
177 float phraseInc = 1f / (float)phraseTotal;
178 string log = string.Empty;
179 for (int v = 0; v < preloadData.voices.Length; v++)
180 {
181 // Get voice data
182 TTSPreloadVoiceData voiceData = preloadData.voices[v];
183 if (voiceData.phrases == null)
184 {
185 continue;
186 }
187
188 // Get voice
189 TTSVoiceSettings voiceSettings = service.GetPresetVoiceSettings(voiceData.presetVoiceID);
190 if (voiceSettings == null)
191 {
192 log += "\n-Missing Voice Setting: " + voiceData.presetVoiceID;
193 phraseCount += voiceData.phrases.Length;
194 continue;
195 }
196
197 // Iterate phrases
198 for (int p = 0; p < voiceData.phrases.Length; p++)
199 {
200 // Iterate progress
201 float progress = (float) phraseCount / (float) phraseTotal;
202 onProgress?.Invoke(progress);
203 phraseCount++;
204
205 // Iterate Load
206 yield return onIterate(service, cacheSettings, voiceSettings, voiceData.phrases[p],
207 (p2) => onProgress?.Invoke(progress + p2 * phraseInc), (l) => log += l);
208
209 // Skip if check fails
210 if (!CheckIterateData(service, preloadData, onIterate, onProgress, onComplete))
211 {
212 yield break;
213 }
214 }
215 }
216
217 // Complete
218 onProgress?.Invoke(1f);
219 onComplete?.Invoke(log);
220 }

319 public static void RefreshPhraseData(TTSService service, TTSDiskCacheSettings cacheSettings, TTSVoiceSettings voiceSettings, TTSPreloadPhraseData phraseData)
320 {
321 // Get voice settings
322 if (service == null || voiceSettings == null || string.IsNullOrEmpty(phraseData.textToSpeak))
323 {
324 phraseData.clipID = string.Empty;
325 phraseData.downloaded = false;
326 phraseData.downloadProgress = 0f;
327 return;
328 }
329 if (cacheSettings == null)
330 {
331 cacheSettings = new TTSDiskCacheSettings()
332 {
333 DiskCacheLocation = TTSDiskCacheLocation.Preload
334 };
335 }
336
337 // Get phrase data
338 phraseData.clipID = service.GetClipID(phraseData.textToSpeak, voiceSettings);
339
340 // Check if file exists
341 string path = service.GetDiskCachePath(phraseData.textToSpeak, phraseData.clipID, voiceSettings, cacheSettings);
342 phraseData.downloaded = File.Exists(path);
343 phraseData.downloadProgress = phraseData.downloaded ? 1f : 0f;

401 public static bool ImportData(TTSPreloadSettings preloadSettings, string textFilePath)
402 {
403 // Check for file
404 if (!File.Exists(textFilePath))
405 {
406 VLog.E($"TTS Preload Utility - Preload file does not exist\nPath: {textFilePath}");
407 return false;
408 }
409 // Load file
410 string textFileContents = File.ReadAllText(textFilePath);
411 if (string.IsNullOrEmpty(textFileContents))
412 {
413 VLog.E($"TTS Preload Utility - Preload file load failed\nPath: {textFilePath}");
414 return false;
415 }
416 // Parse file
417 WitResponseNode node = WitResponseNode.Parse(textFileContents);
418 if (node == null)
419 {
420 VLog.E($"TTS Preload Utility - Preload file parse failed\nPath: {textFilePath}");
421 return false;
422 }
423 // Iterate children for texts
424 WitResponseClass data = node.AsObject;
425 Dictionary<string, List<string>> textsByVoice = new Dictionary<string, List<string>>();
426 foreach (var voiceName in data.ChildNodeNames)
427 {
428 // Get texts list
429 List<string> texts;
430 if (textsByVoice.ContainsKey(voiceName))
431 {
432 texts = textsByVoice[voiceName];
433 }
434 else
435 {
436 texts = new List<string>();
437 }
438
439 // Add text phrases
440 string[] voicePhrases = data[voiceName].AsStringArray;
441 if (voicePhrases != null)
442 {
443 foreach (var phrase in voicePhrases)
444 {
445 if (!string.IsNullOrEmpty(phrase) && !texts.Contains(phrase))
446 {
447 texts.Add(phrase);
448 }
449 }
450 }
451
452 // Apply
453 textsByVoice[voiceName] = texts;
454 }
455 // Import
456 return ImportData(preloadSettings, textsByVoice);

462 public static bool ImportPhrases(TTSPreloadSettings preloadSettings)
463 {
464 // Find phrase providers in all scenes
465 List<ITTSPhraseProvider> phraseProviders = new List<ITTSPhraseProvider>();
466 for (int s = 0; s < SceneManager.sceneCount; s++)
467 {
468 Scene scene = SceneManager.GetSceneAt(s);
469 foreach (var root in scene.GetRootGameObjects())
470 {
471 ITTSPhraseProvider[] found = root.GetComponentsInChildren<ITTSPhraseProvider>(true);
472 if (found != null)
473 {
474 phraseProviders.AddRange(found);
475 }
476 }
477 }
478 // Get all phrases by voice id
479 Dictionary<string, List<string>> textsByVoice = new Dictionary<string, List<string>>();
480 foreach (var phraseProvider in phraseProviders)
481 {
482 // Ignore if no voices are found
483 List<string> voiceIds = phraseProvider.GetVoiceIds();
484 if (voiceIds == null || voiceIds.Count == 0)
485 {
486 continue;
487 }
488
489 // Iterate voice ids
490 foreach (var voiceId in voiceIds)
491 {
492 // Ignore empty voice id
493 if (string.IsNullOrEmpty(voiceId))
494 {
495 continue;
496 }
497
498 // Ignore if phrases are null
499 List<string> phrases = phraseProvider.GetVoicePhrases(voiceId);
500 if (phrases == null || phrases.Count == 0)
501 {
502 continue;
503 }
504
505 // Get phrase list
506 List<string> voicePhrases;
507 if (textsByVoice.ContainsKey(voiceId))
508 {
509 voicePhrases = textsByVoice[voiceId];
510 }
511 else
512 {
513 voicePhrases = new List<string>();
514 }
515
516 // Append unique phrases
517 foreach (var phrase in phrases)
518 {
519 if (!string.IsNullOrEmpty(phrase) && !voicePhrases.Contains(phrase))
520 {
521 voicePhrases.Add(phrase);
522 }
523 }
524
525 // Apply phrase list
526 textsByVoice[voiceId] = voicePhrases;
527 }
528 }
529 // Import with data
530 return ImportData(preloadSettings, textsByVoice);

535 public static bool ImportData(TTSPreloadSettings preloadSettings, Dictionary<string, List<string>> textsByVoice)
536 {
537 // Import
538 if (preloadSettings == null)
539 {
540 VLog.E("TTS Preload Utility - Import Failed - Null Preload Settings");
541 return false;
542 }
543
544 // Whether or not changed
545 bool changed = false;
546
547 // Generate if needed
548 if (preloadSettings.data == null)
549 {
550 preloadSettings.data = new TTSPreloadData();
551 changed = true;
552 }
553
554 // Begin voice list
555 List<TTSPreloadVoiceData> voices = new List<TTSPreloadVoiceData>();
556 if (preloadSettings.data.voices != null)
557 {
558 voices.AddRange(preloadSettings.data.voices);
559 }
560
561 // Iterate voice names
562 foreach (var voiceName in textsByVoice.Keys)
563 {
564 // Get voice index if possible
565 int voiceIndex = voices.FindIndex((v) => string.Equals(v.presetVoiceID, voiceName));
566
567 // Generate voice
568 TTSPreloadVoiceData voice;
569 if (voiceIndex == -1)
570 {
571 voice = new TTSPreloadVoiceData();
572 voice.presetVoiceID = voiceName;
573 voiceIndex = voices.Count;
574 voices.Add(voice);
575 }
576 // Use existing
577 else
578 {
579 voice = voices[voiceIndex];
580 }
581
582 // Get texts & phrases for current voice
583 List<string> texts = new List<string>();
584 List<TTSPreloadPhraseData> phrases = new List<TTSPreloadPhraseData>();
585 if (voice.phrases != null)
586 {
587 foreach (var phrase in voice.phrases)
588 {
589 if (!string.IsNullOrEmpty(phrase.textToSpeak) && !texts.Contains(phrase.textToSpeak))
590 {
591 texts.Add(phrase.textToSpeak);
592 phrases.Add(phrase);
593 }
594 }
595 }
596
597 // Get data
598 List<string> newTexts = textsByVoice[voiceName];
599 if (newTexts != null)
600 {
601 foreach (var newText in newTexts)
602 {
603 if (!string.IsNullOrEmpty(newText) && !texts.Contains(newText))
604 {
605 changed = true;
606 texts.Add(newText);
607 phrases.Add(new TTSPreloadPhraseData()
608 {
609 textToSpeak = newText
610 });
611 }
612 }
613 }
614
615 // Apply voice
616 voice.phrases = phrases.ToArray();
617 voices[voiceIndex] = voice;
618 }
619
620 // Apply data
621 if (changed)
622 {
623 preloadSettings.data.voices = voices.ToArray();
624 EditorUtility.SetDirty(preloadSettings);
625 }
626
627 // Return changed
628 return changed;
629 }
