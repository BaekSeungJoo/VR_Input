145 public OVRPlugin.SpaceQueryInfo ToQueryInfo()
146 {
147 var filterType = OVRPlugin.SpaceQueryFilterType.None;
148 var numIds = 0;
149 var numComponents = 0;
150 if (_uuidFilter != null)
151 {
152 filterType = OVRPlugin.SpaceQueryFilterType.Ids;
153 foreach (var id in _uuidFilter.ToNonAlloc())
154 {
155 if (numIds >= MaxUuidCount)
156 throw new InvalidOperationException(
157 $"{nameof(UuidFilter)} must not contain more than {MaxUuidCount} UUIDs.");
158
159 Ids[numIds++] = id;
160 }
161 }
162 else
163 {
164 filterType = OVRPlugin.SpaceQueryFilterType.Components;
165 ComponentTypes[numComponents++] = _componentType;
166 }
167
168 return new OVRPlugin.SpaceQueryInfo
169 {
170 QueryType = QueryType,
171 MaxQuerySpaces = MaxResults,
172 Timeout = Timeout,
173 Location = Location.ToSpaceStorageLocation(),
174 ActionType = ActionType,
175 FilterType = filterType,
176 IdInfo = new OVRPlugin.SpaceFilterInfoIds
177 {
178 Ids = Ids,
179 NumIds = numIds
180 },
181 ComponentsInfo = new OVRPlugin.SpaceFilterInfoComponents
182 {
183 Components = ComponentTypes,
184 NumComponents = numComponents,
185 }
186 };
187 }
