273 private void ControllerRaycast()
274 {
275 Ray ray = new Ray(_raycastOrigin.position, _raycastOrigin.TransformDirection(Vector3.forward));
276 _lineRenderer.SetPosition(0, _raycastOrigin.position);
277 _lineRenderer.SetPosition(1,
278 _raycastOrigin.position + _raycastOrigin.TransformDirection(Vector3.forward) * 10f);
279
280 RaycastHit hit;
281 if (Physics.Raycast(ray, out hit, Mathf.Infinity))
282 {
283 Anchor anchorObject = hit.collider.GetComponent<Anchor>();
284 if (anchorObject != null)
285 {
286 _lineRenderer.SetPosition(1, hit.point);
287
288 HoverAnchor(anchorObject);
289 return;
290 }
291 }

293 UnhoverAnchor();
294 }
