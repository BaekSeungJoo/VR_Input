62 void Update()
63 {
64 NameValue.text = trackedKeyboard.SystemKeyboardInfo.Name;
65 ConnectedValue.text =
66 ((bool)((trackedKeyboard.SystemKeyboardInfo.KeyboardFlags & OVRPlugin.TrackedKeyboardFlags.Connected) > 0))
67 .ToString();
68 StateValue.text = trackedKeyboard.TrackingState.ToString();
69 SelectKeyboardValue.text = "Select " + trackedKeyboard.KeyboardQueryFlags.ToString() + " Keyboard";
70 TypeValue.text = trackedKeyboard.KeyboardQueryFlags.ToString();
71 switch (trackedKeyboard.TrackingState)
72 {
73 case OVRTrackedKeyboard.TrackedKeyboardState.Uninitialized:
74 case OVRTrackedKeyboard.TrackedKeyboardState.Error:
75 case OVRTrackedKeyboard.TrackedKeyboardState.ErrorExtensionFailed:
76 case OVRTrackedKeyboard.TrackedKeyboardState.StartedNotTracked:
77 case OVRTrackedKeyboard.TrackedKeyboardState.Stale:
78 StateValue.color = BadStateColor;
79 break;
80 default:
81 StateValue.color = GoodStateColor;
82 break;
83 }

106 private IEnumerator SetShaderCoroutine(string shaderName)
107 {
108 bool trackingWasEnabled = trackedKeyboard.TrackingEnabled;
109 trackedKeyboard.TrackingEnabled = false;
110 yield return new WaitWhile(() =>
111 trackedKeyboard.TrackingState != OVRTrackedKeyboard.TrackedKeyboardState.Offline);
112 trackedKeyboard.keyboardModelShader = Shader.Find(shaderName);
113 trackedKeyboard.TrackingEnabled = trackingWasEnabled;
114 }
