81 private class SkeletonTree
82 {
83 public readonly Node Root;
84 public readonly IReadOnlyList<Node> Nodes;
85
86 public SkeletonTree(TSourceJointId root,
87 IReadOnlyDictionary<BodyJointId, JointInfo> mapping)
88 {
89 Dictionary<TSourceJointId, Node> nodes = new Dictionary<TSourceJointId, Node>();
90 foreach (var map in mapping)
91 {
92 BodyJointId jointId = map.Key;
93 JointInfo jointInfo = map.Value;
94 Assert.IsFalse(nodes.ContainsKey(jointInfo.SourceJointId),
95 "Duplicate Joint ID in Mapping");
96 nodes[jointInfo.SourceJointId] =
97 new Node(jointInfo.SourceJointId, jointId);
98 }
99 foreach (var jointInfo in mapping.Values)
100 {
101 Node thisNode = nodes[jointInfo.SourceJointId];
102 thisNode.Parent = nodes[jointInfo.ParentJointId];
103 thisNode.Parent.Children.Add(thisNode);
104 }
105 Nodes = new List<Node>(nodes.Values);
106 Root = nodes[root];
107 }
