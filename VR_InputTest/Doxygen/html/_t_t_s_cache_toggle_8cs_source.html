15namespace Meta.Voice.Samples.TTSVoices
16{
17 public class TTSCacheToggle : MonoBehaviour
18 {
19 // UI references
20 [SerializeField] private TTSDiskCache _diskCache;
21 [SerializeField] private Text _cacheLabel;
22 [SerializeField] private Button _button;
23
24 // Current disk cache location
25 private TTSDiskCacheLocation _cacheLocation = (TTSDiskCacheLocation) (-1);
26
27 // Add listeners
28 private void OnEnable()
29 {
30 // Obtain disk cache if possible
31 if (_diskCache == null)
32 {
33 _diskCache = GameObject.FindObjectOfType<TTSDiskCache>();
34 }
35 // Reset location text
36 RefreshLocation();
37 _button.onClick.AddListener(ToggleCache);
38 }
39 // Current disk cache location
40 private TTSDiskCacheLocation GetCurrentCacheLocation() => _diskCache == null ? TTSDiskCacheLocation.Stream : _diskCache.DiskCacheDefaultSettings.DiskCacheLocation;
41 // Check for changes
42 private void Update()
43 {
44 if (_cacheLocation != GetCurrentCacheLocation())
45 {
46 RefreshLocation();
47 }
48 }
49 // Refresh location & button text
50 private void RefreshLocation()
51 {
52 _cacheLocation = GetCurrentCacheLocation();
53 _cacheLabel.text = $"Disk Cache: {_cacheLocation}";
54 }
55 // Remove listeners
56 private void OnDisable()
57 {
58 _button.onClick.RemoveListener(ToggleCache);
59 }
60 // Toggle cache
61 public void ToggleCache()
62 {
63 // Toggle to next option
64 TTSDiskCacheLocation cacheLocation = GetCurrentCacheLocation();
65 switch (cacheLocation)
66 {
67 case TTSDiskCacheLocation.Stream:
68 cacheLocation = TTSDiskCacheLocation.Temporary;
69 break;
70 case TTSDiskCacheLocation.Temporary:
71 cacheLocation = TTSDiskCacheLocation.Persistent;
72 break;
73 case TTSDiskCacheLocation.Persistent:
74 cacheLocation = TTSDiskCacheLocation.Preload;
75 break;
76 default:
77 cacheLocation = TTSDiskCacheLocation.Stream;
78 break;
79 }
80
81 // Set next option
82 _diskCache.DiskCacheDefaultSettings.DiskCacheLocation = cacheLocation;
83 // Clear runtime cache
84 TTSService.Instance.UnloadAll();
85
86 // Refresh location
87 RefreshLocation();
88 }
89 }
90}
