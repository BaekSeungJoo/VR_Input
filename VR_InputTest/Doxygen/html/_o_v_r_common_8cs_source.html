52 public static OVRPose ToTrackingSpacePose(this Transform transform, Camera camera)
53 {
54 // Initializing to identity, but for all Oculus headsets, down below the pose will be initialized to the runtime's pose value, so identity will never be returned.
55 OVRPose headPose = OVRPose.identity;
56
57 Vector3 pos;
58 Quaternion rot;
59 if (OVRNodeStateProperties.GetNodeStatePropertyVector3(Node.Head, NodeStatePropertyType.Position,
60 OVRPlugin.Node.Head, OVRPlugin.Step.Render, out pos))
61 headPose.position = pos;
62 if (OVRNodeStateProperties.GetNodeStatePropertyQuaternion(Node.Head, NodeStatePropertyType.Orientation,
63 OVRPlugin.Node.Head, OVRPlugin.Step.Render, out rot))
64 headPose.orientation = rot;
65
66 var ret = headPose * transform.ToHeadSpacePose(camera);
67
68 return ret;
69 }

83 public static OVRPose ToWorldSpacePose(this OVRPose trackingSpacePose, Camera mainCamera)
84 {
85 // Transform from tracking-Space to head-Space
86 OVRPose poseInHeadSpace = trackingSpacePose.ToHeadSpacePose();
87
88 // Transform from head space to world space
89 var cameraTransform = mainCamera.transform.localToWorldMatrix;
90 var headSpaceTransform = Matrix4x4.TRS(
91 poseInHeadSpace.position, poseInHeadSpace.orientation, Vector3.one);
92 var worldSpaceTransform = cameraTransform * headSpaceTransform;
93 return new OVRPose
94 {
95 position = worldSpaceTransform.GetColumn(3),
96 orientation = worldSpaceTransform.rotation
97 };
98 }

103 public static OVRPose ToHeadSpacePose(this OVRPose trackingSpacePose)
104 {
105 OVRPose headPose = OVRPose.identity;
106
107 Vector3 pos;
108 Quaternion rot;
109 if (OVRNodeStateProperties.GetNodeStatePropertyVector3(UnityEngine.XR.XRNode.Head,
110 NodeStatePropertyType.Position, OVRPlugin.Node.Head, OVRPlugin.Step.Render, out pos))
111 headPose.position = pos;
112 if (OVRNodeStateProperties.GetNodeStatePropertyQuaternion(UnityEngine.XR.XRNode.Head,
113 NodeStatePropertyType.Orientation, OVRPlugin.Node.Head, OVRPlugin.Step.Render, out rot))
114 headPose.orientation = rot;
115
116 OVRPose poseInHeadSpace = headPose.Inverse() * trackingSpacePose;
117
118 return poseInHeadSpace;
119 }

124 public static OVRPose ToHeadSpacePose(this Transform transform, Camera camera)
125 {
126 // The Quaternion used in OVRPose is not 1-1 mapped with the Transform[0:3,0:3] (which is Scale * Rotation), need to calculate rotation and position separately.
127 Quaternion rotation = Quaternion.Inverse(camera.transform.rotation) * transform.rotation;
128 var position = (camera.transform.worldToLocalMatrix * transform.localToWorldMatrix).GetColumn(3);
129 return new OVRPose
130 {
131 orientation = rotation,
132 position = position
133 };
134 }

335 public static bool Equals(this Gradient gradient, Gradient otherGradient)
336 {
337 if (gradient.colorKeys.Length != otherGradient.colorKeys.Length ||
338 gradient.alphaKeys.Length != otherGradient.alphaKeys.Length)
339 return false;
340
341 for (int i = 0; i < gradient.colorKeys.Length; i++)
342 {
343 GradientColorKey key = gradient.colorKeys[i];
344 GradientColorKey otherKey = otherGradient.colorKeys[i];
345 if (key.color != otherKey.color || key.time != otherKey.time)
346 return false;
347 }
348
349 for (int i = 0; i < gradient.alphaKeys.Length; i++)
350 {
351 GradientAlphaKey key = gradient.alphaKeys[i];
352 GradientAlphaKey otherKey = otherGradient.alphaKeys[i];
353 if (key.alpha != otherKey.alpha || key.time != otherKey.time)
354 return false;
355 }
356
357 return true;
358 }
359
360 public static void CopyFrom(this Gradient gradient, Gradient otherGradient)
361 {
362 GradientColorKey[] colorKeys = new GradientColorKey[otherGradient.colorKeys.Length];
363 for (int i = 0; i < colorKeys.Length; i++)
364 {
365 Color col = otherGradient.colorKeys[i].color;
366 colorKeys[i].color = new Color(col.r, col.g, col.b, col.a);
367 colorKeys[i].time = otherGradient.colorKeys[i].time;
368 }
369
370 GradientAlphaKey[] alphaKeys = new GradientAlphaKey[otherGradient.alphaKeys.Length];
371 for (int i = 0; i < alphaKeys.Length; i++)
372 {
373 alphaKeys[i].alpha = otherGradient.alphaKeys[i].alpha;
374 alphaKeys[i].time = otherGradient.alphaKeys[i].time;
375 }
376
377 gradient.SetKeys(colorKeys, alphaKeys);
378 }

506 private static bool ValidateProperty(Node nodeType, ref NodeState requestedNodeState)
507 {
508 InputTracking.GetNodeStates(nodeStateList);
509
510 if (nodeStateList.Count == 0)
511 return false;
512
513 bool nodeStateFound = false;
514 requestedNodeState = nodeStateList[0];
515
516 for (int i = 0; i < nodeStateList.Count; i++)
517 {
518 if (nodeStateList[i].nodeType == nodeType)
519 {
520 requestedNodeState = nodeStateList[i];
521 nodeStateFound = true;
522 break;
523 }
524 }
525
526 return nodeStateFound;

722{
723 private bool disposed = false;
724 private int m_numBytes = 0;
725 private IntPtr m_ptr = IntPtr.Zero;

730 public OVRNativeBuffer(int numBytes)

734
739 ~OVRNativeBuffer()
740 {
741 Dispose(false);
742 }
743
747 public void Reset(int numBytes)
748 {
749 Reallocate(numBytes);
750 }
751
755 public int GetCapacity()
756 {
757 return m_numBytes;
758 }
759
763 public IntPtr GetPointer(int byteOffset = 0)
764 {
765 if (byteOffset < 0 || byteOffset >= m_numBytes)
766 return IntPtr.Zero;
767 return (byteOffset == 0) ? m_ptr : new IntPtr(m_ptr.ToInt64() + byteOffset);
768 }
769
777 public void Dispose()
778 {
779 Dispose(true);
780 GC.SuppressFinalize(this);
781 }
782
783 private void Dispose(bool disposing)
784 {
785 if (disposed)
786 return;
787
788 if (disposing)
789 {
790 // dispose managed resources
791 }
792
793 // dispose unmanaged resources
794 Release();
795
796 disposed = true;
797 }
798
799 private void Reallocate(int numBytes)
800 {
801 Release();
802
803 if (numBytes > 0)
804 {
805 m_ptr = Marshal.AllocHGlobal(numBytes);
806 m_numBytes = numBytes;
807 }
808 else
809 {
810 m_ptr = IntPtr.Zero;
811 m_numBytes = 0;
812 }
813 }
814
815 private void Release()
816 {
817 if (m_ptr != IntPtr.Zero)
818 {
819 Marshal.FreeHGlobal(m_ptr);
820 m_ptr = IntPtr.Zero;
821 m_numBytes = 0;
822 }
823 }
