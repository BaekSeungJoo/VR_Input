44 protected override void InitializeTeleportDestination()

46 _initialRotation = LocomotionTeleport.GetHeadRotationY();
47 _currentRotation = _initialRotation;

51 protected override void UpdateTeleportDestination()
52 {
53 float magnitude;
54 Vector2 direction;
55 if (Thumbstick == OVRInput.Controller.Touch)
56 {
57 Vector2 leftDir = OVRInput.Get(OVRInput.RawAxis2D.LThumbstick);
58 Vector2 rightDir = OVRInput.Get(OVRInput.RawAxis2D.RThumbstick);
59 float leftMag = leftDir.magnitude;
60 float rightMag = rightDir.magnitude;
61 if (leftMag > rightMag)
62 {
63 magnitude = leftMag;
64 direction = leftDir;
65 }
66 else
67 {
68 magnitude = rightMag;
69 direction = rightDir;
70 }
71 }
72 else
73 {
74 if (Thumbstick == OVRInput.Controller.LTouch) direction = OVRInput.Get(OVRInput.RawAxis2D.LThumbstick);
75 else direction = OVRInput.Get(OVRInput.RawAxis2D.RThumbstick);
76 magnitude = direction.magnitude;
77 }
78
79 if (!AimData.TargetValid)
80 {
81 _lastValidDirection = new Vector2();
82 }
83
84 if (magnitude < RotateStickThreshold)
85 {
86 direction = _lastValidDirection;
87 magnitude = direction.magnitude;
88
89 if (magnitude < RotateStickThreshold)
90 {
91 _initialRotation = LocomotionTeleport.GetHeadRotationY();
92 direction.x = 0;
93 direction.y = 1;
94 }
95 }
96 else
97 {
98 _lastValidDirection = direction;
99 }
100
101 var tracking = LocomotionTeleport.LocomotionController.CameraRig.trackingSpace.rotation;
102
103 if (magnitude > RotateStickThreshold)
104 {
105 direction /= magnitude; // normalize the vector
106 var rot = _initialRotation * Quaternion.LookRotation(new Vector3(direction.x, 0, direction.y), Vector3.up);
107 _currentRotation = tracking * rot;
108 }
109 else
110 {
111 _currentRotation = tracking * LocomotionTeleport.GetHeadRotationY();
112 }
113
114 LocomotionTeleport.OnUpdateTeleportDestination(AimData.TargetValid, AimData.Destination, _currentRotation,
115 GetLandingOrientation(OrientationMode, _currentRotation));
116 }
