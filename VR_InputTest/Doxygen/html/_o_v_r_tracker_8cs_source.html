128 public OVRPose GetPose(int tracker = 0)

133 OVRPose p;

138 break;
139 case 1:

141 break;
142 case 2:
143 p = OVRPlugin.GetNodePose(OVRPlugin.Node.TrackerTwo, OVRPlugin.Step.Render).ToOVRPose();
144 break;
145 case 3:
146 p = OVRPlugin.GetNodePose(OVRPlugin.Node.TrackerThree, OVRPlugin.Step.Render).ToOVRPose();
147 break;
148 default:
149 return OVRPose.identity;
150 }
151
152 return new OVRPose()
