45 public bool Raycast(in Ray ray, out SurfaceHit hit, float maxDistance = 0)
46 {
47 hit = new SurfaceHit();
48 Plane plane = new Plane(_planeSurface.Normal, Transform.position);
49
50 if (plane.Raycast(ray, out float hitDistance))
51 {
52 if (maxDistance > 0 && hitDistance > maxDistance)
53 {
54 return false;
55 }
56
57 Vector3 hitPointWorld = ray.GetPoint(hitDistance);
58 Vector3 hitPointLocal = Transform.InverseTransformPoint(hitPointWorld);
59
60 if (Mathf.Abs(hitPointLocal.x) > _radius ||
61 Mathf.Abs(hitPointLocal.y) > _radius)
62 {
63 return false;
64 }
65
66 hit.Point = hitPointWorld;
67 hit.Normal = plane.normal;
68 hit.Distance = hitDistance;
69 return true;
70 }
71
72 return false;
73 }
74
75 // Closest point to circle is computed by projecting point to the plane
76 // the circle is on and then clamping to the circle
77 public bool ClosestSurfacePoint(in Vector3 point, out SurfaceHit hit, float maxDistance = 0)
78 {
79 hit = new SurfaceHit();
80
81 Vector3 vectorFromPlane = point - Transform.position;
82 Vector3 planeNormal = _planeSurface.Normal;
83 Vector3 projectedPoint = Vector3.ProjectOnPlane(vectorFromPlane, planeNormal);
84
85 float distanceFromCenterSqr = projectedPoint.sqrMagnitude;
86 float worldRadius = Transform.lossyScale.x * _radius;
87 if (distanceFromCenterSqr > worldRadius * worldRadius)
88 {
89 projectedPoint = worldRadius * projectedPoint.normalized;
90 }
91 Vector3 closestPoint = projectedPoint + Transform.position;
92
93 hit.Point = closestPoint;
94 hit.Normal = -1.0f * Transform.forward;
95 hit.Distance = Vector3.Distance(point, closestPoint);
96 return maxDistance <= 0 || hit.Distance <= maxDistance;
97 }
