311 public bool AimCollisionTest(Vector3 start, Vector3 end, LayerMask aimCollisionLayerMask, out RaycastHit hitInfo)
312 {
313 var delta = end - start;
314 var distance = delta.magnitude;
315 var direction = delta / distance;
316
317 switch (AimCollisionType)
318 {
319 case AimCollisionTypes.Capsule:
320 {
321 float r, h;
322 if (UseCharacterCollisionData)
323 {
324 var c = LocomotionController.CharacterController;
325 h = c.height;
326 r = c.radius;
327 }
328 else
329 {
330 h = AimCollisionHeight;
331 r = AimCollisionRadius;
332 }
333
334 return Physics.CapsuleCast(start + new Vector3(0, r, 0),
335 start + new Vector3(0, h + r, 0), r, direction,
336 out hitInfo, distance, aimCollisionLayerMask, QueryTriggerInteraction.Ignore);
337 }
338
339 case AimCollisionTypes.Point:
340 return Physics.Raycast(start, direction, out hitInfo, distance, aimCollisionLayerMask,
341 QueryTriggerInteraction.Ignore);
342
343 case AimCollisionTypes.Sphere:
344 {
345 float r;
346 if (UseCharacterCollisionData)
347 {
348 var c = LocomotionController.CharacterController;
349 //r = c.radius - c.skinWidth;
350 r = c.radius;
351 }
352 else
353 {
354 r = AimCollisionRadius;
355 }
356
357 return Physics.SphereCast(start, r, direction, out hitInfo, distance, aimCollisionLayerMask,
358 QueryTriggerInteraction.Ignore);
359 }
360 }
361
362 // App should never get here.
363 throw new Exception();
364 }

463
464 // yield once so that all the components will have time to process their OnEnable message before this

551 public event Action ExitStateAim;
552
559 protected IEnumerator AimStateCoroutine()
560 {
561 LogState("AimState: Start");
562 CurrentState = States.Aim;
563 EnableMotion(EnableMovementDuringAim, EnableRotationDuringAim);
564 if (EnterStateAim != null)
565 {
566 EnterStateAim();
567 }
568
569 _teleportDestination.gameObject.SetActive(true);
570
571 // Wait until the user is done aiming. The input system will turn this off when the button that triggered aiming is released.
572 while (CurrentIntention == TeleportIntentions.Aim)
573 {
574 yield return null;
575 }
576
577 LogState("AimState: End. Intention: " + CurrentIntention);
578 if (ExitStateAim != null)
579 {
580 ExitStateAim();
581 }
582
583 // Wait until the next frame to proceed to the next state's coroutine.
584 yield return null;
585
586 // If target is valid, enter pre-teleport otherwise cancel the teleport.
587 LogState("AimState: Switch state. Intention: " + CurrentIntention);
588 if ((CurrentIntention == TeleportIntentions.PreTeleport || CurrentIntention == TeleportIntentions.Teleport) &&
589 _teleportDestination.IsValidDestination)
590 {
591 StartCoroutine(PreTeleportStateCoroutine());
592 }
593 else
594 {
595 StartCoroutine(CancelAimStateCoroutine());
596 }
597 }

798 public void DoTeleport()
799 {
800 var character = LocomotionController.CharacterController;
801 var characterTransform = character.transform;
802 var destTransform = _teleportDestination.OrientationIndicator;
803
804 Vector3 destPosition = destTransform.position;
805 destPosition.y += character.height * 0.5f;
806 Quaternion destRotation = _teleportDestination.LandingRotation; // destTransform.rotation;
807#if false
808 Quaternion destRotation = destTransform.rotation;
809
810 //Debug.Log("Rots: " + destRotation + " " + destTransform.rotation * Quaternion.Euler(0, -LocomotionController.CameraRig.trackingSpace.localEulerAngles.y, 0));
811
812 destRotation =
813 destRotation * Quaternion.Euler(0, -LocomotionController.CameraRig.trackingSpace.localEulerAngles.y, 0);
814#endif
815 if (Teleported != null)
816 {
817 Teleported(characterTransform, destPosition, destRotation);
818 }
819
820 characterTransform.position = destPosition;
821 characterTransform.rotation = destRotation;
822 }

838 public Quaternion GetHeadRotationY()
839 {
840 Quaternion headRotation = Quaternion.identity;
841#if UNITY_2019_1_OR_NEWER
842 UnityEngine.XR.InputDevice device = UnityEngine.XR.InputDevices.GetDeviceAtXRNode(UnityEngine.XR.XRNode.Head);
843 if (device.isValid)
844 {
845 device.TryGetFeatureValue(UnityEngine.XR.CommonUsages.deviceRotation, out headRotation);
846 }
847#elif UNITY_2017_2_OR_NEWER
848 List<UnityEngine.XR.XRNodeState> nodeStates = new List<UnityEngine.XR.XRNodeState>();
849 UnityEngine.XR.InputTracking.GetNodeStates(nodeStates);
850 foreach (UnityEngine.XR.XRNodeState n in nodeStates)
851 {
852 if (n.nodeType == UnityEngine.XR.XRNode.Head)
853 {
854 n.TryGetRotation(out headRotation);
855 break;
856 }
857 }
858#else
859 headRotation = InputTracking.GetLocalRotation(VRNode.Head);
860#endif
861 Vector3 euler = headRotation.eulerAngles;
862 euler.x = 0;
863 euler.z = 0;
864 headRotation = Quaternion.Euler(euler);
865 return headRotation;
866 }
