32 [CustomEditor(typeof(OVRCanvasMeshRenderer))]
33 public class OVRCanvasMeshRendererEditor : UnityEditor.Editor
34 {
35 private EditorBase _editorDrawer;
36 private const string DeferredSection = "Deferred";
37
38 public new OVRCanvasMeshRenderer target
39 {
40 get
41 {
42 return base.target as OVRCanvasMeshRenderer;
43 }
44 }
45
46 protected virtual void OnEnable()
47 {
48 _editorDrawer = new EditorBase(serializedObject);
49 _editorDrawer.AddToSection(DeferredSection, baseProps.UseAlphaToMask, baseProps.AlphaCutoutThreshold);
50 var renderingMode = serializedObject.FindProperty(baseProps.RenderingMode);
51
52 bool CheckIsOVR()
53 {
54 return renderingMode.intValue == (int)OVRRenderingMode.Underlay ||
55 renderingMode.intValue == (int)OVRRenderingMode.Overlay;
56 }
57
58 _editorDrawer.Draw(props.RuntimeOffset, (offsetProp) =>
59 {
60 if (CheckIsOVR())
61 {
62 EditorGUILayout.PropertyField(offsetProp);
63 }
64 });
65
66 _editorDrawer.Draw(baseProps.RenderingMode, props.CanvasMesh, (modeProp, meshProp) =>
67 {
68 EditorGUILayout.PropertyField(meshProp);
69 OVRRenderingMode value = (OVRRenderingMode)modeProp.intValue;
70 value = (OVRRenderingMode)EditorGUILayout.EnumPopup("Rendering Mode", value);
71 modeProp.intValue = (int)value;
72 });
73
74 _editorDrawer.Draw(props.EnableSuperSampling, props.EmulateWhileInEditor, props.DoUnderlayAntiAliasing, (sampleProp, emulateProp, aaProp) =>
75 {
76 if (CheckIsOVR())
77 {
78 EditorGUILayout.PropertyField(sampleProp);
79 if (renderingMode.intValue == (int)OVRRenderingMode.Underlay)
80 {
81 EditorGUILayout.PropertyField(aaProp);
82 }
83 EditorGUILayout.PropertyField(emulateProp);
84 }
85 });
86
87 _editorDrawer.Draw(baseProps.UseAlphaToMask, baseProps.AlphaCutoutThreshold, (maskProp, cutoutProp) =>
88 {
89 if (renderingMode.intValue == (int)OVRRenderingMode.AlphaCutout)
90 {
91 EditorGUILayout.PropertyField(maskProp);
92
93 if (maskProp.boolValue == false)
94 {
95 EditorGUILayout.PropertyField(cutoutProp);
96 }
97 }
98 });
99 }
100
101 protected virtual void OnDisable()
102 {
103 }
104
105 protected void BeforeInspector()
106 {
107 AutoFix(AutoFixIsUsingMipMaps(), AutoFixDisableMipMaps,
108 $"{nameof(CanvasRenderTexture)} " +
109 $"is generating mip maps, but these are ignored when using OVR Overlay/Underlay rendering.");
110 }
111
112 public override void OnInspectorGUI()
113 {
114 BeforeInspector();
115 _editorDrawer.DrawFullInspector();
116 }
117
118 private bool AutoFix(bool needsFix, Action fixAction, string message)
119 {
120 if (needsFix)
121 {
122 using (new EditorGUILayout.HorizontalScope())
123 {
124 EditorGUILayout.HelpBox(message, MessageType.Warning);
125 if (GUILayout.Button("Auto-Fix", GUILayout.ExpandHeight(true)))
126 {
127 fixAction();
128 }
129 }
130 }
131
132 return needsFix;
133 }
134
135 private bool AutoFixIsUsingMipMaps()
136 {
137 var modeProp = serializedObject.FindProperty(baseProps.RenderingMode);
138 OVRRenderingMode mode = (OVRRenderingMode)modeProp.intValue;
139 if (mode != OVRRenderingMode.Overlay && mode != OVRRenderingMode.Underlay)
140 {
141 return false;
142 }
143
144 var rtProp = serializedObject.FindProperty(props.CanvasRenderTexture);
145 CanvasRenderTexture canvasRT = rtProp.objectReferenceValue as CanvasRenderTexture;
146 if (canvasRT == null)
147 {
148 return false;
149 }
150
151 var mipProp = new SerializedObject(canvasRT).FindProperty(rtprops.GenerateMipMaps);
152 return mipProp.boolValue;
153 }
154
155 private void AutoFixDisableMipMaps()
156 {
157 var rtProp = serializedObject.FindProperty(props.CanvasRenderTexture);
158 CanvasRenderTexture canvasRT = rtProp.objectReferenceValue as CanvasRenderTexture;
159 var rtSO = new SerializedObject(canvasRT);
160 var mipProp = rtSO.FindProperty(rtprops.GenerateMipMaps);
161 mipProp.boolValue = false;
162 rtSO.ApplyModifiedProperties();
163 }
164 }
165}
