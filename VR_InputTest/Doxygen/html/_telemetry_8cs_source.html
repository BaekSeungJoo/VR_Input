265 private class TelemetryChannel
266 {
267 private const string PluginName = "SDKTelemetry";
268
272 private Dictionary<int, TelemetryEventId> _instanceKeyMap = new Dictionary<int, TelemetryEventId>();
273
279 internal virtual int StartEvent(TelemetryEventId eventId)
280 {
281 _instanceKeyMap[_nextEventSequenceId] = eventId;
282 QplMarkerStart((int)eventId, _nextEventSequenceId, -1);
283 AnnotateEvent(_nextEventSequenceId, AnnotationKey.SessionId, _sessionID.ToString());
284 AnnotateEvent(_nextEventSequenceId, AnnotationKey.StartTimeStamp,
285 DateTimeUtility.ElapsedMilliseconds.ToString());
286
287 LogVerbose($"Started telemetry event {eventId}:{_nextEventSequenceId}");
288 return _nextEventSequenceId++;
289 }
290
297 internal virtual void LogInstantEvent(TelemetryEventId eventId,
298 Dictionary<AnnotationKey, string> annotations = null)
299 {
300 var instanceKey = _nextEventSequenceId;
301 _instanceKeyMap[instanceKey] = eventId;
302 var timeStamp = DateTimeUtility.ElapsedMilliseconds.ToString();
303 QplMarkerStart((int)eventId, instanceKey, -1);
304 LogVerbose($"Started instant telemetry event {eventId}:{instanceKey}");
305 AnnotateEvent(instanceKey, AnnotationKey.SessionId, _sessionID.ToString());
306 AnnotateEvent(instanceKey, AnnotationKey.StartTimeStamp, timeStamp);
307 AnnotateEvent(instanceKey, AnnotationKey.EndTimeStamp, timeStamp);
308
309 if (annotations != null)
310 {
311 foreach (var annotation in annotations)
312 {
313 AnnotateEvent(instanceKey, annotation.Key, annotation.Value);
314 }
315 }
316
317 QplMarkerEnd((int)eventId, ResultType.Success, instanceKey, -1);
318 LogVerbose($"Ended instant telemetry event {eventId}:{instanceKey}");
319 _instanceKeyMap.Remove(instanceKey);
320 ++_nextEventSequenceId;
321 }
322
330 internal virtual void AnnotateEvent(int instanceKey, AnnotationKey key,
331 string value)
332 {
333 if (!this._instanceKeyMap.ContainsKey(instanceKey))
334 {
335 LogWarning($"Attempted to end an event that's invalid or not started. Instance ID: {instanceKey}");
336 return;
337 }
338
339 var eventId = _instanceKeyMap[instanceKey];
340 QplMarkerAnnotation((int)eventId, key.ToString(), value, instanceKey);
341 LogVerbose($"Annotated telemetry event {eventId}:{instanceKey} with {key}:{value}");
342 }
343
351 internal virtual void EndEvent(int instanceKey, ResultType result)
352 {
353 if (!this._instanceKeyMap.ContainsKey(instanceKey))
354 {
355 LogWarning($"Attempted to end an event that's not started or invalid. Instance ID: {instanceKey}");
356 return;
357 }
358
359 var eventId = _instanceKeyMap[instanceKey];
360 AnnotateEvent(instanceKey, AnnotationKey.EndTimeStamp,
361 DateTimeUtility.ElapsedMilliseconds.ToString());
362 QplMarkerEnd((int)eventId, result, instanceKey, -1);
363 _instanceKeyMap.Remove(instanceKey);
364 LogVerbose($"Ended telemetry event {eventId}:{instanceKey}({result})");
365 }
366
367 public void ShutdownTelemetry()
368 {
369 try
370 {
371 foreach (var instanceKey in _instanceKeyMap.Keys)
372 {
373 AnnotateEvent(instanceKey, AnnotationKey.Error, "Telemetry event not ended gracefully");
374 EndEvent(instanceKey, ResultType.Cancel);
375 }
376 }
377 finally
378 {
379 OnEditorShutdown();
380 }
381 }
