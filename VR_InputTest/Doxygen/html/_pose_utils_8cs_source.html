283 public static Pose MirrorPoseRotation(this in Pose pose, Vector3 normal, Vector3 tangent)
284 {
285 Pose mirrorPose = pose;
286 Vector3 forward = pose.rotation * -Vector3.forward;
287 Vector3 projectedForward = Vector3.ProjectOnPlane(forward, normal);
288 float angleForward = Vector3.SignedAngle(projectedForward, tangent, normal);
289 Vector3 mirrorForward = Quaternion.AngleAxis(2 * angleForward, normal) * forward;
290
291 Vector3 up = pose.rotation * -Vector3.up;
292 Vector3 projectedUp = Vector3.ProjectOnPlane(up, normal);
293 float angleUp = Vector3.SignedAngle(projectedUp, tangent, normal);
294 Vector3 mirrorUp = Quaternion.AngleAxis(2 * angleUp, normal) * up;
295
296 mirrorPose.rotation = Quaternion.LookRotation(mirrorForward, mirrorUp);
297 return mirrorPose;
298 }
