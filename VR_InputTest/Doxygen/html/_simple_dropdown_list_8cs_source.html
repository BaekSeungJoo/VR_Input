170 #region LOAD
175 public void LoadDropdown(string[] newOptions)
176 {
177 // Get options
178 _options = newOptions;
179
180 // Disable all previous cells
181 foreach (var cell in _cells)
182 {
183 cell.gameObject.SetActive(false);
184 }
185
186 // Setup if needed, or log error
187 FindCellPrefab();
188 if (_dropdownListCellPrefab == null)
189 {
190 VLog.W($"Cannot load {gameObject.name} without a cell prefab");
191 return;
192 }
193
194 // Iterate all options
195 float height = 0f;
196 int total = _options == null ? 0 : _options.Length;
197 for (int c = 0; c < total; c++)
198 {
199 // Get existing cell if possible
200 Toggle cell = c < _cells.Count ? _cells[c] : null;
201
202 // Instantiate if needed
203 if (cell == null)
204 {
205 // Instantiate
206 cell = Instantiate(_dropdownListCellPrefab.gameObject).GetComponent<Toggle>();
207 cell.name = $"{_dropdownListCellPrefab.gameObject.name}_CELL_{c:000}";
208 cell.transform.SetParent(_dropdownListScrollRect.content, false);
209
210 // Override cell
211 if (c < _cells.Count)
212 {
213 _cells[c] = cell;
214 }
215 // Add new cell
216 else
217 {
218 _cells.Add(cell);
219 }
220 }
221
222 // Enable & load
223 cell.gameObject.SetActive(true);
224 LoadCell(cell, c, _options[c], ref height);
225 }
226
227 // Set height to content of scrolldown list
228 _dropdownListScrollRect.content.SetInsetAndSizeFromParentEdge(RectTransform.Edge.Top, 0f, height);
229 }
235 protected virtual void LoadCell(Toggle cell, int index, string option, ref float y)
236 {
237 // Apply name to cell
238 Text cellText = cell.GetComponentInChildren<Text>();
239 if (cellText != null)
240 {
241 cellText.text = option;
242 }
243
244 // Apply callback to cell
245 cell.onValueChanged.RemoveAllListeners();
246 cell.isOn = false;
247 cell.onValueChanged.AddListener((isSelected) => OnCellClick(index, isSelected));
248
249 // Apply width to cell
250 RectTransform cellRect = cell.GetComponent<RectTransform>();
251 cellRect.anchorMin = new Vector2(0f, 1f);
252 cellRect.anchorMax = new Vector2(1f, 1f);
253 cellRect.pivot = new Vector2(0.5f, 1f);
254 cellRect.SetSizeWithCurrentAnchors(RectTransform.Axis.Horizontal, _dropdownListScrollRect.content.rect.width);
255 cellRect.anchoredPosition = new Vector2(0f, -y);
256
257 // Apply height
258 float height = cellRect.rect.height;
259 if (cellText != null)
260 {
261 float textMargin = Mathf.Max(0f, height - cellText.rectTransform.rect.height);
262 height = Mathf.CeilToInt(cellText.preferredHeight + _dropdownCellTextPadding + textMargin);
263 cellRect.SetSizeWithCurrentAnchors(RectTransform.Axis.Vertical, height);
264 }
265
266 // Add new height
267 y += height;
268 }

276 public void SelectOption(int newIndex)
277 {
278 // Ignore if same
279 if (newIndex == SelectedOptionIndex)
280 {
281 return;
282 }
283
284 // Apply new index
285 _selectedOptionIndex = newIndex;
286
287 // Get option text
288 string optionText = SelectedOption;
289
290 // Apply to button
291 Text buttonText = _dropdownToggle.GetComponentInChildren<Text>();
292 if (buttonText != null)
293 {
294 if (string.IsNullOrEmpty(optionText))
295 {
296 buttonText.text = DropdownToggleUnselectedText;
297 }
298 else
299 {
300 buttonText.text = optionText;
301 }
302 }
303
304 // Adjust cells
305 int total = _cells == null ? 0 : _cells.Count;
306 for (int i = 0; i < total; i++)
307 {
308 _cells[i].isOn = i == SelectedOptionIndex;
309 _cells[i].interactable = i != SelectedOptionIndex;
310 }
311
312 // Call event
313 OnOptionSelected?.Invoke(optionText);
314 OnIndexSelected?.Invoke(SelectedOptionIndex);
315
316 // Selected
317 if (IsShowing)
318 {
319 SetShowing(false);
320 }
321 }

344 public int GetOptionIndex(string option)
345 {
346 if (_options != null)
347 {
348 for (int o = 0; o < _options.Length; o++)
349 {
350 if (string.Equals(_options[o], option))
351 {
352 return o;
353 }
354 }
355 }
356 return -1;
357 }
