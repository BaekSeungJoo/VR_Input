25namespace OculusSampleFramework

27 public class PanelHMDFollower : MonoBehaviour
28 {
29 private const float TOTAL_DURATION = 3.0f;
30 private const float HMD_MOVEMENT_THRESHOLD = 0.3f;
31
32 [SerializeField] private float _maxDistance = 0.3f;
33 [SerializeField] private float _minDistance = 0.05f;
34 [SerializeField] private float _minZDistance = 0.05f;
35
36 private OVRCameraRig _cameraRig;
37 private Vector3 _panelInitialPosition = Vector3.zero;
38 private Coroutine _coroutine = null;
39 private Vector3 _prevPos = Vector3.zero;
40 private Vector3 _lastMovedToPos = Vector3.zero;
41
42 private void Awake()
43 {
44 _cameraRig = FindObjectOfType<OVRCameraRig>();
45 _panelInitialPosition = transform.position;
46 }
47
48 private void Update()
49 {
50 var centerEyeAnchorPos = _cameraRig.centerEyeAnchor.position;
51 var myPosition = transform.position;
52 //Distance from centereye since last time we updated panel position.
53 float distanceFromLastMovement = Vector3.Distance(centerEyeAnchorPos, _lastMovedToPos);
54 float headMovementSpeed = (_cameraRig.centerEyeAnchor.position - _prevPos).magnitude / Time.deltaTime;
55 var currDiffFromCenterEye = transform.position - centerEyeAnchorPos;
56 var currDistanceFromCenterEye = currDiffFromCenterEye.magnitude;
57
58
59 // 1) wait for center eye to stabilize after distance gets too large
60 // 2) check if center eye is too close to panel
61 // 3) check if depth isn't too close
62 if (((distanceFromLastMovement > _maxDistance) || (_minZDistance > currDiffFromCenterEye.z) ||
63 (_minDistance > currDistanceFromCenterEye)) &&
64 headMovementSpeed < HMD_MOVEMENT_THRESHOLD && _coroutine == null)
65 {
66 if (_coroutine == null)
67 {
68 _coroutine = StartCoroutine(LerpToHMD());
69 }
70 }
71
72 _prevPos = _cameraRig.centerEyeAnchor.position;
73 }
74
75 private Vector3 CalculateIdealAnchorPosition()
76 {
77 return _cameraRig.centerEyeAnchor.position + _panelInitialPosition;
78 }
79
80 private IEnumerator LerpToHMD()
81 {
82 Vector3 newPanelPosition = CalculateIdealAnchorPosition();
83 _lastMovedToPos = _cameraRig.centerEyeAnchor.position;
84 float startTime = Time.time;
85 float endTime = Time.time + TOTAL_DURATION;
86
87 while (Time.time < endTime)
88 {
89 transform.position =
90 Vector3.Lerp(transform.position, newPanelPosition, (Time.time - startTime) / TOTAL_DURATION);
91 yield return null;
92 }
93
94 transform.position = newPanelPosition;
95 _coroutine = null;
96 }
97 }
