Detailed Description

Represents a volume described by its Bounds.

This component can be accessed from an OVRAnchor that supports it by calling OVRAnchor.GetComponent<T> from the anchor.

46 {
47 // OpenXR Bounds describe a volume by its position (or offset) and its size (or extents)
48 // https://registry.khronos.org/OpenXR/specs/1.0/man/html/XrRect3DfFB.html
49 // Unity Bounds describe a volume by its center and its size (or extents)
50 // https://docs.unity3d.com/ScriptReference/Bounds.html
51 var extents = openXrBounds.Size.FromSize3f();
52
53 // OpenXR uses a right-handed coordinate system
54 // Unity uses left-handed coordinate system
55 // We therefore need to flip the z axis to convert from OpenXR to Unity
56 // And then, because of the z-axis positive normal, rotate 180 around +y
57 // This ends up being equivalent to flipping x axis
58 var offset = openXrBounds.Pos.FromFlippedXVector3f();
59 // When flipping one axis, position doesn't point to a min corner any more
60 offset.x -= extents.x;
61 // And add half of the extents to find the center
62 var halfExtents = extents * 0.5f;
63 var center = offset + halfExtents;
64
65 // Bounds constructor takes the center and the full size (not half extents)
66 return new Bounds(center, extents);
67 }
