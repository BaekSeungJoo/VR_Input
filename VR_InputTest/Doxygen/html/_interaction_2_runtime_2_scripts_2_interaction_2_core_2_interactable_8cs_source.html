210 public bool CanBeSelectedBy(TInteractor interactor)
211 {

215 }

218 _selectingInteractors.Count == MaxSelectingInteractors)

221 }
222
223 if (MaxInteractors >= 0 &&
224 _interactors.Count == MaxInteractors &&
225 !_interactors.Contains(interactor))
226 {
227 return false;
228 }
229
230 if (InteractorFilters == null)
231 {
232 return true;
233 }
234
235 foreach (IGameObjectFilter interactorFilter in InteractorFilters)
236 {
237 if (!interactorFilter.Filter(interactor.gameObject))
238 {
239 return false;
240 }
241 }

243 return true;
