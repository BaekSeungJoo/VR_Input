105 protected override RayInteractable ComputeCandidate()
106 {
107 CollisionInfo = null;
108
109 RayInteractable closestInteractable = null;
110 float closestDist = float.MaxValue;
111 Vector3 candidatePosition = Vector3.zero;
112 var interactables = RayInteractable.Registry.List(this);
113
114 foreach (RayInteractable interactable in interactables)
115 {
116 if (interactable.Raycast(Ray, out SurfaceHit hit, MaxRayLength, false))
117 {
118 bool equal = Mathf.Abs(hit.Distance - closestDist) < _equalDistanceThreshold;
119 if ((!equal && hit.Distance < closestDist) ||
120 (equal && ComputeCandidateTiebreaker(interactable, closestInteractable) > 0))
121 {
122 closestDist = hit.Distance;
123 closestInteractable = interactable;
124 CollisionInfo = hit;
125 candidatePosition = hit.Point;
126 }
127 }
128 }
129
130 float rayDist = (closestInteractable != null ? closestDist : MaxRayLength);
131 End = Origin + rayDist * Forward;
132
133 _rayCandidateProperties = new RayCandidateProperties(closestInteractable, candidatePosition);
134
135 return closestInteractable;
136 }

149 protected override void InteractableSelected(RayInteractable interactable)
150 {
151 if (interactable != null)
152 {
153 _movedHit = CollisionInfo.Value;
154 Pose hitPose = new Pose(_movedHit.Point, Quaternion.LookRotation(_movedHit.Normal));
155 Pose backHitPose = new Pose(_movedHit.Point, Quaternion.LookRotation(-_movedHit.Normal));
156 _movement = interactable.GenerateMovement(_rayOrigin.GetPose(), backHitPose);
157 if (_movement != null)
158 {
159 _movementHitDelta = PoseUtils.Delta(_movement.Pose, hitPose);
160 }
161 }
162 base.InteractableSelected(interactable);
163 }

175 protected override void DoSelectUpdate()
176 {
177 RayInteractable interactable = _selectedInteractable;
178
179 if (_movement != null)
180 {
181 _movement.UpdateTarget(_rayOrigin.GetPose());
182 _movement.Tick();
183 Pose hitPoint = PoseUtils.Multiply(_movement.Pose, _movementHitDelta);
184 _movedHit.Point = hitPoint.position;
185 _movedHit.Normal = hitPoint.forward;
186 CollisionInfo = _movedHit;
187 End = _movedHit.Point;
188 return;
189 }
190
191 CollisionInfo = null;
192 if (interactable != null &&
193 interactable.Raycast(Ray, out SurfaceHit hit, MaxRayLength, true))
194 {
195 End = hit.Point;
196 CollisionInfo = hit;
197 }
198 else
199 {
200 End = Origin + MaxRayLength * Forward;
201 }
202 }
203
204 protected override Pose ComputePointerPose()
205 {
206 if (_movement != null)
207 {
208 return _movement.Pose;
209 }
210
211 if (CollisionInfo != null)
212 {
213 Vector3 position = CollisionInfo.Value.Point;
214 Quaternion rotation = Quaternion.LookRotation(CollisionInfo.Value.Normal);
215 return new Pose(position, rotation);
216 }
217 return new Pose(Vector3.zero, Quaternion.identity);
218 }
