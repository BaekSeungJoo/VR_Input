88 private void UpdateShapeAndStereo()
89 {
90 if (AutoDetectStereoLayout)
91 {
92 if (overlay.isExternalSurface)
93 {
94 int w = NativeVideoPlayer.VideoWidth;
95 int h = NativeVideoPlayer.VideoHeight;
96 switch (NativeVideoPlayer.VideoStereoMode)
97 {
98 case NativeVideoPlayer.StereoMode.Mono:
99 Stereo = VideoStereo.Mono;
100 break;
101 case NativeVideoPlayer.StereoMode.LeftRight:
102 Stereo = VideoStereo.LeftRight;
103 break;
104 case NativeVideoPlayer.StereoMode.TopBottom:
105 Stereo = VideoStereo.TopBottom;
106 break;
107 case NativeVideoPlayer.StereoMode.Unknown:
108 if (w > h)
109 {
110 Stereo = VideoStereo.LeftRight;
111 }
112 else
113 {
114 Stereo = VideoStereo.TopBottom;
115 }
116
117 break;
118 }
119 }
120 }
121
122 if (Shape != _LastShape || Stereo != _LastStereo || DisplayMono != _LastDisplayMono)
123 {
124 Rect destRect = new Rect(0, 0, 1, 1);
125 switch (Shape)
126 {
127 case VideoShape._360:
128 // set shape to Equirect
129 overlay.currentOverlayShape = OVROverlay.OverlayShape.Equirect;
130 break;
131 case VideoShape._180:
132 overlay.currentOverlayShape = OVROverlay.OverlayShape.Equirect;
133 destRect = new Rect(0.25f, 0, 0.5f, 1.0f);
134 break;
135 case VideoShape.Quad:
136 default:
137 overlay.currentOverlayShape = OVROverlay.OverlayShape.Quad;
138 break;
139 }
140
141 overlay.overrideTextureRectMatrix = true;
142 overlay.invertTextureRects = false;
143
144 Rect sourceLeft = new Rect(0, 0, 1, 1);
145 Rect sourceRight = new Rect(0, 0, 1, 1);
146 switch (Stereo)
147 {
148 case VideoStereo.LeftRight:
149 // set source matrices for left/right
150 sourceLeft = new Rect(0.0f, 0.0f, 0.5f, 1.0f);
151 sourceRight = new Rect(0.5f, 0.0f, 0.5f, 1.0f);
152 break;
153 case VideoStereo.TopBottom:
154 // set source matrices for top/bottom
155 sourceLeft = new Rect(0.0f, 0.5f, 1.0f, 0.5f);
156 sourceRight = new Rect(0.0f, 0.0f, 1.0f, 0.5f);
157 break;
158 case VideoStereo.BottomTop:
159 // set source matrices for top/bottom
160 sourceLeft = new Rect(0.0f, 0.0f, 1.0f, 0.5f);
161 sourceRight = new Rect(0.0f, 0.5f, 1.0f, 0.5f);
162 break;
163 }
164
165 overlay.SetSrcDestRects(sourceLeft, DisplayMono ? sourceLeft : sourceRight, destRect, destRect);
166
167 _LastDisplayMono = DisplayMono;
168 _LastStereo = Stereo;
169 _LastShape = Shape;
170 }
171 }

209 public void Play(string moviePath, string drmLicencesUrl)
210 {
211 if (moviePath != string.Empty)
212 {
213 Debug.Log("Playing Video: " + moviePath);
214 if (overlay.isExternalSurface)
215 {
216 OVROverlay.ExternalSurfaceObjectCreated surfaceCreatedCallback = () =>
217 {
218 Debug.Log("Playing ExoPlayer with SurfaceObject");
219 NativeVideoPlayer.PlayVideo(moviePath, drmLicencesUrl, overlay.externalSurfaceObject);
220 NativeVideoPlayer.SetLooping(LoopVideo);
221 };
222
223 if (overlay.externalSurfaceObject == IntPtr.Zero)
224 {
225 overlay.externalSurfaceObjectCreated = surfaceCreatedCallback;
226 }
227 else
228 {
229 surfaceCreatedCallback.Invoke();
230 }
231 }
232 else
233 {
234 Debug.Log("Playing Unity VideoPlayer");
235 videoPlayer.url = moviePath;
236 videoPlayer.Prepare();
237 videoPlayer.Play();
238 }
239
240 Debug.Log("MovieSample Start");
241 IsPlaying = true;
242 }
243 else
244 {
245 Debug.LogError("No media file name provided");
246 }
247 }

290 void Update()
291 {
292 UpdateShapeAndStereo();
293 if (!overlay.isExternalSurface)
294 {
295 var displayTexture = videoPlayer.texture != null ? videoPlayer.texture : Texture2D.blackTexture;
296 if (overlay.enabled)
297 {
298 if (overlay.textures[0] != displayTexture)
299 {
300 // OVROverlay won't check if the texture changed, so disable to clear old texture
301 overlay.enabled = false;
302 overlay.textures[0] = displayTexture;
303 overlay.enabled = true;
304 }
305 }
306 else
307 {
308 mediaRenderer.material.mainTexture = displayTexture;
309 mediaRenderer.material.SetVector("_SrcRectLeft", overlay.srcRectLeft.ToVector());
310 mediaRenderer.material.SetVector("_SrcRectRight", overlay.srcRectRight.ToVector());
311 }
312
313 IsPlaying = videoPlayer.isPlaying;
314 PlaybackPosition = (long)(videoPlayer.time * 1000L);
315
316#if UNITY_2019_1_OR_NEWER
317 Duration = (long)(videoPlayer.length * 1000L);
318#else
319 Duration = videoPlayer.frameRate > 0 ? (long)(videoPlayer.frameCount / videoPlayer.frameRate * 1000L) : 0L;
320#endif
321 }
322 else
323 {
324 NativeVideoPlayer.SetListenerRotation(Camera.main.transform.rotation);
325 IsPlaying = NativeVideoPlayer.IsPlaying;
326 PlaybackPosition = NativeVideoPlayer.PlaybackPosition;
327 Duration = NativeVideoPlayer.Duration;
328 if (IsPlaying && (int)OVRManager.display.displayFrequency != 60)
329 {
330 OVRManager.display.displayFrequency = 60.0f;
331 }
332 else if (!IsPlaying && (int)OVRManager.display.displayFrequency != 72)
333 {
334 OVRManager.display.displayFrequency = 72.0f;
335 }
336 }
