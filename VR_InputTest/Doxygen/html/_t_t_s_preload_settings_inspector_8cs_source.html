223 private bool LayoutVoiceData(TTSPreloadData preloadData, int voiceIndex, ref bool updated)
224 {
225 // Indent
226 EditorGUI.indentLevel++;
227
228 // Get data
229 TTSPreloadVoiceData voiceData = preloadData.voices[voiceIndex];
230 string voiceID = voiceData.presetVoiceID;
231 if (string.IsNullOrEmpty(voiceID))
232 {
233 voiceID = "No Voice Selected";
234 }
235 voiceID = $"{(voiceIndex+1)} - {voiceID}";
236
237 // Foldout
238 GUILayout.BeginHorizontal();
239 bool show = WitEditorUI.LayoutFoldout(new GUIContent(voiceID), voiceData);
240 if (!show)
241 {
242 GUILayout.EndHorizontal();
243 EditorGUI.indentLevel--;
244 return true;
245 }
246
247 // Delete
248 if (WitEditorUI.LayoutTextButton("Delete Voice"))
249 {
250 DeleteVoice(voiceIndex);
251 GUILayout.EndHorizontal();
252 EditorGUI.indentLevel--;
253 updated = true;
254 return false;
255 }
256
257 // Begin Voice Data
258 GUILayout.EndHorizontal();
259 EditorGUI.indentLevel++;
260
261 // Voice Text Field
262 if (TtsService == null || _ttsVoiceIDs == null || _ttsVoiceIDs.Count == 0)
263 {
264 WitEditorUI.LayoutTextField(new GUIContent("Voice ID"), ref voiceData.presetVoiceID, ref updated);
265 }
266 // Voice Preset Select
267 else
268 {
269 int presetIndex = _ttsVoiceIDs.IndexOf(voiceData.presetVoiceID);

271 WitEditorUI.LayoutPopup("Voice ID", _ttsVoiceIDs.ToArray(), ref presetIndex, ref presetUpdated);
272 if (presetUpdated)
273 {
274 voiceData.presetVoiceID = _ttsVoiceIDs[presetIndex];
275 string l = string.Empty;
276 TTSPreloadUtility.RefreshVoiceData(TtsService, voiceData, null, ref l);
277 updated = true;
278 }
279 }
280
281 // Ensure phrases exist
282 if (voiceData.phrases == null)
283 {
284 voiceData.phrases = new TTSPreloadPhraseData[] { };
285 }
286
287 // Phrase Foldout
288 EditorGUILayout.BeginHorizontal();
289 bool isLayout = WitEditorUI.LayoutFoldout(new GUIContent($"Phrases ({voiceData.phrases.Length})"),
290 voiceData.phrases);
291 if (WitEditorUI.LayoutTextButton("Add Phrase"))
292 {
293 TTSPreloadPhraseData lastPhrase = voiceData.phrases.Length == 0 ? null : voiceData.phrases[voiceData.phrases.Length - 1];
294 voiceData.phrases = AddArrayItem<TTSPreloadPhraseData>(voiceData.phrases, new TTSPreloadPhraseData()
295 {

314 }

322 // Layout phrase data
323 private bool LayoutPhraseData(TTSPreloadVoiceData voiceData, int phraseIndex, ref bool updated)
324 {
325 // Begin Phrase
326 EditorGUI.indentLevel++;
327
328 // Get data
329 TTSPreloadPhraseData phraseData = voiceData.phrases[phraseIndex];
330 string title = $"{(phraseIndex+1)} - {phraseData.textToSpeak}";
331
332 // Foldout
333 GUILayout.BeginHorizontal();
334 bool show = WitEditorUI.LayoutFoldout(new GUIContent(title), phraseData);
335 if (!show)
336 {
337 GUILayout.EndHorizontal();
338 EditorGUI.indentLevel--;
339 return true;
340 }
341
342 // Delete
343 if (WitEditorUI.LayoutTextButton("Delete Phrase"))
344 {
345 voiceData.phrases = DeleteArrayItem<TTSPreloadPhraseData>(voiceData.phrases, phraseIndex);
346 GUILayout.EndHorizontal();
347 EditorGUI.indentLevel--;
348 updated = true;
349 return false;
350 }
351
352 // Begin phrase Data
353 GUILayout.EndHorizontal();
354 EditorGUI.indentLevel++;
355
356 // Phrase
357 bool phraseChange = false;
358 WitEditorUI.LayoutTextField(new GUIContent("Phrase"), ref phraseData.textToSpeak, ref phraseChange);
359 if (phraseChange)
360 {
361 TTSPreloadUtility.RefreshPhraseData(TtsService, new TTSDiskCacheSettings()
362 {
363 DiskCacheLocation = TTSDiskCacheLocation.Preload
364 }, TtsService?.GetPresetVoiceSettings(voiceData.presetVoiceID), phraseData);
365 updated = true;
366 }
367
368 // Clip
369 string clipID = phraseData.clipID;
370 WitEditorUI.LayoutTextField(new GUIContent("Clip ID"), ref clipID, ref phraseChange);
371
372 // State
373 Color col = GUI.color;
374 Color stateColor = Color.green;
375 string stateValue = "Downloaded";
376 if (!phraseData.downloaded)
377 {
378 if (phraseData.downloadProgress <= 0f)
379 {
380 stateColor = Color.red;
381 stateValue = "Missing";
382 }
383 else
384 {
385 stateColor = Color.yellow;
386 stateValue = $"Downloading {(phraseData.downloadProgress * 100f):00.0}%";
387 }
388 }
389 GUI.color = stateColor;
390 WitEditorUI.LayoutKeyLabel("State", stateValue);
391 GUI.color = col;
392
393 // End Phrase
394 EditorGUILayout.Space();
395 EditorGUI.indentLevel--;
396 EditorGUI.indentLevel--;
397 return true;
398 }
399 // Add
400 private T[] AddArrayItem<T>(T[] array, T item) => EditArray<T>(array, (l) => l.Add(item));
401 // Delete
402 private T[] DeleteArrayItem<T>(T[] array, int index) => EditArray<T>(array, (l) => l.RemoveAt(index));
403 // Edit array
404 private T[] EditArray<T>(T[] array, Action<List<T>> edit)
405 {
406 // Generate list
407 List<T> list = new List<T>();
408
409 // Add array to list
410 if (array != null)
411 {
412 list.AddRange(array);
413 }
414
415 // Call edit action
416 edit(list);
417
418 // Set to array
419 T[] result = list.ToArray();
420
421 // Refresh foldout value
422 WitEditorUI.SetFoldoutValue(result, WitEditorUI.GetFoldoutValue(array));
423
424 // Return array
425 return result;
426 }

428 private void AddVoice()
429 {
430 List<TTSPreloadVoiceData> voices = new List<TTSPreloadVoiceData>();
431 if (Settings?.data?.voices != null)
432 {
433 voices.AddRange(Settings.data.voices);
434 }
435 voices.Add(new TTSPreloadVoiceData()
436 {
437 presetVoiceID = _ttsVoiceIDs == null || _ttsVoiceIDs.Count == 0 ? "" : _ttsVoiceIDs[0],
438 phrases = new TTSPreloadPhraseData[] { new TTSPreloadPhraseData() }
439 });
440 Settings.data.voices = voices.ToArray();
441 }
442 // Delete voice
443 private void DeleteVoice(int index)
444 {
445 // Invalid
446 if (Settings?.data?.voices == null || index < 0 || index >= Settings.data.voices.Length)
447 {
448 return;
449 }
450 // Cancelled
451 if (!EditorUtility.DisplayDialog("Delete Voice?",
452 $"Are you sure you would like to remove voice data:\n#{(index + 1)} - {Settings.data.voices[index].presetVoiceID}?",
453 "Okay", "Cancel"))
454 {
455 return;
456 }
457
458 // Remove
459 List<TTSPreloadVoiceData> voices = new List<TTSPreloadVoiceData>(Settings.data.voices);
460 voices.RemoveAt(index);
461 Settings.data.voices = voices.ToArray();
462 }
463 // Get voice ids
464 private List<string> GetVoiceIDs(TTSService service)
465 {
466 List<string> results = new List<string>();
467 if (service != null)
468 {
469 foreach (var voiceSetting in service.GetAllPresetVoiceSettings())
470 {
471 if (voiceSetting != null && !string.IsNullOrEmpty(voiceSetting.SettingsId) &&
472 !results.Contains(voiceSetting.SettingsId))
473 {
474 results.Add(voiceSetting.SettingsId);
475 }
476 }
477 }
478 return results;
479 }
