127 // To create a pressy button visual, we check each near poke interactor's
128 // depth against the base of the button and use the most pressed-in
129 // value as our depth. We cap this at the button base as the stopping
130 // point. If no interactors exist, we sit the button at the original offset
131
132 float closestDistance = _maxOffsetAlongNormal;
133 foreach (PokeInteractor pokeInteractor in _pokeInteractors)
134 {
135 // Scalar project the poke interactor's position onto the button base's normal vector
136 float pokeDistance =
137 Vector3.Dot(pokeInteractor.Origin - _buttonBaseTransform.position,
138 -1f * _buttonBaseTransform.forward);
139 pokeDistance -= pokeInteractor.Radius;
140 if (pokeDistance < 0f)
141 {
142 pokeDistance = 0f;
143 }
144
145 closestDistance = Math.Min(pokeDistance, closestDistance);
146 }
