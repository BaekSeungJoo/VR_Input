35 // Temporary structure used to pass data to and from native components
36 [StructLayout(LayoutKind.Sequential)]
37 public class HandData
38 {
39 private const int NumHandJoints = 24;
40 [MarshalAs(UnmanagedType.ByValArray, SizeConst = NumHandJoints * 7, ArraySubType = UnmanagedType.R4)]
41 private float[] jointValues;
42 private float _rootRotX;
43 private float _rootRotY;
44 private float _rootRotZ;
45 private float _rootRotW;
46 private float _rootPosX;
47 private float _rootPosY;
48 private float _rootPosZ;
49 private int _handedness;
50
51 public HandData()
52 {
53 jointValues = new float[NumHandJoints * 7];
54 }
55
56 public void SetData(IReadOnlyList<Pose> joints, Pose root, Handedness handedness)
57 {
58 Assert.AreEqual(NumHandJoints, joints.Count);
59 int jointValueIndex = 0;
60 for (int jointIndex = 0; jointIndex < NumHandJoints; jointIndex++)
61 {
62 Pose joint = joints[jointIndex];
63 jointValues[jointValueIndex++] = joint.rotation.x;
64 jointValues[jointValueIndex++] = joint.rotation.y;
65 jointValues[jointValueIndex++] = joint.rotation.z;
66 jointValues[jointValueIndex++] = joint.rotation.w;
67 jointValues[jointValueIndex++] = joint.position.x;
68 jointValues[jointValueIndex++] = joint.position.y;
69 jointValues[jointValueIndex++] = joint.position.z;
70 }
71 this._rootRotX = root.rotation.x;
72 this._rootRotY = root.rotation.y;
73 this._rootRotZ = root.rotation.z;
74 this._rootRotW = root.rotation.w;
75 this._rootPosX = root.position.x;
76 this._rootPosY = root.position.y;
77 this._rootPosZ = root.position.z;
78 this._handedness = (int)handedness;
79 }
80 }

147 public void Update(IHand hand)
148 {
149 if (!hand.GetRootPose(out Pose rootPose))
150 {
151 return;
152 }
153
154 if (!hand.GetJointPosesFromWrist(out ReadOnlyHandJointPoses poses))
155 {
156 return;
157 }
158
159 handData_.SetData(poses, rootPose, hand.Handedness);
160 ReturnValue rv = isdk_FingerPalmGrabAPI_UpdateHandData(GetHandle(), handData_);
161 Debug.Assert(rv != ReturnValue.Failure, "FingerPalmGrabAPI: isdk_FingerPalmGrabAPI_UpdateHandData failed");
162 }
