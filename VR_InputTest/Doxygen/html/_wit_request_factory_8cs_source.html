29 private static void HandleWitRequestOptions(WitRequestOptions requestOptions,
30 IDynamicEntitiesProvider[] additionalEntityProviders)
31 {
32 WitResponseClass entities = new WitResponseClass();
33 bool hasEntities = false;
34
35 if (null != additionalEntityProviders)
36 {
37 foreach (var provider in additionalEntityProviders)
38 {
39 foreach (var providerEntity in provider.GetDynamicEntities())
40 {
41 hasEntities = true;
42 MergeEntities(entities, providerEntity);
43 }
44 }
45 }
46
47 if (DynamicEntityKeywordRegistry.HasDynamicEntityRegistry)
48 {
49 foreach (var providerEntity in DynamicEntityKeywordRegistry.Instance.GetDynamicEntities())
50 {
51 hasEntities = true;
52 MergeEntities(entities, providerEntity);
53 }
54 }
55
56 if (null != requestOptions)
57 {
58 if (!string.IsNullOrEmpty(requestOptions.tag))
59 {
60 requestOptions.QueryParams["tag"] = requestOptions.tag;
61 }
62
63 if (null != requestOptions.dynamicEntities)
64 {
65 foreach (var entity in requestOptions.dynamicEntities.GetDynamicEntities())
66 {
67 hasEntities = true;
68 MergeEntities(entities, entity);
69 }
70 }
71 }
72
73 if (hasEntities)
74 {
75 requestOptions.QueryParams["entities"] = entities.ToString();
76 }
77 }

79 private static void MergeEntities(WitResponseClass entities, WitDynamicEntity providerEntity)
80 {
81 if (!entities.HasChild(providerEntity.entity))
82 {
83 entities[providerEntity.entity] = new WitResponseArray();
84 }
85 var mergedArray = entities[providerEntity.entity];
86 Dictionary<string, WitResponseClass> map = new Dictionary<string, WitResponseClass>();
87 HashSet<string> synonyms = new HashSet<string>();
88 var existingKeywords = mergedArray.AsArray;
89 for (int i = 0; i < existingKeywords.Count; i++)
90 {
91 var keyword = existingKeywords[i].AsObject;
92 var key = keyword["keyword"].Value;
93 if(!map.ContainsKey(key))
94 {
95 map[key] = keyword;
96 }
97 }
98 foreach (var keyword in providerEntity.keywords)
99 {
100 if (map.TryGetValue(keyword.keyword, out var keywordObject))
101 {
102 foreach (var synonym in keyword.synonyms)
103 {
104 keywordObject["synonyms"].Add(synonym);
105 }
106 }
107 else
108 {
109 keywordObject = JsonConvert.SerializeToken(keyword).AsObject;
110 map[keyword.keyword] = keywordObject;
111 mergedArray.Add(keywordObject);
112 }
113 }
114 }

116 private static WitRequestOptions GetSetupOptions(WitRequestOptions newOptions,
117 IDynamicEntitiesProvider[] additionalDynamicEntities)
118 {
119 // Generate options exist
120 WitRequestOptions options = newOptions != null ? newOptions : new WitRequestOptions();
121 // Set intents
122 if (-1 != options.nBestIntents)
123 {
124 options.QueryParams["n"] = options.nBestIntents.ToString();
125 }
126 // Set dynamic entities
127 HandleWitRequestOptions(options, additionalDynamicEntities);
128 // Set tag
129 if (!string.IsNullOrEmpty(options.tag))
130 {
131 options.QueryParams["tag"] = options.tag;
132 }
133 return options;
134 }
