39 public override void Init()
40 {
41 // Find base audio source if possible
42 if (AudioSource == null)
43 {
44 _audioSource = gameObject.GetComponentInChildren<AudioSource>();
45 }
46
47 // Duplicate audio source
48 if (CloneAudioSource)
49 {
50 // Create new audio source
51 AudioSource instance = new GameObject($"{gameObject.name}_AudioOneShot").AddComponent<AudioSource>();
52 instance.PreloadCopyData();
53
54 // Move into this transform & default to 3D audio
55 if (AudioSource == null)
56 {
57 instance.transform.SetParent(transform, false);
58 instance.spread = 1f;
59 }
60
61 // Move into audio source & copy source values
62 else
63 {
64 instance.transform.SetParent(AudioSource.transform, false);
65 instance.Copy(AudioSource);
66 }
67
68 // Reset instance's transform
69 instance.transform.localPosition = Vector3.zero;
70 instance.transform.localRotation = Quaternion.identity;
71 instance.transform.localScale = Vector3.one;
72
73 // Apply
74 _audioSource = instance;
75 }
76
77 // Setup audio source settings
78 AudioSource.playOnAwake = false;
