147 public virtual void UpdateCurrentCollisionsBasedOnDepth()
148 {
149 _currInteractableToCollisionInfos.Clear();
150 foreach (InteractableCollisionInfo interactableCollisionInfo in _currentIntersectingObjects)
151 {
152 var interactable = interactableCollisionInfo.InteractableCollider.ParentInteractable;
153 var depth = interactableCollisionInfo.CollisionDepth;
154 InteractableCollisionInfo collisionInfoFromMap = null;
155
156 if (!_currInteractableToCollisionInfos.TryGetValue(interactable, out collisionInfoFromMap))
157 {
158 _currInteractableToCollisionInfos[interactable] = interactableCollisionInfo;
159 }
160 else if (collisionInfoFromMap.CollisionDepth < depth)
161 {
162 collisionInfoFromMap.InteractableCollider = interactableCollisionInfo.InteractableCollider;
163 collisionInfoFromMap.CollisionDepth = depth;
164 }
165 }
166 }

172 public virtual void UpdateLatestCollisionData()
173 {
174 _addedInteractables.Clear();
175 _removedInteractables.Clear();
176 _remainingInteractables.Clear();
177
178 foreach (Interactable key in _currInteractableToCollisionInfos.Keys)
179 {
180 if (!_prevInteractableToCollisionInfos.ContainsKey(key))
181 {
182 _addedInteractables.Add(key);
183 }
184 else
185 {
186 _remainingInteractables.Add(key);
187 }
188 }
189
190 foreach (Interactable key in _prevInteractableToCollisionInfos.Keys)
191 {
192 if (!_currInteractableToCollisionInfos.ContainsKey(key))
193 {
194 _removedInteractables.Add(key);
195 }
196 }
197
198 // tell removed interactables that we are gone
199 foreach (Interactable removedInteractable in _removedInteractables)
200 {
201 removedInteractable.UpdateCollisionDepth(this,
202 _prevInteractableToCollisionInfos[removedInteractable].CollisionDepth,
203 InteractableCollisionDepth.None);
204 }
205
206 // tell added interactable what state we are now in
207 foreach (Interactable addedInteractableKey in _addedInteractables)
208 {
209 var addedInteractable = _currInteractableToCollisionInfos[addedInteractableKey];
210 var collisionDepth = addedInteractable.CollisionDepth;
211 addedInteractableKey.UpdateCollisionDepth(this, InteractableCollisionDepth.None,
212 collisionDepth);
213 }
214
215 // remaining interactables must be updated
216 foreach (Interactable remainingInteractableKey in _remainingInteractables)
217 {
218 var newDepth = _currInteractableToCollisionInfos[remainingInteractableKey].CollisionDepth;
219 var oldDepth = _prevInteractableToCollisionInfos[remainingInteractableKey].CollisionDepth;
220 remainingInteractableKey.UpdateCollisionDepth(this, oldDepth, newDepth);
221 }
222
223 _prevInteractableToCollisionInfos = new Dictionary<Interactable, InteractableCollisionInfo>(
224 _currInteractableToCollisionInfos);
225 }
226 }
