83 public bool GetPositionDelta(HandJointId joint, out Vector3 delta)
84 {
85 UpdateData();
86
87 PoseData prevPose = _poseDataCache[joint][PrevDataIndex];
88 PoseData curPose = _poseDataCache[joint][CurDataIndex];
89
90 if (!prevPose.IsValid || !curPose.IsValid)
91 {
92 delta = Vector3.zero;
93 return false;
94 }
95
96 delta = curPose.Pose.position - prevPose.Pose.position;
97 return true;
98 }
99
106 public bool GetRotationDelta(HandJointId joint, out Quaternion delta)
107 {
108 UpdateData();
109
110 PoseData prevPose = _poseDataCache[joint][PrevDataIndex];
111 PoseData curPose = _poseDataCache[joint][CurDataIndex];
112
113 if (!prevPose.IsValid || !curPose.IsValid)
114 {
115 delta = Quaternion.identity;
116 return false;
117 }
118
119 delta = curPose.Pose.rotation * Quaternion.Inverse(prevPose.Pose.rotation);
120 return true;
121 }
122
129 public bool GetPrevJointPose(HandJointId joint, out Pose pose)
130 {
131 UpdateData();
132
133 PoseData poseData = _poseDataCache[joint][PrevDataIndex];
134 pose = poseData.Pose;
135 return poseData.IsValid;
136 }

138 public void RegisterConfig(JointDeltaConfig config)
139 {
140 bool containsKeyAlready = _requestors.ContainsKey(config.InstanceID);
141
142 this.AssertIsTrue(!containsKeyAlready,
143 $"Trying to register multiple configs with the same id");
144
145 _requestors.Add(config.InstanceID, new List<HandJointId>(config.JointIDs));
146
147 // Check if any new joints added, if so then add to cache
148 foreach (var joint in config.JointIDs)
149 {
150 if (!_poseDataCache.ContainsKey(joint))
151 {
152 _poseDataCache.Add(joint, new PoseData[2]
153 { new PoseData(), new PoseData() });
154
155 // New joint tracked, so write current data
156 PoseData toWrite = _poseDataCache[joint][CurDataIndex];
157 toWrite.IsValid = Hand.GetJointPose(joint, out toWrite.Pose);
158 }
159 }
160 }
161
162 public void UnRegisterConfig(JointDeltaConfig config)
163 {
164 _requestors.Remove(config.InstanceID);
165 }

195 private void UpdateData()
196 {
197 if (Hand.CurrentDataVersion <= _lastUpdateDataVersion)
198 {
199 return;
200 }
201 _lastUpdateDataVersion = Hand.CurrentDataVersion;
202
203 // Swap read and write indices each data version
204 CurDataIndex = 1 - CurDataIndex;
205
206 // Only fetch pose data for currently tracked joints
207 _trackedJoints.Clear();
208 foreach (var key in _requestors.Keys)
209 {
210 IList<HandJointId> joints = _requestors[key];
211 _trackedJoints.UnionWithNonAlloc(joints);
212 }
213
214 // Fetch pose data for tracked joints, and
215 // invalidate data for untracked joints
216 foreach (var joint in _poseDataCache.Keys)
217 {
218 PoseData toWrite = _poseDataCache[joint][CurDataIndex];
219 toWrite.IsValid = _trackedJoints.Contains(joint) &&
220 Hand.GetJointPose(joint, out toWrite.Pose);
221 }
222 }
