327 public static void DrawCrosshair(Transform crosshair, bool isHand = true, Controller hand = Controller.RTouch)
328 {
329 Ray ray = default;
330 // 컨트롤러의 위치와 방향을 이용해 레이 제작
331 if (isHand)
332 {
333#if BUILD_PLATFORM_PC
334 ray = Camera.main.ScreenPointToRay(Input.mousePosition);
335#elif TARGET_DEVICE_OCULUS
336 if(hand == Controller.RTouch)
337 {
338 ray = new Ray(RHandPosition, RHandDirection);
339 } // if : 오른쪽 컨트롤러에서 레이를 쏘는 경우
340 else
341 {
342 ray = new Ray(LHandPosition, LHandDirection);
343 } // else : 왼쪽 컨트롤러에서 레이를 쏘는 경우
344
345#endif // BUILD_PLATFORM_PC
346 }
347 else
348 {
349 // 카메라를 기준으로 화면의 정중앙으로 레이를 제작
350 ray = new Ray(Camera.main.transform.position, Camera.main.transform.forward);
351 }
352
353 // 눈에 안보이는 Plane을 만든다.
354 Plane plane = new Plane(Vector3.up, 0);
355 float distance = 0;
356
357 // plane을 이용해 ray를 쏜다
358 if (plane.Raycast(ray, out distance))
359 {
360 // 레이의 GetPoint 함수를 이용해 충돌 지점의 위치를 가져온다.
361 crosshair.position = ray.GetPoint(distance);
362 crosshair.forward = -Camera.main.transform.forward;
363 // 크로스 헤어의 크기를 최소 기본 크기에서 거리에 따라 더 커지도록 한다.
364 crosshair.localScale = originScale * Mathf.Max(1.0f, distance);
365 } // if : 레이에 충돌한 지점이 있는 경우
366
367 else
368 {
369 crosshair.position = ray.origin + ray.direction * 100.0f;
370 crosshair.forward = -Camera.main.transform.forward;
371 distance = (crosshair.position - ray.origin).magnitude;
372 crosshair.localScale = originScale * Mathf.Max(1.0f, distance);
373 } // else : 레이에 충돌한 지점이 없는 경우
374 } // DrawCrosshair

411#if TARGET_DEVICE_OCULUS
419 private static IEnumerator VibrationCoroutine(float duration, float frequency, float amplitude, Controller hand)
420 {
421 float currentTime = 0f;
422
423 while (currentTime < duration)
424 {
425 currentTime += Time.deltaTime;
426 OVRInput.SetControllerVibration(frequency, amplitude, (OVRInput.Controller)hand);
427 yield return null;
428 } // loop : duration 시간 동안 반복하는 루프
429
430 // 진동 시간이 지나면 진동을 멈추는 로직
431 OVRInput.SetControllerVibration(0, 0, (OVRInput.Controller)hand);
432 } // VibrationCoroutine()
433#endif // TARGET_DEVICE_OCULUS
