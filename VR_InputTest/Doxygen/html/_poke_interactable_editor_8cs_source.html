27 [CustomEditor(typeof(PokeInteractable))]
28 public class PokeInteractableEditor : SimplifiedEditor
29 {
30 private PokeInteractable _interactable;
31
32 private SerializedProperty _surfaceProperty;
33
34 private static readonly float DRAW_RADIUS = 0.02f;
35
36 private void Awake()
37 {
38 _interactable = target as PokeInteractable;
39
40 _surfaceProperty = serializedObject.FindProperty("_surfacePatch");
41 }
42
43 public void OnSceneGUI()
44 {
45 Handles.color = EditorConstants.PRIMARY_COLOR;
46 ISurfacePatch surfacePatch = _surfaceProperty.objectReferenceValue as ISurfacePatch;
47
48 if (surfacePatch == null)
49 {
50 // Currently only supports visualizing planar surfaces
51 return;
52 }
53
54 Transform triggerPlaneTransform = surfacePatch.Transform;
55
56 if (triggerPlaneTransform == null)
57 {
58 return;
59 }
60
61 Vector3 touchPoint = triggerPlaneTransform.position - triggerPlaneTransform.forward * _interactable.EnterHoverNormal;
62 surfacePatch.ClosestSurfacePoint(touchPoint, out SurfaceHit hit);
63 Vector3 proximalPoint = hit.Point;
64
65 Handles.DrawSolidDisc(touchPoint, triggerPlaneTransform.forward, DRAW_RADIUS);
66
67#if UNITY_2020_2_OR_NEWER
68 Handles.DrawLine(touchPoint, proximalPoint, EditorConstants.LINE_THICKNESS);
69
70 Handles.DrawLine(proximalPoint - triggerPlaneTransform.right * DRAW_RADIUS,
71 proximalPoint + triggerPlaneTransform.right * DRAW_RADIUS, EditorConstants.LINE_THICKNESS);
72 Handles.DrawLine(proximalPoint - triggerPlaneTransform.up * DRAW_RADIUS,
73 proximalPoint + triggerPlaneTransform.up * DRAW_RADIUS, EditorConstants.LINE_THICKNESS);
74#else
75 Handles.DrawLine(touchPoint, proximalPoint);
76
77 Handles.DrawLine(proximalPoint - triggerPlaneTransform.right * DRAW_RADIUS,
78 proximalPoint + triggerPlaneTransform.right * DRAW_RADIUS);
79 Handles.DrawLine(proximalPoint - triggerPlaneTransform.up * DRAW_RADIUS,
80 proximalPoint + triggerPlaneTransform.up * DRAW_RADIUS);
81#endif
82 }
83 }
