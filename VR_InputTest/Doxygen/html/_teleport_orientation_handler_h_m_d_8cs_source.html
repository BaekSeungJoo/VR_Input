50 protected override void UpdateTeleportDestination()
51 {
52 // Only update the orientation during preteleport, or if configured to do updates during aim.
53 if (AimData.Destination.HasValue && (UpdateOrientationDuringAim ||
54 LocomotionTeleport.CurrentState == LocomotionTeleport.States.PreTeleport))
55 {
56 var t = LocomotionTeleport.LocomotionController.CameraRig.centerEyeAnchor;
57 var destination = AimData.Destination.GetValueOrDefault();
58
59 // create a plane that contains the destination, with the normal pointing to the HMD.
60 var plane = new Plane(Vector3.up, destination);
61
62 // find the point on the plane that the HMD is looking at.
63 float d;
64 bool hit = plane.Raycast(new Ray(t.position, t.forward), out d);
65 if (hit)
66 {
67 var target = t.position + t.forward * d;
68 var local = target - destination;
69 local.y = 0;
70 var distance = local.magnitude;
71 if (distance > AimDistanceThreshold)
72 {
73 local.Normalize();
74
75 // Some debug draw code to visualize what the math is doing.
76
77 //OVRDebugDraw.AddCross(target, 0.2f, 0.01f, Color.yellow, 0.1f);
78 //OVRDebugDraw.AddCross(destination + new Vector3(local.x, 0, local.z), 0.2f, 0.01f, Color.blue, 0.1f);
79
80 //OVRDebugDraw.AddLine(t.position + new Vector3(0, 0.1f, 0), target, 0.01f, Color.yellow, 1.0f);
81 //OVRDebugDraw.AddLine(target + new Vector3(0, 1f, 0), target - new Vector3(0, 1f, 0), 0.01f, Color.blue, 1.0f);
82
83 var rot = Quaternion.LookRotation(new Vector3(local.x, 0, local.z), Vector3.up);
84 _initialRotation = rot;
85
86 if (AimDistanceMaxRange > 0 && distance > AimDistanceMaxRange)
87 {
88 AimData.TargetValid = false;
89 }
90
91 LocomotionTeleport.OnUpdateTeleportDestination(AimData.TargetValid, AimData.Destination, rot,
92 GetLandingOrientation(OrientationMode, rot));
93 return;
94 }
95 }
96 }
97
98 LocomotionTeleport.OnUpdateTeleportDestination(AimData.TargetValid, AimData.Destination, _initialRotation,
99 GetLandingOrientation(OrientationMode, _initialRotation));
100 }
