21 // Maps the expansion status of references foldouts
22 private readonly Dictionary<string, bool> _referencesExpanded = new Dictionary<string, bool>();
23
24 // Use name value for title if possible
25 protected override string GetLocalizedText(SerializedProperty property, string key)
26 {
27 // Determine by ids
28 switch (key)
29 {
30 case LocalizedTitleKey:
31 string title = GetFieldStringValue(property, "name");
32 if (!string.IsNullOrEmpty(title))
33 {
34 return title;
35 }
36 break;
37 case "id":
38 return WitTexts.Texts.ConfigurationIntentsIdLabel;
39 case "entities":
40 return WitTexts.Texts.ConfigurationIntentsEntitiesLabel;
41 }
42
43 // Default to base
44 return base.GetLocalizedText(property, key);
45 }

47 // Layout entity override
48 protected override void LayoutPropertyField(FieldInfo subfield, SerializedProperty subfieldProperty, GUIContent labelContent, bool canEdit)
49 {
50 // Handle all the same except entities
51 if (canEdit || !string.Equals(subfield.Name, "entities"))
52 {
53 base.LayoutPropertyField(subfield, subfieldProperty, labelContent, canEdit);
54 return;
55 }
56
57 // Entity foldout
58 subfieldProperty.isExpanded = WitEditorUI.LayoutFoldout(labelContent, subfieldProperty.isExpanded);
59 if (!subfieldProperty.isExpanded)
60 {
61 return;
62 }
63
64 EditorGUI.indentLevel++;
65 if (subfieldProperty.arraySize == 0)
66 {
67 WitEditorUI.LayoutErrorLabel(WitTexts.Texts.ConfigurationEntitiesMissingLabel);
68 }
69 else
70 {
71 for (int i = 0; i < subfieldProperty.arraySize; i++)
72 {
73 SerializedProperty entityProp = subfieldProperty.GetArrayElementAtIndex(i);
74 string entityPropName = entityProp.FindPropertyRelative("name").stringValue;
75 WitEditorUI.LayoutLabel(entityPropName);
76 }
77 }
78 EditorGUI.indentLevel--;
79 }
80
81 protected override void OnGUIPostFields(Rect position, SerializedProperty property, GUIContent label)
82 {
83 var configuration = property.serializedObject.targetObject as WitConfiguration;
84 if (configuration == null || !configuration.useConduit)
85 {
86 return;
87 }
88
89 var assemblyWalker = WitConfigurationEditor.AssemblyWalker;
90 if (assemblyWalker == null)
91 {
92 return;
93 }
94
95 var manifest = ManifestLoader.LoadManifest(configuration.ManifestLocalPath);
96
97 var intentName = property.FindPropertyRelative("name")?.stringValue;
98
99 if (!manifest.ContainsAction(intentName))
100 {
101 return;
102 }
103
104 var contexts = manifest.GetInvocationContexts(intentName);
105
106 if (!_referencesExpanded.ContainsKey(intentName))
107 {
108 _referencesExpanded[intentName] = false;
109 }
110 _referencesExpanded[intentName] = WitEditorUI.LayoutFoldout(new GUIContent("References"), _referencesExpanded[intentName]);
111 if (!_referencesExpanded[intentName])
112 {
113 return;
114 }
115
116 EditorGUI.indentLevel++;
117 foreach (var context in contexts)
118 {
119 GUILayout.BeginHorizontal();
120 {
121 GUILayout.Space(EditorGUI.indentLevel * WitStyles.IndentationSpaces);
122 if (WitEditorUI.LayoutTextLink($"{context.Type.Name}::{context.MethodInfo.Name}()"))
123 {
124 assemblyWalker.GetSourceCode(context.Type, out var sourceCodeFile, out _);
125 UnityEditorInternal.InternalEditorUtility.OpenFileAtLineExternal(sourceCodeFile, 1);
126 }
127 }
128 GUILayout.EndHorizontal();
129 }
130 EditorGUI.indentLevel--;
131 }
132
133 // Determine if should layout field
134 protected override bool ShouldLayoutField(SerializedProperty property, FieldInfo subfield)
135 {
136 switch (subfield.Name)
137 {
138 case "name":
139 return false;
140 }
141 return base.ShouldLayoutField(property, subfield);
142 }
143 }
