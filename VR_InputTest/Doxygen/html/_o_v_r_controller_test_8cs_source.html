29 public class BoolMonitor
30 {
31 public delegate bool BoolGenerator();
32
33 private string m_name = "";
34 private BoolGenerator m_generator;
35 private bool m_prevValue = false;
36 private bool m_currentValue = false;
37 private bool m_currentValueRecentlyChanged = false;
38 private float m_displayTimeout = 0.0f;
39 private float m_displayTimer = 0.0f;
40
41 public BoolMonitor(string name, BoolGenerator generator, float displayTimeout = 0.5f)
42 {
43 m_name = name;
44 m_generator = generator;
45 m_displayTimeout = displayTimeout;
46 }
47
48 public void Update()
49 {
50 m_prevValue = m_currentValue;
51 m_currentValue = m_generator();
52
53 if (m_currentValue != m_prevValue)
54 {
55 m_currentValueRecentlyChanged = true;
56 m_displayTimer = m_displayTimeout;
57 }
58
59 if (m_displayTimer > 0.0f)
60 {
61 m_displayTimer -= Time.deltaTime;
62
63 if (m_displayTimer <= 0.0f)
64 {
65 m_currentValueRecentlyChanged = false;
66 m_displayTimer = 0.0f;
67 }
68 }
69 }
70
71 public void AppendToStringBuilder(ref StringBuilder sb)
72 {
73 sb.Append(m_name);
74
75 if (m_currentValue && m_currentValueRecentlyChanged)
76 sb.Append(": *True*\n");
77 else if (m_currentValue)
78 sb.Append(": True \n");
79 else if (!m_currentValue && m_currentValueRecentlyChanged)
80 sb.Append(": *False*\n");
81 else if (!m_currentValue)
82 sb.Append(": False \n");
83 }
84 }
