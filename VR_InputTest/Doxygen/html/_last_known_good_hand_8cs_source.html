23 public class LastKnownGoodHand : Hand
24 {
25 private readonly HandDataAsset _lastState = new HandDataAsset();
26
27 protected override void Apply(HandDataAsset data)
28 {
29 bool shouldUseData = data.IsHighConfidence ||
30 data.RootPoseOrigin == PoseOrigin.FilteredTrackedPose ||
31 data.RootPoseOrigin == PoseOrigin.SyntheticPose;
32 if (data.IsDataValid && data.IsTracked && shouldUseData)
33 {
34 _lastState.CopyFrom(data);
35 }
36 else if (_lastState.IsDataValid && data.IsConnected)
37 {
38 // No high confidence data, use last known good.
39 // Only copy pose data, not confidence/tracked flags.
40 data.CopyPosesFrom(_lastState);
41 data.RootPoseOrigin = PoseOrigin.SyntheticPose;
42 data.IsDataValid = true;
43 data.IsTracked = true;
44 data.IsHighConfidence = true;
45 }
46 else
47 {
48 // This hand is not connected, or has never seen valid data.
49 data.IsTracked = false;
50 data.IsHighConfidence = false;
51 data.RootPoseOrigin = PoseOrigin.None;
52 }
53 }
54 }
