191 #region Listen Wrapping
192 // Listeners
193 private HashSet<SpeechEvents> _listeners = new HashSet<SpeechEvents>();
194
195 // Adds all listener events
196 public void AddListener(SpeechEvents listener)
197 {
198 // Ignore if null or already set
199 if (listener == null || _listeners.Contains(listener))
200 {
201 return;
202 }
203
204 // Add all events
205 if (_listeners.Count == 0)
206 {
207 SetEvents(true);
208 }
209
210 // Add listener
211 _listeners.Add(listener);
212 }
213 // Removes all listener events
214 public void RemoveListener(SpeechEvents listener)
215 {
216 // Ignore if null or not already set
217 if (listener == null || !_listeners.Contains(listener))
218 {
219 return;
220 }
221
222 // Remove listener
223 _listeners.Remove(listener);
224
225 // Remove all events
226 if (_listeners.Count == 0)
227 {
228 SetEvents(false);
229 }
230 }

256 // Set UnityEvent with no parameter
257 protected void SetEvent(Func<SpeechEvents, UnityEvent> getEvent, bool add)
258 {
259 // Get source event
260 UnityEvent sourceEvent = getEvent(this);
261
262 // Add event
263 if (!add)
264 {
265 sourceEvent?.RemoveAllListeners();
266 return;
267 }
268
269 // Add listener
270 sourceEvent?.AddListener(() =>
271 {
272 foreach (var listener in _listeners)
273 {
274 getEvent(listener)?.Invoke();
275 }
276 });
277 }
