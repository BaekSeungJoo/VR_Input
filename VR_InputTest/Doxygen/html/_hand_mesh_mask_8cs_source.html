3/*
4 * This script creates a custom mesh, specifically for hand masking in the Passthrough SDK:
5 *
6 * 1. Created with 2D screen space in mind, since it's 2D triangles facing the camera.
7 * Not advised to use this mesh in any other way.
8 * 2. The look of it should be coupled with maskMaterial, which defines the falloff of the fade and
9 * also how it blends Passthrough.
10 * 3. The mesh has UV.x of 0 where the hand is fully opaque, and UV.x of 1 where faded out.
11 * This way, the gradient curve can be tuned in a user-friendly way (instead of in shader).
12 * There may be an optimization of avoiding a texture read, if the fade is all in shader (remapping UV.x)
13 * 4. Requires knowledge of the hand bones and indices
14 */
15

87 // do not edit any numbers below unless you like pain
88 // (it's EASY to break mesh creation and HARD to debug it)
89 // if you do, you need to become intimately aware of the OVRSkeleton.Bone numbers
90 void CreateHandMesh()
91 {
92 int knuckleVerts = 8 + (radialDivisions - 2) * 2;
93 int knuckleCount = 25; // five fingers, three knuckles per finger, then 10 more as palm borders
94
95 int palmVerts = 12;
96 int palmTriIndices = 66; // 22 triangles, each triangle has three verts
97
98 handVertices = new Vector3[knuckleVerts * knuckleCount + palmVerts];
99 handUVs = new Vector2[handVertices.Length];
100 handColors = new Color32[handVertices.Length];
101 handTriangles = new int[handVertices.Length * 3 + palmTriIndices];
102 // these counters are incremented during mesh construction, at each step, to ensure a valid mesh
103 vertCounter = 0;
104 triCounter = 0;
105
106 // make knuckle meshes for each finger
107 for (int i = 0; i < 5; i++)
108 {
109 int pinkyOffset = i < 4 ? 0 : 1; // pinky bone numbering is slightly different than other fingers
110 int baseId = 3 + i * 3 + pinkyOffset;
111 int tipId = 19 + i;
112 float k0taper = 1.0f;
113 float k1taper = 1.0f - fingerTaper;
114 float k2taper = 1.0f - (fingerTaper * 2);
115 float k3taper = 1.0f - (fingerTaper * 3);
116 // thumb also gets a bit wider thickness at the base
117 if (i == 0)
118 {
119 k0taper *= 1.2f;
120 k1taper *= 1.1f;
121 }
122
123 AddKnuckleMesh(knuckleVerts, k0taper, k1taper, referenceHand.Bones[baseId].Transform.position,
124 referenceHand.Bones[baseId + 1].Transform.position);
125 AddKnuckleMesh(knuckleVerts, k1taper, k2taper, referenceHand.Bones[baseId + 1].Transform.position,
126 referenceHand.Bones[baseId + 2].Transform.position);
127
128 // for the tip of the finger, the mask needs to be a bit different:
129 // the final joint of the skeleton's finger is at the tip, but
130 // we need the final joint to be somewhat inside the finger tip, so the radial mask matches
131 Vector3 lastKnucklePos = referenceHand.Bones[baseId + 2].Transform.position;
132 Vector3 tipPos = referenceHand.Bones[tipId].Transform.position;
133 Vector3 offsetTipPos = (tipPos - lastKnucklePos) * fingerTipLength + lastKnucklePos;
134 AddKnuckleMesh(knuckleVerts, k2taper, k3taper, lastKnucklePos, offsetTipPos);
135 }
136
137 // add palm mesh, which is very different than fingers
138 // it uses the same concept and parameters ("fade out from center")
139 AddPalmMesh(knuckleVerts);
140
141 // final step: combine all the arrays to make the mesh object
142 maskMesh.Clear();
143 maskMesh.name = "HandMeshMask";
144 maskMesh.vertices = handVertices;
145 maskMesh.uv = handUVs;
146 maskMesh.colors32 = handColors;
147 maskMesh.triangles = handTriangles;
148 }

150 // a "knuckle" is a camera-facing mesh from two bones
151 // the first two verts are at the bones, the rest border them like a 2D capsule
152 void AddKnuckleMesh(int knuckleVerts, float point1scale, float point2scale, Vector3 point1, Vector3 point2)
153 {
154 int baseVertId = vertCounter;
155
156 Vector3 camPos = Camera.main.transform.position;
157 Vector3 fwdVec = (point1 + point2) * 0.5f - camPos; // use the center of the two points
158 Vector3 upVec = point2 - point1;
159 Vector3.OrthoNormalize(ref fwdVec, ref upVec);
160 Vector3 rightVec = Vector3.Cross(upVec, fwdVec);
161
162 AddVertex(point2, Vector2.zero, Color.black);
163 AddVertex(point1, Vector2.zero, Color.black);
164
165 int fanVerts = radialDivisions + 1;
166
167 // rotate this vector counter clockwise, making verts along the way
168 Vector3 windingVec = rightVec;
169
170 for (int i = 0; i < fanVerts * 2; i++)
171 {
172 int basePoint = (i / fanVerts) + baseVertId;
173 Vector3 vertPos = handVertices[basePoint] + windingVec * borderSize * handScale *
174 (basePoint != baseVertId ? point1scale : point2scale);
175 AddVertex(vertPos, new Vector2(1, 0), Color.black);
176 if (i != radialDivisions) // after making the first fan, don't wind for one vert
177 {
178 windingVec = Quaternion.AngleAxis(180.0f / radialDivisions, fwdVec) * windingVec;
179 }
180 }
181
182 // after vertices have been made, assign their indices to create triangles
183 handTriangles[triCounter++] = baseVertId + 0;
184 handTriangles[triCounter++] = baseVertId + 1;
185 handTriangles[triCounter++] = baseVertId + radialDivisions + 3;
186 handTriangles[triCounter++] = baseVertId + 0;
187 handTriangles[triCounter++] = baseVertId + radialDivisions + 3;
188 handTriangles[triCounter++] = baseVertId + radialDivisions + 2;
189
190 handTriangles[triCounter++] = baseVertId + 2;
191 handTriangles[triCounter++] = baseVertId + knuckleVerts - 1;
192 handTriangles[triCounter++] = baseVertId + 0;
193 handTriangles[triCounter++] = baseVertId + 0;
194 handTriangles[triCounter++] = baseVertId + knuckleVerts - 1;
195 handTriangles[triCounter++] = baseVertId + 1;
196
197 for (int i = 0; i < radialDivisions; i++)
198 {
199 handTriangles[triCounter++] = baseVertId + 2 + i;
200 handTriangles[triCounter++] = baseVertId + 0;
201 handTriangles[triCounter++] = baseVertId + 3 + i;
202 }
203
204 for (int i = 0; i < radialDivisions; i++)
205 {
206 handTriangles[triCounter++] = baseVertId + fanVerts + 2 + i;
207 handTriangles[triCounter++] = baseVertId + 1;
208 handTriangles[triCounter++] = baseVertId + fanVerts + 3 + i;
209 }
210 }

377 void AddVertex(Vector3 position, Vector2 uv, Color color)
378 {
379 handVertices[vertCounter] = position;
380 // UV.x will be the distance from center
381 // this way, the shader could decide how best to remap: either with a gradient texture, or directly on UV.x
382 handUVs[vertCounter] = uv;

384 handColors[vertCounter] = color;
