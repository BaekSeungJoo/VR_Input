78 public static TActivate ActivateCategory<TCategory, TActivate>(GameObject target)
79 where TCategory : MonoBehaviour where TActivate : MonoBehaviour
80 {
81 var components = target.GetComponents<TCategory>();
82 Log("Activate " + typeof(TActivate) + " derived from " + typeof(TCategory) + "[" + components.Length + "]");
83 TActivate result = null;
84 for (int i = 0; i < components.Length; i++)
85 {
86 var c = (MonoBehaviour)components[i];
87 var active = c.GetType() == typeof(TActivate);
88 Log(c.GetType() + " is " + typeof(TActivate) + " = " + active);
89 if (active)
90 {
91 result = (TActivate)c;
92 }
93
94 if (c.enabled != active)
95 {
96 c.enabled = active;
97 }
98 }
99
100 return result;
101 }
