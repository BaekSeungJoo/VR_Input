283 protected virtual void OnAudioRequestCreated(VoiceServiceRequest audioRequest)
284 {
285 if (audioRequest == null)
286 {
287 return;
288 }
289 if (!audioRequest.IsActive)
290 {
291 HandleRequestResults(audioRequest);
292 return;
293 }
294 audioRequest.Events.OnPartialResponse.AddListener((response) => OnAudioPartialResponse(audioRequest));
295 audioRequest.Events.OnCancel.AddListener(HandleRequestResults);
296 audioRequest.Events.OnFailed.AddListener(HandleRequestResults);
297 audioRequest.Events.OnSuccess.AddListener(HandleRequestResults);
298 Requests.Add(audioRequest);
299 }
300 // Callback for early validation
301 protected virtual void OnAudioPartialResponse(VoiceServiceRequest audioRequest)
302 {
303 // Ignore unless can be validated
304 if (VoiceEvents.OnValidatePartialResponse == null || audioRequest == null || audioRequest.State != VoiceRequestState.Transmitting)
305 {
306 return;
307 }
308
309 // Create short response data
310 WitResponseNode response = audioRequest?.Results?.ResponseData;
311 VoiceSession validationData = GetVoiceSession(response);
312
313 // Call short response
314 VoiceEvents.OnValidatePartialResponse.Invoke(validationData);
315
316 // Invoke
317 if (UseConduit)
318 {
319 // Ignore without an intent
320 WitIntentData intent = response.GetFirstIntentData();
321 if (intent != null)
322 {
323 _conduitParameterProvider.PopulateParametersFromNode(response);
324 _conduitParameterProvider.AddParameter(ParameterProvider.VoiceSessionReservedName,
325 validationData);
326 _conduitParameterProvider.AddParameter(ParameterProvider.WitResponseNodeReservedName, response);
327 ConduitDispatcher.InvokeAction(_conduitParameterProvider, intent.name, _witConfiguration.relaxedResolution, intent.confidence, true);
328 }
329 }
330
331 // Deactivate & abort immediately but use the response data as results
332 if (validationData.validResponse)
333 {
334 VLog.D("Validated Early");
335 audioRequest.CompleteEarly();
336 }
337 }

452 private void HandleIntent(WitIntentData intent, WitResponseNode response)
453 {
454 if (UseConduit)
455 {
456 _conduitParameterProvider.PopulateParametersFromNode(response);
457 _conduitParameterProvider.AddParameter(ParameterProvider.WitResponseNodeReservedName, response);
458 ConduitDispatcher.InvokeAction(_conduitParameterProvider, intent.name,
459 _witConfiguration.relaxedResolution, intent.confidence, false);
460 }
461 else
462 {
463 var methods = MatchIntentRegistry.RegisteredMethods[intent.name];
464 foreach (var method in methods)
465 {
466 ExecuteRegisteredMatch(method, intent, response);
467 }
468 }
469 }
470
471
472
473 private void ExecuteRegisteredMatch(RegisteredMatchIntent registeredMethod,
474 WitIntentData intent, WitResponseNode response)
475 {
476 if (intent.confidence >= registeredMethod.matchIntent.MinConfidence &&
477 intent.confidence <= registeredMethod.matchIntent.MaxConfidence)
478 {
479 foreach (var obj in GetObjectsOfType(registeredMethod.type))
480 {
481 var parameters = registeredMethod.method.GetParameters();
482 if (parameters.Length == 0)
483 {
484 registeredMethod.method.Invoke(obj, Array.Empty<object>());
485 continue;
486 }
487 if (parameters[0].ParameterType != typeof(WitResponseNode) || parameters.Length > 2)
488 {
489 VLog.E("Match intent only supports methods with no parameters or with a WitResponseNode parameter. Enable Conduit or adjust the parameters");
490 continue;
491 }
492 if (parameters.Length == 1)
493 {
494 registeredMethod.method.Invoke(obj, new object[] {response});
495 }
496 }
497 }
