386 void UpdateResolutionEyeTexture()
387 {
388 OVRDisplay.EyeRenderDesc leftEyeDesc = OVRManager.display.GetEyeRenderDesc(UnityEngine.XR.XRNode.LeftEye);
389 OVRDisplay.EyeRenderDesc rightEyeDesc = OVRManager.display.GetEyeRenderDesc(UnityEngine.XR.XRNode.RightEye);
390
391 float scale = UnityEngine.XR.XRSettings.renderViewportScale;
392 float w = (int)(scale * (float)(leftEyeDesc.resolution.x + rightEyeDesc.resolution.x));
393 float h = (int)(scale * (float)Mathf.Max(leftEyeDesc.resolution.y, rightEyeDesc.resolution.y));
394
395 strResolutionEyeTexture = System.String.Format("Resolution : {0} x {1}", w, h);
396 }

401 void UpdateLatencyValues()
402 {
403#if !UNITY_ANDROID || UNITY_EDITOR
404 OVRDisplay.LatencyData latency = OVRManager.display.latency;
405 if (latency.render < 0.000001f && latency.timeWarp < 0.000001f && latency.postPresent < 0.000001f)
406 strLatencies = System.String.Format("Latency values are not available.");
407 else
408 strLatencies = System.String.Format(
409 "Render: {0:F3} TimeWarp: {1:F3} Post-Present: {2:F3}\nRender Error: {3:F3} TimeWarp Error: {4:F3}",
410 latency.render,
411 latency.timeWarp,
412 latency.postPresent,
413 latency.renderError,
414 latency.timeWarpError);
415#endif
416 }

421 void UpdateFPS()
422 {
423 timeLeft -= Time.unscaledDeltaTime;
424 accum += Time.unscaledDeltaTime;
425 ++frames;
426
427 // Interval ended - update GUI text and start new interval
428 if (timeLeft <= 0.0)
429 {
430 // display two fractional digits (f2 format)
431 float fps = frames / accum;
432
433 strFPS = System.String.Format("FPS: {0:F2}", fps);
434
435 timeLeft += updateInterval;
436 accum = 0.0f;
437 frames = 0;
438 }
