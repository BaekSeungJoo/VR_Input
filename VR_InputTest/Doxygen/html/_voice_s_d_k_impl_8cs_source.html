184 // Obtains a VoiceSDKImplRequest with specified parameters
185 private VoiceServiceRequest GetRequest(WitRequestOptions requestOptions,
186 VoiceServiceRequestEvents requestEvents,
187 NLPRequestInputType inputType,
188 bool audioImmediate = false)
189 {
190 VoiceSDKImplRequest request = new VoiceSDKImplRequest(service, inputType, audioImmediate, requestOptions, requestEvents);
191 Requests.Add(request);
192 request.Events.OnCancel.AddListener(OnRequestCanceled);
193 request.Events.OnFailed.AddListener(OnRequestFailed);
194 request.Events.OnSuccess.AddListener(OnRequestSuccess);
195 request.Events.OnComplete.AddListener(OnRequestComplete);
196 VoiceEvents?.OnRequestInitialized?.Invoke(request);
197 return request;
198 }

200 // Cancelation callback
201 private void OnRequestCanceled(VoiceServiceRequest request)
202 {
203 // Ignore if missing
204 if (!Requests.Contains(request))
205 {
206 return;
207 }
208
209 // Canceled
210 VLog.D($"Request Canceled\nReason: {request.Results.Message}");
211 VoiceEvents?.OnCanceled?.Invoke(request.Results.Message);
212 if (!string.Equals(request.Results.Message, WitConstants.CANCEL_MESSAGE_PRE_SEND))
213 {
214 VoiceEvents?.OnAborted?.Invoke();
215 }
216 }
217
218 // Failure callback
219 private void OnRequestFailed(VoiceServiceRequest request)
220 {
221 // Ignore if missing
222 if (!Requests.Contains(request))
223 {
224 return;
225 }
226
227 // Failed
228 VLog.D($"Request Failed\nError: {request.Results.Message}");
229 VoiceEvents?.OnError?.Invoke("HTTP Error " + request.Results.StatusCode, request.Results.Message);
230 VoiceEvents?.OnRequestCompleted?.Invoke();
231 }
232
233 // Success callback
234 private void OnRequestSuccess(VoiceServiceRequest request)
235 {
236 // Ignore if missing
237 if (!Requests.Contains(request))
238 {
239 return;
240 }
241
242 // Success
243 VLog.D("Request Success");
244 VoiceEvents?.OnResponse?.Invoke(request.Results.ResponseData);
245 VoiceEvents?.OnRequestCompleted?.Invoke();
246 }
247
248 // Request completed
249 private void OnRequestComplete(VoiceServiceRequest request)
250 {
251 // Ignore if missing
252 if (!Requests.Contains(request))
253 {
254 return;
255 }
256
257 // Remove listeners
258 request.Events.OnCancel.RemoveListener(OnRequestCanceled);
259 request.Events.OnFailed.RemoveListener(OnRequestFailed);
260 request.Events.OnSuccess.RemoveListener(OnRequestSuccess);
261 request.Events.OnComplete.RemoveListener(OnRequestComplete);
262
263 // Remove request
264 Requests.Remove(request);
265 VoiceEvents?.OnComplete?.Invoke(request);
266 }
