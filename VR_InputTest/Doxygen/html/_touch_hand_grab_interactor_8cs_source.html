213 protected override void DoHoverUpdate()
214 {
215 TouchHandGrabInteractable closestInteractable = _interactable;
216 if (closestInteractable == null) return;
217
218 TouchShadowHand.GrabTouchInfo output = new TouchShadowHand.GrabTouchInfo();
219
220 _touchShadowHand.GrabTouch(_fromShadow, _toShadow, closestInteractable.ColliderGroup,
221 false, output);
222 if (!output.grabbing)
223 {
224 _touchShadowHand.GrabTouch(_fromShadow, _toShadow, closestInteractable.ColliderGroup,
225 true, output);
226 }
227
228 if (!output.grabbing)
229 {
230 return;
231 }
232
233 _touchShadowHand.SetShadowRootFromHands(_fromShadow, _toShadow, output.grabT);
234 for (int i = 0; i < _fingerStatuses.Length; i++)
235 {
236 FingerStatus fingerStatus = _fingerStatuses[i];
237 ComputeNewTouching(i, _interactable.ColliderGroup, output.offset);
238
239 // We are overlapping at start, try pushout
240 if (output.grabbingFingers[i] && !_fingerStatuses[i].Locked)
241 {
242 _openShadow.FromHand(OpenHand, OpenHand.Handedness != Hand.Handedness);
243 if (!_touchShadowHand.PushoutFinger(i, _fromShadow, _openShadow,
244 _interactable.ColliderGroup, output.offset))
245 {
246 continue;
247 }
248
249 // Save the pre-touching locations
250 for (int j = 0; j < fingerStatus.Joints.Length; j++)
251 {
252 HandJointId jointId = fingerStatus.Joints[j];
253 _fromShadow.SetLocalPose(jointId, _touchShadowHand.ShadowHand.GetLocalPose(jointId));
254 }
255
256 ComputeNewTouching(i, _interactable.ColliderGroup, output.offset);
257 }
258 }
259
260 if (!HandStatusSelecting())
261 {
262 ClearFingerLockStatuses();
263 }
264 else
265 {
266 GrabOffset = Vector3.zero;
267 _saveOffset = Quaternion.Inverse(GrabRotation) * output.offset;
268 _firstSelect = true;
269 }
270
271 WhenFingerLocked();
272 }

293 // Given a touch hand root, conform fingers from _fromShadow to _toShadow until touching
294 private void ComputeNewTouching(int idx, ColliderGroup colliderGroup, Vector3 offset)
295 {
296 FingerStatus fingerStatus = _fingerStatuses[idx];
297
298 // Ignore locked
299 if (fingerStatus.Locked)
300 {
301 return;
302 }
303
304 _touchShadowHand.SetShadowFingerFrom(idx, _fromShadow);
305
306 // Check if finger is starting within
307 if (_touchShadowHand.CheckFingerTouch(idx, 0, colliderGroup, offset, null))
308 {
309 return;
310 }
311
312 if (!_touchShadowHand.GrabConformFinger(idx, _fromShadow, _toShadow, colliderGroup, offset))
313 {
314 return;
315 }
316
317 // We are touching, lock the finger
318 fingerStatus.Locked = true;
319 fingerStatus.Selecting = true;
320 fingerStatus.Timer = 0f;
321
322 // Save the locked joints
323 _touchShadowHand.GetJointsFromShadow(fingerStatus.Joints, fingerStatus.LocalJoints, true);
324
325 // Save the curl value of the conformed finger
326 Pose[] worldPoses = new Pose[fingerStatus.Joints.Length];
327 for (int i = 0; i < fingerStatus.Joints.Length; i++)
328 {
329 worldPoses[i] = _touchShadowHand.ShadowHand.GetWorldPose(fingerStatus.Joints[i]);
330 }
331 fingerStatus.CurlValueAtLock = FingerShapes.PosesListCurlValue(worldPoses);
332
333 // Save the touching locations
334 for (int i = 0; i < fingerStatus.Joints.Length; i++)
335 {
336 HandJointId jointId = fingerStatus.Joints[i];
337 _fromShadow.SetLocalPose(jointId, _touchShadowHand.ShadowHand.GetLocalPose(jointId));
338 }
339 }

341 private void ComputeNewRelease(int idx, ColliderGroup colliderGroup, Vector3 offset)
342 {
343 FingerStatus fingerStatus = _fingerStatuses[idx];
344 // Ignore unlocked
345 if (!fingerStatus.Locked)
346 {
347 return;
348 }
349
350 Pose[] worldPoses = new Pose[fingerStatus.Joints.Length];
351 for (int i = 0; i < fingerStatus.Joints.Length; i++)
352 {
353 worldPoses[i] = _toShadow.GetWorldPose(fingerStatus.Joints[i]);
354 }
355
356 // If not curling out, return
357 float toCurl = FingerShapes.PosesListCurlValue(worldPoses);
358 if (toCurl >= fingerStatus.CurlValueAtLock - _curlDeltaThreshold)
359 {
360 fingerStatus.Timer = 0f;
361 return;
362 }
363
364 // Check if finger releases
365 if (!_touchShadowHand.GrabReleaseFinger(idx, _fromShadow, _toShadow, colliderGroup, offset))
366 {
367 fingerStatus.Timer = 0f;
368 return;
369 }
370
371 fingerStatus.Timer += _deltaTime;
372 if (fingerStatus.Timer < _curlTimeThreshold)
373 {
374 return;
375 }
376
377 // If so, unlock
378 fingerStatus.Locked = false;
379 fingerStatus.Selecting = false;
380 }

382 protected override void DoSelectUpdate()
383 {
384 if (_firstSelect)
385 {
386 GrabOffset = _saveOffset;
387 _saveOffset = Vector3.zero;
388 _firstSelect = false;
389 return;
390 }
391
392 TouchHandGrabInteractable interactable = _selectedInteractable;
393 if (interactable == null)
394 {
395 for (int i = 0; i < _fingerStatuses.Length; i++)
396 {
397 FingerStatus fingerStatus = _fingerStatuses[i];
398 if (!fingerStatus.Locked)
399 {
400 continue;
401 }
402
403 fingerStatus.Selecting = true;
404 fingerStatus.Locked = false;
405 fingerStatus.Timer = 0f;
406
407 Pose[] worldPoses = new Pose[fingerStatus.Joints.Length];
408 for (int j = 0; j < fingerStatus.Joints.Length; j++)
409 {
410 worldPoses[j] = _toShadow.GetWorldPose(fingerStatus.Joints[j]);
411 }
412 fingerStatus.CurlValueAtLock = FingerShapes.PosesListCurlValue(worldPoses);
413 }
414
415 for (int i = 0; i < _fingerStatuses.Length; i++)
416 {
417 FingerStatus fingerStatus = _fingerStatuses[i];
418 if (!fingerStatus.Selecting)
419 {
420 continue;
421 }
422
423 Pose[] worldPoses = new Pose[fingerStatus.Joints.Length];
424 for (int j = 0; j < fingerStatus.Joints.Length; j++)
425 {
426 worldPoses[j] = _toShadow.GetWorldPose(fingerStatus.Joints[j]);
427 }
428
429 float curlValue = FingerShapes.PosesListCurlValue(worldPoses);
430 if (curlValue >= fingerStatus.CurlValueAtLock - _curlDeltaThreshold)
431 {
432 fingerStatus.Timer = 0f;
433 continue;
434 }
435
436 fingerStatus.Timer += _deltaTime;
437 if (fingerStatus.Timer < _curlTimeThreshold)
438 {
439 return;
440 }
441
442 fingerStatus.Selecting = false;
443 }
444
445 return;
446 }
447
448 _touchShadowHand.ShadowHand.Copy(_fromShadow);
449
450 _touchShadowHand.SetShadowRootFromHand(_fromShadow);
451 if (MeetsGrabPrerequisite())
452 {
453 for (int i = 0; i < _fingerStatuses.Length; i++)
454 {
455 if (_fingerStatuses[i].Locked)
456 {
457 ComputeNewRelease(i, interactable.ColliderGroup, Vector3.zero);
458 }
459 else
460 {
461 ComputeNewTouching(i, interactable.ColliderGroup, Vector3.zero);
462 }
463 }
464 }
465
466 WhenFingerLocked();

495 protected override TouchHandGrabInteractable ComputeCandidate()
496 {
497 TouchHandGrabInteractable closest = null;
498 float minSqrDist = float.MaxValue;
499 foreach (TouchHandGrabInteractable interactable in TouchHandGrabInteractable.Registry
500 .List())
501 {
502 foreach (Collider collider in interactable.ColliderGroup.Colliders)
503 {
504 Vector3 closestPoint = collider.ClosestPoint(_hoverLocation.position);
505 float sqrDist = (closestPoint - _hoverLocation.position).sqrMagnitude;
506 if (sqrDist < minSqrDist && sqrDist < _minHoverDistance * _minHoverDistance)
507 {
508 minSqrDist = sqrDist;
509 closest = interactable;
510 }
511 }
512 }
513
514 return closest;
515 }
