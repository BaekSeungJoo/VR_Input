61 private bool ShouldFireEvent(PointableCanvasEventArgs args)
62 {
63 if (args.Canvas != PointableCanvas.Canvas)
64 {
65 return false;
66 }
67 if (_suppressWhileDragging && args.Dragging)
68 {
69 return false;
70 }
71 return true;

74 private void PointableCanvasModule_WhenSelectableHoverEnter(PointableCanvasEventArgs args)
75 {
76 if (ShouldFireEvent(args))

78 _whenBeginHighlight.Invoke();
79 }
80 }
81
82 private void PointableCanvasModule_WhenSelectableHoverExit(PointableCanvasEventArgs args)
83 {
84 if (ShouldFireEvent(args))
85 {
86 _whenEndHighlight.Invoke();
87 }
88 }
89
90 private void PointableCanvasModule_WhenSelectableSelected(PointableCanvasEventArgs args)
91 {
92 if (ShouldFireEvent(args))
93 {
94 if (args.Hovered == null)
95 _whenSelectedEmpty.Invoke();
96 else
97 _whenSelectedHovered.Invoke();
98 }
99 }
100
101 private void PointableCanvasModule_WhenSelectableUnselected(PointableCanvasEventArgs args)
102 {
103 if (ShouldFireEvent(args))
104 {
105 if (args.Hovered == null)
106 _whenUnselectedEmpty.Invoke();
107 else
108 _whenUnselectedHovered.Invoke();
109 }
110 }
