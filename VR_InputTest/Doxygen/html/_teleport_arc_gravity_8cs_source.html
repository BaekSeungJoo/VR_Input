171 private Vector3 EvaluateGravityArc(Pose origin, float speed, float t)
172 {
173 Vector3 point = origin.position
174 + origin.forward * speed * t
175 + 0.5f * t * t * GRAVITY * _gravityModifier;
176 if (t >= 1f
177 && point.y > origin.position.y - GROUND_MARGIN)
178 {
179 point.y = origin.position.y - GROUND_MARGIN;
180 }
181 return point;
182 }

200 Vector3 delta = pose.position - _stabilizationPoint.position;

203 return _rangeCurve.Evaluate(distance);

206 private void StabilizeDirection(ref Pose pose)
207 {
208 Vector3 up = _stabilizationPoint.up;
209 Vector3 direction = (pose.position - _stabilizationPoint.position).normalized;
210 if (direction.sqrMagnitude == 0f)
211 {
212 direction = _stabilizationPoint.forward;
213 }
214 Quaternion stabilizedRotation = Quaternion.LookRotation(direction);
215
216 float mixing = Vector3.Dot(direction, up) * 0.5f + 0.5f;
217 mixing = _stabilizationMixCurve.Evaluate(mixing);
218 Quaternion mixedRotation = Quaternion.Lerp(pose.rotation, stabilizedRotation, mixing);
219 pose.rotation = mixedRotation;
220 }
221
222 private void RemapPitch(ref Pose pose)
223 {
224 Vector3 up = _stabilizationPoint.up;
225 Vector3 direction = pose.forward;
226 Vector3 flatDir = Vector3.ProjectOnPlane(direction, up).normalized;
227 Vector3 right = Vector3.Cross(flatDir, up).normalized;
228 float angle = Vector3.SignedAngle(flatDir, direction, right);
229 angle = _pitchCurve.Evaluate(angle);
230 Quaternion delta = Quaternion.AngleAxis(angle, right);
231
232 Vector3 dir = delta * flatDir;
233 if (dir.sqrMagnitude != 0)
234 {
235 pose.rotation = Quaternion.LookRotation(dir, pose.up);
236 }
237 }
