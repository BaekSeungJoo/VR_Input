24 public WitMultiSelectionPopupContent(IList<string> options, HashSet<string> disabledOptions, Action<IList<string>> callback)
25 {
26 this.options = new Dictionary<string, bool>();
27 this.callback = callback;
28
29 options.ToList().ForEach(optName => this.options[optName] = !disabledOptions.Contains(optName));
30 }
31
32 public override Vector2 GetWindowSize()
33 {
34 var lineHeight = 20;
35 var height = 25 + (options.Count() * lineHeight);
36 return new Vector2(WINDOW_WIDTH, height);
37 }
38
39 public override void OnGUI(Rect rect)
40 {
41 var initialLabelWidth = EditorGUIUtility.labelWidth;
42 EditorGUIUtility.labelWidth = WINDOW_WIDTH - 30;
43
44 GUILayout.Label("Assemblies to include in the generated Manifest", EditorStyles.boldLabel);
45 var keys = new List<string>(options.Keys);
46 foreach (var optName in keys) {
47 options[optName] = EditorGUILayout.Toggle(optName, options[optName], WitStyles.Toggle);
48 }
49
50 EditorGUIUtility.labelWidth = initialLabelWidth;
51 }
52
53 public override void OnClose()
54 {
55 var deselectedValues = this.options.Keys.Where(optName => false == this.options[optName]).ToList();
56 callback(deselectedValues);
57 }
58 }
