50 private List<HandJointMap> AutoAsignBones(SkinnedMeshRenderer skinnedHand)
51 {
52 List<HandJointMap> maps = new List<HandJointMap>();
53 Transform root = skinnedHand.rootBone;
54 Regex regEx = new Regex(@"Hand(\w*)(\d)");
55 foreach (var bone in FingersMetadata.HAND_JOINT_IDS)
56 {
57 Match match = regEx.Match(bone.ToString());
58 if (match != Match.Empty)
59 {
60 string boneName = match.Groups[1].Value.ToLower();
61 string boneNumber = match.Groups[2].Value;
62 Transform skinnedBone = RecursiveSearchForChildrenContainingPattern(root, "col", boneName, boneNumber);
63 if (skinnedBone != null)
64 {
65 maps.Add(new HandJointMap()
66 {
67 id = bone,
68 transform = skinnedBone,
69 rotationOffset = Vector3.zero
70 });
71 }
72 }
73 }
74 return maps;
75 }
76
77 private Transform RecursiveSearchForChildrenContainingPattern(Transform root, string ignorePattern, params string[] args)
78 {
79 if (root == null)
80 {
81 return null;
82 }
83
84 for (int i = 0; i < root.childCount; i++)
85 {
86 Transform child = root.GetChild(i);
87 string childName = child.name.ToLower();
88
89 bool shouldCheck = string.IsNullOrEmpty(ignorePattern)|| !childName.Contains(ignorePattern);
90 if (shouldCheck)
91 {
92 bool containsAllArgs = args.All(a => childName.Contains(a));
93 Transform result = containsAllArgs ? child
94 : RecursiveSearchForChildrenContainingPattern(child, ignorePattern, args);
95 if (result != null)
96 {
97 return result;
98 }
99 }
100 }
101 return null;
102 }

104 private static void SetPrivateValue(object instance, string fieldName, object value)
105 {
106 FieldInfo fieldData = GetPrivateField(instance, fieldName);
107 fieldData.SetValue(instance, value);

110 private static FieldInfo GetPrivateField(object instance, string fieldName)
111 {
112 Type type = instance.GetType();
113 FieldInfo fieldData = null;
114 while (type != null && fieldData == null)
115 {
116 fieldData = type.GetField(fieldName, BindingFlags.NonPublic | BindingFlags.Instance);
117 type = type.BaseType;
118 }
119 return fieldData;
120 }
