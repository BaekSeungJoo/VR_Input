128 private bool GetOverlayParameters(out OverlayShape shape,
129 out Vector3 position,
130 out Vector3 scale)
131 {
132 if (_canvasMesh is CanvasCylinder canvasCylinder)
133 {
134 shape = OverlayShape.Cylinder;
135 Vector2Int resolution = _canvasRenderTexture.GetBaseResolutionToUse();
136 position = new Vector3(0, 0, -canvasCylinder.Radius) - _runtimeOffset;
137 scale = new Vector3(_canvasRenderTexture.PixelsToUnits(resolution.x) /
138 canvasCylinder.transform.lossyScale.x,
139 _canvasRenderTexture.PixelsToUnits(resolution.y) /
140 canvasCylinder.transform.lossyScale.y,
141 canvasCylinder.Radius);
142 return true;
143 }
144 else if (_canvasMesh is CanvasRect canvasRect)
145 {
146 shape = OverlayShape.Quad;
147 Vector2Int resolution = _canvasRenderTexture.GetBaseResolutionToUse();
148 position = -_runtimeOffset;
149 scale = new Vector3(_canvasRenderTexture.PixelsToUnits(resolution.x),
150 _canvasRenderTexture.PixelsToUnits(resolution.y),
151 1);
152 return true;
153 }
154 else // Unsupported
155 {
156 shape = OverlayShape.Quad;
157 position = Vector3.zero;
158 scale = Vector3.zero;
159 return false;
160 }
161 }

172 protected void UpdateOverlay(Texture texture)
173 {
174 Profiler.BeginSample("InterfaceRenderer.UpdateOverlay");
175 try
176 {
177 if (!ShouldUseOVROverlay)
178 {
179 _overlay?.gameObject?.SetActive(false);
180 return;
181 }
182
183 if (_overlay == null)
184 {
185 GameObject overlayObj = CreateChildObject("__Overlay");
186 _overlay = overlayObj.AddComponent<OVROverlay>();
187 _overlay.isAlphaPremultiplied = !Application.isMobilePlatform;
188 }
189 else
190 {
191 _overlay.gameObject.SetActive(true);
192 }
193
194 if (!GetOverlayParameters(out OverlayShape shape,
195 out Vector3 pos,
196 out Vector3 scale))
197 {
198 _overlay.gameObject.SetActive(false);
199 return;
200 }
201
202 bool useUnderlayRendering = RenderingMode == OVRRenderingMode.Underlay;
203 _overlay.textures = new Texture[1] { texture };
204 _overlay.noDepthBufferTesting = useUnderlayRendering;
205 _overlay.currentOverlayType = useUnderlayRendering ? OverlayType.Underlay : OverlayType.Overlay;
206 _overlay.currentOverlayShape = shape;
207 _overlay.useExpensiveSuperSample = _enableSuperSampling;
208 _overlay.transform.localPosition = pos;
209 _overlay.transform.localScale = scale;
210 }
211 finally
212 {
213 Profiler.EndSample();
214 }
215 }
