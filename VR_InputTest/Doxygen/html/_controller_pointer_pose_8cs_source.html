28 public class ControllerPointerPose : MonoBehaviour, IActiveState

34 [SerializeField]

37 protected bool _started = false;

39 public bool Active { get; private set; }
40
41 protected virtual void Awake()
42 {
43 Controller = _controller as IController;

45
46 protected virtual void Start()
47 {

49 this.AssertField(Controller, nameof(Controller));

52
53 protected virtual void OnEnable()
54 {
55 if (_started)
56 {
57 Controller.WhenUpdated += HandleUpdated;
58 }
59 }

62 {
63 if (_started)
64 {
65 Controller.WhenUpdated -= HandleUpdated;
66 }
67 }
68
69 private void HandleUpdated()
70 {
71 IController controller = Controller;
72 if (controller.TryGetPointerPose(out Pose pose))
73 {
74 pose.position += pose.rotation * (Controller.Scale * _offset);
75 transform.SetPose(pose);
76 Active = true;
77 }
78 else
79 {
80 Active = false;
81 }
82 }

92 public void InjectOffset(Vector3 offset)

96
97 public void InjectAllHandPointerPose(IController controller,
