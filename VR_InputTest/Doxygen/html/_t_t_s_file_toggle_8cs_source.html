18namespace Meta.Voice.Samples.TTSVoices

20 public class TTSFileToggle : MonoBehaviour
21 {
22 // UI references
23 [SerializeField] private TTSWit _service;
24 [SerializeField] private Button _button;
25 [SerializeField] private Text _label;
26 [SerializeField] private string _labelFormat = "File Type: {0}";
27
28 // Current audio file type
29 private TTSWitAudioType _fileType = (TTSWitAudioType) (-1);
30 private List<TTSWitAudioType> _fileTypes = new List<TTSWitAudioType>(Enum.GetValues(typeof(TTSWitAudioType)) as TTSWitAudioType[]);
31
32 // Add listeners
33 private void OnEnable()
34 {
35 if (_service == null)
36 {
37 _service = GameObject.FindObjectOfType<TTSWit>();
38 }
39 _button.onClick.AddListener(Toggle);
40 }
41 // Check for changes
42 private void Update()
43 {
44 if (_fileType != _service.RequestSettings.audioType)
45 {
46 Refresh();
47 }
48 }
49 // Remove listeners
50 private void OnDisable()
51 {
52 _button.onClick.RemoveListener(Toggle);
53 }
54
55 // Toggle cache
56 private void Toggle()
57 {
58 // Increment to next
59 int index = _fileTypes.IndexOf(_fileType);
60 index++;
61 if (index >= _fileTypes.Count)
62 {
63 index = 0;
64 }
65
66 // Apply file type
67 _service.RequestSettings.audioType = _fileTypes[index];
68
69 // Clear runtime cache
70 _service.UnloadAll();
71
72 // Refresh
73 Refresh();
74 }
75
76 // Refresh location & button text
77 private void Refresh()
78 {
79 _fileType = _service.RequestSettings.audioType;
80 _label.text = string.Format(_labelFormat, _fileType.ToString());
81 }
82 }
83}
