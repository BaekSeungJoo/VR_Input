54 private void DrawClippers()
55 {
56 IEnumerable<ICylinderClipper> activeClippers =
57 ClippedCylinder.GetClippers()
58 .Where(c => c != null);
59
60 if (activeClippers.Count() > 0)
61 {
62 foreach (var clipper in activeClippers)
63 {
64 if (clipper.GetCylinderSegment(out CylinderSegment segment))
65 {
66 Handles.color = Color.gray;
67 SurfaceDrawing.DrawCylinderSegment(
68 ClippedCylinder.Cylinder, segment);
69 }
70 }
71 }
72 }
73
74 private void DrawClippedArea()

81 ClippedCylinder.Cylinder, clipped);
