107 public virtual bool CalculateBestPose(Pose userPose,
108 Handedness handedness, PoseMeasureParameters scoringModifier,
109 Transform relativeTo, ref HandGrabResult result)
110 {
111 result.HasHandPose = false;
112 if (HandPose != null
113 && HandPose.Handedness != handedness)
114 {
115 return false;
116 }
117
118 result.Score = CompareNearPoses(userPose,
119 scoringModifier, relativeTo, out Pose worldPose);
120 result.RelativePose = PoseUtils.Delta(relativeTo, worldPose);
121 if (HandPose != null)
122 {
123 result.HasHandPose = true;
124 result.HandPose.CopyFrom(HandPose);
125 }

139 private GrabPoseScore CompareNearPoses(in Pose worldPoint,

152 }
