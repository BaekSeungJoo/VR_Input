102 public bool TryGetMesh(NativeArray<Vector3> positions, NativeArray<int> indices)
103 {
104 if (!TryGetMeshRawUntransformed(positions, indices)) return false;
105
106 for (var i = 0; i < positions.Length; i++)
107 {
108 var p = positions[i];
109 // Necessary due to the coordinate space difference between OpenXR (right-handed) and Unity (left-handed)
110 positions[i] = new Vector3(-p.x, p.y, p.z);
111 }
112
113 var triangles = indices.Reinterpret<Triangle>(
114 expectedTypeSize: sizeof(int));
115 for (var i = 0; i < triangles.Length; i++)
116 {
117 var triangle = triangles[i];
118 triangles[i] = new Triangle
119 {
120 A = triangle.A,
121 B = triangle.C,
122 C = triangle.B
123 };
124 }
125
126 return true;
127 }

149 public JobHandle ScheduleGetMeshJob(NativeArray<Vector3> positions, NativeArray<int> indices,
