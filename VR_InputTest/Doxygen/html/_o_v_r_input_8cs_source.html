401 public static void Update()
402 {
403 connectedControllerTypes = Controller.None;
404 stepType = OVRPlugin.Step.Render;
405 fixedUpdateCount = 0;
406
407 if (OVRManager.loadedXRDevice == OVRManager.XRDevice.OpenVR)
408 {
409 UpdateXRControllerNodeIds();
410 UpdateXRControllerHaptics();
411 }
412
413 for (int i = 0; i < controllers.Count; i++)
414 {
415 OVRControllerBase controller = controllers[i];
416
417 connectedControllerTypes |= controller.Update();
418
419 if ((connectedControllerTypes & controller.controllerType) != 0)
420 {
421 RawButton rawButtonMask = RawButton.Any;
422 RawTouch rawTouchMask = RawTouch.Any;
423
424 if (Get(rawButtonMask, controller.controllerType)
425 || Get(rawTouchMask, controller.controllerType))
426 {
427 activeControllerType = controller.controllerType;
428 }
429 }
430 }
431
432 if ((activeControllerType == Controller.LTouch) || (activeControllerType == Controller.RTouch))
433 {
434 if ((connectedControllerTypes & Controller.Touch) == Controller.Touch)
435 {
436 // If either Touch controller is Active and both Touch controllers are connected, set both to Active.
437 activeControllerType = Controller.Touch;
438 }
439 }
440
441 if ((activeControllerType == Controller.LHand) || (activeControllerType == Controller.RHand))
442 {
443 if ((connectedControllerTypes & Controller.Hands) == Controller.Hands)
444 {
445 // If either Hand controller is Active and both Hand controllers are connected, set both to Active.
446 activeControllerType = Controller.Hands;
447 }
448 }
449
450 if ((connectedControllerTypes & activeControllerType) == 0)
451 {
452 activeControllerType = Controller.None;
453 }
454
455 bool UsePluginActiveAndConnectedControllers = (OVRManager.loadedXRDevice == OVRManager.XRDevice.Oculus && pluginSupportsActiveController);
456 if(OVRManager.instance != null && OVRManager.instance.IsSimultaneousHandsAndControllersSupported) {
457 UsePluginActiveAndConnectedControllers = OVRManager.loadedXRDevice == OVRManager.XRDevice.Oculus;
458 }
459
460 if (UsePluginActiveAndConnectedControllers)
461 {
462 Controller localActiveController = activeControllerType;
463
464 // override locally derived active and connected controllers if plugin provides more accurate data
465 connectedControllerTypes = (OVRInput.Controller)OVRPlugin.GetConnectedControllers();
466 activeControllerType = (OVRInput.Controller)OVRPlugin.GetActiveController();
467
468 // unless the plugin reports none and we locally detected hands as the active controller
469 if (activeControllerType == Controller.None && ((localActiveController & Controller.Hands) != 0))
470 {
471 activeControllerType = localActiveController;
472 }
473 }
474 else if (OVRManager.loadedXRDevice == OVRManager.XRDevice.OpenVR)
475 {
476 activeControllerType = connectedControllerTypes;
477 }
478 }
