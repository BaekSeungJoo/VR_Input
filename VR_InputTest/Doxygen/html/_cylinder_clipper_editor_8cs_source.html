30 [CustomEditor(typeof(CylinderClipper))]
31 public class CylinderClipperEditor : UnityEditor.Editor
32 {
33 private bool _visualize = false;
34
35 private IEnumerable<IRemoteDrawable> _drawables;
36
37 private CylinderClipper Clipper => target as CylinderClipper;
38
39 public override void OnInspectorGUI()
40 {
41 if (GUILayout.Button(_visualize ? "Hide Surface Visuals" : "Show Surface Visuals"))
42 {
43 _visualize = !_visualize;
44 _drawables = null;
45 SceneView.RepaintAll();
46 }
47
48 EditorGUILayout.Space();
49 base.OnInspectorGUI();
50 }
51
52 private void UpdateDrawables()
53 {
54 _drawables = SceneManager.GetActiveScene()
55 .GetRootGameObjects()
56 .Union(new[] { Clipper.transform.root.gameObject })
57 .SelectMany(root => root
58 .GetComponentsInChildren<IClippedSurface<ICylinderClipper>>(false))
59 .Where((s) => s.GetClippers().Contains(Clipper))
60 .Select(s => CreateEditor(s as Object) as IRemoteDrawable);
61 }
62
63 private void OnSceneGUI()
64 {
65 if (!Clipper.GetCylinderSegment(out CylinderSegment curvedPlane))
66 {
67 return;
68 }
69
70 if (_visualize)
71 {
72 if (_drawables == null)
73 {
74 UpdateDrawables();
75 }
76 foreach (var drawer in _drawables)
77 {
78 drawer.DrawRemote();
79 }
80 }
81 }
82 }
