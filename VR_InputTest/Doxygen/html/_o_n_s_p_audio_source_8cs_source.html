196 void Update()
197 {
198 // We might iterate through multiple sources / game object
199 var source = GetComponent<AudioSource>();
200
201 // READ-ONLY PARAMETER TEST
202#if TEST_READONLY_PARAMETERS
203 float rfl_enabled = 0.0f;
204 source.GetSpatializerFloat(readOnly_GlobalRelectionOn, out rfl_enabled);
205 float num_voices = 0.0f;
206 source.GetSpatializerFloat(readOnly_NumberOfUsedSpatializedVoices, out num_voices);
207
208 String readOnly = System.String.Format
209 ("Read only values: refl enabled: {0:F0} num voices: {1:F0}", rfl_enabled, num_voices);
210 Debug.Log(readOnly);
211#endif
212
213 // Check to see if we should disable spatializion
214 if ((Application.isPlaying == false) ||
215 (AudioListener.pause == true) ||
216 (source.isPlaying == false) ||
217 (source.isActiveAndEnabled == false)
218 )
219 {
220 source.spatialize = false;
221 return;
222 }
223 else
224 {
225 SetParameters(ref source);
226 }
227 }

229 enum Parameters : int
230 {
231 P_GAIN = 0,

240 P_SENDLEVEL,
241 P_NUM

248 public void SetParameters(ref AudioSource source)
249 {
250 // See if we should enable spatialization
251 source.spatialize = enableSpatialization;
252
253 source.SetSpatializerFloat((int)Parameters.P_GAIN, gain);
254 // All inputs are floats; convert bool to 0.0 and 1.0
255 if(useInvSqr == true)
256 source.SetSpatializerFloat((int)Parameters.P_USEINVSQR, 1.0f);
257 else
258 source.SetSpatializerFloat((int)Parameters.P_USEINVSQR, 0.0f);
259
260 source.SetSpatializerFloat((int)Parameters.P_NEAR, near);
261 source.SetSpatializerFloat((int)Parameters.P_FAR, far);
262
263 source.SetSpatializerFloat((int)Parameters.P_RADIUS, volumetricRadius);
264
265 if(enableRfl == true)
266 source.SetSpatializerFloat((int)Parameters.P_DISABLE_RFL, 0.0f);
267 else
268 source.SetSpatializerFloat((int)Parameters.P_DISABLE_RFL, 1.0f);
269
270 source.SetSpatializerFloat((int)Parameters.P_SENDLEVEL, reverbSend);
271 }

278 void OnDrawGizmos()
279 {
280 // Are we the first one created? make sure to set our static ONSPAudioSource
281 // for drawing out room parameters once
282 if(RoomReflectionGizmoAS == null)
283 {
284 RoomReflectionGizmoAS = this;
285 }
286
287 Color c;
288 const float colorSolidAlpha = 0.1f;
289
290 // Draw the near/far spheres
291
292 // Near (orange)
293 c.r = 1.0f;
294 c.g = 0.5f;
295 c.b = 0.0f;
296 c.a = 1.0f;
297 Gizmos.color = c;
298 Gizmos.DrawWireSphere(transform.position, Near);
299 c.a = colorSolidAlpha;
300 Gizmos.color = c;
301 Gizmos.DrawSphere(transform.position, Near);
302
303 // Far (red)
304 c.r = 1.0f;
305 c.g = 0.0f;
306 c.b = 0.0f;
307 c.a = 1.0f;
308 Gizmos.color = Color.red;
309 Gizmos.DrawWireSphere(transform.position, Far);
310 c.a = colorSolidAlpha;
311 Gizmos.color = c;
312 Gizmos.DrawSphere(transform.position, Far);
313
314 // VolumetricRadius (purple)
315 c.r = 1.0f;
316 c.g = 0.0f;
317 c.b = 1.0f;
318 c.a = 1.0f;
319 Gizmos.color = c;
320 Gizmos.DrawWireSphere(transform.position, VolumetricRadius);
321 c.a = colorSolidAlpha;
322 Gizmos.color = c;
323 Gizmos.DrawSphere(transform.position, VolumetricRadius);
324
325 // Draw room parameters ONCE only, provided reflection engine is on
326 if (RoomReflectionGizmoAS == this)
327 {
328 // Get global room parameters (write new C api to get reflection values)
329 bool reflOn = false;
330 bool reverbOn = false;
331 float width = 1.0f;
332 float height = 1.0f;
333 float length = 1.0f;
334
335 ONSP_GetGlobalRoomReflectionValues(ref reflOn, ref reverbOn, ref width, ref height, ref length);
336
337 // TO DO: Get the room reflection values and render those out as well (like we do in the VST)
338
339 if((Camera.main != null) && (reflOn == true))
340 {
341 // Set color of cube (cyan is early reflections only, white is with reverb on)
342 if(reverbOn == true)
343 c = Color.white;
344 else
345 c = Color.cyan;
346
347 Gizmos.color = c;
348 Gizmos.DrawWireCube(Camera.main.transform.position, new Vector3(width, height, length));
349 c.a = colorSolidAlpha;
350 Gizmos.color = c;
351 Gizmos.DrawCube(Camera.main.transform.position, new Vector3(width, height, length));
352 }
353 }
354 }
