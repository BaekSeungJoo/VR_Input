323 protected Vector3 NearestPointInSurface(Vector3 targetPosition, Transform relativeTo)
324 {
325 Vector3 start = GetStartPoint(relativeTo);
326 Vector3 dir = GetDirection(relativeTo);
327 Vector3 projectedVector = Vector3.Project(targetPosition - start, dir);
328 float height = GetHeight(relativeTo);
329 if (projectedVector.magnitude > height)
330 {
331 projectedVector = projectedVector.normalized * height;
332 }
333 if (Vector3.Dot(projectedVector, dir) < 0f)
334 {
335 projectedVector = Vector3.zero;
336 }
337
338 Vector3 projectedPoint = start + projectedVector;
339 Vector3 targetDirection = Vector3.ProjectOnPlane((targetPosition - projectedPoint), dir).normalized;
340
341 Vector3 startArcDir = GetStartArcDir(relativeTo);
342 float desiredAngle = Mathf.Repeat(Vector3.SignedAngle(startArcDir, targetDirection, dir), 360f);
343 if (desiredAngle > ArcLength)
344 {
345 if (Mathf.Abs(desiredAngle - ArcLength) >= Mathf.Abs(360f - desiredAngle))
346 {
347 targetDirection = startArcDir;
348 }
349 else
350 {
351 targetDirection = GetEndArcDir(relativeTo);
352 }
353 }
354 Vector3 surfacePoint = projectedPoint + targetDirection * GetRadius(relativeTo);
355 return surfacePoint;
356 }

358 public bool CalculateBestPoseAtSurface(Ray targetRay, out Pose bestPose,
359 Transform relativeTo)
360 {
361 Pose recordedPose = GetReferencePose(relativeTo);
362 Vector3 start = GetStartPoint(relativeTo);
363 Vector3 direction = GetDirection(relativeTo);
364 Vector3 lineToCylinder = start - targetRay.origin;
365
366 float perpendiculiarity = Vector3.Dot(targetRay.direction, direction);
367 float rayToLineDiff = Vector3.Dot(lineToCylinder, targetRay.direction);
368 float cylinderToLineDiff = Vector3.Dot(lineToCylinder, direction);
369
370 float determinant = 1f / (perpendiculiarity * perpendiculiarity - 1f);
371
372 float lineOffset = (perpendiculiarity * cylinderToLineDiff - rayToLineDiff) * determinant;
373 float cylinderOffset = (cylinderToLineDiff - perpendiculiarity * rayToLineDiff) * determinant;
374
375 float radius = GetRadius(relativeTo);
376 Vector3 pointInLine = targetRay.origin + targetRay.direction * lineOffset;
377 Vector3 pointInCylinder = start + direction * cylinderOffset;
378 float distanceToSurface = Mathf.Max(Vector3.Distance(pointInCylinder, pointInLine) - radius);
379 if (distanceToSurface < radius)
380 {
381 float adjustedDistance = Mathf.Sqrt(radius * radius - distanceToSurface * distanceToSurface);
382 pointInLine -= targetRay.direction * adjustedDistance;
383 }
384 Vector3 surfacePoint = NearestPointInSurface(pointInLine, relativeTo);
385 Pose desiredPose = new Pose(surfacePoint, recordedPose.rotation);
386 bestPose = MinimalTranslationPoseAtSurface(desiredPose, relativeTo);
387
388 return true;
389 }

391 protected Pose MinimalRotationPoseAtSurface(in Pose userPose, Transform relativeTo)
392 {
393 Pose referencePose = GetReferencePose(relativeTo);
394 Vector3 direction = GetDirection(relativeTo);
395 Quaternion rotation = GetRotation(relativeTo);
396 float radius = GetRadius(relativeTo);
397 Vector3 desiredPos = userPose.position;
398 Quaternion desiredRot = userPose.rotation;
399 Quaternion baseRot = referencePose.rotation;
400 Quaternion rotDif = (desiredRot) * Quaternion.Inverse(baseRot);
401 Vector3 desiredDirection = (rotDif * rotation) * Vector3.forward;
402 Vector3 projectedDirection = Vector3.ProjectOnPlane(desiredDirection, direction).normalized;
403 Vector3 altitudePoint = PointAltitude(desiredPos, relativeTo);
404 Vector3 surfacePoint = NearestPointInSurface(altitudePoint + projectedDirection * radius, relativeTo);
405 Quaternion surfaceRotation = CalculateRotationOffset(surfacePoint, relativeTo) * baseRot;
406 return new Pose(surfacePoint, surfaceRotation);
407 }
408
409 protected Pose MinimalTranslationPoseAtSurface(in Pose userPose, Transform relativeTo)
410 {
411 Pose referencePose = GetReferencePose(relativeTo);
412 Vector3 desiredPos = userPose.position;
413 Quaternion baseRot = referencePose.rotation;
414 Vector3 surfacePoint = NearestPointInSurface(desiredPos, relativeTo);
415 Quaternion surfaceRotation = CalculateRotationOffset(surfacePoint, relativeTo) * baseRot;
416
417 return new Pose(surfacePoint, surfaceRotation);
418 }
419
420 protected Quaternion CalculateRotationOffset(Vector3 surfacePoint, Transform relativeTo)
421 {
422 Vector3 start = GetStartPoint(relativeTo);
423 Vector3 direction = GetDirection(relativeTo);
424 Vector3 referenceDir = GetPerpendicularDir(relativeTo);
425 Vector3 recordedDirection = Vector3.ProjectOnPlane(referenceDir, direction);
426 Vector3 desiredDirection = Vector3.ProjectOnPlane(surfacePoint - start, direction);
427 return Quaternion.FromToRotation(recordedDirection, desiredDirection);
428 }
