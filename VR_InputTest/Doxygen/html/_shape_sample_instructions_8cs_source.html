79 private void UpdateStep()
80 {
81#if UNITY_EDITOR
82 var appVoiceExperience = FindObjectOfType<AppVoiceExperience>();
83 string[] guids = AssetDatabase.FindAssets("t:WitConfiguration");
84 if (guids.Length == 0)
85 {
86 currentStep = Step.SetupWit;
87 }
88 else if (!appVoiceExperience)
89 {
90 currentStep = Step.AddVoiceExperiences;
91 }
92 else if (!appVoiceExperience.RuntimeConfiguration.witConfiguration)
93 {
94 currentStep = Step.SetConfig;
95 appVoiceExperience.RuntimeConfiguration.witConfiguration =
96 AssetDatabase.LoadAssetAtPath<WitConfiguration>(
97 AssetDatabase.GUIDToAssetPath(guids[0]));
98 }
99 else if (!WitAuthUtility.IsServerTokenValid())
100 {
101 currentStep = Step.MissingServerToken;
102 }
103 else if (string.IsNullOrEmpty(appVoiceExperience.RuntimeConfiguration?.witConfiguration?
104 .GetClientAccessToken()))
105 {
106 currentStep = Step.MissingClientToken;
107 }
108 else
109 {
110 currentStep = Step.Ready;
111 }
112
113
114 instructionText.text = steps[(int) currentStep];
115#endif
116 }
