86 private IEnumerator AttachTriggerLogic()
87 {
88 while (!HandsManager.Instance || !HandsManager.Instance.IsInitialized())
89 {
90 yield return null;
91 }
92
93 OVRSkeleton handSkeleton = IsRightHandedTool
94 ? HandsManager.Instance.RightHandSkeleton
95 : HandsManager.Instance.LeftHandSkeleton;
96
97 OVRSkeleton.BoneId boneToTestCollisions = OVRSkeleton.BoneId.Hand_Pinky3;
98 switch (_fingerToFollow)
99 {
100 case OVRPlugin.HandFinger.Thumb:
101 boneToTestCollisions = OVRSkeleton.BoneId.Hand_Thumb3;
102 break;
103 case OVRPlugin.HandFinger.Index:
104 boneToTestCollisions = OVRSkeleton.BoneId.Hand_Index3;
105 break;
106 case OVRPlugin.HandFinger.Middle:
107 boneToTestCollisions = OVRSkeleton.BoneId.Hand_Middle3;
108 break;
109 case OVRPlugin.HandFinger.Ring:
110 boneToTestCollisions = OVRSkeleton.BoneId.Hand_Ring3;
111 break;
112 default:
113 boneToTestCollisions = OVRSkeleton.BoneId.Hand_Pinky3;
114 break;
115 }
116
117 List<BoneCapsuleTriggerLogic> boneCapsuleTriggerLogic = new List<BoneCapsuleTriggerLogic>();
118 List<OVRBoneCapsule> boneCapsules = HandsManager.GetCapsulesPerBone(handSkeleton, boneToTestCollisions);
119 foreach (var ovrCapsuleInfo in boneCapsules)
120 {
121 var boneCapsuleTrigger =
122 ovrCapsuleInfo.CapsuleRigidbody.gameObject.AddComponent<BoneCapsuleTriggerLogic>();
123 ovrCapsuleInfo.CapsuleCollider.isTrigger = true;
124 boneCapsuleTrigger.ToolTags = ToolTags;
125 boneCapsuleTriggerLogic.Add(boneCapsuleTrigger);
126 }
127
128 _boneCapsuleTriggerLogic = boneCapsuleTriggerLogic.ToArray();
129 // finger tip should have only one capsule
130 if (boneCapsules.Count > 0)
131 {
132 _capsuleToTrack = boneCapsules[0];
133 }
134
135 _isInitialized = true;

138 private void Update()
139 {
140 if (!HandsManager.Instance || !HandsManager.Instance.IsInitialized() || !_isInitialized ||
141 _capsuleToTrack == null)
142 {
143 return;
144 }
145
146 OVRHand hand = IsRightHandedTool ? HandsManager.Instance.RightHand : HandsManager.Instance.LeftHand;
147 float currentScale = hand.HandScale;
148 // push tool into the tip based on how wide it is. so negate the direction
149 Transform capsuleTransform = _capsuleToTrack.CapsuleCollider.transform;
150 // NOTE: use time settings 0.0111111/0.02 to make collisions work correctly!
151 Vector3 capsuleDirection = capsuleTransform.right;
152 Vector3 capsuleTipPosition = capsuleTransform.position + _capsuleToTrack.CapsuleCollider.height * 0.5f
153 * capsuleDirection;
154 Vector3 toolSphereRadiusOffsetFromTip = currentScale * _fingerTipPokeToolView.SphereRadius *
155 capsuleDirection;
156 // push tool back so that it's centered on transform/bone
157 Vector3 toolPosition = capsuleTipPosition + toolSphereRadiusOffsetFromTip;
158 transform.position = toolPosition;
159 transform.rotation = capsuleTransform.rotation;
160 InteractionPosition = capsuleTipPosition;
161
162 UpdateAverageVelocity();
163
164 CheckAndUpdateScale();
165 }

167 private void UpdateAverageVelocity()
168 {
169 var prevPosition = _position;
170 var currPosition = transform.position;
171 var currentVelocity = (currPosition - prevPosition) / Time.deltaTime;
172 _position = currPosition;
173 _velocityFrames[_currVelocityFrame] = currentVelocity;
174 // if sampled more than allowed, loop back toward the beginning
175 _currVelocityFrame = (_currVelocityFrame + 1) % NUM_VELOCITY_FRAMES;
176
177 Velocity = Vector3.zero;
178 // edge case; when we first start up, we will have only sampled less than the
179 // max frames. so only compute the average over that subset. After that, the
180 // frame samples will act like an array that loops back toward to the beginning
181 if (!_sampledMaxFramesAlready && _currVelocityFrame == NUM_VELOCITY_FRAMES - 1)
182 {
183 _sampledMaxFramesAlready = true;
184 }
185
186 int numFramesToSamples = _sampledMaxFramesAlready ? NUM_VELOCITY_FRAMES : _currVelocityFrame + 1;
187 for (int frameIndex = 0; frameIndex < numFramesToSamples; frameIndex++)
188 {
189 Velocity += _velocityFrames[frameIndex];
190 }
191
192 Velocity /= numFramesToSamples;
193 }

195 private void CheckAndUpdateScale()
196 {
197 float currentScale = IsRightHandedTool
198 ? HandsManager.Instance.RightHand.HandScale
199 : HandsManager.Instance.LeftHand.HandScale;
200 if (Mathf.Abs(currentScale - _lastScale) > Mathf.Epsilon)
201 {
202 transform.localScale = new Vector3(currentScale, currentScale, currentScale);
203 _lastScale = currentScale;
204 }
