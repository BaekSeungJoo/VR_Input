31 public static float GetIndexCurl(OVRInput.Controller ovrController)
32 {
33 float indexCurl;
34 if (SupportsAnalogIndex(ovrController))
35 {
36 indexCurl = OVRInput.Get(OVRInput.Axis1D.PrimaryIndexTriggerCurl, ovrController);
37 }
38 else
39 {
40 // Fallback to binary capsense
41 bool isPointing = !OVRInput.Get(OVRInput.NearTouch.PrimaryIndexTrigger, ovrController)
42 && OVRInput.Get(OVRInput.Axis1D.PrimaryIndexTrigger, ovrController) == 0f;
43
44 indexCurl = isPointing ? 0 : 1;
45 }
46 return indexCurl;
47 }
48
49 public static float GetIndexSlide(OVRInput.Controller ovrController)
50 {
51 float indexSlide;
52 if (SupportsAnalogIndex(ovrController))
53 {
54 indexSlide = OVRInput.Get(OVRInput.Axis1D.PrimaryIndexTriggerSlide, ovrController);
55 }
56 else
57 {
58 indexSlide = 0;
59 }
60 return indexSlide;
61 }

63 private static bool SupportsAnalogIndex(OVRInput.Controller ovrController)
64 {
65 bool isTouchController = (ovrController == OVRInput.Controller.LTouch) || (ovrController == OVRInput.Controller.RTouch);
66 if (!isTouchController)
67 {
68 return false;
69 }
70
71 OVRInput.Hand ovrHandedness = (ovrController == OVRInput.Controller.LTouch) ? OVRInput.Hand.HandLeft : OVRInput.Hand.HandRight;
72 OVRInput.InteractionProfile ovrProfile = OVRInput.GetCurrentInteractionProfile(ovrHandedness);
73 return ovrProfile == OVRInput.InteractionProfile.TouchPro;
74 }
