261 private void PerformLutTextureCheck(Texture2D texture)
262 {
263 if (texture != null)
264 {
265 if (!OVRPassthroughColorLut.IsTextureSupported(texture, out var message))
266 {
267 EditorGUILayout.HelpBox(message, MessageType.Error);
268 }
269
270 CheckLutImportSettings(texture);
271 }

274 private void CheckLutImportSettings(Texture lut)
275 {
276 if (lut != null)
277 {
278 var importer = AssetImporter.GetAtPath(AssetDatabase.GetAssetPath(lut)) as TextureImporter;
279
280 // Fails when using an internal texture as you can't change import settings on
281 // builtin resources, thus the check for null
282 if (importer != null)
283 {
284 bool isReadable = importer.isReadable == true;
285 bool isUncompressed = importer.textureCompression == TextureImporterCompression.Uncompressed;
286 bool valid = isReadable && isUncompressed;
287
288 if (!valid)
289 {
290 string warningMessage = ""
291 + (isReadable ? "" : "Texture is not readable. ")
292 + (isUncompressed ? "" : "Texture is compressed.");
293 DrawFixMeBox(warningMessage, () => SetLutImportSettings(importer));
294 }
295 }
296 }
297 }
298
299 private void SetLutImportSettings(TextureImporter importer)
300 {
301 importer.isReadable = true;
302 importer.textureCompression = TextureImporterCompression.Uncompressed;
303 importer.SaveAndReimport();
304 AssetDatabase.Refresh();
305 }
306
307 private void DrawFixMeBox(string text, Action action)
308 {
309 EditorGUILayout.HelpBox(text, MessageType.Warning);
310
311 GUILayout.Space(-32);
312 using (new EditorGUILayout.HorizontalScope())
313 {
314 GUILayout.FlexibleSpace();
315
316 if (GUILayout.Button("Fix", GUILayout.Width(60)))
317 action();
318
319 GUILayout.Space(8);
320 }
321
322 GUILayout.Space(11);
323 }
