108 public void UpdatePose()
109 {
110 _jointPosesLocal.Clear();
111 _jointPosesFromRoot.Clear();
112
113 foreach (var joint in Body.SkeletonMapping.Joints)
114 {
115 if (Body.GetJointPoseLocal(joint,
116 out Pose localPose))
117 {
118 _jointPosesLocal[joint] = localPose;
119 }
120 if (Body.GetJointPoseFromRoot(joint,
121 out Pose poseFromRoot))
122 {
123 _jointPosesFromRoot[joint] = poseFromRoot;
124 }
125 }
126
127 WhenBodyPoseUpdated.Invoke();
