357 internal async void OnApplicationPause(bool isPaused)
358 {
359 // if we haven't loaded scene, we won't check anchor status
360 if (isPaused || !_hasLoadedScene) return;
361
362 using (new OVRObjectPool.ListScope<OVRAnchor>(out var anchors))
363 {
364 var success = await OVRAnchor.FetchAnchorsAsync<OVRRoomLayout>(anchors);
365 if (!success)
366 {
367 Verbose?.Log(nameof(OVRSceneManager), "Failed to retrieve scene model information on resume.");
368 return;
369 }
370
371 // check whether room anchors have changed
372 foreach (var anchor in anchors)
373 {
374 if (!OVRSceneAnchor.SceneAnchors.ContainsKey(anchor.Uuid))
375 {
376 Verbose?.Log(nameof(OVRSceneManager),
377 $"Scene model changed. Invoking {nameof(NewSceneModelAvailable)} event.");
378 NewSceneModelAvailable?.Invoke();
379 break;
380 }
381 }

426 private void OnAnchorsFetchCompleted(bool success, List<OVRAnchor> roomLayoutAnchors)
427 {
428 if (success)
429 {
430 if (roomLayoutAnchors.Any())
431 {
432 InstantiateSceneRooms(roomLayoutAnchors);
433 }
434 else
435 {

437 "Loading the Scene definition yielded no result. "
438 + "Typically, this means the user has not captured the room they are in yet. "
439 + "Alternatively, an internal error may be preventing this app from accessing scene. "
440 + $"Invoking {nameof(NoSceneModelToLoad)}.");
441

443 }
444 }
445 OVRObjectPool.Return(roomLayoutAnchors);

448 private void InstantiateSceneRooms(List<OVRAnchor> roomLayoutAnchors)
449 {
450 _roomCounter = roomLayoutAnchors.Count;
451 foreach (var roomAnchor in roomLayoutAnchors)
452 {
453 // Check if anchor already exists
454 if (OVRSceneAnchor.SceneAnchors.TryGetValue(roomAnchor.Uuid, out var sceneAnchor))
455 {
456 sceneAnchor.IsTracked = true;
457 continue;
458 }
459
460 if (!(roomAnchor.TryGetComponent(out OVRRoomLayout roomLayoutComponent) &&
461 roomLayoutComponent.IsEnabled))
462 {
463 continue;
464 }
465
466 var roomGO = new GameObject("Room " + roomAnchor.Uuid);
467 roomGO.transform.parent = _initialAnchorParent;
468
469 sceneAnchor = roomGO.AddComponent<OVRSceneAnchor>();
470 sceneAnchor.Initialize(roomAnchor);
471
472 var sceneRoom = roomGO.AddComponent<OVRSceneRoom>();
473 sceneRoom.LoadRoom();
474 }
475 }

524 public OVRTask<bool> DoesRoomSetupExist(IEnumerable<string> requestedAnchorClassifications)
525 {
526 var task = OVRTask.FromGuid<bool>(Guid.NewGuid());

564 private static void CheckClassificationsInRooms(bool success, List<OVRAnchor> rooms, IEnumerable<string> requestedAnchorClassifications, OVRTask<bool> task)
565 {
566 if (!success)
567 {
568 Development.Log(nameof(OVRSceneManager),
569 $"{nameof(OVRAnchor.FetchAnchorsAsync)} failed on {nameof(DoesRoomSetupExist)}() request to fetch room anchors.");
570 return;
571 }
572
573 using (new OVRObjectPool.HashSetScope<Guid>(out var uuidsToQuery))
574 using (new OVRObjectPool.ListScope<Guid>(out var anchorUuids))
575 {
576 for (int i = 0; i < rooms.Count; i++)
577 {
578 GetUuidsToQuery(rooms[i], uuidsToQuery);
579 anchorUuids.AddRange(uuidsToQuery);
580 uuidsToQuery.Clear();
581 }
582
583 using (new OVRObjectPool.ListScope<OVRAnchor>(out var roomAnchors))
584 {
585 OVRAnchor.FetchAnchorsAsync(anchorUuids, roomAnchors)
586 .ContinueWith(result => CheckIfAnchorsContainClassifications(result, roomAnchors, requestedAnchorClassifications, task));
587 }
588 }
589 }

591 private static void CheckIfAnchorsContainClassifications(bool success, List<OVRAnchor> roomAnchors, IEnumerable<string> requestedAnchorClassifications, OVRTask<bool> task)
592 {
593 if (!success)
594 {
595 Development.Log(nameof(OVRSceneManager),
596 $"{nameof(OVRAnchor.FetchAnchorsAsync)} failed on {nameof(DoesRoomSetupExist)}() request to fetch anchors in rooms.");
597 return;
598 }
599
600 using (new OVRObjectPool.ListScope<string>(out var labels))
601 {
602 CollectLabelsFromAnchors(roomAnchors, labels);
603
604 foreach (var classification in requestedAnchorClassifications)
605 {
606 var labelIndex = labels.IndexOf(classification);
607 if (labelIndex >= 0)
608 {
609 labels.RemoveAt(labelIndex);
610 }
611 else
612 {
613 task.SetResult(false);
614 return;
615 }
616 }
617 }
618 task.SetResult(true);
619 }

621 private static void CollectLabelsFromAnchors(List<OVRAnchor> anchors, List<string> labels)
622 {
623 for (int i = 0; i < anchors.Count; i++)
624 {
625 var anchor = anchors[i];
626
627 if (anchor.TryGetComponent<OVRSemanticLabels>(out var classification))
628 {
629 labels.AddRange(classification.Labels.Split(OVRSemanticClassification.LabelSeparator));
630 }
631 }

646 {

669 var anchor = OVRSceneAnchor.SceneAnchorsList[_sceneAnchorUpdateIndex++];
670 anchor.TryUpdateTransform(false);
671 }

747 private void OVRManager_SceneCaptureComplete(UInt64 requestId, bool result)
748 {
749 if (requestId != _sceneCaptureRequestId)
750 {
751 Verbose?.LogWarning(nameof(OVRSceneManager),
752 $"Scene Room Setup with requestId: [{requestId}] was ignored, as it was not issued by this Scene Load request.");
753 return;
754 }
755
756 Development.Log(nameof(OVRSceneManager),
757 $"{nameof(OVRManager_SceneCaptureComplete)}() requestId: [{requestId}] result: [{result}]");
758
759 if (result)
760 {
761 // Either the user created a room, or they confirmed that the existing room is up to date. We can now load it.
762 Development.Log(nameof(OVRSceneManager),
763 $"The Room Setup returned without errors. Invoking {nameof(SceneCaptureReturnedWithoutError)}.");
764 SceneCaptureReturnedWithoutError?.Invoke();
765 }
766 else
767 {
768 Development.LogError(nameof(OVRSceneManager),
769 $"An error occurred when sending the user to the Room Setup. Invoking {nameof(UnexpectedErrorWithSceneCapture)}.");
770 UnexpectedErrorWithSceneCapture?.Invoke();
771 }

774 internal OVRSceneAnchor InstantiateSceneAnchor(OVRAnchor anchor, OVRSceneAnchor prefab)
775 {
776 var space = (OVRSpace)anchor.Handle;
777 var uuid = anchor.Uuid;
778
779 // Query for the semantic classification of the object
780 var hasSemanticLabels = OVRPlugin.GetSpaceSemanticLabels(space, out var labelString);
781 var labels = hasSemanticLabels
782 ? labelString.Split(',')
783 : Array.Empty<string>();
784
785 // Search the prefab override for a matching label, and if found override the prefab
786 if (PrefabOverrides.Count > 0)
787 {
788 foreach (var label in labels)
789 {
790 // Skip empty labels
791 if (string.IsNullOrEmpty(label)) continue;
792
793 // Search the prefab override for an entry matching the label
794 foreach (var @override in PrefabOverrides)
795 {
796 if (@override.ClassificationLabel == label)
797 {
798 prefab = @override.Prefab;
799 break;
800 }
801 }
802 }
803 }
804
805 // This can occur if neither the prefab nor any matching override prefab is set in the inspector
806 if (prefab == null)
807 {
808 Verbose?.Log(nameof(OVRSceneManager),
809 $"No prefab was provided for space: [{space}]"
810 + (labels.Length > 0 ? $" with semantic label {labels[0]}" : ""));
811 return null;
812 }
813
814 var sceneAnchor = Instantiate(prefab, Vector3.zero, Quaternion.identity, _initialAnchorParent);
815 sceneAnchor.gameObject.SetActive(true);
816 sceneAnchor.Initialize(anchor);
817
818 return sceneAnchor;
819 }
