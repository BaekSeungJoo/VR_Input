74 private void UpdateHandPose(IHandGrabState snapper)
75 {
76 HandGrabTarget snap = snapper.HandGrabTarget;
77
78 if (snap == null)
79 {
80 FreeFingers();
81 FreeWrist();
82 return;
83 }
84
85 if (snap.HandPose != null)
86 {
87 UpdateFingers(snap.HandPose, snapper.GrabbingFingers());
88 _areFingersFree = false;
89 }
90 else
91 {
92 FreeFingers();
93 }
94
95 Pose wristLocalPose = snapper.GetVisualWristPose();
96 Pose wristPose = Transformer != null
97 ? Transformer.ToTrackingPose(wristLocalPose)
98 : wristLocalPose;
99 _syntheticHand.LockWristPose(wristPose, 1f);
100 _isWristFree = false;
101 }
102
103 private void UpdateFingers(HandPose handPose, HandFingerFlags grabbingFingers)
104 {
105 Quaternion[] desiredRotations = handPose.JointRotations;
106 _syntheticHand.OverrideAllJoints(desiredRotations, 1f);
107
108 for (int fingerIndex = 0; fingerIndex < Constants.NUM_FINGERS; fingerIndex++)
109 {
110 int fingerFlag = 1 << fingerIndex;
111 JointFreedom fingerFreedom = handPose.FingersFreedom[fingerIndex];
112 if (fingerFreedom == JointFreedom.Constrained
113 && ((int)grabbingFingers & fingerFlag) != 0)
114 {
115 fingerFreedom = JointFreedom.Locked;
116 }
117 _syntheticHand.SetFingerFreedom((HandFinger)fingerIndex, fingerFreedom);
118 }
119 }

140 return false;
