111 protected override void InteractableUnselected(DistanceGrabInteractable interactable)
112 {
113 interactable.WhenPointerEventRaised -= HandleOtherPointerEventRaised;
114 _movement?.StopAndSetPose(_movement.Pose);
115 base.InteractableUnselected(interactable);
116 _movement = null;
117
118 ReleaseVelocityInformation throwVelocity = VelocityCalculator != null ?
119 VelocityCalculator.CalculateThrowVelocity(interactable.transform) :
120 new ReleaseVelocityInformation(Vector3.zero, Vector3.zero, Vector3.zero);
121 interactable.ApplyVelocities(throwVelocity.LinearVelocity, throwVelocity.AngularVelocity);
122 }

124 private void HandleOtherPointerEventRaised(PointerEvent evt)
125 {
126 if (SelectedInteractable == null)
127 {
128 return;
129 }
130
131 if (evt.Type == PointerEventType.Select || evt.Type == PointerEventType.Unselect)
132 {
133 Pose toPose = _grabTarget.GetPose();
134 if (SelectedInteractable.ResetGrabOnGrabsUpdated)
135 {
136 _movement = SelectedInteractable.GenerateMovement(toPose);
137 SelectedInteractable.PointableElement.ProcessPointerEvent(
138 new PointerEvent(Identifier, PointerEventType.Move, _movement.Pose, Data));
139 }
140 }
141
142 if (evt.Identifier == Identifier && evt.Type == PointerEventType.Cancel)
143 {
144 SelectedInteractable.WhenPointerEventRaised -= HandleOtherPointerEventRaised;
145 }
146 }
