44 public static bool TryCalculateBestGrab(this IHandGrabInteractor handGrabInteractor,
45 IHandGrabInteractable interactable, GrabTypeFlags grabTypes,
46 out HandGrabTarget.GrabAnchor anchorMode, ref HandGrabResult handGrabResult)
47 {
48 grabTypes = grabTypes & interactable.SupportedGrabTypes;
49 anchorMode = HandGrabInteractionPoses.GetBestAvailableAnchor(handGrabInteractor, grabTypes);
50 Pose handPose = HandGrabInteractionPoses.GetHandAlignmentPoint(handGrabInteractor, interactable.UsesHandPose, anchorMode);
51
52 if (interactable.CalculateBestPose(handPose,

168 public static float ComputeHandGrabScore(IHandGrabInteractor handGrabInteractor,
169 IHandGrabInteractable handGrabInteractable, out GrabTypeFlags handGrabTypes)
170 {
171 HandGrabAPI api = handGrabInteractor.HandGrabApi;
172 handGrabTypes = GrabTypeFlags.None;
173 float handGrabScore = 0f;
174
175 if (SupportsPinch(handGrabInteractor, handGrabInteractable))
176 {
177 float pinchStrength = api.GetHandPinchScore(handGrabInteractable.PinchGrabRules, false);
178 if (pinchStrength > handGrabScore)
179 {
180 handGrabScore = pinchStrength;
181 handGrabTypes = GrabTypeFlags.Pinch;
182 }
183 }
184
185 if (SupportsPalm(handGrabInteractor, handGrabInteractable))
186 {
187 float palmStrength = api.GetHandPalmScore(handGrabInteractable.PalmGrabRules, false);
188 if (palmStrength > handGrabScore)
189 {
190 handGrabScore = palmStrength;
191 handGrabTypes = GrabTypeFlags.Palm;
192 }
193 }
194
195 return handGrabScore;
196 }

198 public static bool ComputeShouldSelect(this IHandGrabInteractor handGrabInteractor,
199 IHandGrabInteractable handGrabInteractable, out GrabTypeFlags selectingGrabTypes)
200 {
201 if (handGrabInteractable == null)
202 {
203 selectingGrabTypes = GrabTypeFlags.None;
204 return false;
205 }
206
207 HandGrabAPI api = handGrabInteractor.HandGrabApi;
208 selectingGrabTypes = GrabTypeFlags.None;
209 if (SupportsPinch(handGrabInteractor, handGrabInteractable) &&
210 api.IsHandSelectPinchFingersChanged(handGrabInteractable.PinchGrabRules))
211 {
212 selectingGrabTypes |= GrabTypeFlags.Pinch;
213 }
214
215 if (SupportsPalm(handGrabInteractor, handGrabInteractable) &&
216 api.IsHandSelectPalmFingersChanged(handGrabInteractable.PalmGrabRules))
217 {
218 selectingGrabTypes |= GrabTypeFlags.Palm;
219 }
220
221 return selectingGrabTypes != GrabTypeFlags.None;
222 }

224 public static bool ComputeShouldUnselect(this IHandGrabInteractor handGrabInteractor,
225 IHandGrabInteractable handGrabInteractable)
226 {
227 HandGrabAPI api = handGrabInteractor.HandGrabApi;
228 HandFingerFlags pinchFingers = api.HandPinchGrabbingFingers();
229 HandFingerFlags palmFingers = api.HandPalmGrabbingFingers();
230
231 if (handGrabInteractable.SupportedGrabTypes == GrabTypeFlags.None)
232 {
233 if (!api.IsSustainingGrab(GrabbingRule.FullGrab, pinchFingers) &&
234 !api.IsSustainingGrab(GrabbingRule.FullGrab, palmFingers))
235 {
236 return true;
237 }
238 return false;
239 }
240
241 bool pinchHolding = false;
242 bool palmHolding = false;
243 bool pinchReleased = false;
244 bool palmReleased = false;
245
246 if (SupportsPinch(handGrabInteractor, handGrabInteractable.SupportedGrabTypes))
247 {
248 pinchHolding = api.IsSustainingGrab(handGrabInteractable.PinchGrabRules, pinchFingers);
249 if (api.IsHandUnselectPinchFingersChanged(handGrabInteractable.PinchGrabRules))
250 {
251 pinchReleased = true;
252 }
253 }
254
255 if (SupportsPalm(handGrabInteractor, handGrabInteractable.SupportedGrabTypes))
256 {
257 palmHolding = api.IsSustainingGrab(handGrabInteractable.PalmGrabRules, palmFingers);
258 if (api.IsHandUnselectPalmFingersChanged(handGrabInteractable.PalmGrabRules))
259 {
260 palmReleased = true;
261 }
262 }
263
264 return !pinchHolding && !palmHolding && (pinchReleased || palmReleased);
265 }

267 public static HandFingerFlags GrabbingFingers(this IHandGrabInteractor handGrabInteractor,
268 IHandGrabInteractable handGrabInteractable)
269 {
270 HandGrabAPI api = handGrabInteractor.HandGrabApi;
271 if (handGrabInteractable == null)
272 {
273 return HandFingerFlags.None;
274 }
275
276 HandFingerFlags fingers = HandFingerFlags.None;
277
278 if (SupportsPinch(handGrabInteractor, handGrabInteractable))
279 {
280 HandFingerFlags pinchingFingers = api.HandPinchGrabbingFingers();
281 handGrabInteractable.PinchGrabRules.StripIrrelevant(ref pinchingFingers);
282 fingers = fingers | pinchingFingers;
283 }
284
285 if (SupportsPalm(handGrabInteractor, handGrabInteractable))
286 {
287 HandFingerFlags grabbingFingers = api.HandPalmGrabbingFingers();
288 handGrabInteractable.PalmGrabRules.StripIrrelevant(ref grabbingFingers);
289 fingers = fingers | grabbingFingers;
290 }
291
292 return fingers;
293 }

324 GrabTypeFlags grabTypes)

327 && (grabTypes & GrabTypeFlags.Palm) != 0)
328 {
329 return HandGrabTarget.GrabAnchor.Palm;
