188 public class ReadOnlyBodyJointPoses : IReadOnlyList<Pose>
189 {
190 private Pose[] _poses;
191
192 public ReadOnlyBodyJointPoses(Pose[] poses)
193 {
194 _poses = poses;
195 }
196
197 public IEnumerator<Pose> GetEnumerator()
198 {
199 foreach (var pose in _poses)
200 {
201 yield return pose;
202 }
203 }
204
205 IEnumerator IEnumerable.GetEnumerator()
206 {
207 return GetEnumerator();
208 }
209
210 public static ReadOnlyBodyJointPoses Empty { get; } = new ReadOnlyBodyJointPoses(Array.Empty<Pose>());
211
212 public int Count => _poses.Length;
213
214 public Pose this[int index] => _poses[index];
215
216 public ref readonly Pose this[BodyJointId index] => ref _poses[(int)index];
217 }
