77 public void SetBodyPose(IBody body)
78 {
79 _jointData.Clear();
80 foreach (var joint in body.SkeletonMapping.Joints)
81 {
82 if (body.GetJointPoseLocal(joint, out Pose local) &&
83 body.GetJointPoseFromRoot(joint, out Pose fromRoot) &&
84 body.SkeletonMapping.TryGetParentJointId(joint, out BodyJointId parent))
85 {
86 _jointData.Add(new JointData()
87 {
88 JointId = joint,
89 ParentId = parent,
90 PoseFromRoot = fromRoot,
91 LocalPose = local,
92 });
93 }

117 _localPoses[_jointData[i].JointId] =
118 _jointData[i].LocalPose;
119 _posesFromRoot[_jointData[i].JointId] =
120 _jointData[i].PoseFromRoot;
121 _mapping.Joints.Add(

123 _mapping.JointToParent.Add(
