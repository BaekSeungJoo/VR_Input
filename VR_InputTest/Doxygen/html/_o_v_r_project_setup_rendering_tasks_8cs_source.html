48 // Call action for all UniversalRendererData being used, return true if all the return value of action is true
49 private static bool ForEachRendererData(Func<UniversalRendererData, bool> action)
50 {
51 var ret = true;
52 var pipelineAssets = new System.Collections.Generic.List<RenderPipelineAsset>();
53 QualitySettings.GetAllRenderPipelineAssetsForPlatform("Android", ref pipelineAssets);
54 foreach (var pipelineAsset in pipelineAssets)
55 {
56 var urpPipelineAsset = pipelineAsset as UniversalRenderPipelineAsset;
57 // If using URP pipeline
58 if (urpPipelineAsset)
59 {
60 var path = AssetDatabase.GetAssetPath(urpPipelineAsset);
61 var dependency = AssetDatabase.GetDependencies(path);
62 for (int i = 0; i < dependency.Length; i++)
63 {
64 // Try to read the dependency as UniversalRendererData
65 if (AssetDatabase.GetMainAssetTypeAtPath(dependency[i]) != typeof(UniversalRendererData))
66 continue;
67
68 UniversalRendererData renderData =
69 (UniversalRendererData)AssetDatabase.LoadAssetAtPath(dependency[i],
70 typeof(UniversalRendererData));
71 if (renderData)
72 {
73 ret = ret && action(renderData);
74 }
75
76 if (!ret)
77 {
78 break;
79 }
80 }
81 }
82 }
83
84 return ret;
85 }

88 private static GraphicsDeviceType[] GetGraphicsAPIs(BuildTargetGroup buildTargetGroup)
89 {
90 var buildTarget = buildTargetGroup.GetBuildTarget();
91 if (PlayerSettings.GetUseDefaultGraphicsAPIs(buildTarget))
92 {
93 return Array.Empty<GraphicsDeviceType>();
94 }
95
96 // Recommends OpenGL ES 3 or Vulkan
97 return PlayerSettings.GetGraphicsAPIs(buildTarget);
98 }

118#if USING_XR_SDK_OCULUS && OCULUS_XR_EYE_TRACKED_FOVEATED_RENDERING && UNITY_2021_3_OR_NEWER
119 //[Required] Use Vulkan and IL2CPP/ARM64 when using ETFR
120 OVRProjectSetup.AddTask(
121 level: OVRProjectSetup.TaskLevel.Required,
122 group: targetGroup,
123 isDone: buildTargetGroup =>
124 {
125 var useIL2CPP = PlayerSettings.GetScriptingBackend(buildTargetGroup) == ScriptingImplementation.IL2CPP;
126 var useARM64 = PlayerSettings.Android.targetArchitectures == AndroidArchitecture.ARM64;
127 var useVK = GetGraphicsAPIs(buildTargetGroup).Any(item => item == GraphicsDeviceType.Vulkan);
128 return useVK && useARM64 && useIL2CPP;
129 },
130 message: "Need to use Vulkan for Graphics APIs, IL2CPP for scripting backend, and ARM64 for target architectures when using eye-tracked foveated rendering",
131 fix: buildTargetGroup =>
132 {
133 var buildTarget = buildTargetGroup.GetBuildTarget();
134 PlayerSettings.Android.targetArchitectures = AndroidArchitecture.ARM64;
135 PlayerSettings.SetScriptingBackend(buildTargetGroup, ScriptingImplementation.IL2CPP);
136 PlayerSettings.SetGraphicsAPIs(buildTarget, new[] { GraphicsDeviceType.Vulkan });
137 },
138 fixMessage: "Set target architectures to ARM64, scripting backend to IL2CPP, and Graphics APIs to Vulkan for this build.",
139 conditionalValidity: buildTargetGroup => OculusSettings?.FoveatedRenderingMethod == Unity.XR.Oculus.OculusSettings.FoveationMethod.EyeTrackedFoveatedRendering
140 );
