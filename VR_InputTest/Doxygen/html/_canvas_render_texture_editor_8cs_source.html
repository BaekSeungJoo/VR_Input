98 protected void BeforeInspector()
99 {
100 bool isEmpty;

102 AutoFix(AutoFixIsUsingScreenSpaceCanvas(), AutoFixSetToWorldSpaceCamera,

126 {
127 if (needsFix)

142 private bool AutoFixIsUsingScreenSpaceCanvas()
143 {
144 Canvas canvas = target.GetComponent<Canvas>();
145 if (canvas == null)
146 {
147 return false;
148 }
149
150 return canvas.renderMode != RenderMode.WorldSpace;
151 }
152
153 private void AutoFixSetToWorldSpaceCamera()
154 {
155 Canvas canvas = target.GetComponent<Canvas>();
156 if (canvas != null)
157 {
158 Undo.RecordObject(canvas, "Set Canvas To World Space");
159 canvas.renderMode = RenderMode.WorldSpace;
160 }
161 }
162
163 private bool AutoFixIsMaskEmpty()
164 {
165 var layerProp = serializedObject.FindProperty(props.RenderLayers);
166 return layerProp.intValue == 0;
167 }
168
169 public void AutoFixAssignUIToMask()
170 {
171 Undo.RecordObject(target, "Set Overlay Mask");
172 var layerProp = serializedObject.FindProperty(props.RenderLayers);
173 layerProp.intValue = CanvasRenderTexture.DEFAULT_UI_LAYERMASK;
174 serializedObject.ApplyModifiedProperties();
175 }
176
177 private bool AutoFixAnyRenderersOnUnrenderedLayers()
178 {
179 var canvasProp = serializedObject.FindProperty(props.Canvas);
180 Canvas canvas = canvasProp.objectReferenceValue as Canvas;
181
182 if (canvas == null)
183 {
184 return false;
185 }
186
187 canvas.gameObject.GetComponentsInChildren(_tmpRenderers);
188 foreach (var renderer in _tmpRenderers)
189 {
190 int layer = renderer.gameObject.layer;
191 if (((1 << layer) & target.RenderingLayers) == 0)
192 {
193 return true;
194 }
195 }
196
197 return false;
198 }
199
200 private void AutoFixMoveRenderersToMaskedLayers()
201 {
202 var canvasProp = serializedObject.FindProperty(props.Canvas);
203 Canvas canvas = canvasProp.objectReferenceValue as Canvas;
204
205 if (canvas == null)
206 {
207 return;
208 }
209
210 var maskedLayers = AutoFixGetMaskedLayers();
211 var targetLayer = maskedLayers.FirstOrDefault();
212
213 canvas.gameObject.GetComponentsInChildren(_tmpRenderers);
214 foreach (var renderer in _tmpRenderers)
215 {
216 int layer = renderer.gameObject.layer;
217 if ((layer & target.RenderingLayers) == 0)
218 {
219 Undo.RecordObject(renderer.gameObject, "Set Overlay Layer");
220 renderer.gameObject.layer = targetLayer;
221 }
222 }
223 }
224
225 private bool AutoFixAnyCamerasRenderingTargetLayers()
226 {
227 var cameras = FindObjectsOfType<Camera>();
228 foreach (var camera in cameras)
229 {
230 //Ignore the special camera we create to render to the overlay
231 if (camera == target.OverlayCamera)
232 {
233 continue;
234 }
235
236 if ((camera.cullingMask & target.RenderingLayers) != 0)
237 {
238 return true;
239 }
240 }
241
242 return false;
243 }
244
245 private void AutoFixRemoveRenderingMaskFromCameras()
246 {
247 var cameras = FindObjectsOfType<Camera>();
248 foreach (var camera in cameras)
249 {
250 Undo.RecordObject(camera, "Set Camera Culling Mask");
251 camera.cullingMask = camera.cullingMask & ~target.RenderingLayers;
252 }
253 }
254
255 private List<int> AutoFixGetMaskedLayers()
256 {
257 List<int> maskedLayers = new List<int>();
258 for (int i = 0; i < 32; i++)
259 {
260 if ((target.RenderingLayers & (1 << i)) != 0)
261 {
262 maskedLayers.Add(i);
263 }
264 }
265 return maskedLayers;
266 }
267 #endregion
268 }
