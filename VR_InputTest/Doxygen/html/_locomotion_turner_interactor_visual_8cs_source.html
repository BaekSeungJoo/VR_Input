201 private void UpdatePose()
202 {
203 Pose origin = _turner.MidPoint;
204 Vector3 forward = origin.forward;
205 if (_lookAt != null)
206 {
207 forward = Vector3.ProjectOnPlane(origin.position - _lookAt.position, origin.up).normalized;
208 }
209 Vector3 position = origin.position - forward * _radius + origin.up * _verticalOffset;
210 Quaternion rotation = Quaternion.LookRotation(forward, origin.up);
211
212 _root.SetPositionAndRotation(position, rotation);
213 }
214
215 private void UpdateArrows()
216 {
217 float value = _turner.Value();
218 float angle = Mathf.Lerp(0f, _maxAngle, Mathf.Abs(value));
219 bool isLeft = value < 0;
220 bool follow = ShouldFollowArrow();
221 float squeeze = 0f;
222 if (Progress != null)
223 {
224 squeeze = Mathf.Lerp(0f, _squeezeLength, Progress.Value());
225 }
226
227 _rightRail.enabled = !isLeft;
228 _leftRail.enabled = isLeft;
229
230 angle = Mathf.Max(angle, _trailLength);
231
232 UpdateArrowPosition(isLeft ? _trailLength : angle + squeeze, _rightArrow.transform);
233 RotateTrail(follow && !isLeft ? angle - _trailLength : 0f, _rightTrail);
234 UpdateTrail(isLeft ? _trailLength : (follow ? _trailLength : angle) + squeeze, _rightTrail);
235
236 UpdateArrowPosition(!isLeft ? -_trailLength : -angle - squeeze, _leftArrow.transform);
237 RotateTrail(follow && isLeft ? -angle + _trailLength : 0f, _leftTrail);
238 UpdateTrail(!isLeft ? _trailLength : (follow ? _trailLength : angle) + squeeze, _leftTrail);
239
240 UpdateRail(angle, squeeze, isLeft ? _leftRail : _rightRail);
241 }

243 private void UpdateArrowPosition(float angle, Transform arrow)
244 {
245 Quaternion rotation = Quaternion.AngleAxis(angle, Vector3.up);
246 arrow.localPosition = rotation * Vector3.forward * _radius;
247 arrow.localRotation = rotation * _rotationCorrectionLeft;
248 }
249
250 private void RotateTrail(float angle, TubeRenderer trail)
251 {
252 trail.transform.localRotation = Quaternion.AngleAxis(angle, Vector3.up);
253 }
254
255 private void UpdateTrail(float angle, TubeRenderer trail)
256 {
257 float max = _maxAngle + _squeezeLength;
258 float segmentLenght = trail.TotalLength;
259 float start = -100;
260 float end = (max - angle - _margin) / max;
261
262 trail.StartFadeThresold = segmentLenght * start;
263 trail.EndFadeThresold = segmentLenght * end;
264 trail.InvertThreshold = false;
265 trail.RedrawFadeThresholds();
266 }
267
268 private void UpdateRail(float angle, float extra, TubeRenderer rail)
269 {
270 float segmentLenght = rail.TotalLength;
271 float start = (angle - _trailLength - _margin) / _maxAngle;
272 float end = (_maxAngle - angle - extra - _margin) / _maxAngle;
273
274 float gap = _railGap + rail.Feather;
275 rail.StartFadeThresold = segmentLenght * start - gap;
276 rail.EndFadeThresold = segmentLenght * end - gap;
277 rail.InvertThreshold = true;
278 rail.RedrawFadeThresholds();
279 }

281 private void UpdateColors()
282 {
283 bool isSelection = _turner.State == InteractorState.Select;
284 if (_turner.Value() > 0)
285 {
286 _leftMaterialBlock.MaterialPropertyBlock.SetColor(_colorShaderPropertyID, _disabledColor);
287 _rightMaterialBlock.MaterialPropertyBlock.SetColor(_colorShaderPropertyID, isSelection ? _highligtedColor : _enabledColor);
288 }
289 else
290 {
291 _leftMaterialBlock.MaterialPropertyBlock.SetColor(_colorShaderPropertyID, isSelection ? _highligtedColor : _enabledColor);
292 _rightMaterialBlock.MaterialPropertyBlock.SetColor(_colorShaderPropertyID, _disabledColor);
293 }
294 _leftMaterialBlock.UpdateMaterialPropertyBlock();
295 _rightMaterialBlock.UpdateMaterialPropertyBlock();
296 }

307 private TubePoint[] InitializeSegment(Vector2 minMax)
308 {
309 float lowLimit = minMax.x;
310 float upLimit = minMax.y;
311 int segments = Mathf.RoundToInt(Mathf.Repeat(upLimit - lowLimit, 360f) / _degreesPerSegment);
312 TubePoint[] tubePoints = new TubePoint[segments];
313 float segmentLenght = 1f / segments;
314 for (int i = 0; i < segments; i++)
315 {
316 Quaternion rotation = Quaternion.AngleAxis(-i * _degreesPerSegment - lowLimit, Vector3.up);
317 tubePoints[i] = new TubePoint()
318 {
319 position = rotation * Vector3.forward * _radius,
320 rotation = rotation * _rotationCorrectionLeft,
321 relativeLength = i * segmentLenght
322 };
323 }
324 return tubePoints;
