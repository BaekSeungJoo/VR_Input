39 private void InitializeEventCategories(Type eventsType)
40 {
41 // Get all category events in type & base type
42 Dictionary<string, List<string>> categoryLists = new Dictionary<string, List<string>>();
43 foreach (var field in eventsType.GetFields(FLAGS))
44 {
45 AddCustomField(field, categoryLists);
46 }
47 foreach (var baseField in eventsType.BaseType.GetFields(FLAGS))
48 {
49 AddCustomField(baseField, categoryLists);
50 }
51
52 // Apply
53 _eventCategories = new Dictionary<string, string[]>();
54 foreach (var category in categoryLists.Keys)
55 {
56 _eventCategories[category] = categoryLists[category].ToArray();
57 }
58 }
59 private void AddCustomField(FieldInfo field, Dictionary<string, List<string>> categoryLists)
60 {
61 if (!ShouldShowField(field))
62 {
63 return;
64 }
65 EventCategoryAttribute[] attributes = field.GetCustomAttributes(
66 typeof(EventCategoryAttribute), false) as EventCategoryAttribute[];
67 if (attributes == null || attributes.Length == 0)
68 {
69 return;
70 }
71 foreach (var eventCategory in attributes)
72 {
73 List<string> values = categoryLists.ContainsKey(eventCategory.Category) ? categoryLists[eventCategory.Category] : new List<string>();
74 string fieldName = GetDisplayFieldName(field);
75 if (!values.Contains(fieldName))
76 {
77 values.Add(fieldName);
78 }
79 categoryLists[eventCategory.Category] = values;
80 }
81 }
82 private bool ShouldShowField(FieldInfo field)
83 {
84 if (field.IsStatic)
85 {
86 return false;
87 }
88 if (!field.IsPublic && !Attribute.IsDefined(field, typeof(SerializeField)))
89 {
90 return false;
91 }
92 if (Attribute.IsDefined(field, typeof(HideInInspector)))
93 {
94 return false;
95 }
96 return Attribute.IsDefined(field, typeof(EventCategoryAttribute));
97 }
98 private string GetDisplayFieldName(FieldInfo field)
99 {
100 string result = field.Name.TrimStart('_');
101 return result[0].ToString().ToUpper() + result.Substring(1, result.Length - 1);
102 }
103 private string GetFieldNameFromDisplay(string fieldDisplayName)
104 {
105 return "_" + fieldDisplayName[0].ToString().ToLower() + fieldDisplayName.Substring(1, fieldDisplayName.Length - 1);
106 }

108 public override float GetPropertyHeight(SerializedProperty property, GUIContent label)
109 {
110 var eventObject = fieldInfo.GetValue(property.serializedObject.targetObject) as EventRegistry;
111
112 var lineHeight = EditorGUIUtility.singleLineHeight;
113 var lines = 1;
114 var height = 0;
115
116 // Allocate enough lines to display dropdown elements depending on which ones are showing.
117 if (showEvents && Selection.activeTransform)
118 lines++;
119
120 if (showEvents && selectedCategoryIndex != UNSELECTED)
121 lines++;
122
123 height = Mathf.RoundToInt(lineHeight * lines);
124
125 // By default, the property elements appear directly below the dropdowns.
126 propertyOffset = height + (int)WitStyles.TextButtonPadding;
127
128 // If the Events foldout is expanded and there are overridden properties, allocate space for them.
129 if (eventObject != null && eventObject.OverriddenCallbacks.Count != 0 && showEvents)
130 {
131 var callbacksArray = eventObject.OverriddenCallbacks.ToArray();
132
133 foreach (var callback in callbacksArray)
134 {
135 var fieldProperty = GetPropertyFromDisplayFieldName(property, callback);
136 if (fieldProperty != null)
137 {
138 height += Mathf.RoundToInt(EditorGUI.GetPropertyHeight(fieldProperty, true) + CONTROL_SPACING);
139 }
140 }
141
142 // Add some extra space so the last property field's +/- buttons don't overlap the next control.
143 height += PROPERTY_FIELD_SPACING;
144 }
145
146 return height;
147 }

149 public override void OnGUI(Rect position, SerializedProperty property, GUIContent label)
150 {
151 showEvents = EditorGUI.Foldout(new Rect(position.x, position.y, position.width, EditorGUIUtility.singleLineHeight), showEvents, "Events");
152
153 string url = DocumentationUrl;
154 if (!string.IsNullOrEmpty(url))
155 {
156 Texture texture = WitStyles.HelpIcon.image;
157 if (texture != null)
158 {
159 // Add a ? button
160 Vector2 textureSize = WitStyles.IconButton.CalcSize(WitStyles.HelpIcon);
161 Rect buttonRect = new Rect(position.x + position.width - textureSize.x, position.y, textureSize.x, textureSize.y);
162 if (GUI.Button(buttonRect,
163 new GUIContent(WitStyles.HelpIcon.image, DocumentationTooltip), WitStyles.IconButton))
164 {
165 Application.OpenURL(url);
166 }
167 // Add a tooltip
168 if (!string.IsNullOrEmpty(DocumentationTooltip))
169 {
170 GUI.Label(buttonRect, GUI.tooltip);
171 }
172 }
173 }
174
175 if (showEvents && Selection.activeTransform)
176 {
177 if (_eventCategories == null)
178 InitializeEventCategories(fieldInfo.FieldType);
179
180 var eventObject = fieldInfo.GetValue(property.serializedObject.targetObject) as EventRegistry;
181
182 var eventCategoriesKeyArray = _eventCategories.Keys.ToArray();
183
184 EditorGUI.indentLevel++;
185
186 // Shift the control rectangle down one line to accomodate the category dropdown.
187 position.y += EditorGUIUtility.singleLineHeight;
188 position.height = EditorGUIUtility.singleLineHeight;
189
190 selectedCategoryIndex = EditorGUI.Popup(position, "Event Category",
191 selectedCategoryIndex, eventCategoriesKeyArray);
192
193 if (selectedCategoryIndex != UNSELECTED)
194 {
195 var eventsArray = _eventCategories[eventCategoriesKeyArray[selectedCategoryIndex]];
196
197 if (selectedEventIndex >= eventsArray.Length)
198 selectedEventIndex = 0;
199
200 // Create a new rectangle to position the events dropdown and Add button.
201 var selectedEventDropdownPosition = new Rect(position);
202
203 selectedEventDropdownPosition.y += EditorGUIUtility.singleLineHeight + 2;
204 selectedEventDropdownPosition.width = position.width - (BUTTON_WIDTH + (int)WitStyles.TextButtonPadding);
205
206 selectedEventIndex = EditorGUI.Popup(selectedEventDropdownPosition, "Event", selectedEventIndex,
207 eventsArray);
208
209 var selectedEventButtonPosition = new Rect(selectedEventDropdownPosition);
210
211 selectedEventButtonPosition.width = BUTTON_WIDTH;
212 selectedEventButtonPosition.x =
213 selectedEventDropdownPosition.x + selectedEventDropdownPosition.width + CONTROL_SPACING;
214
215 if (GUI.Button(selectedEventButtonPosition, "Add"))
216 {
217 var eventName = _eventCategories[eventCategoriesKeyArray[selectedCategoryIndex]][
218 selectedEventIndex];
219
220 if (eventObject != null && selectedEventIndex != UNSELECTED &&
221 !eventObject.IsCallbackOverridden(eventName))
222 {
223 var fieldName = GetFieldNameFromDisplay(eventName);
224 if (eventObject.IsCallbackOverridden(fieldName))
225 {
226 eventObject.RemoveOverriddenCallback(fieldName);
227 }
228 fieldName = GetFieldNameFromDisplay(eventName).Substring(1);
229 if (eventObject.IsCallbackOverridden(fieldName))
230 {
231 eventObject.RemoveOverriddenCallback(fieldName);
232 }
233 eventObject.RegisterOverriddenCallback(eventName);
234 }
235 }
236 }
237
238 // If any overrides have been added to the property, allow them to be edited
239 if (eventObject != null && eventObject.OverriddenCallbacks.Count != 0)
240 {
241 var propertyRect = new Rect(position.x, position.y + propertyOffset, position.width, 0);
242
243 foreach (var callback in eventObject.OverriddenCallbacks)
244 {
245 var fieldProperty = GetPropertyFromDisplayFieldName(property, callback);
246 if (fieldProperty == null)
247 {
248 continue;
249 }
250
251 propertyRect.height = EditorGUI.GetPropertyHeight(fieldProperty, true);
252
253 EditorGUI.PropertyField(propertyRect, fieldProperty);
254
255 propertyRect.y += propertyRect.height + CONTROL_SPACING;
256 }
257 }
258
259 EditorGUI.indentLevel--;
260 }
261 }
262 private SerializedProperty GetPropertyFromDisplayFieldName(SerializedProperty property, string fieldName)
263 {
264 SerializedProperty result = property.FindPropertyRelative(fieldName);
265 if (result == null)
266 {
267 string fieldName2 = GetFieldNameFromDisplay(fieldName);
268 result = property.FindPropertyRelative(fieldName2);
269 if (result == null)
270 {
271 Debug.LogError($"Could not find serialized property field: {fieldName} ({fieldName2})");
272 }
273 }
274 return result;
275 }
