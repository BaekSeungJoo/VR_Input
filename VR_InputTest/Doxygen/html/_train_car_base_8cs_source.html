27 public abstract class TrainCarBase : MonoBehaviour
28 {
29 private static Vector3 OFFSET = new Vector3(0f, 0.0195f, 0f);
30 private const float WHEEL_RADIUS = 0.027f;
31 private const float TWO_PI = Mathf.PI * 2.0f;
32
33 [SerializeField] protected Transform _frontWheels = null;
34 [SerializeField] protected Transform _rearWheels = null;
35 [SerializeField] protected TrainTrack _trainTrack = null;
36 [SerializeField] protected Transform[] _individualWheels = null;
37
38 public float Distance { get; protected set; }
39 protected float scale = 1.0f;
40
41 private Pose _frontPose = new Pose(), _rearPose = new Pose();
42
43 public float Scale
44 {
45 get { return scale; }
46 set { scale = value; }
47 }
48
49 protected virtual void Awake()
50 {
51 Assert.IsNotNull(_frontWheels);
52 Assert.IsNotNull(_rearWheels);
53 Assert.IsNotNull(_trainTrack);
54 Assert.IsNotNull(_individualWheels);
55 }
56
57 public void UpdatePose(float distance, TrainCarBase train, Pose pose)
58 {
59 // distance could be negative; add track length to it in case that happens
60 distance = (train._trainTrack.TrackLength + distance) % train._trainTrack.TrackLength;
61 if (distance < 0)
62 {
63 distance += train._trainTrack.TrackLength;
64 }
65
66 var currentSegment = train._trainTrack.GetSegment(distance);
67 var distanceInto = distance - currentSegment.StartDistance;
68
69 currentSegment.UpdatePose(distanceInto, pose);
70 }
71
72 protected void UpdateCarPosition()
73 {
74 UpdatePose(Distance + _frontWheels.transform.localPosition.z * scale,
75 this, _frontPose);
76 UpdatePose(Distance + _rearWheels.transform.localPosition.z * scale,
77 this, _rearPose);
78
79 var midPoint = 0.5f * (_frontPose.Position + _rearPose.Position);
80 var carLookDirection = _frontPose.Position - _rearPose.Position;
81
82 transform.position = midPoint + OFFSET;
83 transform.rotation = Quaternion.LookRotation(carLookDirection, transform.up);
84 _frontWheels.transform.rotation = _frontPose.Rotation;
85 _rearWheels.transform.rotation = _rearPose.Rotation;
86 }
87
88 protected void RotateCarWheels()
89 {
90 // divide by radius to get angle
91 float angleOfRot = (Distance / WHEEL_RADIUS) % TWO_PI;
92
93 foreach (var individualWheel in _individualWheels)
94 {
95 individualWheel.localRotation = Quaternion.AngleAxis(Mathf.Rad2Deg * angleOfRot,
96 Vector3.right);
97 }
98 }
