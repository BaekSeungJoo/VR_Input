136 private void SetChildScale()
137 {
138 // this will scale all children unless they specifically ask not to
139 // be scaled, using a TransformType that is not Volume.
140 for (var i = 0; i < transform.childCount; i++)
141 {
142 var child = transform.GetChild(i);
143 if (child.TryGetComponent<OVRSceneObjectTransformType>(out var transformType))
144 {
145 if (transformType.TransformType != OVRSceneObjectTransformType.Transformation.Volume)
146 continue;
147 }
148
149 child.localScale = Dimensions;
150 }
151 }
152
153 private void SetChildOffset()
154 {
155 // this will offset all children unless they specifically ask not to
156 // be offset, using a TransformType that is not Volume.
157 for (var i = 0; i < transform.childCount; i++)
158 {
159 var child = transform.GetChild(i);
160 if (child.TryGetComponent<OVRSceneObjectTransformType>(out var transformType))
161 {
162 if (transformType.TransformType != OVRSceneObjectTransformType.Transformation.Volume)
163 continue;
164 }
165
166 child.localPosition = Offset;
167 }

170 internal void UpdateTransform()
171 {
172 if (OVRPlugin.GetSpaceBoundingBox3D(_sceneAnchor.Space, out var bounds))
173 {
174 Width = bounds.Size.w;
175 Height = bounds.Size.h;
176 Depth = bounds.Size.d;
177
178 // calculate the offset as the difference between the
179 // volume pivot and anchor pivot, in Unity coordinate system
180 var anchorPivot = transform.position;
181 var maxPoint = transform.TransformPoint(
182 bounds.Pos.FromVector3f() + bounds.Size.FromSize3f());
183 var volumePivot = transform.TransformPoint(
184 bounds.Pos.FromVector3f() + (bounds.Size.FromSize3f() / 2));
185 volumePivot.y = maxPoint.y;
186
187 Offset = new Vector3(
188 volumePivot.x - anchorPivot.x,
189 volumePivot.z - anchorPivot.z,
190 volumePivot.y - anchorPivot.y);
191
192 OVRSceneManager.Development.Log(nameof(OVRSceneVolume),
193 $"[{_sceneAnchor.Uuid}] Volume has dimensions {Dimensions} " +
194 $"and offset {Offset}.", gameObject);
195
196 if (ScaleChildren)
197 SetChildScale();
198 if (OffsetChildren)
199 SetChildOffset();
200 }
201 else
202 {
203 OVRSceneManager.Development.LogError(nameof(OVRSceneVolume),
204 $"[{_sceneAnchor.Space}] Failed to retrieve volume's information.",
205 gameObject);
206 }
207 }
