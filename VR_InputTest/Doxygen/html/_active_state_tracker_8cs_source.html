26namespace Oculus.Interaction

36 [DefaultExecutionOrder(1)]
37 public class ActiveStateTracker : MonoBehaviour
38 {
39 [Tooltip("The IActiveState to be tracked.")]
40 [SerializeField, Interface(typeof(IActiveState))]
41 private UnityEngine.Object _activeState;
42
43 private IActiveState ActiveState;
44
45 [Header("Active state dependents")]
46 [SerializeField]
47 [Tooltip("If true, all children of this object will be included as dependents.")]
48 private bool _includeChildrenAsDependents = false;
49
50 [SerializeField, Optional]
51 [Tooltip("Sets the `active` field on whole GameObjects.")]
52 private List<GameObject> _gameObjects;
53
54 [SerializeField, Optional]
55 [Tooltip("Sets the `enabled` field on individual components.")]
56 private List<MonoBehaviour> _monoBehaviours;
57
58 protected virtual void Awake()
59 {
60 ActiveState = _activeState as IActiveState;
61 }
62
63 private bool _active = false;
64
65 protected virtual void Start()
66 {
67 this.AssertField(ActiveState, nameof(ActiveState));
68
69 if (_includeChildrenAsDependents)
70 {
71 for(int i = 0; i < transform.childCount; i ++)
72 {
73 _gameObjects.Add(transform.GetChild(i).gameObject);
74 }
75 }
76
77 SetDependentsActive(false);
78 }
79
80 protected virtual void Update()
81 {
82 if (_active == ActiveState.Active) return;
83
84 _active = ActiveState.Active;
85 SetDependentsActive(ActiveState.Active);
86 }
87
88 private void SetDependentsActive(bool active)
89 {
90 for (int i = 0; i < _gameObjects.Count; ++i)
91 {
92 _gameObjects[i].SetActive(active);
93 }
94
95 for (int i = 0; i < _monoBehaviours.Count; ++i)
96 {
97 _monoBehaviours[i].enabled = active;
98 }
99 }
100

114 public void InjectOptionalIncludeChildrenAsDependents(bool includeChildrenAsDependents)

117 }
118
119 public void InjectOptionalGameObjects(List<GameObject> gameObjects)
120 {

122 }
123
124 public void InjectOptionalMonoBehaviours(List<MonoBehaviour> monoBehaviours)

126 _monoBehaviours = monoBehaviours;
127 }
