76 public void MoveTrackedElement(int id, Pose pose)
77 {
78 int desiredIndex = FindBestSlotIndex(pose.position);
79 if (TryFindIndexForInteractor(id, out int index))
80 {
81 if (desiredIndex != index)
82 {
83 _slotInteractors[index] = 0;
84 if (TryOccupySlot(desiredIndex))
85 {
86 _slotInteractors[desiredIndex] = id;
87 }
88 }
89 }
90 else if (TryOccupySlot(desiredIndex))

92 _slotInteractors[desiredIndex] = id;
93 }

96 private bool TryFindIndexForInteractor(int id, out int index)
97 {
98 //FindIndex is not ideal, but this single line simplifies this sample SlotsProvider a lot.
99 index = Array.FindIndex(_slotInteractors, i => i == id);
100 return index >= 0;
101 }

103 public bool SnapPoseForElement(int id, Pose pose, out Pose result)
104 {
105 if (TryFindIndexForInteractor(id, out int index))
106 {
107 result = _slots[index].GetPose();
108 return true;
109 }
110 result = Pose.identity;
111 return false;
112 }

114 private bool TryOccupySlot(int index)
115 {
116 if (IsSlotFree(index))
117 {
118 return true;
119 }
120
121 int freeSlot = FindBestSlotIndex(_slots[index].position, true);
122 if (freeSlot < 0)
123 {
124 return false;
125 }
126
127 PushSlots(index, freeSlot);
128 return true;
129 }
130
131 private bool IsSlotFree(int index)
132 {
133 return _slotInteractors[index] == 0;
134 }
135
136 private int FindBestSlotIndex(in Vector3 target, bool freeOnly = false)
137 {
138 int bestIndex = -1;
139 float minDistance = float.PositiveInfinity;
140 for (int i = 0; i < _slots.Count; i++)
141 {
142 if (freeOnly && !IsSlotFree(i))
143 {
144 continue;
145 }
146
147 float distance = (target - _slots[i].position).sqrMagnitude;
148 if (distance < minDistance)
149 {
150 minDistance = distance;
151 bestIndex = i;
152 }
153
154 }
155 return bestIndex;
156 }
157
158 private void PushSlots(int index, int freeSlot)
159 {
160 bool forwardDirection = index > freeSlot;
161 for (int i = freeSlot; i != index; i = Next(i))
162 {
163 int nextIndex = Next(i);
164 SwapSlot(i, nextIndex);
165 }
166
167 int Next(int value)
168 {
169 return value + (forwardDirection ? 1 : -1);
170 }
171 }
172
173 private void SwapSlot(int index, int freeSlot)
174 {
175 (_slotInteractors[index], _slotInteractors[freeSlot]) = (_slotInteractors[freeSlot], _slotInteractors[index]);
176 }
