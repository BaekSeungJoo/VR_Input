155 public void UpdatePose(float t, bool applyDeadZone = true)
156 {
157 if (applyDeadZone)
158 {
159 const float deadZone = 0.05f;
160 if (Math.Abs(m_inputNodeState.t - t) < deadZone)
161 return;
162 }
163
164 m_inputNodeState.t = t;
165
166 if (m_translations.Count > 1)
167 m_gameObj.transform.localPosition = Vector3.Lerp(m_translations[0], m_translations[1], t);
168 if (m_rotations.Count > 1)
169 m_gameObj.transform.localRotation = Quaternion.Lerp(m_rotations[0], m_rotations[1], t);
170 if (m_scales.Count > 1)
171 m_gameObj.transform.localScale = Vector3.Lerp(m_scales[0], m_scales[1], t);
172 if (m_morphTargetHandler != null && m_weights.Count > 0)
173 {
174 // TODO: t assumes an animation channel input of [0,1].
175 // Changes will be necessary if a model has animations with more keyframes for different timescales
176 var stride = m_morphTargetHandler.Weights.Length;
177 if (m_additiveWeightIndex == -1)
178 {
179 for (int i = 0; i < stride; i++)
180 {
181 m_morphTargetHandler.Weights[i] = Mathf.Lerp(m_weights[i], m_weights[i + stride], t);
182 }
183 }
184 else
185 {
186 m_morphTargetHandler.Weights[m_additiveWeightIndex] += Mathf.Lerp(m_weights[m_additiveWeightIndex],
187 m_weights[m_additiveWeightIndex + stride], t);
188 }
189
190 // mark the geo as dirty
191 m_morphTargetHandler.MarkModified();
192 }
193 }

195 public void UpdatePose(Vector2 joystick)
196 {
197 const float deadZone = 0.05f;
198 if (Math.Abs((m_inputNodeState.vecT - joystick).magnitude) < deadZone)
199 return;
200 m_inputNodeState.vecT.x = joystick.x;
201 m_inputNodeState.vecT.y = joystick.y;
202
203 if (m_rotations.Count != (int)ThumbstickDirection.NorthWest + 1)
204 {
205 Debug.LogError("Wrong joystick animation data.");
206 return;
207 }
208
209 Tuple<ThumbstickDirection, ThumbstickDirection> dir = GetCardinalThumbsticks(joystick);
210 Vector2 weights = GetCardinalWeights(joystick, dir);
211 Quaternion a = CloneQuaternion(m_rotations[0]);
212 for (int i = 0; i < 2; i++)
213 {
214 float t = weights[i];
215 if (t != 0)
216 {
217 int poseIndex = (i == 0 ? (int)dir.Item1 : (int)dir.Item2) - (int)ThumbstickDirection.North;
218 Quaternion b = m_rotations[poseIndex + 1];
219 a = Quaternion.Slerp(a, b, t);
220 }
221 }
222
223 m_gameObj.transform.localRotation = a;
224 if (m_translations.Count > 1 || m_scales.Count > 1)
225 Debug.LogWarning("Unsupported pose.");
226 }

228 // We will blend the 2 closest animations, this picks which 2.
229 private Tuple<ThumbstickDirection, ThumbstickDirection> GetCardinalThumbsticks(Vector2 joystick)
230 {
231 const float deadZone = 0.005f;
232 if (joystick.magnitude < deadZone)
233 {
234 return new Tuple<ThumbstickDirection, ThumbstickDirection>(ThumbstickDirection.None,
235 ThumbstickDirection.None);
236 }
237
238 // East half
239 if (joystick.x >= 0.0f)
240 {
241 // Northeast quadrant
242 if (joystick.y >= 0.0f)
243 {
244 // North-Northeast
245 if (joystick.y > joystick.x)
246 {
247 return new Tuple<ThumbstickDirection, ThumbstickDirection>(ThumbstickDirection.North,
248 ThumbstickDirection.NorthEast);
249 }
250 // East-Northeast
251 else
252 {
253 return new Tuple<ThumbstickDirection, ThumbstickDirection>(ThumbstickDirection.NorthEast,
254 ThumbstickDirection.East);
255 }
256 }
257 // Southeast quadrant
258 else
259 {
260 // East-Southeast
261 if (joystick.x > -joystick.y)
262 {
263 return new Tuple<ThumbstickDirection, ThumbstickDirection>(ThumbstickDirection.East,
264 ThumbstickDirection.SouthEast);
265 }
266 // South-southeast
267 else
268 {
269 return new Tuple<ThumbstickDirection, ThumbstickDirection>(ThumbstickDirection.SouthEast,
270 ThumbstickDirection.South);
271 }
272 }
273 }
274 // West half
275 else
276 {
277 // Southwest quadrant
278 if (joystick.y < 0.0f)
279 {
280 // South-Southwest
281 if (joystick.x > joystick.y)
282 {
283 return new Tuple<ThumbstickDirection, ThumbstickDirection>(ThumbstickDirection.South,
284 ThumbstickDirection.SouthWest);
285 }
286 // West-Southwest
287 else
288 {
289 return new Tuple<ThumbstickDirection, ThumbstickDirection>(ThumbstickDirection.SouthWest,
290 ThumbstickDirection.West);
291 }
292 }
293 // Northwest quadrant
294 else
295 {
296 // West-Northwest
297 if (-joystick.x > joystick.y)
298 {
299 return new Tuple<ThumbstickDirection, ThumbstickDirection>(ThumbstickDirection.West,
300 ThumbstickDirection.NorthWest);
301 }
302 // North-Northwest
303 else
304 {
305 return new Tuple<ThumbstickDirection, ThumbstickDirection>(ThumbstickDirection.NorthWest,
306 ThumbstickDirection.North);
307 }
308 }
309 }

312 // This figures out how much of each of the 2 animations to blend, based on where in between the 2
313 // cardinal directions the user is actually pushing the thumbstick, and how far they are pushing
314 // the thumbstick("animations" themselves are a fixed pose for a "maximum" push.
315 private Vector2 GetCardinalWeights(Vector2 joystick, Tuple<ThumbstickDirection, ThumbstickDirection> cardinals)
316 {
317 // follows ThumbstickDirection, can use ThumbstickDirection to directly index into this
318
319 if (cardinals.Item1 == ThumbstickDirection.None || cardinals.Item2 == ThumbstickDirection.None)
320 {
321 return new Vector2(0.0f, 0.0f);
322 }
323
324 // Compute the barycentric coordinates of the joystick position in the triangle formed by the 2
325 // cardinal directions
326 Vector2 triangleEdge1 = CardDirections[(int)(cardinals.Item1)];
327 Vector2 triangleEdge2 = CardDirections[(int)(cardinals.Item2)];
328 float dot11 = Vector2.Dot(triangleEdge1, triangleEdge1);
329 float dot12 = Vector2.Dot(triangleEdge1, triangleEdge2);
330 float dot1j = Vector2.Dot(triangleEdge1, joystick);
331 float dot22 = Vector2.Dot(triangleEdge2, triangleEdge2);
332 float dot2j = Vector2.Dot(triangleEdge2, joystick);
333
334 float invDenom = 1.0f / (dot11 * dot22 - dot12 * dot12);
335 float weight1 = (dot22 * dot1j - dot12 * dot2j) * invDenom;
336 float weight2 = (dot11 * dot2j - dot12 * dot1j) * invDenom;
337
338 return new Vector2(weight1, weight2);
339 }

341 private void ProcessAnimationSampler(JSONNode samplerNode, int nodeId, OVRGLTFTransformType transformType,
342 JSONNode extras)
343 {
344 int outputId = samplerNode["output"].AsInt;
345 OVRInterpolationType interpolationId = ToOVRInterpolationType(samplerNode["interpolation"].Value);
346 if (interpolationId == OVRInterpolationType.None)
347 {
348 Debug.LogError("Unsupported interpolation type: " + samplerNode["interpolation"].Value);
349 return;
350 }
351
352
353 var jsonOutputAccessor = m_jsonData["accessors"][outputId];
354 OVRGLTFAccessor outputReader = new OVRGLTFAccessor(jsonOutputAccessor, m_jsonData);
355
356 int inputId = samplerNode["input"].AsInt;
357 var jsonInputAccessor = m_jsonData["accessors"][inputId];
358 OVRGLTFAccessor inputReader = new OVRGLTFAccessor(jsonInputAccessor, m_jsonData);
359 float[] inputFloats = new float[inputReader.GetDataCount()];
360 inputReader.ReadAsFloat(m_binaryChunk, ref inputFloats, 0);
361 // implementation assumes inputFloats = [0, 1]
362 // Changes will be necessary if a model has animations with more keyframes for different timescales
363
364 switch (transformType)
365 {
366 case OVRGLTFTransformType.Translation:
367 Vector3[] translations = new Vector3[outputReader.GetDataCount()];
368 outputReader.ReadAsVector3(m_binaryChunk, ref translations, 0, OVRGLTFLoader.GLTFToUnitySpace);
369 CopyData(ref m_translations, translations);
370 break;
371 case OVRGLTFTransformType.Rotation:
372 Vector4[] rotations = new Vector4[outputReader.GetDataCount()];
373 outputReader.ReadAsVector4(m_binaryChunk, ref rotations, 0, OVRGLTFLoader.GLTFToUnitySpace_Rotation);
374 List<Vector4> rotationDest = new List<Vector4>();
375 CopyData(ref rotationDest, rotations);
376 foreach (Vector4 v in rotationDest)
377 {
378 m_rotations.Add(new Quaternion(v.x, v.y, v.z, v.w));
379 }
380
381 break;
382 case OVRGLTFTransformType.Scale:
383 Vector3[] scales = new Vector3[outputReader.GetDataCount()];
384 outputReader.ReadAsVector3(m_binaryChunk, ref scales, 0, new Vector3(1, 1, 1));
385 CopyData(ref m_scales, scales);
386 break;
387 case OVRGLTFTransformType.Weights:
388 float[] weights = new float[outputReader.GetDataCount()];
389 outputReader.ReadAsFloat(m_binaryChunk, ref weights, 0);
390 CopyData(ref m_weights, weights);
391 if (extras != null && extras["additiveWeightIndex"] != null)
392 {
393 m_additiveWeightIndex = extras["additiveWeightIndex"].AsInt;
394 }
395
396 if (m_morphTargetHandler != null)
397 {
398 m_morphTargetHandler.Weights = new float[weights.Length / inputFloats.Length];
399 }
400
401 break;
402 default:
403 Debug.LogError("Unsupported transform type: " + transformType.ToString());
404 break;
405 }
406 }

408 private OVRGLTFTransformType GetTransformType(string transform)
409 {
410 switch (transform)
411 {
412 case "translation":
413 return OVRGLTFTransformType.Translation;
414 case "rotation":
415 return OVRGLTFTransformType.Rotation;
416 case "scale":
417 return OVRGLTFTransformType.Scale;
418 case "weights":
419 return OVRGLTFTransformType.Weights;
420 case "none":
421 return OVRGLTFTransformType.None;
422 default:
423 Debug.LogError("Unsupported transform type: " + transform);
424 return OVRGLTFTransformType.None;

446 private void CopyData<T>(ref List<T> dest, T[] src)
447 {
448 if (m_intputNodeType == OVRGLTFInputNode.None)
449 {
450 dest = src.ToList();
451 }
452 else if (m_intputNodeType == OVRGLTFInputNode.ThumbStick)
453 {
454 foreach (int idx in ThumbStickKeyFrames)
455 {
456 if (idx < src.Length)
457 dest.Add(src[idx]);
458 }
459 }
460 else
461 {
462 int idx = InputNodeKeyFrames[m_intputNodeType];
463 if (idx < src.Length)
464 dest.Add(src[idx]);
465 }
466 }
