30 void LateUpdate()
31 {
32 // face camera
33 transform.rotation = Quaternion.LookRotation(transform.position - Camera.main.transform.position);
34
35 if (!(controllerHand == OVRInput.Controller.LTouch || controllerHand == OVRInput.Controller.RTouch))

39
40 Vector3 tipPosition = transform.position;
41 switch (brushStatus)
42 {
43 case BrushState.Idle:
44 if (OVRInput.GetUp(OVRInput.Button.One, controllerHand))

46 UndoInkLine();
47 }
48
49 if (OVRInput.GetDown(OVRInput.Button.PrimaryIndexTrigger, controllerHand))

52 brushStatus = BrushState.Inking;
53 }
54
55 break;
56 case BrushState.Inking:
57 // every frame, draw line
58 UpdateLine(tipPosition);
59 if (OVRInput.GetUp(OVRInput.Button.PrimaryIndexTrigger, controllerHand))
60 {
61 brushStatus = BrushState.Idle;
62 }
63
64 break;
65 }
