79 {
80 foreach (var entry in sourceFrequencies)
81 {
82 if (!targetFrequencies.ContainsKey(entry.Key))
83 {
84 targetFrequencies[entry.Key] = entry.Value;

86 else

88 targetFrequencies[entry.Key] += entry.Value;
