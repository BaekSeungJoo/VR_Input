168 float currentLight = sceneLight ? sceneLight.intensity : 0;
169 while (timer <= fadeTime)

171 timer += Time.deltaTime;
172 float normTimer = Mathf.Clamp01(timer / fadeTime);
173 Camera.main.backgroundColor = Color.Lerp(currentColor, newColor, normTimer);
