88 private static void Log(LogType logType, string logCategory, object log)
89 {
90 #if UNITY_EDITOR
91 // Skip logs with higher log type then global log level
92 if ((int) logType > (int)EditorLogLevel)
93 {
94 return;
95 }
96 #endif
97 // Suppress logs if desired
98 if (SuppressLogs)
99 {
100 return;
101 }
102
103 // Use calling category if null
104 string category = logCategory;
105 if (string.IsNullOrEmpty(category))
106 {
107 category = GetCallingCategory();
108 }
109
110 // String builder
111 StringBuilder result = new StringBuilder();
112
113 #if !UNITY_EDITOR && !UNITY_ANDROID
114 {
115 // Start with datetime if not done so automatically
116 DateTime now = DateTime.Now;
117 result.Append($"[{now.ToShortDateString()} {now.ToShortTimeString()}] ");
118 }
119 #endif
120
121 // Insert log type
122 int start = result.Length;
123 result.Append($"[{logType.ToString().ToUpper()}] ");
124 WrapWithLogColor(result, start, logType);
125
126 // Append VDSK & Category
127 start = result.Length;
128 result.Append("[VSDK");
129 if (!string.IsNullOrEmpty(category))
130 {
131 result.Append($" {category}");
132 }
133 result.Append("] ");
134 WrapWithCallingLink(result, start);
135
136 // Append the actual log
137 result.Append(log == null ? string.Empty : log.ToString());
138
139 // Final log append
140 OnPreLog?.Invoke(result, logCategory, logType);
141
142 // Log
143 switch (logType)
144 {
145 case LogType.Error:
146 UnityEngine.Debug.LogError(result);
147 break;
148 case LogType.Warning:
149 UnityEngine.Debug.LogWarning(result);
150 break;
151 default:
152 UnityEngine.Debug.Log(result);
153 break;
154 }
155 }

161 private static string GetCallingCategory()
162 {
163 // Get stack trace method
164 string path = new StackTrace()?.GetFrame(3)?.GetMethod().DeclaringType.Name;
165 if (string.IsNullOrEmpty(path))
166 {
167 return "NoStacktrace";
168 }
169 // Return path
170 return path;
171 }

177 private static void WrapWithCallingLink(StringBuilder builder, int startIndex)
178 {
179 #if UNITY_EDITOR && UNITY_2021_2_OR_NEWER
180 StackTrace stackTrace = new StackTrace(true);
181 StackFrame stackFrame = stackTrace.GetFrame(3);
182 string callingFileName = stackFrame.GetFileName().Replace('\\', '/');
183 int callingFileLine = stackFrame.GetFileLineNumber();
184 builder.Insert(startIndex, $"<a href=\"{callingFileName}\" line=\"{callingFileLine}\">");
185 builder.Append("</a>");
186 #endif
187 }

192 private static void WrapWithLogColor(StringBuilder builder, int startIndex, LogType logType)
193 {
194 #if UNITY_EDITOR
195 string hex;
196 switch (logType)
197 {
198 case LogType.Error:
199 hex = "FF0000";
200 break;
201 case LogType.Warning:
202 hex = "FFFF00";
203 break;
204 default:
205 hex = "00FF00";
206 break;
207 }
208 builder.Insert(startIndex, $"<color=#{hex}>");
209 builder.Append("</color>");
210 #endif
211 }
