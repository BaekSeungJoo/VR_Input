61 protected virtual void HandleInteractorPostProcessed()
62 {
63 int pointsCount = _interactor.TeleportArc.PointsCount;
64 if (_positions == null
65 || _positions.Length != pointsCount)
66 {
67 _positions = new Vector3[pointsCount];
68 _arcRenderer.positionCount = pointsCount;
69 }
70
71 for (int i = 0; i < pointsCount; i++)
72 {
73 _positions[i] = _interactor.TeleportArc.PointAtIndex(i);
74 }
75
76 _arcRenderer.SetPositions(_positions);
77 }
