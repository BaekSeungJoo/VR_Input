57 public void RefreshCameraRig(GameObject parentObject, Camera mainCamera)
58 {
59 OVRCameraRig cameraRig = mainCamera.GetComponentInParent<OVRCameraRig>();
60 if (cameraRig == null)
61 {
62 cameraRig = parentObject.GetComponent<OVRCameraRig>();
63 }
64
65 cameraInTrackingSpace = (cameraRig != null && cameraRig.trackingSpace != null);
66 this.cameraRig = cameraRig;

70 public OVRPose ComputeCameraWorldSpacePose(OVRPlugin.CameraExtrinsics extrinsics, Camera mainCamera)
71 {
72 OVRPose trackingSpacePose = ComputeCameraTrackingSpacePose(extrinsics);
73 OVRPose worldSpacePose = OVRExtensions.ToWorldSpacePose(trackingSpacePose, mainCamera);

77 public OVRPose ComputeCameraTrackingSpacePose(OVRPlugin.CameraExtrinsics extrinsics)
78 {
79 OVRPose trackingSpacePose = new OVRPose();
80
81 OVRPose cameraTrackingSpacePose = extrinsics.RelativePose.ToOVRPose();
82#if OVR_ANDROID_MRC
83 OVRPose stageToLocalPose =
84 OVRPlugin.GetTrackingTransformRelativePose(OVRPlugin.TrackingOrigin.Stage).ToOVRPose();
85 cameraTrackingSpacePose = stageToLocalPose * cameraTrackingSpacePose;
86#endif
87 trackingSpacePose = cameraTrackingSpacePose;
88
89 if (extrinsics.AttachedToNode != OVRPlugin.Node.None && OVRPlugin.GetNodePresent(extrinsics.AttachedToNode))
90 {
91 if (usingLastAttachedNodePose)
92 {
93 Debug.Log("The camera attached node get tracked");
94 usingLastAttachedNodePose = false;
95 }
96
97 OVRPose attachedNodePose =
98 OVRPlugin.GetNodePose(extrinsics.AttachedToNode, OVRPlugin.Step.Render).ToOVRPose();
99 lastAttachedNodePose = attachedNodePose;
100 trackingSpacePose = attachedNodePose * trackingSpacePose;
101 }
102 else
103 {
104 if (extrinsics.AttachedToNode != OVRPlugin.Node.None)
105 {
106 if (!usingLastAttachedNodePose)
107 {
108 Debug.LogWarning("The camera attached node could not be tracked, using the last pose");
109 usingLastAttachedNodePose = true;
110 }
111
112 trackingSpacePose = lastAttachedNodePose * trackingSpacePose;
113 }
114 }
115
116 return trackingSpacePose;
117 }
118}
