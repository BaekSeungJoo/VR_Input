537 public static void Share(ICollection<OVRSpatialAnchor> anchors, ICollection<OVRSpaceUser> users,
538 Action<ICollection<OVRSpatialAnchor>, OperationResult> onComplete = null)
539 {
540 if (anchors == null)
541 throw new ArgumentNullException(nameof(anchors));
542
543 using var spaces = ToNativeArray(anchors);
544
545 var handles = new NativeArray<ulong>(users.Count, Allocator.Temp);
546 using var disposer = handles;
547 int i = 0;
548 foreach (var user in users)
549 {
550 handles[i++] = user._handle;
551 }
552
553 var shareResult = OVRPlugin.ShareSpaces(spaces, handles, out var requestId);
554 if (shareResult.IsSuccess())
555 {
556 Development.LogRequest(requestId, $"Sharing {(uint)spaces.Length} spatial anchors...");
557
558 MultiAnchorCompletionDelegates[requestId] = new MultiAnchorDelegatePair
559 {
560 Anchors = CopyAnchorListIntoListFromPool(anchors),
561 Delegate = onComplete
562 };
563 }
564 else
565 {
566 Development.LogError(
567 $"{nameof(OVRPlugin)}.{nameof(OVRPlugin.ShareSpaces)} failed with error {shareResult}.");
568 onComplete?.Invoke(anchors, (OperationResult)shareResult);

581 private List<OVRSpatialAnchor> GetListToStoreTheShareRequest(List<OVRSpaceUser> users)
582 {
583 users.Sort((x, y) => x.Id.CompareTo(y.Id));
584 foreach (var (shareRequestUsers, shareRequestAnchors) in ShareRequests)
585 {
586 if (!AreSortedUserListsEqual(users, shareRequestUsers))
587 {
588 continue;
589 }
590
591 // reuse the current request
592 return shareRequestAnchors;
593 }
594
595 // add a new request
596 var anchorList = OVRObjectPool.List<OVRSpatialAnchor>();
597 ShareRequests.Add((users, anchorList));
598 return anchorList;
599 }

601 private static bool AreSortedUserListsEqual(IReadOnlyList<OVRSpaceUser> sortedList1,
602 IReadOnlyList<OVRSpaceUser> sortedList2)
603 {
604 if (sortedList1.Count != sortedList2.Count)
605 {
606 return false;
607 }
608
609 for (var i = 0; i < sortedList1.Count; i++)
610 {
611 if (sortedList1[i].Id != sortedList2[i].Id)
612 {
613 return false;
614 }
615 }
616
617 return true;
618 }

819 {
820 var tryLocateSpace = OVRPlugin.TryLocateSpace(space, OVRPlugin.GetTrackingOriginType(), out var posef, out var locationFlags);
821 if (!tryLocateSpace || !locationFlags.IsOrientationValid() || !locationFlags.IsPositionValid())

823 pose = OVRPose.identity;
824 return false;
825 }
826
827 pose = posef.ToOVRPose();

829 if (mainCamera)
830 {
831 pose = pose.ToWorldSpacePose(mainCamera);
832 }
833
834 return true;

900 private static void InvokeMultiAnchorDelegate(ulong requestId, OperationResult result,
901 MultiAnchorActionType actionType)
902 {
903 if (!TryExtractValue(MultiAnchorCompletionDelegates, requestId, out var value))
904 {
905 return;
906 }
907
908 if (actionType == MultiAnchorActionType.Save)
909 {
910 OVRTelemetry.Client.MarkerEnd(OVRTelemetryConstants.Scene.MarkerId.SpatialAnchorSave,
911 result == OperationResult.Success ? OVRPlugin.Qpl.ResultType.Success : OVRPlugin.Qpl.ResultType.Fail,
912 requestId.GetHashCode());
913 }
914
915 value.Delegate?.Invoke(value.Anchors, result);
916
917 try
918 {
919 foreach (var anchor in value.Anchors)
920 {
921 switch (actionType)
922 {
923 case MultiAnchorActionType.Save:
924 {
925 if (result != OperationResult.Success)
926 {
927 Development.LogError(
928 $"[{anchor.Uuid}] {nameof(OVRPlugin)}.{nameof(OVRPlugin.SaveSpaceList)} failed with result: {result}.");
929 }
930
931 if (AsyncRequestTaskIds.TryGetValue(anchor, out var taskId))
932 {
933 AsyncRequestTaskIds.Remove(anchor);
934 OVRTask.GetExisting<bool>(taskId).SetResult(result == OperationResult.Success);
935 }
936
937 break;
938 }
939 case MultiAnchorActionType.Share:
940 {
941 if (result != OperationResult.Success)
942 {
943 Development.LogError(
944 $"[{anchor.Uuid}] {nameof(OVRPlugin)}.{nameof(OVRPlugin.ShareSpaces)} failed with result: {result}.");
945 }
946
947 if (AsyncRequestTaskIds.TryGetValue(anchor, out var taskId))
948 {
949 AsyncRequestTaskIds.Remove(anchor);
950 OVRTask.GetExisting<OperationResult>(taskId).SetResult(result);
951 }
952
953 break;
954 }
955 default:
956 throw new ArgumentOutOfRangeException(nameof(actionType), actionType, null);
957 }
958 }
959 }
960 finally
961 {
962 OVRObjectPool.Return(value.Anchors);
963 }
964 }

966 private static void OnSpatialAnchorCreateComplete(ulong requestId, bool success, OVRSpace space, Guid uuid)
967 {
968 Development.LogRequestResult(requestId, success,
969 $"[{uuid}] Spatial anchor created.",
970 $"Failed to create spatial anchor. Destroying {nameof(OVRSpatialAnchor)} component.");
971
972 if (!TryExtractValue(CreationRequests, requestId, out var anchor)) return;
973
974 OVRTelemetry.Client.MarkerEnd(OVRTelemetryConstants.Scene.MarkerId.SpatialAnchorCreate,
975 success ? OVRPlugin.Qpl.ResultType.Success : OVRPlugin.Qpl.ResultType.Fail,
976 requestId.GetHashCode());
977
978 if (success && anchor)
979 {
980 // All good; complete setup of OVRSpatialAnchor component.
981 anchor.InitializeUnchecked(space, uuid);
982 return;
983 }
984
985 if (success && !anchor)
986 {
987 // Creation succeeded, but the OVRSpatialAnchor component was destroyed before the callback completed.
988 OVRPlugin.DestroySpace(space);
989 }
990 else if (!success && anchor)
991 {
992 // The OVRSpatialAnchor component exists but creation failed.
993 Destroy(anchor);
994 }
995 // else if creation failed and the OVRSpatialAnchor component was destroyed, nothing to do.
996 }
