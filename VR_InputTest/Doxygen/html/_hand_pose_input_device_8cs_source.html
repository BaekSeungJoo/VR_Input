74 public HandJointPoseMetaData(HandFinger finger,
75 HandJointId joint, int bufferLength)

85 public void BufferNewValue(Pose newPose, float delta)
86 {
87 Vector3 newPosition = newPose.position;
88 Vector3 newVelocity = Vector3.zero;
89 if (delta > Mathf.Epsilon && _previousPosition.HasValue)
90 {
91 newVelocity = (newPosition - _previousPosition.Value)
92 / delta;
93 }
94 int nextWritePos = (_lastWritePos < 0) ? 0 :
95 (_lastWritePos + 1) % _bufferLength;
96 if (Velocities.Count <= nextWritePos)
97 {
98 Velocities.Add(newVelocity);
99 }
100 else
101 {
102 Velocities[nextWritePos] = newVelocity;
103 }
104
105 _previousPosition = newPosition;
106 _lastWritePos = nextWritePos;
107 }
108
109 public Vector3 GetAverageVelocityVector()
110 {
111 int numVelocities = Velocities.Count;
112 if (numVelocities == 0)
113 {
114 return Vector3.zero;
115 }
116 Vector3 average = Vector3.zero;
117 foreach (var speed in Velocities)
118 {
119 average += speed;
120 }
121 average /= numVelocities;
122 return average;
123 }
124
125 public void ResetSpeedsBuffer()
126 {
127 Velocities.Clear();
128 _lastWritePos = -1;
129 _previousPosition = null;
130 }

244 private void BufferFingerBoneVelocities()
245 {
246 float deltaValue = Time.deltaTime;
247 foreach (var jointPoseInfo in _jointPoseInfoArray)
248 {
249 if (!GetFingerIsHighConfidence(jointPoseInfo.Finger))
250 {
251 continue;
252 }
253
254 Pose jointPose;
255 if (!GetJointPose(jointPoseInfo.JointId, out jointPose))
256 {
257 continue;
258 }
259
260 jointPoseInfo.BufferNewValue(jointPose, deltaValue);
261 }

264 public (Vector3, Vector3) GetExternalVelocities()
265 {
266 if (_jointPoseInfoArray == null ||
267 _jointPoseInfoArray.Length == 0)
268 {
269 return (Vector3.zero, Vector3.zero);
270 }
271
272 Vector3 averageVelocityAllFingers = Vector3.zero;
273 foreach (var fingerMetaInfo in _jointPoseInfoArray)
274 {
275 averageVelocityAllFingers += fingerMetaInfo.GetAverageVelocityVector();
276 }
277 averageVelocityAllFingers /= _jointPoseInfoArray.Length;
278 foreach (var item in _jointPoseInfoArray)
279 {
280 item.ResetSpeedsBuffer();
281 }
282
283 return (averageVelocityAllFingers, Vector3.zero);
284 }
