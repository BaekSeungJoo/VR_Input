145 private bool TryGetHand(out IHand hand)
146 {
147 hand = null;
148
149 if (_handVisual == null)
150 {
151 return false;
152 }
153
154 if (_handVisual.Hand != null)
155 {
156 hand = _handVisual.Hand;
157 return true;
158 }
159
160 System.Type targetObjectClassType = _handVisual.GetType();
161 System.Reflection.FieldInfo field = targetObjectClassType.GetField("_hand",
162 System.Reflection.BindingFlags.NonPublic | System.Reflection.BindingFlags.Instance);
163 if (field != null)
164 {
165 object value = field.GetValue(_handVisual);
166 hand = value as IHand;
167 return hand != null;
168 }
169 return false;
170 }

265 private AnimationClip GenerateClipAsset(string title)
266 {
267 GetHandPrefixes(out string prefix, out string mirrorPrefix);
268 AnimationClip clip = new AnimationClip();
269
270 for (int i = (int)HandJointId.HandStart; i < (int)HandJointId.HandEnd; ++i)
271 {
272 Transform jointTransform = _handVisual.Joints[i];
273 string path = GetGameObjectPath(jointTransform, Root);
274 RegisterLocalPose(ref clip, jointTransform.GetPose(Space.Self), path);
275 }
276
277 StoreAsset(clip, $"{title}{prefix}.anim");
278 return clip;
279 }

281 private AvatarMask GenerateMaskAsset(List<HandJointId> maskData, string title)
282 {
283 GetHandPrefixes(out string prefix, out string mirrorPrefix);
284 AvatarMask mask = new AvatarMask();
285 List<string> paths = new List<string>(maskData.Count);
286
287 foreach (var maskJoints in maskData)
288 {
289 Transform jointTransform = _handVisual.Joints[(int)maskJoints];
290 string localPath = GetGameObjectPath(jointTransform, Root);
291 paths.Add(localPath);
292 }
293
294 mask.transformCount = paths.Count;
295 for (int i = 0; i < paths.Count; ++i)
296 {
297 mask.SetTransformPath(i, paths[i]);
298 mask.SetTransformActive(i, true);
299 }
300
301 StoreAsset(mask, $"{title}{prefix}.mask");
302 return mask;
303 }

305 private AnimationClip GenerateMirrorClipAsset(AnimationClip originalClip)
306 {
307 if (originalClip == null)
308 {
309 Debug.LogError("Please generate a valid Clip first");
310 return null;
311 }
312 GetHandPrefixes(out string prefix, out string mirrorPrefix);
313
314 AnimationClip mirrorClip = new AnimationClip();
315
316 EditorCurveBinding[] curveBindings = AnimationUtility.GetCurveBindings(originalClip);
317 foreach (EditorCurveBinding curveBinding in curveBindings)
318 {
319 string mirrorPath = curveBinding.path.Replace(prefix, mirrorPrefix);
320 AnimationCurve curve = AnimationUtility.GetEditorCurve(originalClip, curveBinding);
321 float invertFactor = curveBinding.propertyName.Contains("LocalPosition") ? -1f : 1f;
322
323 AnimationCurve mirrorCurve = new AnimationCurve();
324
325 for (int i = 0; i < curve.length; i++)
326 {
327 mirrorCurve.AddKey(curve[i].time, curve[i].value * invertFactor);
328 }
329 mirrorClip.SetCurve(mirrorPath, curveBinding.type, curveBinding.propertyName, mirrorCurve);
330 }
331 StoreAsset(mirrorClip, $"{originalClip.name.Replace(prefix, mirrorPrefix)}.anim");
332 return mirrorClip;
333 }

335 private AvatarMask GenerateMirrorMaskAsset(AvatarMask originalMask)
336 {
337 if (originalMask == null)

339 Debug.LogError("Please generate a valid mask first");

341 }
342 GetHandPrefixes(out string prefix, out string mirrorPrefix);
343
344 AvatarMask mirrorMask = new AvatarMask();
345
346 mirrorMask.transformCount = originalMask.transformCount;
347 for (int i = 0; i < originalMask.transformCount; ++i)
348 {
349 string mirrorPath = originalMask.GetTransformPath(i).Replace(prefix, mirrorPrefix);
350 bool active = originalMask.GetTransformActive(i);
351 mirrorMask.SetTransformPath(i, mirrorPath);
352 mirrorMask.SetTransformActive(i, active);
353 }
354
355 StoreAsset(mirrorMask, $"{originalMask.name.Replace(prefix, mirrorPrefix)}.mask");
356
357 return mirrorMask;

360 private void RegisterLocalPose(ref AnimationClip clip, Pose pose, string path)
361 {
362 Vector3 euler = pose.rotation.eulerAngles;
363 clip.SetCurve(path, typeof(Transform), "localEulerAngles.x", AnimationCurve.Constant(0f, 0.01f, euler.x));
364 clip.SetCurve(path, typeof(Transform), "localEulerAngles.y", AnimationCurve.Constant(0f, 0.01f, euler.y));
365 clip.SetCurve(path, typeof(Transform), "localEulerAngles.z", AnimationCurve.Constant(0f, 0.01f, euler.z));
366
367 Vector3 pos = pose.position;
368 clip.SetCurve(path, typeof(Transform), "localPosition.x", AnimationCurve.Constant(0f, 0.01f, pos.x));
369 clip.SetCurve(path, typeof(Transform), "localPosition.y", AnimationCurve.Constant(0f, 0.01f, pos.y));
370 clip.SetCurve(path, typeof(Transform), "localPosition.z", AnimationCurve.Constant(0f, 0.01f, pos.z));
371 }

373 private AnimatorController CreateAnimator(string path, HandClips clips)
374 {
375 if (!clips.IsComplete())
376 {
377 Debug.LogError("Missing clips and masks to generate the animator");
378 return null;
379 }
380 AnimatorController animator = AnimatorController.CreateAnimatorControllerAtPath(path);
381
382 animator.AddParameter(FLEX_PARAM, AnimatorControllerParameterType.Float);
383 animator.AddParameter(PINCH_PARAM, AnimatorControllerParameterType.Float);
384
385 animator.RemoveLayer(0);
386
387 CreateLayer(animator, "Flex Layer", null);
388 CreateLayer(animator, "Thumb Layer", clips.thumbMask);
389 CreateLayer(animator, "Point Layer", clips.indexMask);
390
391 CreateFlexStates(animator, 0, clips);
392 CreateThumbUpStates(animator, 1, clips);
393 CreatePointStates(animator, 2, clips);
394
395 return animator;
396 }

398 private AnimatorControllerLayer CreateLayer(AnimatorController animator, string layerName, AvatarMask mask = null)
399 {
400 AnimatorControllerLayer layer = new AnimatorControllerLayer();
401 layer.name = layerName;
402 AnimatorStateMachine stateMachine = new AnimatorStateMachine();
403 stateMachine.name = layer.name;
404 AssetDatabase.AddObjectToAsset(stateMachine, animator);
405 stateMachine.hideFlags = HideFlags.HideInHierarchy;
406 layer.stateMachine = stateMachine;
407 layer.avatarMask = mask;
408 animator.AddLayer(layer);
409 return layer;
410 }

412 private void CreateFlexStates(AnimatorController animator, int layerIndex, HandClips clips)
413 {
414 BlendTree blendTree;
415 AnimatorState flexState = animator.CreateBlendTreeInController("Flex", out blendTree, layerIndex);
416 blendTree.blendType = BlendTreeType.FreeformCartesian2D;
417 blendTree.blendParameter = FLEX_PARAM;
418 blendTree.blendParameterY = PINCH_PARAM;
419 blendTree.AddChild(clips.handCap, new Vector2(0f, 0f));
420 blendTree.AddChild(clips.handPinch, new Vector2(0f, 0.835f));
421 blendTree.AddChild(clips.handPinch, new Vector2(0f, 1f));
422 blendTree.AddChild(clips.handMidFist, new Vector2(0.5f, 0f));
423 blendTree.AddChild(clips.handMidFist, new Vector2(0.5f, 1f));
424 blendTree.AddChild(clips.hand3qtrFist, new Vector2(0.835f, 0f));
425 blendTree.AddChild(clips.hand3qtrFist, new Vector2(0.835f, 1f));
426 blendTree.AddChild(clips.handFist, new Vector2(1f, 0f));
427 blendTree.AddChild(clips.handFist, new Vector2(1f, 1f));
428 animator.layers[layerIndex].stateMachine.defaultState = flexState;
429 }

482 private static string GetGameObjectPath(Transform transform, Transform root)
483 {
484 string path = transform.name;
485 while (transform.parent != null
486 && transform.parent != root)
487 {
488 transform = transform.parent;
489 path = $"{transform.name}/{path}";
490 }

494 private void GetHandPrefixes(out string prefix, out string mirrorPrefix)
495 {
496 if (!TryGetHand(out IHand hand))
497 {
498 Debug.LogError("Hand not found");
499 prefix = mirrorPrefix = string.Empty;
500 return;
501 }
502
503 Handedness originalHandedness = hand.Handedness;
504 prefix = originalHandedness == Handedness.Left ? _handLeftPrefix : _handRightPrefix;
505 mirrorPrefix = originalHandedness == Handedness.Left ? _handRightPrefix : _handLeftPrefix;
506 }
