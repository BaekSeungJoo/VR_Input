38 // Update is called once per frame
39 void Update()
40 {
41 if (_character == null)
42 {
43 _character = GetComponentInParent<CharacterController>();
44 if (_character == null)
45 {
46 return;
47 }
48 }
49
50 if (_height == _character.height
51 && _radius == _character.radius
52 && _subdivisionU == SubdivisionsU
53 && _subdivisionV == SubdivisionsV)
54 {
55 return;
56 }
57
58 _height = _character.height;
59 _radius = _character.radius;
60 _subdivisionU = SubdivisionsU;
61 _subdivisionV = SubdivisionsV;
62
63 List<Vector3> verts = new List<Vector3>();
64
65 var vector = new Vector3(1, 0, 0);
66
67 // Generate the mesh
68 var topOffset = new Vector3(0, _height / 2.0f - _radius, 0);
69 var bottomOffset = new Vector3(0, _radius - _height / 2.0f, 0);
70
71 // Add all the necessary vertices
72 verts.Add(new Vector3(0, _height / 2.0f, 0));
73
74 for (int u = SubdivisionsU - 1; u >= 0; u--)
75 {
76 float uf = (float)u / (float)SubdivisionsU;
77 for (int v = 0; v < SubdivisionsV; v++)
78 {
79 float vf = (float)v / (float)SubdivisionsV;
80 var q = Quaternion.Euler(0, vf * 360.0f, uf * 90.0f);
81 var vert = q * vector;
82 vert *= _radius;
83 var v1 = vert + topOffset;
84 verts.Add(v1);
85 }
86 }
87
88 for (int u = 0; u < SubdivisionsU; u++)
89 {
90 float uf = (float)u / (float)SubdivisionsU;
91 for (int v = 0; v < SubdivisionsV; v++)
92 {
93 float vf = (float)v / (float)SubdivisionsV;
94 var q = Quaternion.Euler(0, vf * 360.0f + 180.0f, uf * 90.0f);
95 var vert = q * vector;
96 vert *= _radius;
97 var v2 = bottomOffset - vert;
98 verts.Add(v2);
99 }
100 }
101
102 verts.Add(new Vector3(0, -_height / 2.0f, 0));
103
104
105 // Setup all the triangles
106
107 List<int> tris = new List<int>();
108 int index;
109 int i;
110
111 // top cap
112 for (int v = 0; v < SubdivisionsV; v++)
113 {
114 i = 0;
115 tris.Add(i);
116 tris.Add(v);
117 tris.Add(v + 1);
118 }
119
120 tris.Add(0);
121 tris.Add(SubdivisionsV);
122 tris.Add(1);
123
124 // top hemisphere
125 for (int u = 0; u < SubdivisionsU - 1; u++)
126 {
127 index = u * SubdivisionsV + 1;
128 for (int v = 0; v < SubdivisionsV - 1; v++)
129 {
130 i = index + v;
131 tris.Add(i);
132 tris.Add(i + SubdivisionsV);
133 tris.Add(i + 1);
134
135 tris.Add(i + 1);
136 tris.Add(i + SubdivisionsV);
137 tris.Add(i + SubdivisionsV + 1);
138 }
139
140 i = index + SubdivisionsV - 1;
141 tris.Add(i);

143 tris.Add(i + 1 - SubdivisionsV);
144
145 tris.Add(i + 1 - SubdivisionsV);
146 tris.Add(i + SubdivisionsV);
147 tris.Add(i + 1);
148 }
149
150 // center tube
151 index = (SubdivisionsU - 1) * SubdivisionsV + 1;
152 for (int v = 0; v < SubdivisionsV - 1; v++)
153 {
154 i = index + v;
155 tris.Add(i);
156 tris.Add(i + SubdivisionsV);
157 tris.Add(i + 1);
158
159 tris.Add(i + 1);
160 tris.Add(i + SubdivisionsV);
161 tris.Add(i + SubdivisionsV + 1);
162 }
163
164 i = index + SubdivisionsV - 1;
165 tris.Add(i);
166 tris.Add(i + SubdivisionsV);
167 tris.Add(i + 1 - SubdivisionsV);
168
169 tris.Add(i + 1 - SubdivisionsV);
170 tris.Add(i + SubdivisionsV);
171 tris.Add(i + 1);
172
173 // bottom hemisphere
174 for (int u = 0; u < SubdivisionsU - 1; u++)
175 {
176 index = u * SubdivisionsV + (SubdivisionsU * SubdivisionsV) + 1;
177 for (int v = 0; v < SubdivisionsV - 1; v++)
178 {
179 i = index + v;
180 tris.Add(i);
181 tris.Add(i + SubdivisionsV);
182 tris.Add(i + 1);
183
184 tris.Add(i + 1);
185 tris.Add(i + SubdivisionsV);
186 tris.Add(i + SubdivisionsV + 1);
187 }
188
189 i = index + SubdivisionsV - 1;
190 tris.Add(i);
191 tris.Add(i + SubdivisionsV);
192 tris.Add(i + 1 - SubdivisionsV);
193
194 tris.Add(i + 1 - SubdivisionsV);
195 tris.Add(i + SubdivisionsV);
196 tris.Add(i + 1);
197 }
198
199 // bottom cap
200 var last = verts.Count - 1;
201 var lastRow = last - SubdivisionsV;
202 for (int v = 0; v < SubdivisionsV; v++)
203 {
204 i = 0;
205 tris.Add(last);
206 tris.Add(lastRow + v + 1);
207 tris.Add(lastRow + v);
208 }
209
210 tris.Add(last);
211 tris.Add(lastRow);
212 tris.Add(last - 1);
213

219 _meshFilter.mesh = new Mesh();
