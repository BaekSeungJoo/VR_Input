103 public static GrabPoseScore CollidersScore(Vector3 position, Collider[] colliders,
104 out Vector3 hitPoint)
105 {
106 GrabPoseScore bestScore = GrabPoseScore.Max;
107 GrabPoseScore score;
108 hitPoint = position;
109 foreach (Collider collider in colliders)
110 {
111 bool isPointInsideCollider = Collisions.IsPointWithinCollider(position, collider);
112 Vector3 measuringPoint = isPointInsideCollider ? collider.bounds.center
113 : collider.ClosestPoint(position);
114
115 score = new GrabPoseScore(position, measuringPoint,
116 isPointInsideCollider);
117
118 if (score.IsBetterThan(bestScore))
119 {
120 hitPoint = isPointInsideCollider ? position : measuringPoint;
121 bestScore = score;
122 }
123 }
124
125 return bestScore;
126 }
