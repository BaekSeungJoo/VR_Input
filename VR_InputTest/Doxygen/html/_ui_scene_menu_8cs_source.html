19 private Scene m_activeScene;

21 private void Awake()
22 {
23 m_activeScene = SceneManager.GetActiveScene();
24
25 // Build labels
26 for (int i = 0; i < SceneManager.sceneCountInBuildSettings; ++i)
27 {
28 string scenePath = SceneUtility.GetScenePathByBuildIndex(i);
29 CreateLabel(i, scenePath);
30 }
31 }
32
33 private void Update()
34 {
35 int sceneCount = SceneManager.sceneCountInBuildSettings;
36 if (InputPrevScene())
37 {
38 ChangeScene((m_activeScene.buildIndex - 1 + sceneCount) % sceneCount);
39 }
40 else if (InputNextScene())
41 {
42 ChangeScene((m_activeScene.buildIndex + 1) % sceneCount);
43 }
44
45 s_lastThumbstickL = OVRInput.Get(OVRInput.Axis2D.PrimaryThumbstick, OVRInput.Controller.LTouch);
46 s_lastThumbstickR = OVRInput.Get(OVRInput.Axis2D.PrimaryThumbstick, OVRInput.Controller.RTouch);
47 }
48
49 private bool InputPrevScene()
50 {
51 return KeyboardPrevScene() || ThumbstickPrevScene(OVRInput.Controller.LTouch) ||
52 ThumbstickPrevScene(OVRInput.Controller.RTouch);
53 }
54
55 private bool InputNextScene()
56 {
57 return KeyboardNextScene() || ThumbstickNextScene(OVRInput.Controller.LTouch) ||
58 ThumbstickNextScene(OVRInput.Controller.RTouch);
59 }
60
61 private bool KeyboardPrevScene()
62 {
63 return Input.GetKeyDown(KeyCode.UpArrow);
64 }
65
66 private bool KeyboardNextScene()
67 {
68 return Input.GetKeyDown(KeyCode.DownArrow);
69 }
70
71 private bool ThumbstickPrevScene(OVRInput.Controller controller)
72 {
73 return (OVRInput.Get(OVRInput.Axis2D.PrimaryThumbstick, controller).y >= 0.9f) &&
74 (GetLastThumbstickValue(controller).y < 0.9f);
75 }
76
77 private bool ThumbstickNextScene(OVRInput.Controller controller)
78 {
79 return (OVRInput.Get(OVRInput.Axis2D.PrimaryThumbstick, controller).y <= -0.9f) &&
80 (GetLastThumbstickValue(controller).y > -0.9f);
81 }
82
83 private Vector2 GetLastThumbstickValue(OVRInput.Controller controller)
84 {
85 return controller == OVRInput.Controller.LTouch ? s_lastThumbstickL : s_lastThumbstickR;
86 }
87
88 private void ChangeScene(int nextScene)
89 {
90 SceneManager.LoadScene(nextScene);
91 }
92
93 private void CreateLabel(int sceneIndex, string scenePath)
94 {
95 // Get the scene name
96 string sceneName = System.IO.Path.GetFileNameWithoutExtension(scenePath);
97
98 // Add spaces after capital letters
99 sceneName = Regex.Replace(sceneName, "[A-Z]", " $0").Trim();
100
101 // Call attention to the active scene
102 bool isActiveScene = m_activeScene.buildIndex == sceneIndex;
103 if (isActiveScene)
104 {
105 sceneName = $"Open: {sceneName}";
106 }
107
108 // Create and set the label
109 TextMeshProUGUI label = GameObject.Instantiate(m_labelPf);
110 label.SetText($"{sceneIndex + 1}. {sceneName}");
111 label.transform.SetParent(m_layoutGroup.transform, false);
112 }
113}
