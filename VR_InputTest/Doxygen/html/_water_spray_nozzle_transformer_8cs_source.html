21using UnityEngine;
22
23namespace Oculus.Interaction.Demo
24{
25 public class WaterSprayNozzleTransformer : MonoBehaviour, ITransformer
26 {
27 [SerializeField]
28 private float _factor = 3f;
29 [SerializeField]
30 private float _snapAngle = 90;
31 [SerializeField]
32 [Range(0f, 1f)]
33 private float _snappiness = 0.8f;
34 [SerializeField]
35 private int _maxSteps = 1;
36
37 private float _relativeAngle = 0f;
38 private int _stepsCount = 0;
39 private IGrabbable _grabbable;
40
41 private Pose _previousGrabPose;
42
43 public void Initialize(IGrabbable grabbable)
44 {
45 _grabbable = grabbable;
46 }
47
48 public void BeginTransform()
49 {
50 _previousGrabPose = _grabbable.GrabPoints[0];
51 _relativeAngle = 0.0f;
52 _stepsCount = 0;
53 }
54
55 public void UpdateTransform()
56 {
57 Pose grabPoint = _grabbable.GrabPoints[0];
58 Transform targetTransform = _grabbable.Transform;
59 Vector3 localAxis = Vector3.forward;
60 Vector3 worldAxis = targetTransform.TransformDirection(localAxis);
61
62 Vector3 initialVector = Vector3.ProjectOnPlane(_previousGrabPose.right, worldAxis).normalized;
63 Vector3 targetVector = Vector3.ProjectOnPlane(grabPoint.right, worldAxis).normalized;
64
65 float angleDelta = Vector3.SignedAngle(initialVector, targetVector, worldAxis) * _factor;
66
67 _relativeAngle += angleDelta;
68
69 if (Mathf.Abs(_relativeAngle) > _snapAngle * (1 - _snappiness)
70 && Mathf.Abs(_stepsCount + Mathf.Sign(_relativeAngle)) <= _maxSteps)
71 {
72 float currentAngle = targetTransform.localEulerAngles.z;
73 int turns = Mathf.FloorToInt((currentAngle + _snappiness * 0.5f) / _snapAngle);
74 float sign = Mathf.Sign(_relativeAngle);
75 float angle = sign > 0 ? _snapAngle * (turns + 1) : _snapAngle * turns;
76 Vector3 rot = targetTransform.localEulerAngles;
77 rot.z = angle;
78 targetTransform.localEulerAngles = rot;
79
80 _relativeAngle = 0;
81 _stepsCount += (int)sign;
82 }
83 else
84 {
85 targetTransform.Rotate(worldAxis, angleDelta, Space.World);
86 }
87
88 _previousGrabPose = grabPoint;
89 }
90
91 public void EndTransform() { }
92 }
93}
