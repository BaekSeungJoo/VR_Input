29using UnityEngine;
30using UnityEditor;
31
32[CustomEditor(typeof(ONSPPropagationGeometry))]
33public class ONSPPropagationGeometryEditor : Editor
34{
35 public override void OnInspectorGUI()
36 {
37 ONSPPropagationGeometry mesh = (ONSPPropagationGeometry)target;
38
39 EditorGUI.BeginChangeCheck();
40
41 bool newIncludeChildMeshes = EditorGUILayout.Toggle( new GUIContent("Include Child Meshes","Include all child meshes into single geometry instance"), mesh.includeChildMeshes );
42
43 Separator();
44
45 #if UNITY_EDITOR
46 string newFilePath = mesh.filePath;
47 bool editedPath = false;
48 bool writeMesh = false;
49 EditorGUI.BeginDisabledGroup( Application.isPlaying );
50 bool newFileEnabled = EditorGUILayout.Toggle( new GUIContent("File Enabled","If set, the serialized mesh file is used as the mesh data source"), mesh.fileEnabled );
51 EditorGUILayout.LabelField( new GUIContent("File Path:","The path to the serialized mesh file, relative to the StreamingAssets directory" ),
52 new GUIContent(mesh.filePathRelative != null ? mesh.filePathRelative : ""));
53
54 EditorGUILayout.BeginHorizontal();
55 EditorGUILayout.PrefixLabel( " " );
56 if ( GUILayout.Button("Set Path") )
57 {
58 if (!System.IO.Directory.Exists(Application.streamingAssetsPath))
59 {
60 System.IO.Directory.CreateDirectory(Application.streamingAssetsPath);
61 }
62
63 string directory = Application.streamingAssetsPath;
64 string fileName = mesh.gameObject.name + "." + ONSPPropagationGeometry.GEOMETRY_FILE_EXTENSION;
65
66 if (newFilePath != "")
67 {
68 directory = System.IO.Path.GetDirectoryName(newFilePath);
69 fileName = System.IO.Path.GetFileName(newFilePath);
70 }
71
72 newFilePath = EditorUtility.SaveFilePanel(
73 "Save baked mesh to file", directory, fileName,
74 ONSPPropagationGeometry.GEOMETRY_FILE_EXTENSION);
75
76 // If the user canceled, use the old path.
77 if ( newFilePath == null || newFilePath.Length == 0 )
78 newFilePath = mesh.filePath;
79 else
80 editedPath = true;
81 }
82
83 if ( GUILayout.Button("Bake Mesh to File") )
84 writeMesh = true;
85
86 EditorGUILayout.EndHorizontal();
87
88#if ENABLE_DEBUG_EXPORT_OBJ
89 // this allows you to export the geometry to a .obj for viewing
90 // in an external model viewer for debugging/validation
91 if ( GUILayout.Button("Write to .obj (debug)") )
92 mesh.WriteToObj();
93#endif
94
95 EditorGUI.EndDisabledGroup();
96
97 #endif
98
99 if ( EditorGUI.EndChangeCheck() )
100 {
101 Undo.RecordObject( mesh, "Edited OVRAudioMesh" );
102
103 mesh.includeChildMeshes = newIncludeChildMeshes;
104 mesh.fileEnabled = newFileEnabled;
105
106 newFilePath = newFilePath.Replace(Application.streamingAssetsPath + "/", "");
107
108 if ( editedPath )
109 mesh.filePathRelative = newFilePath;
110
111 if ( editedPath || writeMesh )
112 mesh.WriteFile();
113 }
114
115 if ( Application.isPlaying && GUILayout.Button("Upload Mesh") )
116 mesh.UploadGeometry();
117 }
118 void Separator()
119 {
120 GUI.color = new Color(1, 1, 1, 0.25f);
121 GUILayout.Box("", "HorizontalSlider", GUILayout.Height(16));
122 GUI.color = Color.white;
123 }
124
125}
