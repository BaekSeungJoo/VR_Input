20namespace Meta.WitAi
21{
22 public static class ThreadUtility
23 {
24 // Default timeout to off
25 public const float THREAD_DEFAULT_TIMEOUT = -1f;
26
27 // Perform in background & return on complete
28 public static ThreadPerformer<T> PerformInBackground<T>(Func<T> workerAction, Action<T, string> onComplete, float timeout = THREAD_DEFAULT_TIMEOUT)
29 {
30 return new ThreadPerformer<T>(workerAction, onComplete, timeout);
31 }
32
33 // Performer
34 public class ThreadPerformer<T>
35 {
39 public bool IsRunning { get; private set; }
40
41 // Complete callback items
42 #if THREADING_ENABLED
43 private Thread _thread;
44 #endif
45 private Func<T> _worker;
46 private Action<T, string> _complete;
47 private float _timeout;
48 private T _result;
49 private string _error;
50 private CoroutineUtility.CoroutinePerformer _coroutine;
51
55 public ThreadPerformer(Func<T> worker, Action<T, string> onComplete, float timeout)
56 {
57 // Begin
58 IsRunning = true;
59
60 // Wait for thread completion
61 _result = default(T);
62 _error = string.Empty;
63 _worker = worker;
64 _complete = onComplete;
65 _timeout = timeout;
66 _coroutine = CoroutineUtility.StartCoroutine(WaitForCompletion(), true);
67
68 // Start thread
69 #if THREADING_ENABLED
70 _thread = new Thread(Work);
71 _thread.Start();
72 #endif
73 }
74
75 // Work
76 private void Work()
77 {
78 // Perform action
79 try
80 {
81 _result = _worker.Invoke();
82 }
83 // Catch exceptions
84 catch (Exception e)
85 {
86 _error = $"Background thread error thrown\n{e}";
87 }
88
89 // Complete
90 IsRunning = false;
91 }
92
93 // Wait for completion
94 private IEnumerator WaitForCompletion()
95 {
96 #if !THREADING_ENABLED
97 yield return null;
98 Work();
99 #endif
100
101 // Wait while running
102 DateTime start = DateTime.Now;
103 while (IsRunning && !IsTimedOut(start))
104 {
105 yield return null;
106 }
107
108 // Timed out
109 if (IsTimedOut(start))
110 {
111 _error = "Timed out";
112 }
113
114 // Complete
115 _complete?.Invoke(_result, _error);
116
117 // Quit
118 Quit();
119 }
120 // Check if timed out
121 private bool IsTimedOut(DateTime start)
122 {
123 // Ignore if no timeout
124 if (_timeout <= 0)
125 {
126 return false;
127 }
128 // Timed out
129 return (DateTime.Now - start).TotalSeconds >= _timeout;
130 }
131
132 // Quit running thread
133 public void Quit()
134 {
135 if (_coroutine != null)
136 {
137 GameObject.DestroyImmediate(_coroutine);
138 _coroutine = null;
139 }
140 #if THREADING_ENABLED
141 if (_thread != null)
142 {
143 if (IsRunning)
144 {
145 _thread.Join();
146 }
147 _thread = null;
148 }
149 #endif
150 if (IsRunning)
151 {
152 IsRunning = false;
153 }
154 }
155 }
156 }
