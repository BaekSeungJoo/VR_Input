56 public void UpdateColor(string[] colorNames, string[] shapes)
57 {
58 if (shapes.Length != 0 && colorNames.Length != shapes.Length)
59 {
60 return;
61 }
62 if (shapes.Length == 0 || shapes[0] == "color"){
63 UpdateColorAllShapes(colorNames);
64 return;
65 }
66
67 for(var entity = 0; entity < colorNames.Length; entity++)
68 {
69 if (!ColorUtility.TryParseHtmlString(colorNames[entity], out var color)) return;
70
71 for (int i = 0; i < transform.childCount; i++)
72 {
73 Transform child = transform.GetChild(i);
74 if (String.Equals(shapes[entity], child.name,
75 StringComparison.CurrentCultureIgnoreCase))
76 {
77 SetColor(child, color);
78 break;
79 }
80 }
81 }
82 }
87 public void UpdateColorAllShapes(string[] colorNames)
88 {
89 var unspecifiedShape = 0;
90 for(var entity = 0; entity < colorNames.Length; entity++)
91 {
92 if (!ColorUtility.TryParseHtmlString(colorNames[entity], out var color)) return;
93
94 var splitLimit = (transform.childCount/colorNames.Length) * (entity+1);
95 while (unspecifiedShape < splitLimit)
96 {
97 SetColor(transform.GetChild(unspecifiedShape), color);
98 unspecifiedShape++;
99 }
100 }
