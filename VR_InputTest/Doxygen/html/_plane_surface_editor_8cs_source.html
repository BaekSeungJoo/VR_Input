30 private const int NUM_SEGMENTS = 40;
31 private const float FADE_DISTANCE = 10f;
32
33 private static readonly Color Color = EditorConstants.PRIMARY_COLOR_DISABLED;
34
35 private static float Interval => FADE_DISTANCE / NUM_SEGMENTS;
36
37 public void OnSceneGUI()
38 {
39 PlaneSurface plane = target as PlaneSurface;
40 Draw(plane);
41 }

43 public static void Draw(PlaneSurface plane)
44 {
45 Vector3 origin = plane.transform.position;
46
47 if (SceneView.lastActiveSceneView?.camera != null)
48 {
49 Transform camTransform = SceneView.lastActiveSceneView.camera.transform;
50 if (plane.ClosestSurfacePoint(camTransform.position, out SurfaceHit hit, 0))
51 {
52 Vector3 hitDelta = PoseUtils.Delta(plane.transform, new Pose(hit.Point, plane.transform.rotation)).position;
53 hitDelta.x = Mathf.RoundToInt(hitDelta.x / Interval) * Interval;
54 hitDelta.y = Mathf.RoundToInt(hitDelta.y / Interval) * Interval;
55 hitDelta.z = 0f;
56 origin = PoseUtils.Multiply(plane.transform.GetPose(), new Pose(hitDelta, Quaternion.identity)).position;
57 }
58 }
59
60 DrawLines(origin, plane.Normal, plane.transform.up, Color);
61 DrawLines(origin, plane.Normal, -plane.transform.up, Color);
62 DrawLines(origin, plane.Normal, plane.transform.right, Color);
63 DrawLines(origin, plane.Normal, -plane.transform.right, Color);
64 }
65
66 private static void DrawLines(in Vector3 origin,
67 in Vector3 normal,
68 in Vector3 direction,
69 in Color baseColor)
70 {
71 Color prevColor = Handles.color;
72 Color color = baseColor;
73 Vector3 offsetOrigin = origin;
74
75 for (int i = 0; i < NUM_SEGMENTS; ++i)
76 {
77 Handles.color = color;
78
79 Vector3 cross = Vector3.Cross(normal, direction).normalized;
80 float interval = Interval;
81
82 for (int j = -NUM_SEGMENTS; j < NUM_SEGMENTS; ++j)
83 {
84 float horizStart = interval * j;
85 float horizEnd = horizStart + interval;
86
87 Vector3 start = offsetOrigin + cross * horizStart;
88 Vector3 end = offsetOrigin + cross * horizEnd;
89
90 color.a = 1f - Mathf.Abs((float)j / NUM_SEGMENTS);
91 color.a *= 1f - ((float)i / NUM_SEGMENTS);
92 color.a *= color.a;
93
94 Handles.color = color;
95 Handles.DrawLine(start, end);
96 }
97
98 offsetOrigin += direction * interval;
99 color = baseColor;
100 }
101
102 Handles.color = prevColor;
103 }
104 }
