71 void Start()

73 // Add a listener to the OVRTouchpad for touch events

90 void HandleKeyboard()
91 {
92 // Turn loopback on/off
93 if (Input.GetKeyDown(loopbackKey))
94 {
95 ToggleAudioLoopback();
96 }
97 else if (Input.GetKeyDown(debugVisemesKey))
98 {
99 showVisemes = !showVisemes;
100
101 if (showVisemes)
102 {
103 if (hasDebugConsole)
104 {
105 Debug.Log("DEBUG SHOW VISEMES: ENABLED");
106 }
107 else
108 {
109 Debug.LogWarning("Warning: No OVRLipSyncDebugConsole in the scene!");
110 showVisemes = false;
111 }
112 }
113 else
114 {
115 if (hasDebugConsole)
116 {
117 OVRLipSyncDebugConsole.Clear();
118 }
119 Debug.Log("DEBUG SHOW VISEMES: DISABLED");
120 }
121 }
122 else if (Input.GetKeyDown(debugLaughterKey))
123 {
124 showLaughter = !showLaughter;
125
126 if (showLaughter)
127 {
128 if (hasDebugConsole)
129 {
130 Debug.Log("DEBUG SHOW LAUGHTER: ENABLED");
131 }
132 else
133 {
134 Debug.LogWarning("Warning: No OVRLipSyncDebugConsole in the scene!");
135 showLaughter = false;
136 }
137 }
138 else
139 {
140 if (hasDebugConsole)
141 {
142 OVRLipSyncDebugConsole.Clear();
143 }
144 Debug.Log("DEBUG SHOW LAUGHTER: DISABLED");
145 }
146 }
147 else if (Input.GetKeyDown(KeyCode.LeftArrow))
148 {
149 gain -= 1.0f;
150 if (gain < 1.0f) gain = 1.0f;
151
152 string g = "LINEAR GAIN: ";
153 g += gain;
154
155 if (hasDebugConsole)
156 {
157 OVRLipSyncDebugConsole.Clear();
158 OVRLipSyncDebugConsole.Log(g);
159 OVRLipSyncDebugConsole.ClearTimeout(1.5f);
160 }
161 }
162 else if (Input.GetKeyDown(KeyCode.RightArrow))
163 {
164 gain += 1.0f;
165 if (gain > 15.0f)
166 gain = 15.0f;
167
168 string g = "LINEAR GAIN: ";
169 g += gain;
170
171 if (hasDebugConsole)
172 {
173 OVRLipSyncDebugConsole.Clear();
174 OVRLipSyncDebugConsole.Log(g);
175 OVRLipSyncDebugConsole.ClearTimeout(1.5f);
176 }
177 }
178 }

198 public void PreprocessAudioSamples(float[] data, int channels)
199 {
200 // Increase the gain of the input
201 for (int i = 0; i < data.Length; ++i)
202 {
203 data[i] = data[i] * gain;
204 }

212 public void PostprocessAudioSamples(float[] data, int channels)
213 {
214 // Turn off output (so that we don't get feedback from mics too close to speakers)
215 if (!audioLoopback)
216 {
217 for (int i = 0; i < data.Length; ++i)
218 data[i] = data[i] * 0.0f;
219 }
220 }

227 public void ProcessAudioSamplesRaw(float[] data, int channels)
228 {
229 // Send data into Phoneme context for processing (if context is not 0)
230 lock (this)

232 if (Context == 0 || OVRLipSync.IsInitialized() != OVRLipSync.Result.Success)

238 }

266 public void ProcessAudioSamples(float[] data, int channels)
267 {
268 // Do not process if we are not initialized, or if there is no
269 // audio source attached to game object
270 if ((OVRLipSync.IsInitialized() != OVRLipSync.Result.Success) || audioSource == null)

295 void DebugShowVisemesAndLaughter()
296 {
297 if (hasDebugConsole)
298 {
299 string seq = "";
300 if (showLaughter)
301 {
302 seq += "Laughter:";
303 int count = (int)(50.0f * this.Frame.laughterScore);
304 for (int c = 0; c < count; c++)
305 seq += "*";
306 seq += "\n";
307 }
308 if (showVisemes)
309 {
310 for (int i = 0; i < this.Frame.Visemes.Length; i++)
311 {
312 seq += ((OVRLipSync.Viseme)i).ToString();
313 seq += ":";
314
315 int count = (int)(50.0f * this.Frame.Visemes[i]);
316 for (int c = 0; c < count; c++)
317 seq += "*";
318
319 seq += "\n";
320 }
321 }
322
323 OVRLipSyncDebugConsole.Clear();
324
325 if (seq != "")
326 {
327 OVRLipSyncDebugConsole.Log(seq);
328 }
329 }

348 // LocalTouchEventCallback
349 void LocalTouchEventCallback(OVRTouchpad.TouchEvent touchEvent)
350 {
351 string g = "LINEAR GAIN: ";
352
353 switch (touchEvent)
354 {
355 case (OVRTouchpad.TouchEvent.SingleTap):
356 ToggleAudioLoopback();
357 break;
358
359 case (OVRTouchpad.TouchEvent.Up):
360 gain += 1.0f;
361 if (gain > 15.0f)
362 gain = 15.0f;
363
364 g += gain;
365
366 if (hasDebugConsole)
367 {
368 OVRLipSyncDebugConsole.Clear();
369 OVRLipSyncDebugConsole.Log(g);
370 OVRLipSyncDebugConsole.ClearTimeout(1.5f);
371 }
372
373 break;
374
375 case (OVRTouchpad.TouchEvent.Down):
376 gain -= 1.0f;
377 if (gain < 1.0f) gain = 1.0f;
378
379 g += gain;
380
381 if (hasDebugConsole)
382 {
383 OVRLipSyncDebugConsole.Clear();
384 OVRLipSyncDebugConsole.Log(g);
385 OVRLipSyncDebugConsole.ClearTimeout(1.5f);
386 }
387
388 break;
389 }
390 }
