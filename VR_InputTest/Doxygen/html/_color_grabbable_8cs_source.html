22using UnityEngine;
23
24namespace OculusSampleFramework
25{
26 // Simple component that changes color based on grab state.
27 public class ColorGrabbable : OVRGrabbable
28 {
29 public static readonly Color COLOR_GRAB = new Color(1.0f, 0.5f, 0.0f, 1.0f);
30 public static readonly Color COLOR_HIGHLIGHT = new Color(1.0f, 0.0f, 1.0f, 1.0f);
31
32 private Color m_color = Color.black;
33 private MeshRenderer[] m_meshRenderers = null;
34 private bool m_highlight;
35
36 public bool Highlight
37 {
38 get { return m_highlight; }
39 set
40 {
41 m_highlight = value;
42 UpdateColor();
43 }
44 }
45
46 protected void UpdateColor()
47 {
48 if (isGrabbed) SetColor(COLOR_GRAB);
49 else if (Highlight) SetColor(COLOR_HIGHLIGHT);
50 else SetColor(m_color);
51 }
52
53 override public void GrabBegin(OVRGrabber hand, Collider grabPoint)
54 {
55 base.GrabBegin(hand, grabPoint);
56 UpdateColor();
57 }
58
59 override public void GrabEnd(Vector3 linearVelocity, Vector3 angularVelocity)
60 {
61 base.GrabEnd(linearVelocity, angularVelocity);
62 UpdateColor();
63 }
64
65 void Awake()
66 {
67 if (m_grabPoints.Length == 0)
68 {
69 // Get the collider from the grabbable
70 Collider collider = this.GetComponent<Collider>();
71 if (collider == null)
72 {
73 throw new System.ArgumentException(
74 "Grabbables cannot have zero grab points and no collider -- please add a grab point or collider.");
75 }
76
77 // Create a default grab point
78 m_grabPoints = new Collider[1] { collider };
79
80 // Grab points are doing double-duty as a way to identify submeshes that should be colored.
81 // If unspecified, just color self.
82 m_meshRenderers = new MeshRenderer[1];
83 m_meshRenderers[0] = this.GetComponent<MeshRenderer>();
84 }
85 else
86 {
87 m_meshRenderers = this.GetComponentsInChildren<MeshRenderer>();
88 }
89
90 m_color = new Color(
91 Random.Range(0.1f, 0.95f),
92 Random.Range(0.1f, 0.95f),
93 Random.Range(0.1f, 0.95f),
94 1.0f
95 );
96 SetColor(m_color);
97 }
98
99 private void SetColor(Color color)
100 {
101 for (int i = 0; i < m_meshRenderers.Length; ++i)
102 {
103 MeshRenderer meshRenderer = m_meshRenderers[i];
104 for (int j = 0; j < meshRenderer.materials.Length; ++j)
105 {
106 Material meshMaterial = meshRenderer.materials[j];
107 meshMaterial.color = color;
108 }
109 }
110 }
111 }
