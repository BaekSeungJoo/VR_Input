31 public struct HandData
32 {
33 private const int NumHandJoints = 24;
34 [MarshalAs(UnmanagedType.ByValArray, SizeConst = NumHandJoints * 4, ArraySubType = UnmanagedType.R4)]
35 private float[] jointValues;
36 private float _rootRotX;
37 private float _rootRotY;
38 private float _rootRotZ;
39 private float _rootRotW;
40 private float _rootPosX;
41 private float _rootPosY;
42 private float _rootPosZ;
43
44 public void Init()
45 {
46 jointValues = new float[NumHandJoints * 4];
47 }
48
49 public void SetData(Quaternion[] joints, Pose root)
50 {
51 Assert.AreEqual(NumHandJoints, joints.Length);
52 for (int jointIndex = 0; jointIndex < NumHandJoints; jointIndex++)
53 {
54 Quaternion joint = joints[jointIndex];
55 int jointValueIndex = jointIndex * 4;
56 jointValues[jointValueIndex + 0] = joint.x;
57 jointValues[jointValueIndex + 1] = joint.y;
58 jointValues[jointValueIndex + 2] = joint.z;
59 jointValues[jointValueIndex + 3] = joint.w;
60 }
61 this._rootRotX = root.rotation.x;
62 this._rootRotY = root.rotation.y;
63 this._rootRotZ = root.rotation.z;
64 this._rootRotW = root.rotation.w;
65 this._rootPosX = root.position.x;
66 this._rootPosY = root.position.y;
67 this._rootPosZ = root.position.z;
68 }
69
70 public void GetData(ref Quaternion[] joints, out Pose root)
71 {
72 Assert.AreEqual(NumHandJoints, joints.Length);
73 for (int jointIndex = 0; jointIndex < NumHandJoints; jointIndex++)
74 {
75 int jointValueIndex = jointIndex * 4;
76 joints[jointIndex].x = jointValues[jointValueIndex + 0];
77 joints[jointIndex].y = jointValues[jointValueIndex + 1];
78 joints[jointIndex].z = jointValues[jointValueIndex + 2];
79 joints[jointIndex].w = jointValues[jointValueIndex + 3];
80 }
81
82 root = new Pose(new Vector3(_rootPosX, _rootPosY, _rootPosZ),
83 new Quaternion(_rootRotX, _rootRotY, _rootRotZ, _rootRotW));
84 }
85 }

275 protected bool UpdateHandData(HandDataAsset handDataAsset)
276 {
277 // null parameters implies don't filter
278 if (_filterParameters == null)
279 return true;
280
281 // pipe data asset into temp struct
282 _handData.SetData(handDataAsset.Joints,handDataAsset.Root);
283
284 // Send it
285 int result = isdk_ExternalHandSource_SetData(_dataSourceHandle, _handData);
286 if (result != _isdkSuccess)
287 {
288 return false;
289 }
290
291 // Update
292 result = isdk_DataSource_Update(_handModifierHandle);
293 if (result != _isdkSuccess)
294 {
295 return false;
296 }
297
298 // Get result
299 result = isdk_DataSource_GetData(_handModifierHandle, ref _handData);
300 if (result != _isdkSuccess)
301 {
302 return false;
303 }
304
305 // Copy results into our hand data asset
306 _handData.GetData(ref handDataAsset.Joints, out handDataAsset.Root);
307
308 return true;
309 }
