124 public override LocomotionTeleport.TeleportIntentions GetIntention()
125 {
126 if (!isActiveAndEnabled)
127 {
128 return global::LocomotionTeleport.TeleportIntentions.None;
129 }
130
131 // If capacitive touch isn't being used, the base implementation will do the work.
132 if (InputMode == InputModes.SeparateButtonsForAimAndTeleport)
133 {
134 return base.GetIntention();
135 }
136
137 // ThumbstickTeleport will begin aiming when the thumbstick is pushed.
138 if (InputMode == InputModes.ThumbstickTeleport || InputMode == InputModes.ThumbstickTeleportForwardBackOnly)
139 {
140 // Note there's a bit of wasted work here if you're only using 1 thumbstick to trigger teleport.
141 // Feel free to delete the extra code for the unnecessary stick.
142 Vector2 leftStick = OVRInput.Get(OVRInput.RawAxis2D.LThumbstick);
143 Vector2 rightStick = OVRInput.Get(OVRInput.RawAxis2D.RThumbstick);
144 float leftMag = 0.0f;
145 float rightMag = 0.0f;
146 float bestMag = 0.0f;
147 OVRInput.Controller bestController = OVRInput.Controller.Touch;
148 bool leftTouched = OVRInput.Get(OVRInput.RawTouch.LThumbstick);
149 bool rightTouched = OVRInput.Get(OVRInput.RawTouch.RThumbstick);
150
151 if (InputMode == InputModes.ThumbstickTeleportForwardBackOnly &&
152 LocomotionTeleport.CurrentIntention != LocomotionTeleport.TeleportIntentions.Aim)
153 {
154 // If user is aiming, ThumbstickTeleport and ThumbstickTeleportForwardBackOnly are identical. But if not, we only want magnitude along the forward or back vector.
155 leftMag = Mathf.Abs(Vector2.Dot(leftStick, Vector2.up));
156 rightMag = Mathf.Abs(Vector2.Dot(rightStick, Vector2.up));
157 }
158 else
159 {
160 leftMag = leftStick.magnitude;
161 rightMag = rightStick.magnitude;
162 }
163
164 if (AimingController == OVRInput.Controller.LTouch)
165 {
166 bestMag = leftMag;
167 bestController = OVRInput.Controller.LTouch;
168 }
169 else if (AimingController == OVRInput.Controller.RTouch)
170 {
171 bestMag = rightMag;
172 bestController = OVRInput.Controller.RTouch;
173 }
174 else
175 {
176 if (leftMag > rightMag)
177 {
178 bestMag = leftMag;
179 bestController = OVRInput.Controller.LTouch;
180 }
181 else
182 {
183 bestMag = rightMag;
184 bestController = OVRInput.Controller.RTouch;
185 }
186 }
187
188 bool touching = bestMag > ThumbstickTeleportThreshold
189 || (AimingController == OVRInput.Controller.Touch && (leftTouched || rightTouched))
190 || (AimingController == OVRInput.Controller.LTouch && leftTouched)
191 || (AimingController == OVRInput.Controller.RTouch && rightTouched);
192 if (!touching)
193 {
194 if (LocomotionTeleport.CurrentIntention == LocomotionTeleport.TeleportIntentions.Aim)
195 {
196 // If the user has released the thumbstick, enter the PreTeleport state unless FastTeleport is enabled,
197 // in which case enter the Teleport state.
198 return FastTeleport
199 ? LocomotionTeleport.TeleportIntentions.Teleport
200 : LocomotionTeleport.TeleportIntentions.PreTeleport;
201 }
202
203 // If the user is already in the preteleport state, the intention will be to either remain in this state or switch to Teleport
204 if (LocomotionTeleport.CurrentIntention == LocomotionTeleport.TeleportIntentions.PreTeleport)
205 {
206 return LocomotionTeleport.TeleportIntentions.Teleport;
207 }
208 }
209 else
210 {
211 if (LocomotionTeleport.CurrentIntention == LocomotionTeleport.TeleportIntentions.Aim)
212 {
213 return LocomotionTeleport.TeleportIntentions.Aim;
214 }
215 }
216
217 if (bestMag > ThumbstickTeleportThreshold)
218 {
219 InitiatingController = bestController;
220 return LocomotionTeleport.TeleportIntentions.Aim;
221 }
222
223 return LocomotionTeleport.TeleportIntentions.None;
224 }
225
226 // Capacitive touch logic is essentially the same as the base logic, except the button types are different
227 // so different methods need to be used.
228 var teleportButton = _rawButtons[(int)CapacitiveAimAndTeleportButton];
229
230 if (LocomotionTeleport.CurrentIntention == LocomotionTeleport.TeleportIntentions.Aim)
231 {
232 // If the user has actually pressed the teleport button, enter the preteleport state.
233 if (OVRInput.GetDown(teleportButton))
234 {
235 // If the user has released the thumbstick, enter the PreTeleport state unless FastTeleport is enabled,
236 // in which case enter the Teleport state.
237 return FastTeleport
238 ? LocomotionTeleport.TeleportIntentions.Teleport
239 : LocomotionTeleport.TeleportIntentions.PreTeleport;
240 }
241 }
242
243 // If the user is already in the PreTeleport state, the intention will be to either remain in this state or switch to Teleport
244 if (LocomotionTeleport.CurrentIntention == LocomotionTeleport.TeleportIntentions.PreTeleport)
245 {
246 // If they released the button, switch to Teleport.
247 if (FastTeleport || OVRInput.GetUp(teleportButton))
248 {
249 // Button released, enter the Teleport state.
250 return LocomotionTeleport.TeleportIntentions.Teleport;
251 }
252
253 // Button still down, remain in PreTeleport so they can orient the destination if an orientation handler supports it.
254 return LocomotionTeleport.TeleportIntentions.PreTeleport;
255 }
256
257 // If it made it this far, then we need to determine if the user intends to be aiming with the capacitive touch.
258 // The first check is if cap touch has been triggered.
259 if (OVRInput.GetDown(_rawTouch[(int)CapacitiveAimAndTeleportButton]))
260 {
261 return LocomotionTeleport.TeleportIntentions.Aim;
262 }
263
264 if (LocomotionTeleport.CurrentIntention == LocomotionTeleport.TeleportIntentions.Aim)
265 {
266 if (!OVRInput.GetUp(_rawTouch[(int)CapacitiveAimAndTeleportButton]))
267 {
268 return LocomotionTeleport.TeleportIntentions.Aim;
269 }
270 }
271
272 return LocomotionTeleport.TeleportIntentions.None;
273 }

275 public override void GetAimData(out Ray aimRay)
276 {
277 OVRInput.Controller sourceController = AimingController;
278 if (sourceController == OVRInput.Controller.Touch)
279 {
280 sourceController = InitiatingController;
281 }
282
283 Transform t = (sourceController == OVRInput.Controller.LTouch) ? LeftHand : RightHand;
284 aimRay = new Ray(t.position, t.forward);
285 }
