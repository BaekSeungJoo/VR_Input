91 // VR controllers: primary buttons scrolls forward, secondary backwards
92#if UNITY_2019_1_OR_NEWER
93 var inputDevices = new List<InputDevice>();
94#if UNITY_2019_3_OR_NEWER
95 InputDevices.GetDevicesWithCharacteristics(InputDeviceCharacteristics.HeldInHand, inputDevices);
96#else
97 InputDevices.GetDevicesWithRole(InputDeviceRole.RightHanded, inputDevices);
98#endif
99 var primaryButtonPressed = false;
100 var secondaryButtonPressed = false;
101 foreach (var device in inputDevices)
102 {
103 bool boolValue;
104 if (device.TryGetFeatureValue(CommonUsages.primaryButton, out boolValue) && boolValue)
105 {
106 primaryButtonPressed = true;
107 }
108 if (device.TryGetFeatureValue(CommonUsages.secondaryButton, out boolValue) && boolValue)
109 {
110 secondaryButtonPressed = true;
111 }
112 }
113 if (primaryButtonPressed && !XRButtonBeingPressed)
114 {
115 targetSet++;
116 if (targetSet >= maxTarget)
117 {
118 targetSet = 0;
119 }
120 SetCurrentTarget();
121 }
122 if (secondaryButtonPressed && !XRButtonBeingPressed)
123 {
124 targetSet--;
125 if (targetSet < 0)
126 {
127 targetSet = maxTarget - 1;
128 }
129 SetCurrentTarget();
130 }
131 XRButtonBeingPressed = primaryButtonPressed || secondaryButtonPressed;
132#endif
133 }

138 void SetCurrentTarget()
139 {
140 switch(targetSet)
141 {
142 case(0):
143 SwitchTargets[0].SetActive<OVRLipSyncContextMorphTarget>(0);
144 SwitchTargets[1].SetActive<OVRLipSyncContextMorphTarget>(0);
145 break;
146 case(1):
147 SwitchTargets[0].SetActive<OVRLipSyncContextTextureFlip>(0);
148 SwitchTargets[1].SetActive<OVRLipSyncContextTextureFlip>(1);
149 break;
150 case(2):
151 SwitchTargets[0].SetActive<OVRLipSyncContextMorphTarget>(1);
152 SwitchTargets[1].SetActive<OVRLipSyncContextMorphTarget>(2);
153 break;
154 case(3):
155 SwitchTargets[0].SetActive<OVRLipSyncContextTextureFlip>(1);
156 SwitchTargets[1].SetActive<OVRLipSyncContextTextureFlip>(3);
157 break;
158 case(4):
159 SwitchTargets[0].SetActive<OVRLipSyncContextMorphTarget>(2);
160 SwitchTargets[1].SetActive<OVRLipSyncContextMorphTarget>(4);
161 break;
162 case(5):
163 SwitchTargets[0].SetActive<OVRLipSyncContextTextureFlip>(2);
164 SwitchTargets[1].SetActive<OVRLipSyncContextTextureFlip>(5);
165 break;
166 }
167 OVRLipSyncDebugConsole.Clear();
168 }

174 void LocalTouchEventCallback(OVRTouchpad.TouchEvent touchEvent)
175 {
176 switch(touchEvent)
177 {
178 case(OVRTouchpad.TouchEvent.Left):
179
180 targetSet--;
181 if(targetSet < 0)
182 targetSet = maxTarget - 1;
183
184 SetCurrentTarget();
185
186 break;
187
188 case(OVRTouchpad.TouchEvent.Right):
189
190 targetSet++;
191 if(targetSet >= maxTarget)
192 targetSet = 0;
193
194 SetCurrentTarget();
195
196 break;
197 }
198 }
