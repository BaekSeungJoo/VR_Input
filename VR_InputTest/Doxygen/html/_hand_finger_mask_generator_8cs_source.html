51 private static float HandednessMultiplier(Handedness hand) =>
52 hand != Handedness.Right ? -1.0f : 1.0f;

54 private static List<Vector2> GenerateModelUV(Handedness hand, Mesh sharedHandMesh,
55 out Vector2 minPosition,
56 out Vector2 maxPosition)
57 {
58 List<Vector3> mVertices = new List<Vector3>();
59 sharedHandMesh.GetVertices(mVertices);
60 minPosition = new Vector2(mVertices[0].x, mVertices[0].z);
61 maxPosition = new Vector2(mVertices[0].x, mVertices[0].z);
62 for (int i = 0; i < mVertices.Count; i++)
63 {
64 var vertex = mVertices[i] * HandednessMultiplier(hand);
65 var vertex2d = new Vector2(vertex.x, vertex.z);
66 minPosition = Vector2.Min(minPosition, vertex2d);
67 maxPosition = Vector2.Max(maxPosition, vertex2d);
68 mVertices[i] = vertex;
69 }
70
71 List<Vector2> mUVs = new List<Vector2>();
72 Vector2 regionSize = maxPosition - minPosition;
73 float maxLength = Mathf.Max(regionSize.x, regionSize.y);
74 foreach (var vertex in mVertices)
75 {
76 var vertex2d = new Vector2(vertex.x, vertex.z);
77 var vertexUV = (vertex2d - minPosition) / maxLength;
78 mUVs.Add(vertexUV);
79 }
80
81 return mUVs;
82 }

84 private static Vector2 getPositionOnRegion(HandVisual handVisual, HandJointId jointId,
85 Vector2 minRegion,
86 float sideLength)
87 {
88 var lineStartPose = handVisual.GetJointPose(jointId, Space.World);
89 var lineStartLocalPosition =
90 handVisual.transform.InverseTransformPoint(lineStartPose.position);
91 Vector2 point = new Vector2(lineStartLocalPosition.x, lineStartLocalPosition.z);
92 point *= HandednessMultiplier(handVisual.Hand.Handedness);
93 return (point - minRegion) / sideLength;
94 }

96 private static Vector4 GenerateLineData(HandVisual handVisual, HandJointId jointIdStart,
97 HandJointId jointIdEnd,
98 Vector2 minRegion, float sideLength, float lineScale)
99 {
100 Vector2 startPosition =
101 getPositionOnRegion(handVisual, jointIdStart, minRegion, sideLength);
102 Vector2 endPosition =
103 getPositionOnRegion(handVisual, jointIdEnd, minRegion, sideLength);
104 endPosition = Vector2.LerpUnclamped(startPosition, endPosition, lineScale);
105 return new Vector4(startPosition.x, startPosition.y, endPosition.x, endPosition.y);
106 }

108 private static Vector4[] GenerateFingerLines(HandVisual handVisual, Vector2 minPosition,
109 float maxLength, float[] lineScale)
110 {
111 Vector4 thumbLine = GenerateLineData(handVisual, HandJointId.HandThumbTip,
112 HandJointId.HandThumb1, minPosition, maxLength, lineScale[0]);
113 Vector4 indexLine = GenerateLineData(handVisual, HandJointId.HandIndexTip,
114 HandJointId.HandIndex1, minPosition, maxLength, lineScale[1]);
115 Vector4 middleLine = GenerateLineData(handVisual, HandJointId.HandMiddleTip,
116 HandJointId.HandMiddle1, minPosition, maxLength, lineScale[2]);
117 Vector4 ringLine = GenerateLineData(handVisual, HandJointId.HandRingTip,
118 HandJointId.HandRing1, minPosition, maxLength, lineScale[3]);
119 Vector4 pinkyLine = GenerateLineData(handVisual, HandJointId.HandPinkyTip,
120 HandJointId.HandPinky1, minPosition, maxLength, lineScale[4]);
121
122 return new Vector4[5] { thumbLine, indexLine, middleLine, ringLine, pinkyLine };
123 }

138 private static void SetFingerMaskUniforms(HandVisual handVisual, MaterialPropertyBlock materialPropertyBlock, Vector2 minPosition, Vector2 maxPosition)
139 {
140 Vector2 regionSize = maxPosition - minPosition;
141 float maxLength = Mathf.Max(regionSize.x, regionSize.y);
142
143 //The following numbers generate good looking lines for the current hand model
144 var fingerLineScales = new float[5] { 0.9f, 0.91f, 0.9f, 0.87f, 0.87f };
145 var fingerLines = GenerateFingerLines(handVisual, minPosition, maxLength, fingerLineScales);
146 //The thumb line is going to be perpendicularly aligned to the wrist direction
147 fingerLines[0].z = Mathf.Lerp(fingerLines[0].z, fingerLines[0].x, 0.3f);
148 fingerLines[0].x = fingerLines[0].z;
149
150 var palmFingerLineScales = new float[5] { 1.2f, 1.25f, 1.25f, 1.25f, 1.25f };
151 var palmFingerLines =
152 GenerateFingerLines(handVisual, minPosition, maxLength, palmFingerLineScales);
153 //The thumb line is going to be perpendicularly aligned to the wrist direction
154 float thumbOffset = Mathf.Abs(palmFingerLines[0].x - palmFingerLines[0].z) * 0.1f;
155 palmFingerLines[0].z += thumbOffset;
156
157 for (int i = 0; i < 5; i++)
158 {
159 materialPropertyBlock.SetVector(_fingerLinesID[i], fingerLines[i]);
160 materialPropertyBlock.SetVector(_palmFingerLinesID[i], palmFingerLines[i]);
161 }
162 }

164 public static void GenerateFingerMask(SkinnedMeshRenderer handRenderer, HandVisual handVisual, MaterialPropertyBlock materialPropertyBlock)
165 {
166 SetGlowModelUV(handRenderer, handVisual, out Vector2 minPosition,
167 out Vector2 maxPosition);
168 SetFingerMaskUniforms(handVisual, materialPropertyBlock, minPosition, maxPosition);
