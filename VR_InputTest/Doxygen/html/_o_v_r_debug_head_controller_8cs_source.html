79 void Awake()
80 {
81 // locate the camera rig so we can use it to get the current camera transform each frame
82 OVRCameraRig[] CameraRigs = gameObject.GetComponentsInChildren<OVRCameraRig>();
83
84 if (CameraRigs.Length == 0)
85 Debug.LogWarning("OVRCamParent: No OVRCameraRig attached.");
86 else if (CameraRigs.Length > 1)
87 Debug.LogWarning("OVRCamParent: More then 1 OVRCameraRig attached.");
88 else
89 CameraRig = CameraRigs[0];
90 }

92 // Use this for initialization
93 void Start()

97 // Update is called once per frame
98 void Update()
99 {
100 if (AllowMovement)
101 {
102 float gamePad_FwdAxis = OVRInput.Get(OVRInput.RawAxis2D.LThumbstick).y;
103 float gamePad_StrafeAxis = OVRInput.Get(OVRInput.RawAxis2D.LThumbstick).x;
104
105 Vector3 fwdMove = (CameraRig.centerEyeAnchor.rotation * Vector3.forward) * gamePad_FwdAxis *
106 Time.deltaTime * ForwardSpeed;
107 Vector3 strafeMove = (CameraRig.centerEyeAnchor.rotation * Vector3.right) * gamePad_StrafeAxis *
108 Time.deltaTime * StrafeSpeed;
109 transform.position += fwdMove + strafeMove;
110 }
111
112 bool hasDevice = false;
113#if USING_XR_SDK
114 XRDisplaySubsystem currentDisplaySubsystem = OVRManager.GetCurrentDisplaySubsystem();
115 if (currentDisplaySubsystem != null)
116 hasDevice = currentDisplaySubsystem.running;
117#elif REQUIRES_XR_SDK
118 hasDevice = false;
119#else
120 hasDevice = UnityEngine.XR.XRDevice.isPresent;
121#endif
122
123 if (!hasDevice && (AllowYawLook || AllowPitchLook))
124 {
125 Quaternion r = transform.rotation;
126 if (AllowYawLook)
127 {
128 float gamePadYaw = OVRInput.Get(OVRInput.RawAxis2D.RThumbstick).x;
129 float yawAmount = gamePadYaw * Time.deltaTime * GamePad_YawDegreesPerSec;
130 Quaternion yawRot = Quaternion.AngleAxis(yawAmount, Vector3.up);
131 r = yawRot * r;
132 }
133
134 if (AllowPitchLook)
135 {
136 float gamePadPitch = OVRInput.Get(OVRInput.RawAxis2D.RThumbstick).y;
137 if (Mathf.Abs(gamePadPitch) > 0.0001f)
138 {
139 if (InvertPitch)
140 {
141 gamePadPitch *= -1.0f;
142 }
143
144 float pitchAmount = gamePadPitch * Time.deltaTime * GamePad_PitchDegreesPerSec;
145 Quaternion pitchRot = Quaternion.AngleAxis(pitchAmount, Vector3.left);
146 r = r * pitchRot;
147 }
148 }
149
150 transform.rotation = r;
151 }
152 }
153}
