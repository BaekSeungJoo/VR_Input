65 public static float GetMaximumBoundaryDistance(Camera camera, OVRBoundary.BoundaryType boundaryType)
66 {
67 if (!OVRManager.boundary.GetConfigured())
68 {
69 return float.MaxValue;
70 }
71
72 Vector3[] geometry = OVRManager.boundary.GetGeometry(boundaryType);
73 if (geometry.Length == 0)
74 {
75 return float.MaxValue;
76 }
77
78 float maxDistance = -float.MaxValue;
79 foreach (Vector3 v in geometry)
80 {
81 Vector3 pos = GetWorldPosition(camera, v);
82 float distance = Vector3.Dot(camera.transform.forward, pos);
83 if (maxDistance < distance)
84 {
85 maxDistance = distance;
86 }
87 }
88
89 return maxDistance;
90 }

92 public static Mesh BuildBoundaryMesh(OVRBoundary.BoundaryType boundaryType, float topY, float bottomY)
93 {
94 if (!OVRManager.boundary.GetConfigured())
95 {
96 return null;
97 }
98
99 List<Vector3> geometry = new List<Vector3>(OVRManager.boundary.GetGeometry(boundaryType));
100 if (geometry.Count == 0)
101 {
102 return null;
103 }
104
105 geometry.Add(geometry[0]);
106 int numPoints = geometry.Count;
107
108 Vector3[] vertices = new Vector3[numPoints * 2];
109 Vector2[] uvs = new Vector2[numPoints * 2];
110 for (int i = 0; i < numPoints; ++i)
111 {
112 Vector3 v = geometry[i];
113 vertices[i] = new Vector3(v.x, bottomY, v.z);
114 vertices[i + numPoints] = new Vector3(v.x, topY, v.z);
115 uvs[i] = new Vector2((float)i / (numPoints - 1), 0.0f);
116 uvs[i + numPoints] = new Vector2(uvs[i].x, 1.0f);
117 }
118
119 int[] triangles = new int[(numPoints - 1) * 2 * 3];
120 for (int i = 0; i < numPoints - 1; ++i)
121 {
122 // the geometry is built clockwised. only the back faces should be rendered in the camera frame mask
123
124 triangles[i * 6 + 0] = i;
125 triangles[i * 6 + 1] = i + numPoints;
126 triangles[i * 6 + 2] = i + 1 + numPoints;
127
128 triangles[i * 6 + 3] = i;
129 triangles[i * 6 + 4] = i + 1 + numPoints;
130 triangles[i * 6 + 5] = i + 1;
131 }
132
133 Mesh mesh = new Mesh();
134 mesh.vertices = vertices;
135 mesh.uv = uvs;
136 mesh.triangles = triangles;
137 return mesh;
138 }
