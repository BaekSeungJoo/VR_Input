32 [Tooltip("The renderer used for the color change.")]

40
41 [Tooltip("The renderer will be set to this color " +

45
46 private Material _material;
47 private bool _lastActiveValue = false;
48
49 protected virtual void Awake()
50 {
51 ActiveState = _activeState as IActiveState;
52 this.AssertField(ActiveState, nameof(ActiveState));
53 this.AssertField(_target, nameof(_target));
54 _material = _target.material;
55
56 SetMaterialColor(_lastActiveValue ? _activeColor : _normalColor);
57 }
58
59 private void OnDestroy()
60 {
61 Destroy(_material);
62 }
63
64 protected virtual void Update()
65 {
66 bool isActive = ActiveState.Active;
67 if (_lastActiveValue != isActive)
68 {
69 SetMaterialColor(isActive ? _activeColor : _normalColor);
70 _lastActiveValue = isActive;
71 }
72 }
73
74 private void SetMaterialColor(Color activeColor)
75 {
76 _material.color = activeColor;
77 _target.enabled = _material.color.a > 0.0f;
