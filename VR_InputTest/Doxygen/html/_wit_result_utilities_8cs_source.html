337 public static string GetPathValue(this WitResponseNode response, string path)
338 {
339
340 string[] nodes = path.Trim('.').Split('.');
341
342 var node = response;
343
344 foreach (var nodeName in nodes)
345 {
346 string[] arrayElements = SplitArrays(nodeName);
347
348 node = node[arrayElements[0]];
349 for (int i = 1; i < arrayElements.Length; i++)
350 {
351 node = node[int.Parse(arrayElements[i])];
352 }
353 }
354
355 return node.Value;
356 }
357 public static void SetString(this WitResponseNode response, string path, string value)
358 {
359
360 string[] nodes = path.Trim('.').Split('.');
361
362 var node = response;
363 int nodeIndex;
364
365 for(nodeIndex = 0; nodeIndex < nodes.Length - 1; nodeIndex++)
366 {
367 var nodeName = nodes[nodeIndex];
368 string[] arrayElements = SplitArrays(nodeName);
369
370 node = node[arrayElements[0]];
371 for (int i = 1; i < arrayElements.Length; i++)
372 {
373 node = node[int.Parse(arrayElements[i])];
374 }
375 }
376
377
378 node[nodes[nodeIndex]] = value;
379 }
380 public static void RemovePath(this WitResponseNode response, string path)
381 {
382 string[] nodes = path.Trim('.').Split('.');
383
384 var node = response;
385 WitResponseNode parent = null;
386
387 foreach (var nodeName in nodes)
388 {
389 string[] arrayElements = SplitArrays(nodeName);
390
391 parent = node;
392 node = node[arrayElements[0]];
393 for (int i = 1; i < arrayElements.Length; i++)
394 {
395 node = node[int.Parse(arrayElements[i])];
396 }
397 }
398
399 if (null != parent) parent.Remove(node);
400 }

402 public static WitResponseReference GetWitResponseReference(string path)
403 {
404
405 string[] nodes = path.Trim('.').Split('.');
406
407 var rootNode = new WitResponseReference()
408 {
409 path = path
410 };
411 var node = rootNode;
412
413 foreach (var nodeName in nodes)
414 {
415 string[] arrayElements = SplitArrays(nodeName);
416
417 var childObject = new ObjectNodeReference()
418 {
419 path = path
420 };
421 childObject.key = arrayElements[0];
422 node.child = childObject;
423 node = childObject;
424 for (int i = 1; i < arrayElements.Length; i++)
425 {
426 var childIndex = new ArrayNodeReference()
427 {
428 path = path
429 };
430 childIndex.index = int.Parse(arrayElements[i]);
431 node.child = childIndex;
432 node = childIndex;
433 }
434 }
435
436 return rootNode;
437 }

439 public static string GetCodeFromPath(string path)
440 {
441 string[] nodes = path.Trim('.').Split('.');
442 string code = "witResponse";
443 foreach (var nodeName in nodes)
444 {
445 string[] arrayElements = SplitArrays(nodeName);
446
447 code += $"[\"{arrayElements[0]}\"]";
448 for (int i = 1; i < arrayElements.Length; i++)
449 {
450 code += $"[{arrayElements[i]}]";
451 }
452 }
453
454 code += ".Value";
455 return code;

458 private static string[] SplitArrays(string nodeName)
459 {
460 var nodes = nodeName.Split('[');
461 for (int i = 0; i < nodes.Length; i++)
462 {
463 nodes[i] = nodes[i].Trim(']');
464 }
