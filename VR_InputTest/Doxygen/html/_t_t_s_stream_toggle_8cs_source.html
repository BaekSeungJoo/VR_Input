9using UnityEngine;
10using UnityEngine.UI;
11using Meta.WitAi.TTS.Integrations;

13namespace Meta.WitAi.TTS.Samples
14{
15 public class TTSStreamToggle : MonoBehaviour
16 {
17 // UI references
18 [SerializeField] private TTSWit _service;
19 [SerializeField] private Text _label;
20 [SerializeField] private Button _button;
21
22 // Current stream
23 private bool _streamEnabled = true;
24
25 // Add listeners
26 private void OnEnable()
27 {
28 // Obtain disk cache if possible
29 if (_service == null)
30 {
31 _service = GameObject.FindObjectOfType<TTSWit>();
32 }
33 // Log for missing service
34 if (_service == null)
35 {
36 VLog.E("TTS Stream Toggle - Cannot work without a TTSWit reference");
37 }
38 // Reset
39 RefreshStreaming();
40 _button.onClick.AddListener(ToggleStreaming);
41 }
42 // Remove listeners
43 private void OnDisable()
44 {
45 _button.onClick.RemoveListener(ToggleStreaming);
46 }
47
48 // Refresh location & button text
49 private void RefreshStreaming()
50 {
51 _streamEnabled = GetStreaming();
52 _label.text = $"Streaming: {(_streamEnabled ? "ON" : "OFF")}";
53 }
54 // Toggle streaming
55 public void ToggleStreaming()
56 {
57 SetStreaming(!_streamEnabled);
58 RefreshStreaming();
59 }
60
61 // Get streaming option from service
62 private bool GetStreaming()
63 {
64 return _service && _service.RequestSettings.audioStream;
65 }
66 // Set streaming option to
67 private void SetStreaming(bool toStreaming)
68 {
69 if (_service != null)
70 {
71 _service.RequestSettings.audioStream = toStreaming;
72 _service.UnloadAll();
73 }
74 }
75 // Update if changed externally
76 private void Update()
77 {
78 if (_streamEnabled != GetStreaming())
79 {
80 RefreshStreaming();
81 }
82 }
83 }
84}
