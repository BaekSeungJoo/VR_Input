98 private string GenerateManifest(IEnumerable<IConduitAssembly> assemblies, string domain, string id)
99 {
100 VLog.D($"Generating manifest.");
101
102 var (entities, actions, errorHandlers) = ExtractAssemblyData(assemblies);
103
104 var manifest = new Manifest()
105 {
106 ID = id,
107 Version = CurrentVersion,
108 Domain = domain,
109 Entities = entities,
110 Actions = actions,
111 ErrorHandlers = errorHandlers
112 };
113
114 return JsonConvert.SerializeObject(manifest);
115 }
116
117 private (List<ManifestEntity>, List<ManifestAction>, List<ManifestErrorHandler>) ExtractAssemblyData(IEnumerable<IConduitAssembly> assemblies)
118 {
119 var entities = new List<ManifestEntity>();
120 var actions = new List<ManifestAction>();
121 var errorHandlers = new List<ManifestErrorHandler>();
122 _assemblyMiner.Initialize();
123 foreach (var assembly in assemblies)
124 {
125 actions.AddRange(this._assemblyMiner.ExtractActions(assembly));
126 entities.AddRange(this._assemblyMiner.ExtractEntities(assembly));
127 try
128 {
129 errorHandlers.AddRange(this._assemblyMiner.ExtractErrorHandlers(assembly));
130 }
131 catch (Exception)
132 {
133 VLog.W("Conduit App found no error handlers");
134 }
135 }
136
137 this.PruneUnreferencedEntities(ref entities, actions);
138
139 return (entities, actions, errorHandlers);
140 }
141
147 private IEnumerable<Assembly> GetTargetAssemblies()
148 {
149 return AppDomain.CurrentDomain.GetAssemblies().Where(assembly => assembly.IsDefined(typeof(ConduitAssemblyAttribute)));
150 }
151
157 private void PruneUnreferencedEntities(ref List<ManifestEntity> entities, List<ManifestAction> actions)
158 {
159 var referencedEntities = new HashSet<string>();
160
161 foreach (var action in actions)
162 {
163 foreach (var parameter in action.Parameters)
164 {
165 referencedEntities.Add(parameter.EntityType);
166 }
167 }
168
169 for (var i = 0; i < entities.Count; ++i)
170 {
171 if (referencedEntities.Contains(entities[i].ID))
172 {
173 continue;
174 }
175
176 entities.RemoveAt(i--);
177 }
178 }
179 }
