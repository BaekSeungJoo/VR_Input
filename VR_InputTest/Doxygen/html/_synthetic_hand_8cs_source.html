96 protected override void Apply(HandDataAsset data)
97 {
98 if (!Started || !data.IsDataValid || !data.IsTracked || !data.IsHighConfidence)
99 {
100 data.IsConnected = false;
101 data.RootPoseOrigin = PoseOrigin.None;
102 _hasConnectedData = false;
103 return;
104 }
105
106 UpdateRequired.Invoke();
107 _lastStates.CopyFrom(data);
108
109 if (!_hasConnectedData)
110 {
111 _constrainedWristPose.CopyFrom(data.Root);
112 _hasConnectedData = true;
113 }
114
115 UpdateJointsRotation(data);
116 UpdateRootPose(ref data.Root);
117
118 data.RootPoseOrigin = PoseOrigin.SyntheticPose;
119 }

132 private void UpdateRootPose(ref Pose root)
133 {
134 float smoothPositionFactor = _wristPositionLocked ? _wristPositionLockCurve.Progress() : _wristPositionUnlockCurve.Progress();
135 Vector3 position = Vector3.Lerp(root.position, _desiredWristPose.position, _wristPositionOverrideFactor);
136 root.position = Vector3.Lerp(_constrainedWristPose.position, position, smoothPositionFactor);
137
138 float smoothRotationFactor = _wristRotationLocked ? _wristRotationLockCurve.Progress() : _wristRotationUnlockCurve.Progress();
139 Quaternion rotation = Quaternion.Lerp(root.rotation, _desiredWristPose.rotation, _wristRotationOverrideFactor);
140 root.rotation = Quaternion.Lerp(_constrainedWristPose.rotation, rotation, smoothRotationFactor);
141
142 _lastWristPose.CopyFrom(root);
143 }

158 private void UpdateJointsRotation(HandDataAsset data)
159 {
160 float extraRotationAllowance = 0f;
161 Quaternion[] jointRotations = data.Joints;
162
163 for (int i = 0; i < FingersMetadata.HAND_JOINT_IDS.Length; ++i)
164 {
165 JointFreedom freedomLevel = _jointsFreedomLevels[i];
166 Quaternion desiredRotation = _desiredJointsRotation[i];
167 float overrideFactor = _jointsOverrideFactor[i];
168 int rawJointIndex = (int)FingersMetadata.HAND_JOINT_IDS[i];
169
170 if (freedomLevel == JointFreedom.Free)
171 {
172 //nothing to do, we move the finger freely
173 }
174 else if (freedomLevel == JointFreedom.Locked)
175 {
176 jointRotations[rawJointIndex] = Quaternion.Slerp(
177 jointRotations[rawJointIndex],
178 desiredRotation,
179 overrideFactor);
180 }
181 else if (freedomLevel == JointFreedom.Constrained)
182 {
183 bool jointCanSpread = false;
184 if (FingersMetadata.HAND_JOINT_CAN_SPREAD[i])
185 {
186 jointCanSpread = true;
187 extraRotationAllowance = 0f;
188 }
189
190 Quaternion maxRotation = desiredRotation * Quaternion.Euler(0f, 0f, -90f * extraRotationAllowance);
191
192 float overRotation = OverFlex(jointRotations[rawJointIndex], maxRotation);
193 extraRotationAllowance = Mathf.Max(extraRotationAllowance, overRotation);
194
195 if (overRotation < 0f)
196 {
197 jointRotations[rawJointIndex] = Quaternion.Slerp(
198 jointRotations[rawJointIndex],
199 maxRotation,
200 overrideFactor);
201 }
202 else if (jointCanSpread)
203 {
204 Quaternion trackedRotation = jointRotations[rawJointIndex];
205 float spreadAngle = Vector3.SignedAngle(
206 trackedRotation * Vector3.forward,
207 maxRotation * Vector3.forward,
208 trackedRotation * Vector3.up);
209
210 float spreadFactor = 1f - Mathf.Clamp01(overRotation * _spreadAllowance);
211 trackedRotation = trackedRotation * Quaternion.Euler(0f, spreadAngle * spreadFactor, 0f);
212 jointRotations[rawJointIndex] = trackedRotation;
213 }
214 }
215
216 float smoothFactor = _jointsFreedomLevels[i] == JointFreedom.Free ?
217 _jointUnlockProgressCurves[i].Progress()
218 : _jointLockProgressCurves[i].Progress();
219
220 jointRotations[rawJointIndex] = Quaternion.Slerp(
221 _constrainedJointRotations[i],
222 jointRotations[rawJointIndex],
223 smoothFactor);
224
225 _lastSyntheticRotation[i] = jointRotations[rawJointIndex];
226 }

338 private void SetJointFreedomAtIndex(int jointId, in JointFreedom freedomLevel, bool skipAnimation = false)
339 {
340 JointFreedom currentFreedom = _jointsFreedomLevels[jointId];
341 if (currentFreedom != freedomLevel)
342 {
343 bool locked = freedomLevel == JointFreedom.Locked
344 || freedomLevel == JointFreedom.Constrained;
345 UpdateProgressCurve(ref _jointLockProgressCurves[jointId],
346 ref _jointUnlockProgressCurves[jointId],
347 locked, skipAnimation);
348 _constrainedJointRotations[jointId] = _lastSyntheticRotation[jointId];
349 }
350
351 _jointsFreedomLevels[jointId] = freedomLevel;
352 }

450 private static float OverFlex(in Quaternion desiredLocalRot, in Quaternion maxLocalRot)
451 {
452 Vector3 jointDir = desiredLocalRot * Vector3.right;
453 Vector3 jointTan = desiredLocalRot * Vector3.back;
454 Vector3 maxDir = maxLocalRot * Vector3.right;
455 Vector3 difference = Vector3.Cross(jointDir, maxDir);
456 return Vector3.Dot(jointTan, difference);
457 }

459 private static void UpdateProgressCurve(ref ProgressCurve lockProgress, ref ProgressCurve unlockProgress, bool locked, bool skipAnimation)
