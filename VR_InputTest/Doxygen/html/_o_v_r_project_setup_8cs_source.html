168 public static void AddTask(
169 OVRProjectSetup.TaskGroup group,
170 Func<BuildTargetGroup, bool> isDone,
171 BuildTargetGroup platform = BuildTargetGroup.Unknown,
172 Action<BuildTargetGroup> fix = null,
173 OVRProjectSetup.TaskLevel level = OVRProjectSetup.TaskLevel.Recommended,
174 Func<BuildTargetGroup, OVRProjectSetup.TaskLevel> conditionalLevel = null,
175 string message = null,
176 Func<BuildTargetGroup, string> conditionalMessage = null,
177 string fixMessage = null,
178 Func<BuildTargetGroup, string> conditionalFixMessage = null,
179 string url = null,
180 Func<BuildTargetGroup, string> conditionalUrl = null,
181 bool validity = true,
182 Func<BuildTargetGroup, bool> conditionalValidity = null
183 )
184 {
185 var optionalLevel =
186 OptionalLambdaType<BuildTargetGroup, OVRProjectSetup.TaskLevel>.Create(level, conditionalLevel, true);
187 var optionalMessage = OptionalLambdaType<BuildTargetGroup, string>.Create(message, conditionalMessage, true);
188 var optionalFixMessage =
189 OptionalLambdaType<BuildTargetGroup, string>.Create(fixMessage, conditionalFixMessage, true);
190 var optionalUrl = OptionalLambdaType<BuildTargetGroup, string>.Create(url, conditionalUrl, true);
191 var optionalValidity = OptionalLambdaType<BuildTargetGroup, bool>.Create(validity, conditionalValidity, true);
192 AddTask(new OVRConfigurationTask(group, platform, isDone, fix, optionalLevel, optionalMessage,
193 optionalFixMessage, optionalUrl, optionalValidity));
194 }
