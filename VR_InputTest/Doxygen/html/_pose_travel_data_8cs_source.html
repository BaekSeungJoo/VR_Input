32 [Tooltip("When attracting the object, indicates the rate (in m/s, or seconds if UseFixedTravelTime is enabled) for the object to realign with the hand after a grab.")]

38 [Tooltip("Changes the units of the TravelSpeed, disabled means m/s while enabled is fixed seconds")]

48 private const float DEGREES_TO_PERCEIVED_METERS = 0.5f / 360f;

52 _travelSpeed = 1f,

63
64 public Tween CreateTween(in Pose from, in Pose to)
65 {
66 float tweenTime = _travelSpeed;
67 if (!_useFixedTravelTime && _travelSpeed != 0f)
68 {
69 float travelDistance = PerceivedDistance(from, to);
70 tweenTime = travelDistance / _travelSpeed;
71 }
72 Tween tween = new Tween(from, tweenTime, tweenTime * 0.5f, _travelCurve);
73 tween.MoveTo(to);
74 return tween;
75 }
76
77 private static float PerceivedDistance(in Pose from, in Pose to)
78 {
79 Pose grabOffset = PoseUtils.Delta(from, to);
80 float translationDistance = grabOffset.position.magnitude;
81
82 float rotationDistance = DEGREES_TO_PERCEIVED_METERS * Mathf.Max(
83 Mathf.Max(Vector3.Angle(from.forward, to.forward),
84 Vector3.Angle(from.up, to.up),
85 Vector3.Angle(from.right, to.right)));
86
87 float travelDistance = Mathf.Max(translationDistance, rotationDistance);
88
89 return travelDistance;
90 }
91 }
