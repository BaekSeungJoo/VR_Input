43[CustomEditor(typeof(OVRCustomFace))]
44public class OVRCustomFaceEditor : Editor
45{
46 private SerializedProperty _expressionsProp;

48 private SerializedProperty _strengthMultiplier;
49 private SerializedProperty _allowDuplicateMapping;
50 private bool _showBlendshapes = true;

52 protected virtual void OnEnable()
53 {
54 _expressionsProp = serializedObject.FindProperty(nameof(OVRCustomFace._faceExpressions));
55 _mappings = serializedObject.FindProperty(nameof(OVRCustomFace._mappings));
56 _strengthMultiplier = serializedObject.FindProperty(nameof(OVRCustomFace._blendShapeStrengthMultiplier));

60 public override void OnInspectorGUI()
61 {
62 var face = (OVRCustomFace)target;
63
64 serializedObject.Update();
65
66 if (_expressionsProp.objectReferenceValue == null)
67 {
68 _expressionsProp.objectReferenceValue = face.SearchFaceExpressions();
69 }
70
71 if (!IsFaceExpressionsConfigured(face))
72 {
73 if (OVREditorUIElements.RenderWarningWithButton(
74 "OVRFaceExpressions is required.", "Configure OVRFaceExpressions"))
75 {
76 FixFaceExpressions(face);
77 }
78 }
79
80 EditorGUILayout.PropertyField(_expressionsProp, new GUIContent(nameof(OVRFaceExpressions)));
81
82 EditorGUILayout.PropertyField(_strengthMultiplier, new GUIContent("Blend Shape Strength Multiplier"));
83
84
85 //need to pass out some property to find the component from
86 SkinnedMeshRenderer renderer = GetSkinnedMeshRenderer(_expressionsProp);
87
88 if (renderer == null || renderer.sharedMesh == null)
89 {
90 if (_mappings.arraySize > 0)
91 {
92 _mappings.ClearArray();
93 }
94
95 serializedObject.ApplyModifiedProperties();
96 return;
97 }
98
99 if (_mappings.arraySize != renderer.sharedMesh.blendShapeCount)
100 {
101 _mappings.ClearArray();
102 _mappings.arraySize = renderer.sharedMesh.blendShapeCount;
103 for (int i = 0; i < renderer.sharedMesh.blendShapeCount; ++i)
104 {
105 _mappings.GetArrayElementAtIndex(i).intValue = (int)OVRFaceExpressions.FaceExpression.Invalid;
106 }
107 }
108
109 EditorGUILayout.Space();
110
111 var enumValues = Enum.GetNames(typeof(OVRCustomFace.RetargetingType));
112 face.retargetingType = (OVRCustomFace.RetargetingType)
113 EditorGUILayout.Popup("Custom face structure", (int)face.retargetingType, enumValues);
114
115 if (face.retargetingType == OVRCustomFace.RetargetingType.OculusFace
116 || face.retargetingType == OVRCustomFace.RetargetingType.Custom
117 )
118 {
119 _showBlendshapes = EditorGUILayout.BeginFoldoutHeaderGroup(_showBlendshapes, "Blendshapes");
120 if (_showBlendshapes)
121 {
122 if (GUILayout.Button("Auto Generate Mapping"))
123 {
124 face.AutoMapBlendshapes();
125 Refresh(face);
126 }
127
128 if (GUILayout.Button("Clear Mapping"))
129 {
130 face.ClearBlendshapes();
131 Refresh(face);
132 }
133
134 EditorGUILayout.Space();
135
136 for (int i = 0; i < renderer.sharedMesh.blendShapeCount; ++i)
137 {
138 EditorGUILayout.PropertyField(_mappings.GetArrayElementAtIndex(i),
139 new GUIContent(renderer.sharedMesh.GetBlendShapeName(i)));
140 }
141 }
142 }
143
144 EditorGUILayout.EndFoldoutHeaderGroup();
145
146 EditorGUILayout.PropertyField(_allowDuplicateMapping,
147 new GUIContent("Allow duplicate mapping"));
148
149 serializedObject.ApplyModifiedProperties();

158 internal static bool IsFaceExpressionsConfigured(OVRFace face)
159 {
160 return face._faceExpressions != null;
161 }
162
163 internal static void FixFaceExpressions(OVRFace face)
164 {
165 Undo.IncrementCurrentGroup();
166 var gameObject = face.gameObject;
167
168 var faceExpressions = face.SearchFaceExpressions();
169 if (!faceExpressions)
170 {
171 faceExpressions = gameObject.AddComponent<OVRFaceExpressions>();
172 Undo.RegisterCreatedObjectUndo(faceExpressions, "Create OVRFaceExpressions component");
173 }
174
175 Undo.RecordObject(face, "Linked OVRFaceExpression");
176 face._faceExpressions = faceExpressions;
177
178 EditorUtility.SetDirty(face);
179 EditorSceneManager.MarkSceneDirty(gameObject.scene);
180
181 Undo.SetCurrentGroupName("Configure OVRFaceExpressions for OVRCustomFace");
182 }

184 private static SkinnedMeshRenderer GetSkinnedMeshRenderer(SerializedProperty property)
185 {
186 GameObject targetObject = GetGameObject(property);
187
188 if (!targetObject)
189 return null;
190
191 return targetObject.GetComponent<SkinnedMeshRenderer>();
192 }

194 private static GameObject GetGameObject(SerializedProperty property)
195 {
196 Component targetComponent = property.serializedObject.targetObject as Component;
197
198 if (targetComponent && targetComponent.gameObject)
199 {
200 return targetComponent.gameObject;
201 }
202
203 return null;
204 }
