47 public override bool OnGUI(IAudioEffectPlugin plugin)
48 {
49 float fval = 0.0f;
50 bool bval = false;
51
52 Separator();
53 Label("Voice limit (1 - 1024)");
54 ONSPSettings.Instance.voiceLimit = EditorGUILayout.IntField(" ", ONSPSettings.Instance.voiceLimit);
55 if (GUI.changed)
56 {
57 GUI.changed = false;
58 EditorUtility.SetDirty(ONSPSettings.Instance);
59 }
60
61 Separator();
62 Label ("GLOBAL SCALE (0.00001 - 10000.0)");
63 plugin.GetFloatParameter("GScale", out fval);
64 plugin.SetFloatParameter("GScale", EditorGUILayout.FloatField(" ", Mathf.Clamp (fval, 0.00001f, 10000.0f)));
65
66 Separator();
67
68 Label ("REFLECTION ENGINE");
69 Label("");
70 // Treat these floats as bools in the inspector
71 plugin.GetFloatParameter("E.Rflt On", out fval);
72 bval = (fval == 0.0f) ? false : true;
73 bval = EditorGUILayout.Toggle("Enable Early Reflections", bval);
74 plugin.SetFloatParameter("E.Rflt On", (bval == false) ? 0.0f : 1.0f);
75
76 plugin.GetFloatParameter("E.Rflt Rev On", out fval);
77 bval = (fval == 0.0f) ? false : true;
78 bval = EditorGUILayout.Toggle("Enable Reverberation", bval);
79 plugin.SetFloatParameter("E.Rflt Rev On", (bval == false) ? 0.0f : 1.0f);
80
81 Separator();
82 Label("ROOM DIMENSIONS (meters)");
83 Label("");
84 plugin.GetFloatParameter("Room X", out fval);
85 plugin.SetFloatParameter("Room X", EditorGUILayout.Slider("Width", fval, 1.0f, 200.0f));
86 plugin.GetFloatParameter("Room Y", out fval);
87 plugin.SetFloatParameter("Room Y", EditorGUILayout.Slider("Height", fval, 1.0f, 200.0f));
88 plugin.GetFloatParameter("Room Z", out fval);
89 plugin.SetFloatParameter("Room Z", EditorGUILayout.Slider("Length", fval, 1.0f, 200.0f));
90
91 Separator();
92 Label("WALL REFLECTION COEFFICIENTS (0.0 - 0.97)");
93 Label("");
94
95 plugin.GetFloatParameter("Left", out fval);
96 plugin.SetFloatParameter("Left", EditorGUILayout.Slider("Left", fval, 0.0f, 0.97f));
97 plugin.GetFloatParameter("Right", out fval);
98 plugin.SetFloatParameter("Right", EditorGUILayout.Slider("Right", fval, 0.0f, 0.97f));
99 plugin.GetFloatParameter("Up", out fval);
100 plugin.SetFloatParameter("Up", EditorGUILayout.Slider("Up", fval, 0.0f, 0.97f));
101 plugin.GetFloatParameter("Down", out fval);
102 plugin.SetFloatParameter("Down", EditorGUILayout.Slider("Down", fval, 0.0f, 0.97f));
103 plugin.GetFloatParameter("Behind", out fval);
104 plugin.SetFloatParameter("Behind", EditorGUILayout.Slider("Back", fval, 0.0f, 0.97f));
105 plugin.GetFloatParameter("Front", out fval);
106 plugin.SetFloatParameter("Front", EditorGUILayout.Slider("Front", fval, 0.0f, 0.97f));
107
108 Separator();
109 Label("SHARED REVERB ATTENUATION RANGE (1.0 - 10000.0 meters)");
110 Label("");
111 plugin.GetFloatParameter("Shared Rev Min", out fval);
112 plugin.SetFloatParameter("Shared Rev Min", EditorGUILayout.Slider("Minimum", fval, 1.0f, 10000.0f));
113 plugin.GetFloatParameter("Shared Rev Max", out fval);
114 plugin.SetFloatParameter("Shared Rev Max", EditorGUILayout.Slider("Maximum", fval, 1.0f, 10000.0f));
115
116 Separator();
117 Label("SHARED REVERB WET MIX (-60.0 - 20.0 dB)");
118 Label("");
119 plugin.GetFloatParameter("Shared Rev Wet", out fval);
120 plugin.SetFloatParameter("Shared Rev Wet", EditorGUILayout.Slider(" ", fval, -60.0f, 20.0f));
121
122 Separator();
123 Label("PROPAGATION QUALITY LEVEL (0.0 - 200.0%)");
124 Label("");
125 plugin.GetFloatParameter("Prop Quality", out fval);
126 plugin.SetFloatParameter("Prop Quality", EditorGUILayout.Slider(" ", fval, 0.0f, 200.0f));
127
128 Separator();
129
130 // We will override the controls with our own, so return false
131 return false;
132 }
