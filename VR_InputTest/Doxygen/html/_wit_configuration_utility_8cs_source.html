71 // Refresh configuration asset list
72 public static void ReloadConfigurationData()
73 {
74 // Reloaded
75 _needsConfigReload = false;
76
77 // Find all Wit Configurations
78 List<WitConfiguration> loaded = GetLoadedConfigurations();
79 List<WitConfiguration> found = new List<WitConfiguration>();
80 string[] guids = AssetDatabase.FindAssets("t:WitConfiguration");
81 foreach (var guid in guids)
82 {
83 string path = AssetDatabase.GUIDToAssetPath(guid);
84 WitConfiguration config = AssetDatabase.LoadAssetAtPath<WitConfiguration>(path);
85 if (!config.isDemoOnly || loaded.Contains(config))
86 {
87 found.Add(config);
88 }
89 }
90
91 // Store wit configuration data
92 _witConfigs = found.ToArray();
93 // Obtain all names
94 _witConfigNames = new string[_witConfigs.Length];
95 for (int i = 0; i < _witConfigs.Length; i++)
96 {
97 _witConfigNames[i] = _witConfigs[i].name;
98 }
99 }

112 // Return all configurations referenced in loaded scenes
113 public static List<WitConfiguration> GetLoadedConfigurations()
114 {
115 // Get results
116 List<WitConfiguration> results = new List<WitConfiguration>();
117
118 // Iterate loaded scenes
119 for (int sceneIndex = 0; sceneIndex < SceneManager.sceneCount; sceneIndex++)
120 {
121 Scene scene = SceneManager.GetSceneAt(sceneIndex);
122 foreach (var rootGameObject in scene.GetRootGameObjects())
123 {
124 IWitConfigurationProvider[] providers = rootGameObject.GetComponentsInChildren<IWitConfigurationProvider>(true);
125 if (providers != null)
126 {
127 foreach (var provider in providers)
128 {
129 WitConfiguration config = provider.Configuration;
130 if (config != null && !results.Contains(config))
131 {
132 results.Add(config);
133 }
134 }
135 }
136 }
137 }
138
139 // Return results list
140 return results;
141 }

192 // Save configuration to selected location
193 public static int SaveConfiguration(string savePath, string serverToken, WitConfiguration configurationAsset)
194 {
195 // Ensure valid save path
196 if (string.IsNullOrEmpty(savePath))
197 {
198 return -1;
199 }
200 // Must be in assets
201 string unityPath = savePath.Replace("\\", "/");
202 if (!unityPath.StartsWith(Application.dataPath))
203 {
204 VLog.E($"Configuration Utility - Cannot Create Configuration Outside of Assets Directory\nPath: {unityPath}");
205 return -1;
206 }
207
208 // Determine local unity path
209 unityPath = unityPath.Replace(Application.dataPath, "Assets");
210 AssetDatabase.CreateAsset(configurationAsset, unityPath);
211 AssetDatabase.SaveAssets();
212
213 // Refresh configurations
214 ReloadConfigurationData();
215
216 // Get new index following reload
217 string name = System.IO.Path.GetFileNameWithoutExtension(unityPath);
218 int index = GetConfigurationIndex(name);
219
220 // Set server token
221 if (!string.IsNullOrEmpty(serverToken))
222 {
223 _witConfigs[index].SetServerToken(serverToken);
224 }
225
226 // Return index
227 return index;
228 }

276 // On complete

279 // Sets server token for specified configuration by updating it's application data
280 public static void SetServerToken(this WitConfiguration configuration, string serverToken, Action<string> onSetComplete = null)
281 {
282 var instanceKey = Telemetry.StartEvent(Telemetry.TelemetryEventId.SupplyToken);
283
284 // Invalid
285 if (!IsServerTokenValid(serverToken))
286 {
287 onSetComplete?.Invoke("Invalid Token");
288 return;
289 }
290 // Perform a list app request to get app for token
291 WitAppInfoUtility.GetAppInfo(serverToken, (clientToken, info, error) =>
292 {
293 // Set client access token
294 configuration.SetClientAccessToken(clientToken);
295 // Set application info
296 configuration.SetApplicationInfo(info);
297 // Set server token
298 if (!string.IsNullOrEmpty(info.id))
299 {
300 WitAuthUtility.SetAppServerToken(info.id, serverToken);
301 }
302
303 // Complete
304 if (!string.IsNullOrEmpty(error))
305 {
306 Telemetry.EndEventWithFailure(instanceKey, error);
307 }
308 else
309 {
310 Telemetry.EndEvent(instanceKey, Telemetry.ResultType.Success);
311 }
312
313 onSetComplete?.Invoke(error);
314 });
315 }

325 internal static void ImportData(this WitConfiguration configuration, Manifest manifest, VRequest.RequestCompleteDelegate<bool> onComplete = null, bool suppressLogs = false)
326 {
327 var manifestData = GetSanitizedManifestString(manifest);
328 var request = new WitSyncVRequest(configuration);

331 {
332 VLog.SuppressLogs = false;
333 if (!string.IsNullOrEmpty(error))
334 {
335 onComplete?.Invoke(false, error);
336 }
337 else
338 {
339 onComplete?.Invoke(true, string.Empty);
340 }
341 });
342 }

348 private static string GetSanitizedManifestString(Manifest manifest)
349 {
350 var filter = new WitParameterFilter();
351 foreach (var action in manifest.Actions)
352 {
353 action.Parameters.RemoveAll(a => filter.ShouldFilterOut(a.EntityType));
354 }
355
356 return JsonConvert.SerializeObject(manifest);
357 }
