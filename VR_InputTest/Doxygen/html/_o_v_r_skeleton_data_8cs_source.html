28 // This method can be used to generate a serialized skeleton in C# format.
29 public static string CreateString(OVRPlugin.Skeleton2 skeleton)
30 {
31 string bonesJson = "";
32 for (int i = 0; i < skeleton.NumBones; i++)
33 {
34 var bone = skeleton.Bones[i];
35 bonesJson +=
36 "new OVRPlugin.Bone() { " +
37 "Id=OVRPlugin.BoneId." + bone.Id.ToString() + ", " +
38 "ParentBoneIndex=" + bone.ParentBoneIndex + ", " +
39 "Pose=new OVRPlugin.Posef() { " +
40 "Position=new OVRPlugin.Vector3f() {x=" + bone.Pose.Position.x + "f,y=" + bone.Pose.Position.y + "f,z=" + bone.Pose.Position.z + "f}, " +
41 "Orientation=new OVRPlugin.Quatf(){x=" + bone.Pose.Orientation.x + "f,y=" + bone.Pose.Orientation.y + "f,z=" + bone.Pose.Orientation.z + "f,w=" + bone.Pose.Orientation.w + "f}}}";
42 if (i != skeleton.Bones.Length - 1)
43 {
44 bonesJson += ",";
45 }
46 bonesJson += "\n";
47 }
48
49 string boneCapsulesJson = "";
50 for (int i = 0; i < skeleton.NumBoneCapsules; i++)
51 {
52 var cap = skeleton.BoneCapsules[i];
53 boneCapsulesJson += "new OVRPlugin.BoneCapsule() { BoneIndex=" + cap.BoneIndex + ", Radius=" + cap.Radius + "f, " +
54 "StartPoint=new OVRPlugin.Vector3f() {x=" + cap.StartPoint.x + "f,y=" + cap.StartPoint.y + "f,z=" + cap.StartPoint.z + "f}, " +
55 "EndPoint=new OVRPlugin.Vector3f() {x=" + cap.EndPoint.x + "f,y=" + cap.EndPoint.y + "f,z=" + cap.EndPoint.z + "f}}";
56 if (i != skeleton.Bones.Length - 1)
57 {
58 boneCapsulesJson += ",";
59 }
60 boneCapsulesJson += "\n";
61 }
62
63 return "new OVRPlugin.Skeleton2() { " +
64 "Type=OVRPlugin.SkeletonType." + skeleton.Type.ToString() + ", " +
65 "NumBones=" + skeleton.NumBones + ", " +
66 "NumBoneCapsules=" + skeleton.NumBoneCapsules + ", " +
67 "Bones=new OVRPlugin.Bone[] {" + bonesJson + "}, " +
68 "BoneCapsules=new OVRPlugin.BoneCapsule[] {" + boneCapsulesJson + "}" +
69 "};";
70 }
