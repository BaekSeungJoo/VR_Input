84 private bool CheckDirtyBit(int i) => ((_dirtyMap >> i) & 1) == 1;
85 private void SetDirtyBit(int i) => _dirtyMap = _dirtyMap | (1 << i);
86 private void ClearDirtyBit(int i) => _dirtyMap = _dirtyMap & ~(1 << i);
87
88 private void MarkDirty(HandJointId jointId)
89 {
90 if (CheckDirtyBit((int)jointId))
91 {
92 return;
93 }
94
95 SetDirtyBit((int)jointId);
96 foreach (HandJointId child in HandJointUtils.JointChildrenList[(int)jointId])
97 {
98 MarkDirty(child);
99 }
100 }
101
102 private void UpdateDirty(HandJointId jointId)
103 {
104 if (!CheckDirtyBit((int)jointId))
105 {
106 return;
107 }
108
109 HandJointId parent = HandJointUtils.JointParentList[(int)jointId];
110 if (parent != HandJointId.Invalid)
111 {
112 UpdateDirty(parent);
113 }
114
115 ClearDirtyBit((int)jointId);
116
117 Pose parentWorldPose = (int)jointId == 0 ? _rootPose : GetWorldPose(parent);
118 Pose localPose = _localJointMap[(int)jointId];
119 _worldJointMap[(int)jointId] = new Pose(
120 parentWorldPose.position + parentWorldPose.rotation * localPose.position * _rootScale,
121 parentWorldPose.rotation * localPose.rotation
122 );
123 }

125 public void Copy(ShadowHand hand)
126 {
127 SetRoot(hand.GetRoot());
128 SetRootScale(hand.GetRootScale());
129 for (int i = 0; i < (int)HandJointId.HandEnd; i++)
130 {
131 HandJointId jointId = (HandJointId)i;
132 SetLocalPose(jointId, hand.GetLocalPose(jointId));
133 }
134 }

146 public static void FromHandFingers(this ShadowHand shadow, IHand hand, bool flipHandedness)
147 {
148 hand.GetJointPosesLocal(out ReadOnlyHandJointPoses localJointPoses);
149 if (localJointPoses.Count != (int)HandJointId.HandEnd)
150 {
151 return;
152 }
153
154 for (int i = 0; i < (int)HandJointId.HandEnd; i++)
155 {
156 Pose pose = localJointPoses[i];
157 pose.position *= flipHandedness ? -1f : 1f;
158 shadow.SetLocalPose((HandJointId)i, pose);
159 }
160 }
