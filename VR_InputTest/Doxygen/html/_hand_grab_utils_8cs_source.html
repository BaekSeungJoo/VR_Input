93 public static void MirrorHandGrabPose(HandGrabPose originalPoint, HandGrabPose mirrorPoint, Transform relativeTo)
94 {
95 HandPose handPose = originalPoint.HandPose;
96
97 Handedness oppositeHandedness = handPose.Handedness == Handedness.Left ? Handedness.Right : Handedness.Left;
98
99 HandGrabPoseData mirrorData = SaveHandGrabPoseData(originalPoint);
100 mirrorData.handPose.Handedness = oppositeHandedness;
101
102 if (originalPoint.SnapSurface != null)
103 {
104 mirrorData.gripPose = originalPoint.SnapSurface.MirrorPose(mirrorData.gripPose, relativeTo);
105 }
106 else
107 {
108 mirrorData.gripPose = mirrorData.gripPose.MirrorPoseRotation(Vector3.forward, Vector3.up);
109 Vector3 translation = Vector3.Project(mirrorData.gripPose.position, Vector3.right);
110 mirrorData.gripPose.position = mirrorData.gripPose.position - 2f * translation;
111 }
112
113 LoadHandGrabPoseData(mirrorPoint, mirrorData, relativeTo);
114 if (originalPoint.SnapSurface != null)
115 {
116 Grab.GrabSurfaces.IGrabSurface mirroredSurface = originalPoint.SnapSurface.CreateMirroredSurface(mirrorPoint.gameObject);
117 mirrorPoint.InjectOptionalSurface(mirroredSurface);
118 }
119 }

121 private static HandGrabPoseData SaveHandGrabPoseData(HandGrabPose handGrabPose)
122 {

124 {

133 private static void LoadHandGrabPoseData(HandGrabPose handGrabPose, HandGrabPoseData data, Transform relativeTo)
134 {
135 handGrabPose.transform.localScale = Vector3.one * data.scale;
136 handGrabPose.transform.SetPose(PoseUtils.GlobalPoseScaled(relativeTo, data.gripPose));
137 if (data.handPose != null)
138 {
139 handGrabPose.InjectOptionalHandPose(new HandPose(data.handPose));
140 }
141 }

148 public static HandGrabInteractableData SaveData(HandGrabInteractable interactable)
149 {
150 List<HandGrabPoseData> poses = new List<HandGrabPoseData>();
151 foreach (HandGrabPose pose in interactable.HandGrabPoses)
152 {
153 poses.Add(SaveHandGrabPoseData(pose));
154 }
155
156 return new HandGrabInteractableData()
157 {
158 poses = poses,
159 scoringModifier = interactable.ScoreModifier,
160 grabType = interactable.SupportedGrabTypes,
161 handAlignment = interactable.HandAlignment,
162 pinchGrabRules = interactable.PinchGrabRules,
163 palmGrabRules = interactable.PalmGrabRules
164 };
165 }

171 public static void LoadData(HandGrabInteractable interactable, HandGrabInteractableData data)
172 {
173 interactable.InjectSupportedGrabTypes(data.grabType);
174 interactable.InjectPinchGrabRules(data.pinchGrabRules);
175 interactable.InjectPalmGrabRules(data.palmGrabRules);
176 interactable.InjectOptionalScoreModifier(data.scoringModifier);
177 interactable.HandAlignment = data.handAlignment;
178
179 if (data.poses == null)
180 {
181 return;
182 }
183 List<HandGrabPose> poses = new List<HandGrabPose>();
184 foreach (HandGrabPoseData handGrabPoseData in data.poses)
185 {
186 poses.Add(LoadHandGrabPose(interactable, handGrabPoseData));
187 }
188 interactable.InjectOptionalHandGrabPoses(poses);
189 }

191 public static HandGrabPose LoadHandGrabPose(HandGrabInteractable interactable,
192 HandGrabPoseData poseData)
193 {
194 HandGrabPose point = CreateHandGrabPose(interactable.transform, interactable.RelativeTo);
195 LoadHandGrabPoseData(point, poseData, interactable.RelativeTo);
196 interactable.HandGrabPoses.Add(point);
197 return point;
198 }
