179 private void HandleStateChanged(InteractorStateChangeArgs obj)
180 {

187 private void HandlePostprocessed()
188 {
189 if (_wasSelecting && _fireSnapOnUnselect)
190 {
191 _wasSelecting = false;
192 if ((Interactor.State == InteractorState.Hover || Interactor.State == InteractorState.Normal)
193 && _turnMethod == TurnMode.Snap)
194 {
195 ProcessSnapTurn(Axis.Value());
196 }
197 }
198
199 if (Interactor.State == InteractorState.Select)
200 {
201 if (_turnMethod == TurnMode.Smooth)
202 {
203 ProcessSmoothTurn(Axis.Value());
204 }
205 else if (_turnMethod == TurnMode.Snap
206 && !_fireSnapOnUnselect
207 && !_wasSelecting)
208 {
209 _wasSelecting = true;
210 ProcessSnapTurn(Axis.Value());
211 }
212 }

216 private void ProcessSnapTurn(float pointerOffset)
217 {
218 float sign = Mathf.Sign(pointerOffset);
219 Quaternion rot = Quaternion.Euler(0f, sign * _snapTurnDegrees, 0f);
220
221 LocomotionEvent locomotionEvent = new LocomotionEvent(
222 Identifier, rot, LocomotionEvent.RotationType.Relative);
223 _whenLocomotionEventRaised.Invoke(locomotionEvent);
224 }
225
226 private void ProcessSmoothTurn(float pointerOffset)
227 {
228 float sign = Mathf.Sign(pointerOffset);
229 float vel = _smoothTurnCurve.Evaluate(Mathf.Abs(pointerOffset));
230 Quaternion rot = Quaternion.Euler(0f, sign * vel, 0f);
231 LocomotionEvent locomotionEvent = new LocomotionEvent(
232 Identifier, rot, LocomotionEvent.RotationType.Velocity);
233 _whenLocomotionEventRaised.Invoke(locomotionEvent);
234 }
