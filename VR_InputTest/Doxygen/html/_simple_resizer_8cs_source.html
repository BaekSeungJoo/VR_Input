35 private Mesh ProcessVertices(SimpleResizable resizable, Vector3 newSize)
36 {
37 Mesh originalMesh = resizable.Mesh;
38 Vector3 originalBounds = originalMesh.bounds.size;
39
40 // Force scaling if newSize is smaller than the original mesh
41 SimpleResizable.Method methodX = (originalBounds.x < newSize.x)
42 ? resizable.ScalingX
43 : SimpleResizable.Method.Scale;
44 SimpleResizable.Method methodY = (originalBounds.y < newSize.y)
45 ? resizable.ScalingY
46 : SimpleResizable.Method.Scale;
47 SimpleResizable.Method methodZ = (originalBounds.z < newSize.z)
48 ? resizable.ScalingZ
49 : SimpleResizable.Method.Scale;
50
51 Vector3[] resizedVertices = originalMesh.vertices;
52
53 float pivotX = (1 / resizable.DefaultSize.x) * resizable.PivotPosition.x;
54 float pivotY = (1 / resizable.DefaultSize.y) * resizable.PivotPosition.y;
55 float pivotZ = (1 / resizable.DefaultSize.z) * resizable.PivotPosition.z;
56
57 for (int i = 0; i < resizedVertices.Length; i++)
58 {
59 Vector3 vertexPosition = resizedVertices[i];
60 vertexPosition.x = CalculateNewVertexPosition(
61 methodX,
62 vertexPosition.x,
63 originalBounds.x,
64 newSize.x,
65 resizable.PaddingX,
66 resizable.PaddingXMax,
67 pivotX);
68
69 vertexPosition.y = CalculateNewVertexPosition(
70 methodY,
71 vertexPosition.y,
72 originalBounds.y,
73 newSize.y,
74 resizable.PaddingY,
75 resizable.PaddingYMax,
76 pivotY);
77
78 vertexPosition.z = CalculateNewVertexPosition(
79 methodZ,
80 vertexPosition.z,
81 originalBounds.z,
82 newSize.z,
83 resizable.PaddingZ,
84 resizable.PaddingZMax,
85 pivotZ);
86 resizedVertices[i] = vertexPosition;
87 }
88
89 Mesh clonedMesh = MonoBehaviour.Instantiate(originalMesh);
90 clonedMesh.vertices = resizedVertices;
91
92 return clonedMesh;
93 }

95 private float CalculateNewVertexPosition(
96 SimpleResizable.Method resizeMethod,
97 float currentPosition,
98 float currentSize,
99 float newSize,
100 float padding,
101 float paddingMax,
102 float pivot)
103 {
104 float resizedRatio = currentSize / 2
105 * (newSize / 2 * (1 / (currentSize / 2)))
106 - currentSize / 2;
107
108 switch (resizeMethod)
109 {
110 case SimpleResizable.Method.Adapt:
111 if (Mathf.Abs(currentPosition) >= padding)
112 currentPosition = resizedRatio * Mathf.Sign(currentPosition) + currentPosition;
113 break;
114
115 case SimpleResizable.Method.AdaptWithAsymmetricalPadding:
116 if (currentPosition >= padding)
117 currentPosition = resizedRatio * Mathf.Sign(currentPosition) + currentPosition;
118 if (currentPosition <= paddingMax)
119 currentPosition = resizedRatio * Mathf.Sign(currentPosition) + currentPosition;
120 break;
121
122 case SimpleResizable.Method.Scale:
123 currentPosition = newSize / (currentSize / currentPosition);
124 break;
125
126 case SimpleResizable.Method.None:
127 break;
128 }
129
130 float pivotPos = newSize * (-pivot);
131 currentPosition += pivotPos;
132
133 return currentPosition;
134 }
