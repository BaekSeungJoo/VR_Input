89 protected override void Start()
90 {
91 base.Start();
92
93 // Basic hack to guess at max grab distance based on player size.
94 // Note that there's no major downside to making this value too high, as objects
95 // outside the player's grabbable trigger volume will not be eligible targets regardless.
96 Collider sc = m_player.GetComponentInChildren<Collider>();
97 if (sc != null)
98 {
99 m_maxGrabDistance = sc.bounds.size.z * 0.5f + 3.0f;
100 }
101 else
102 {
103 m_maxGrabDistance = 12.0f;
104 }
105
106 if (m_parentHeldObject == true)
107 {
108 Debug.LogError("m_parentHeldObject incompatible with DistanceGrabber. Setting to false.");
109 m_parentHeldObject = false;
110 }
111
112 DistanceGrabber[] grabbers = FindObjectsOfType<DistanceGrabber>();
113 for (int i = 0; i < grabbers.Length; ++i)
114 {
115 if (grabbers[i] != this) m_otherHand = grabbers[i];
116 }
117
118 Debug.Assert(m_otherHand != null);
119
120#if UNITY_EDITOR
121 OVRPlugin.SendEvent("distance_grabber", (SceneManager.GetActiveScene().name == "DistanceGrab").ToString(),
122 "sample_framework");
123#endif

152 protected override void GrabBegin()
153 {
154 DistanceGrabbable closestGrabbable = m_target;
155 Collider closestGrabbableCollider = m_targetCollider;
156
157 GrabVolumeEnable(false);
158
159 if (closestGrabbable != null)
160 {
161 if (closestGrabbable.isGrabbed)
162 {
163 ((DistanceGrabber)closestGrabbable.grabbedBy).OffhandGrabbed(closestGrabbable);
164 }
165
166 m_grabbedObj = closestGrabbable;
167 m_grabbedObj.GrabBegin(this, closestGrabbableCollider);
168 SetPlayerIgnoreCollision(m_grabbedObj.gameObject, true);
169
170 m_movingObjectToHand = true;
171 m_lastPos = transform.position;
172 m_lastRot = transform.rotation;
173
174 // If it's within a certain distance respect the no-snap.
175 Vector3 closestPointOnBounds = closestGrabbableCollider.ClosestPointOnBounds(m_gripTransform.position);
176 if (!m_grabbedObj.snapPosition && !m_grabbedObj.snapOrientation && m_noSnapThreshhold > 0.0f &&
177 (closestPointOnBounds - m_gripTransform.position).magnitude < m_noSnapThreshhold)
178 {
179 Vector3 relPos = m_grabbedObj.transform.position - transform.position;
180 m_movingObjectToHand = false;
181 relPos = Quaternion.Inverse(transform.rotation) * relPos;
182 m_grabbedObjectPosOff = relPos;
183 Quaternion relOri = Quaternion.Inverse(transform.rotation) * m_grabbedObj.transform.rotation;
184 m_grabbedObjectRotOff = relOri;
185 }
186 else
187 {
188 // Set up offsets for grabbed object desired position relative to hand.
189 m_grabbedObjectPosOff = m_gripTransform.localPosition;
190 if (m_grabbedObj.snapOffset)
191 {
192 Vector3 snapOffset = m_grabbedObj.snapOffset.position;
193 if (m_controller == OVRInput.Controller.LTouch) snapOffset.x = -snapOffset.x;
194 m_grabbedObjectPosOff += snapOffset;
195 }
196
197 m_grabbedObjectRotOff = m_gripTransform.localRotation;
198 if (m_grabbedObj.snapOffset)
199 {
200 m_grabbedObjectRotOff = m_grabbedObj.snapOffset.rotation * m_grabbedObjectRotOff;
201 }
202 }
203 }

206 protected override void MoveGrabbedObject(Vector3 pos, Quaternion rot, bool forceTeleport = false)
207 {
208 if (m_grabbedObj == null)
209 {
210 return;
211 }
212
213 Rigidbody grabbedRigidbody = m_grabbedObj.grabbedRigidbody;
214 Vector3 grabbablePosition = pos + rot * m_grabbedObjectPosOff;
215 Quaternion grabbableRotation = rot * m_grabbedObjectRotOff;
216
217 if (m_movingObjectToHand)
218 {
219 float travel = m_objectPullVelocity * Time.deltaTime;
220 Vector3 dir = grabbablePosition - m_grabbedObj.transform.position;
221 if (travel * travel * 1.1f > dir.sqrMagnitude)
222 {
223 m_movingObjectToHand = false;
224 }
225 else
226 {
227 dir.Normalize();
228 grabbablePosition = m_grabbedObj.transform.position + dir * travel;
229 grabbableRotation = Quaternion.RotateTowards(m_grabbedObj.transform.rotation, grabbableRotation,
230 m_objectPullMaxRotationRate * Time.deltaTime);
231 }
232 }
233
234 grabbedRigidbody.MovePosition(grabbablePosition);
235 grabbedRigidbody.MoveRotation(grabbableRotation);
236 }

249 protected bool FindTarget(out DistanceGrabbable dgOut, out Collider collOut)
250 {
251 dgOut = null;
252 collOut = null;
253 float closestMagSq = float.MaxValue;
254
255 // First test for objects within the grab volume, if we're using those.
256 // (Some usage of DistanceGrabber will not use grab volumes, and will only
257 // use spherecasts, and that's supported.)
258 foreach (OVRGrabbable cg in m_grabCandidates.Keys)
259 {
260 DistanceGrabbable grabbable = cg as DistanceGrabbable;
261 bool canGrab = grabbable != null && grabbable.InRange &&
262 !(grabbable.isGrabbed && !grabbable.allowOffhandGrab);
263 if (canGrab && m_grabObjectsInLayer >= 0) canGrab = grabbable.gameObject.layer == m_grabObjectsInLayer;
264 if (!canGrab)
265 {
266 continue;
267 }
268
269 for (int j = 0; j < grabbable.grabPoints.Length; ++j)
270 {
271 Collider grabbableCollider = grabbable.grabPoints[j];
272 // Store the closest grabbable
273 Vector3 closestPointOnBounds = grabbableCollider.ClosestPointOnBounds(m_gripTransform.position);
274 float grabbableMagSq = (m_gripTransform.position - closestPointOnBounds).sqrMagnitude;
275
276 if (grabbableMagSq < closestMagSq)
277 {
278 bool accept = true;
279 if (m_preventGrabThroughWalls)
280 {
281 // NOTE: if this raycast fails, ideally we'd try other rays near the edges of the object, especially for large objects.
282 // NOTE 2: todo optimization: sort the objects before performing any raycasts.
283 Ray ray = new Ray();
284 ray.direction = grabbable.transform.position - m_gripTransform.position;
285 ray.origin = m_gripTransform.position;
286 RaycastHit obstructionHitInfo;
287 Debug.DrawRay(ray.origin, ray.direction, Color.red, 0.1f);
288
289 if (Physics.Raycast(ray, out obstructionHitInfo, m_maxGrabDistance, 1 << m_obstructionLayer,
290 QueryTriggerInteraction.Ignore))
291 {
292 float distToObject = (grabbableCollider.ClosestPointOnBounds(m_gripTransform.position) -
293 m_gripTransform.position).magnitude;
294 if (distToObject > obstructionHitInfo.distance * 1.1)
295 {
296 accept = false;
297 }
298 }
299 }
300
301 if (accept)
302 {
303 closestMagSq = grabbableMagSq;
304 dgOut = grabbable;
305 collOut = grabbableCollider;
306 }
307 }
308 }
309 }
310
311 if (dgOut == null && m_useSpherecast)
312 {
313 return FindTargetWithSpherecast(out dgOut, out collOut);
314 }
315
316 return dgOut != null;
317 }

319 protected bool FindTargetWithSpherecast(out DistanceGrabbable dgOut, out Collider collOut)
320 {
321 dgOut = null;
322 collOut = null;
323 Ray ray = new Ray(m_gripTransform.position, m_gripTransform.forward);
324 RaycastHit hitInfo;
325
326 // If no objects in grab volume, raycast.
327 // Potential optimization:
328 // In DistanceGrabbable.RefreshCrosshairs, we could move the object between collision layers.
329 // If it's in range, it would move into the layer DistanceGrabber.m_grabObjectsInLayer,
330 // and if out of range, into another layer so it's ignored by DistanceGrabber's SphereCast.
331 // However, we're limiting the SphereCast by m_maxGrabDistance, so the optimization doesn't seem
332 // essential.
333 int layer = (m_grabObjectsInLayer == -1) ? ~0 : 1 << m_grabObjectsInLayer;
334 if (Physics.SphereCast(ray, m_spherecastRadius, out hitInfo, m_maxGrabDistance, layer))
335 {
336 DistanceGrabbable grabbable = null;
337 Collider hitCollider = null;
338 if (hitInfo.collider != null)
339 {
340 grabbable = hitInfo.collider.gameObject.GetComponentInParent<DistanceGrabbable>();
341 hitCollider = grabbable == null ? null : hitInfo.collider;
342 if (grabbable)
343 {
344 dgOut = grabbable;
345 collOut = hitCollider;
346 }
347 }
348
349 if (grabbable != null && m_preventGrabThroughWalls)
350 {
351 // Found a valid hit. Now test to see if it's blocked by collision.
352 RaycastHit obstructionHitInfo;
353 ray.direction = hitInfo.point - m_gripTransform.position;
354
355 dgOut = grabbable;
356 collOut = hitCollider;
357 if (Physics.Raycast(ray, out obstructionHitInfo, m_maxGrabDistance, 1 << m_obstructionLayer,
358 QueryTriggerInteraction.Ignore))
359 {
360 DistanceGrabbable obstruction = null;
361 if (hitInfo.collider != null)
362 {
363 obstruction =
364 obstructionHitInfo.collider.gameObject.GetComponentInParent<DistanceGrabbable>();
365 }
366
367 if (obstruction != grabbable && obstructionHitInfo.distance < hitInfo.distance)
368 {
369 dgOut = null;
370 collOut = null;
371 }
372 }
373 }
374 }
375
376 return dgOut != null;
377 }
