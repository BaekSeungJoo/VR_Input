208 public string GetFormattedTimeFromSeconds()
209 {
210 if (_time >= TimeSpan.MaxValue.TotalSeconds)
211 {
212 _time = TimeSpan.MaxValue.TotalSeconds - 1;
213 Log("Error: Hit max time");
214 }
215 TimeSpan span = TimeSpan.FromSeconds(_time);
216 return $"{Math.Floor(span.TotalHours)}:{span.Minutes:00}:{span.Seconds:00}.{Math.Floor(span.Milliseconds/100f)}";
217 }

226 private bool ParseTime(string[] entityValues, out double time)
227 {
228 time = _time;
229 if (entityValues.Length > 0 && double.TryParse(entityValues[0], out time))
230 {
231 if (entityValues.Length < 2)
232 {
233 throw new ArgumentException("Entities being parsed must include time value and unit.");
234 }
235
236 // If entity was not included in the result it will be empty, but the array will still be size 2
237 if (!string.IsNullOrEmpty(entityValues[1]))
238 {
239 switch (entityValues[1])
240 {
241 case "minute":
242 time *= 60;
243 break;
244 case "hour":
245 time *= 60 * 60;
246 break;
247 }
248 }
249
250 return true;
251 }
252
253 return false;
254 }
