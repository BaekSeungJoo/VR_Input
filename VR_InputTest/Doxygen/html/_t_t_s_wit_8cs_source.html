200 public void RequestStreamFromWeb(TTSClipData clipData)
201 {
202 // Stream begin
203 WebStreamEvents?.OnStreamBegin?.Invoke(clipData);
204
205 // Check if valid
206 string validError = IsRequestValid(clipData, RequestSettings.configuration);
207 if (!string.IsNullOrEmpty(validError))
208 {
209 WebStreamEvents?.OnStreamError?.Invoke(clipData, validError);
210 return;
211 }
212 // Ignore if already performing
213 if (_webStreams.ContainsKey(clipData.clipID))
214 {
215 CancelWebStream(clipData);
216 }
217
218 // Begin request
219 WebRequestEvents?.OnRequestBegin?.Invoke(clipData);
220
221 // Whether to stream
222 bool stream = Application.isPlaying && RequestSettings.audioStream;
223
224 // Request tts
225 WitTTSVRequest request = GetTtsRequest(clipData);
226 request.RequestStream(clipData.clipStream,
227 (clipStream, error) =>
228 {
229 // Apply
230 _webStreams.Remove(clipData.clipID);
231
232 // Set new clip stream
233 clipData.clipStream = clipStream;
234
235 // Unloaded
236 if (clipData.loadState == TTSClipLoadState.Unloaded)
237 {
238 error = WitConstants.CANCEL_ERROR;
239 clipStream?.Unload();
240 }
241
242 // Error
243 if (!string.IsNullOrEmpty(error))
244 {
245 if (string.Equals(error, WitConstants.CANCEL_ERROR, StringComparison.CurrentCultureIgnoreCase))
246 {
247 WebStreamEvents?.OnStreamCancel?.Invoke(clipData);
248 WebRequestEvents?.OnRequestCancel?.Invoke(clipData);
249 }
250 else
251 {
252 WebStreamEvents?.OnStreamError?.Invoke(clipData, error);
253 WebRequestEvents?.OnRequestError?.Invoke(clipData, error);
254 }
255 }
256 // Success
257 else
258 {
259 WebStreamEvents?.OnStreamReady?.Invoke(clipData);
260 WebRequestEvents?.OnRequestReady?.Invoke(clipData);
261 if (!RequestSettings.audioStream || !WitTTSVRequest.CanStreamAudio(RequestSettings.audioType))
262 {
263 WebStreamEvents?.OnStreamComplete?.Invoke(clipData);
264 WebRequestEvents?.OnRequestComplete?.Invoke(clipData);
265 }
266 }
267 });
268 _webStreams[clipData.clipID] = request;
269 }
274 public bool CancelWebStream(TTSClipData clipData)
275 {
276 // Ignore without
277 if (!_webStreams.ContainsKey(clipData.clipID))
278 {
279 return false;
280 }
281
282 // Get request
283 VRequest request = _webStreams[clipData.clipID];
284 _webStreams.Remove(clipData.clipID);
285
286 // Destroy immediately
287 request?.Cancel();
288 request = null;
289
290 // Call delegate
291 WebStreamEvents?.OnStreamCancel?.Invoke(clipData);
292 WebRequestEvents?.OnRequestCancel?.Invoke(clipData);
293
294 // Success
295 return true;
296 }

346 WebRequestEvents?.OnRequestError?.Invoke(clipData, error);
347 }
348 WebRequestEvents?.OnRequestComplete?.Invoke(clipData);

413 // Convert voice settings into dictionary to be used with web requests
414 private const string VOICE_KEY = "voice";
415 private const string STYLE_KEY = "style";
416 public Dictionary<string, string> EncodeVoiceSettings(TTSVoiceSettings voiceSettings)
417 {
418 Dictionary<string, string> parameters = new Dictionary<string, string>();
419 if (voiceSettings != null)
420 {
421 foreach (FieldInfo field in GetVoiceSettingsFields(voiceSettings))
422 {
423 // Ensure field value exists
424 object fieldVal = field.GetValue(voiceSettings);
425 if (fieldVal != null)
426 {
427 // Clamp in between range
428 RangeAttribute range = field.GetCustomAttribute<RangeAttribute>();
429 if (range != null && field.FieldType == typeof(int))
430 {
431 int oldFloat = (int) fieldVal;
432 int newFloat = Mathf.Clamp(oldFloat, (int)range.min, (int)range.max);
433 if (oldFloat != newFloat)
434 {
435 fieldVal = newFloat;
436 }
437 }
438
439 // Apply
440 parameters[field.Name] = fieldVal.ToString();
441 }
442 }
443
444 // Set default if no voice is provided
445 if (!parameters.ContainsKey(VOICE_KEY) || string.IsNullOrEmpty(parameters[VOICE_KEY]))
446 {
447 parameters[VOICE_KEY] = TTSWitVoiceSettings.DEFAULT_VOICE;
448 }
449 // Set default if no style is given
450 if (!parameters.ContainsKey(STYLE_KEY) || string.IsNullOrEmpty(parameters[STYLE_KEY]))
451 {
452 parameters[STYLE_KEY] = TTSWitVoiceSettings.DEFAULT_STYLE;
453 }
454 }
455 return parameters;
456 }
457 // Obtain all fields for a specific voice settings
458 private static readonly Dictionary<Type, FieldInfo[]> _settingsFields = new Dictionary<Type, FieldInfo[]>();
459 private static FieldInfo[] GetVoiceSettingsFields(TTSVoiceSettings voiceSettings)
460 {
461 // Return fields if already found
462 Type settingsType = voiceSettings.GetType();
463 if (_settingsFields.ContainsKey(settingsType))
464 {
465 return _settingsFields[settingsType];
466 }
467
468 // Get public/instance fields
469 FieldInfo[] fields = settingsType.GetFields(BindingFlags.Public | BindingFlags.Instance);
470
471 // Remove fields from TTSVoiceSettings
472 Type baseType = typeof(TTSVoiceSettings);
473 fields = fields.ToList().FindAll((field) => field.DeclaringType != baseType).ToArray();
474
475 // Apply & return
476 _settingsFields[settingsType] = fields;
477 return fields;
478 }
