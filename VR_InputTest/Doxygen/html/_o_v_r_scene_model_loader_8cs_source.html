28[RequireComponent(typeof(OVRSceneManager))]
29public class OVRSceneModelLoader : MonoBehaviour
30{
31 private const float RetryingReminderDelay = 10;
32
36 protected OVRSceneManager SceneManager { get; private set; }
37
38 private bool _sceneCaptureRequested;
39
40 void Start()
41 {
42 SceneManager = GetComponent<OVRSceneManager>();
43
44 // Bind the events associated with LoadSceneModel()
45 SceneManager.SceneModelLoadedSuccessfully += OnSceneModelLoadedSuccessfully;
46 SceneManager.NoSceneModelToLoad += OnNoSceneModelToLoad;
47 SceneManager.NewSceneModelAvailable += OnNewSceneModelAvailable;
48
49 // Bind the events associated with RequestSceneCapture()
50 SceneManager.SceneCaptureReturnedWithoutError += OnSceneCaptureReturnedWithoutError;
51 SceneManager.UnexpectedErrorWithSceneCapture += OnUnexpectedErrorWithSceneCapture;
52
53 OnStart();
54 }
55
56 private IEnumerator AttemptToLoadSceneModel()
57 {
58 var timeSinceReminder = 0f;
59 OVRSceneManager.Development.LogWarning(nameof(OVRSceneModelLoader),
60 $"{nameof(OVRSceneManager.LoadSceneModel)} failed. Retrying.");
61 do
62 {
63 timeSinceReminder += Time.deltaTime;
64 if (timeSinceReminder >= RetryingReminderDelay)
65 {
66 timeSinceReminder = 0;
67 OVRSceneManager.Development.LogWarning(nameof(OVRSceneModelLoader),
68 $"{nameof(OVRSceneManager.LoadSceneModel)} failed. Still retrying.");
69 }
70 yield return null;
71 } while (!SceneManager.LoadSceneModel());
72
73 OVRSceneManager.Development.Log(nameof(OVRSceneModelLoader),
74 $"{nameof(OVRSceneManager.LoadSceneModel)} succeeded.");
75 }
76
81 protected virtual void OnStart()
82 {
83 LoadSceneModel();
84 }
85
86 private void LoadSceneModel()
87 {
88 SceneManager.Verbose?.Log(nameof(OVRSceneModelLoader),
89 $"{nameof(OnStart)}() calling {nameof(OVRSceneManager)}.{nameof(OVRSceneManager.LoadSceneModel)}()");
90
91 if (!SceneManager.LoadSceneModel())
92 {
93#if UNITY_EDITOR_WIN || UNITY_STANDALONE_WIN || (UNITY_ANDROID && !UNITY_EDITOR)
94 if (!OVRManager.isHmdPresent)
95 {
96 OVRSceneManager.Development.LogWarning(nameof(OVRSceneModelLoader),
97 $"{nameof(OVRSceneManager.LoadSceneModel)} failed. No link or HMD detected.");
98 }
99 else
100 {
101 StartCoroutine(AttemptToLoadSceneModel());
102 }
103#endif
104 }
105 }
106
110 protected virtual void OnSceneModelLoadedSuccessfully()
111 {
112 // The scene model was captured successfully. At this point all prefabs have been instantiated.
113 SceneManager.Verbose?.Log(nameof(OVRSceneModelLoader),
114 $"{nameof(OVRSceneManager)}.{nameof(OVRSceneManager.LoadSceneModel)}() completed successfully.");
115 }
116
121 protected virtual void OnNoSceneModelToLoad()
122 {
123#if UNITY_EDITOR_WIN
124 UnityEditor.EditorUtility.DisplayDialog("Scene Capture does not work over Link",
125 "There is no scene model available, and scene capture cannot be invoked over Link. " +
126 "Please capture a scene with the HMD in standalone mode, then access the scene model over Link. " +
127 "\n\n" +
128 "If a scene model has already been captured, make sure the HMD is connected via Link and that it is donned.",
129 "Ok");
130#else
131 if (_sceneCaptureRequested)
132 {
133 SceneManager.Verbose?.Log(nameof(OVRSceneModelLoader),
134 $"{nameof(OnSceneCaptureReturnedWithoutError)}() There is no scene model, but we have already requested scene capture once. No further action will be taken.");
135 }
136 else
137 {
138 // There's no Scene model, we have to ask the user to create one
139 SceneManager.Verbose?.Log(nameof(OVRSceneModelLoader),
140 $"{nameof(OnNoSceneModelToLoad)}() calling {nameof(OVRSceneManager)}.{nameof(OVRSceneManager.RequestSceneCapture)}()");
141 _sceneCaptureRequested = SceneManager.RequestSceneCapture();
142 }
143#endif
144 }
145
149 protected virtual void OnNewSceneModelAvailable()
150 {
151 // New scene model detected, reloading the scene.
152 SceneManager.Verbose?.Log(nameof(OVRSceneModelLoader),
153 $"{nameof(OnNewSceneModelAvailable)}() calling {nameof(OVRSceneManager)}.{nameof(OVRSceneManager.LoadSceneModel)}()");
154 SceneManager.LoadSceneModel();
155 }
156
161 protected virtual void OnSceneCaptureReturnedWithoutError()
162 {
163 // The Room Setup successfully returned, we can now load the scene model
164 SceneManager.Verbose?.Log(nameof(OVRSceneModelLoader), $"Room setup returned without errors.");
165 }
166
170 protected virtual void OnUnexpectedErrorWithSceneCapture()
171 {
172 // An unexpected error was returned when invoking the Room Setup. This prevents the user
173 // from capturing their room.
174 SceneManager.Verbose?.LogError(nameof(OVRSceneModelLoader),
175 "Requesting the Room Setup failed. The Scene Model cannot be loaded.");
176 }
177}
