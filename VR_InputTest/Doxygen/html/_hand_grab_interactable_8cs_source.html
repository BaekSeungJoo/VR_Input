175 public bool CalculateBestPose(Pose userPose, float handScale, Handedness handedness,
176 ref HandGrabResult result)
177 {
178 GrabPoseFinder.FindResult findResult = _grabPoseFinder.FindBestPose(userPose,
179 handScale, handedness, _scoringModifier, ref result);
180
181 if (findResult == GrabPoseFinder.FindResult.NotFound)
182 {
183 result.HasHandPose = false;
184 result.Score = GrabPoseHelper.CollidersScore(userPose.position, Colliders, out Vector3 hit);
185 Pose worldSnap = new Pose(hit, userPose.rotation);
186 result.RelativePose = RelativeTo.Delta(worldSnap);
187 }
188
189 return findResult != GrabPoseFinder.FindResult.NotCompatible;
190 }
