21using System;

25public class OVRVirtualKeyboardSampleInputHandler : MonoBehaviour
26{
27 private const float RAY_MAX_DISTANCE = 100.0f;
28 private const float THUMBSTICK_DEADZONE = 0.2f;
29 private const float COLLISION_BOUNDS_ADDED_BLEED_PERCENT = 0.1f;
30 private const float LINEPOINTER_THINNING_THRESHOLD = 0.015f;
31
32 private static float ApplyDeadzone(float value)
33 {
34 if (value > THUMBSTICK_DEADZONE)
35 return (value - THUMBSTICK_DEADZONE) / (1.0f - THUMBSTICK_DEADZONE);
36 else if (value < -THUMBSTICK_DEADZONE)
37 return (value + THUMBSTICK_DEADZONE) / (1.0f - THUMBSTICK_DEADZONE);
38 return 0.0f;
39 }
40
41 public float AnalogStickX => ApplyDeadzone(OVRInput.Get(OVRInput.Axis2D.PrimaryThumbstick).x +
42 OVRInput.Get(OVRInput.Axis2D.SecondaryThumbstick).x);
43
44 public float AnalogStickY => ApplyDeadzone(OVRInput.Get(OVRInput.Axis2D.PrimaryThumbstick).y +
45 OVRInput.Get(OVRInput.Axis2D.SecondaryThumbstick).y);
46
47 public Vector3 InputRayPosition => inputModule.rayTransform.position;
48
49 public Quaternion InputRayRotation =>
50 interactionDevice_ == OVRInput.Controller.LHand
51 ? inputModule.rayTransform.rotation * Quaternion.Euler(Vector3.forward * 180)
52 : // Flip input rotation if left hand input
53 inputModule.rayTransform.rotation;

84 private void UpdateLineRenderer()
85 {
86 leftLinePointer.enabled = false;
87 rightLinePointer.enabled = false;
88
89 UpdateLineRendererFromSource(OVRVirtualKeyboard.InputSource.ControllerLeft);
90 UpdateLineRendererFromSource(OVRVirtualKeyboard.InputSource.ControllerRight);

95 private void UpdateLineRendererFromSource(OVRVirtualKeyboard.InputSource source)
96 {
97 Transform inputTransform = null;
98 switch (source)
99 {
100 case OVRVirtualKeyboard.InputSource.ControllerLeft:
101 inputTransform = OVRInput.IsControllerConnected(OVRInput.Controller.LTouch) &&
102#if UNITY_EDITOR
103 (!OVRVirtualKeyboard.handLeft || !OVRVirtualKeyboard.handLeft.IsTracked)
104#else
105 ((OVRInput.GetControllerIsInHandState(OVRInput.Hand.HandLeft) == OVRInput.ControllerInHandState.NoHand) ||
106 (OVRInput.GetControllerIsInHandState(OVRInput.Hand.HandLeft) == OVRInput.ControllerInHandState.ControllerInHand))
107#endif
108 ? OVRVirtualKeyboard.leftControllerDirectTransform : null;
109 break;
110 case OVRVirtualKeyboard.InputSource.ControllerRight:
111 inputTransform = OVRInput.IsControllerConnected(OVRInput.Controller.RTouch) &&
112#if UNITY_EDITOR
113 (!OVRVirtualKeyboard.handRight || !OVRVirtualKeyboard.handRight.IsTracked)
114#else
115 ((OVRInput.GetControllerIsInHandState(OVRInput.Hand.HandRight) == OVRInput.ControllerInHandState.NoHand) ||
116 (OVRInput.GetControllerIsInHandState(OVRInput.Hand.HandRight) == OVRInput.ControllerInHandState.ControllerInHand))
117#endif
118 ? OVRVirtualKeyboard.rightControllerDirectTransform : null;
119
120 break;
121 case OVRVirtualKeyboard.InputSource.HandLeft:
122 inputTransform = (OVRVirtualKeyboard.handLeft.IsPointerPoseValid)
123 ? OVRVirtualKeyboard.handLeft.PointerPose
124 : null;
125 break;
126 case OVRVirtualKeyboard.InputSource.HandRight:
127 inputTransform = (OVRVirtualKeyboard.handRight.IsPointerPoseValid)
128 ? OVRVirtualKeyboard.handRight.PointerPose
129 : null;
130 break;
131 }
132
133 if (inputTransform == null || inputTransform.position == Vector3.zero)
134 {
135 return;
136 }
137
138 var inputPosition = inputTransform.position;
139
140 bool isLeft = (source == OVRVirtualKeyboard.InputSource.ControllerLeft ||
141 source == OVRVirtualKeyboard.InputSource.HandLeft);
142 LineRenderer linePointer = (isLeft) ? leftLinePointer : rightLinePointer;
143
144 linePointer.startWidth = linePointerInitialWidth_;
145 if (OVRVirtualKeyboard && OVRVirtualKeyboard.isActiveAndEnabled && OVRVirtualKeyboard.Collider)
146 {
147 // get the local input point, but apply scaling to account for the scaled collider bounds
148 var localPoint = OVRVirtualKeyboard.transform.InverseTransformPoint(inputPosition) * OVRVirtualKeyboard.transform.localScale.x;
149 // Check if input ray is within the collider bounds.
150 var interactionBounds = new Bounds
151 {
152 size = OVRVirtualKeyboard.Collider.bounds.size
153 };
154 // grow the interaction bounds beyond the collision bounds
155 interactionBounds.Expand(Vector3.one * COLLISION_BOUNDS_ADDED_BLEED_PERCENT);
156 // if input is with bounds, don't render the line
157 if (interactionBounds.Contains(localPoint))
158 {
159 linePointer.enabled = false;
160 return;
161 }
162
163 var closestPoint = interactionBounds.ClosestPoint(localPoint);
164 Vector3 offset = closestPoint - localPoint;
165 // if the input is outside of bounds, but within the thinning threshold, lerp the width based on distance
166 var distanceToInteractionBounds = offset.magnitude;
167 if (distanceToInteractionBounds < LINEPOINTER_THINNING_THRESHOLD)
168 {
169 linePointer.startWidth = Mathf.Lerp(0, linePointerInitialWidth_, distanceToInteractionBounds / LINEPOINTER_THINNING_THRESHOLD);
170 }
171 }
172 linePointer.endWidth = linePointer.startWidth;
173 linePointer.enabled = true;
174 linePointer.SetPosition(0, inputTransform.position);
175
176 var ray = new Ray(inputPosition, inputTransform.rotation * Vector3.forward);
177 if (OVRVirtualKeyboard.Collider && OVRVirtualKeyboard.Collider.Raycast(ray, out var hit, RAY_MAX_DISTANCE))
178 {
179 linePointer.SetPosition(1, hit.point);
180 }
181 else
182 {
183 linePointer.SetPosition(1, inputPosition + ray.direction * 2.5f);
184 }
185 }

187 private void UpdateInteractionAnchor()
188 {
189 OVRInput.Controller activeController = OVRInput.Controller.None;
190
191 var leftControllerExists = OVRVirtualKeyboard.leftControllerRootTransform != null;
192 var leftControllerActive = leftControllerExists && OVRInput.GetDown(OVRInput.RawButton.LIndexTrigger);
193 activeController = (leftControllerActive) ? OVRInput.Controller.LTouch : activeController;
194
195 var rightControllerExists = OVRVirtualKeyboard.rightControllerRootTransform != null;
196 var rightControllerActive = rightControllerExists && OVRInput.GetDown(OVRInput.RawButton.RIndexTrigger);
197 activeController = (rightControllerActive) ? OVRInput.Controller.RTouch : activeController;
198
199 var handLeftExists = OVRVirtualKeyboard.handLeft != null;
200 var handLeftIsActive =
201 handLeftExists && OVRVirtualKeyboard.handLeft.GetFingerIsPinching(OVRHand.HandFinger.Index);
202 activeController = (handLeftIsActive) ? OVRInput.Controller.LHand : activeController;
203
204 var handRightExists = OVRVirtualKeyboard.handRight != null;
205 var handRightIsActive =
206 handRightExists && OVRVirtualKeyboard.handRight.GetFingerIsPinching(OVRHand.HandFinger.Index);
207 activeController = (handRightIsActive) ? OVRInput.Controller.RHand : activeController;
208
209 if (activeController == OVRInput.Controller.None)
210 {
211 return;
212 }
213
214 // Set transforms for Unity UI interaction
215 var dominantHandIsLeft =
216 (activeController == OVRInput.Controller.LHand || activeController == OVRInput.Controller.LTouch);
217 raycaster.pointer = (dominantHandIsLeft)
218 ? OVRVirtualKeyboard.handLeft.gameObject
219 : OVRVirtualKeyboard.handRight.gameObject;
220 interactionDevice_ = activeController;
221 inputModule.rayTransform = activeController switch
222 {
223 OVRInput.Controller.LHand => OVRVirtualKeyboard.handLeft.PointerPose,
224 OVRInput.Controller.LTouch => OVRVirtualKeyboard.handLeft.transform,
225 OVRInput.Controller.RHand => OVRVirtualKeyboard.handRight.PointerPose,
226 _ => OVRVirtualKeyboard.handRight.transform
227 };
228 }
