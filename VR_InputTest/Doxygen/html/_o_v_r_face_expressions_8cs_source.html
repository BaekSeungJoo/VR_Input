273 public void CopyTo(float[] array, int startIndex = 0)
274 {
275 if (array == null)
276 {
277 throw new ArgumentNullException(nameof(array));
278 }
279
280 if (startIndex < 0 || startIndex >= array.Length)
281 {
282 throw new ArgumentOutOfRangeException(nameof(startIndex),
283 startIndex,
284 $"Value must be between 0 to {array.Length - 1}");
285 }
286
287 if (array.Length - startIndex < (int)FaceExpression.Max)
288 {
289 throw new ArgumentException(
290 $"Capacity is too small - required {(int)FaceExpression.Max}, available {array.Length - startIndex}.",
291 nameof(array));
292 }
293
294 CheckValidity();
295 for (int i = 0; i < (int)FaceExpression.Max; i++)
296 {
297 array[i + startIndex] = _currentFaceState.ExpressionWeights[i];
298 }
299 }
