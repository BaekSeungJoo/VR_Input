53 // Get request for a specified request id
54 private VoiceServiceRequest GetRequest(string requestId)
55 {
56 HashSet<VoiceServiceRequest> requests = _voiceService.Requests;
57 if (requests == null || requests.Count == 0)
58 {
59 return null;
60 }
61 foreach (var request in requests)
62 {
63 string checkRequestId = request?.Options?.RequestId;
64 if (string.Equals(requestId, checkRequestId))
65 {
66 return request;
67 }
68 }
69 return requests.First();
70 }

92 public void onStoppedListening(int reason, string requestId)
93 {
94 // Request callbacks
95 var request = GetRequest(requestId);
96 if (request is VoiceSDKImplRequest implRequest)
97 {
98 implRequest.HandleStopListening();
99 }
100
101 // Event callbacks
102 VoiceEvents.OnStoppedListening?.Invoke();
103 switch((StoppedListeningReason)reason){
104 case StoppedListeningReason.NoReasonProvided:
105 break;
106 case StoppedListeningReason.Inactivity:
107 VoiceEvents.OnStoppedListeningDueToInactivity?.Invoke();
108 break;
109 case StoppedListeningReason.Timeout:
110 VoiceEvents.OnStoppedListeningDueToTimeout?.Invoke();
111 break;
112 case StoppedListeningReason.Deactivation:
113 VoiceEvents.OnStoppedListeningDueToDeactivation?.Invoke();
114 break;
115 }
116 }
