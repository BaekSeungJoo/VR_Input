323 {
324 // Check for changes

328 {
329 checkID = Configuration.GetApplicationId();
330 if (!string.IsNullOrEmpty(checkID))
331 {
332 checkName = Configuration.GetApplicationInfo().name;
333 }
334 }
335 // Reset
336 if (!string.Equals(_appName, checkName) || !string.Equals(_appID, checkID))
337 {
338 // Refresh app data

340 _appID = checkID;

344 if (!string.IsNullOrEmpty(newToken))
345 {

410 {
411 // Application info
412 Data.Info.WitAppInfo appInfo = Configuration.GetApplicationInfo();
413
414 // Indent
415 EditorGUI.indentLevel++;
416
417 // Iterate tabs
418 if (_tabs != null)
419 {
420 GUILayout.BeginHorizontal();
421 for (int i = 0; i < _tabs.Length; i++)
422 {
423 // Enable if not selected
424 GUI.enabled = _requestTab != i;
425 // If valid and clicked, begin selecting
426 if (null != appInfo.id &&_tabs[i].ShouldTabShow(appInfo))
427 {
428 if (WitEditorUI.LayoutTabButton(_tabs[i].GetTabText(true)))
429 {
430 _requestTab = i;
431 }
432 }
433 // If invalid, stop selecting
434 else if (_requestTab == i)
435 {
436 _requestTab = -1;
437 }
438 }
439
440 GUI.enabled = true;
441 GUILayout.EndHorizontal();
442
443 // Layout selected tab using property id
444 string propertyID = _requestTab >= 0 && _requestTab < _tabs.Length
445 ? _tabs[_requestTab].TabID
446 : string.Empty;
447 if (!string.IsNullOrEmpty(propertyID) && Configuration != null)
448 {
449 SerializedObject serializedObj = new SerializedObject(Configuration);
450 SerializedProperty serializedProp = serializedObj.FindProperty(_tabs[_requestTab].GetPropertyName(propertyID));
451 if (serializedProp == null)
452 {
453 WitEditorUI.LayoutErrorLabel(_tabs[_requestTab].GetTabText(false));
454 }
455 else if (!serializedProp.isArray)
456 {
457 EditorGUILayout.PropertyField(serializedProp);
458 }
459 else if (serializedProp.arraySize == 0)
460 {
461 WitEditorUI.LayoutErrorLabel(_tabs[_requestTab].GetTabText(false));
462 }
463 else
464 {
465 for (int i = 0; i < serializedProp.arraySize; i++)
466 {
467 SerializedProperty serializedPropChild = serializedProp.GetArrayElementAtIndex(i);
468 EditorGUILayout.PropertyField(serializedPropChild);
469 }
470 }
471
472 serializedObj.ApplyModifiedProperties();
473 }
474 }

477 EditorGUI.indentLevel--;

531 private static void GenerateManifest(WitConfiguration configuration, bool openManifest)
532 {
533 var instanceKey = Telemetry.StartEvent(Telemetry.TelemetryEventId.GenerateManifest);
534 AssemblyWalker.AssembliesToIgnore = new HashSet<string>(configuration.excludedAssemblies);
535
536 // Generate
537 var startGenerationTime = DateTime.UtcNow;
538 var appInfo = configuration.GetApplicationInfo();
539 var manifest = ManifestGenerator.GenerateManifest(appInfo.name, appInfo.id);
540 var endGenerationTime = DateTime.UtcNow;
541
542 // Get file path
543 var fullPath = configuration.GetManifestEditorPath();
544 if (string.IsNullOrEmpty(fullPath) || !File.Exists(fullPath))
545 {
546 fullPath = GetManifestPullPath(configuration, true);
547 }
548
549 // Write to file
550 try
551 {
552 var writer = new StreamWriter(fullPath);
553 writer.NewLine = "\n";
554 writer.WriteLine(manifest);
555 writer.Close();
556 }
557 catch (Exception e)
558 {
559 VLog.E($"Conduit manifest failed to generate\nPath: {fullPath}\n{e}");
560 Telemetry.EndEventWithFailure(instanceKey, e.Message);
561 return;
562 }
563 try
564 {
565 var incompatibleSignatures = string.Join(" ", AssemblyMiner.IncompatibleSignatureFrequency.Keys);
566 Telemetry.AnnotateEvent(instanceKey, Telemetry.AnnotationKey.IncompatibleSignatures, incompatibleSignatures);
567
568 var compatibleSignatures = string.Join(" ", AssemblyMiner.SignatureFrequency.Keys);
569 Telemetry.AnnotateEvent(instanceKey, Telemetry.AnnotationKey.CompatibleSignatures, compatibleSignatures);
570 }
571 catch (Exception e)
572 {
573 VLog.W($"Failed to collect signature telemetry. Exception: {e}");
574 }
575
576
577 Telemetry.EndEvent(instanceKey, Telemetry.ResultType.Success);
578 Statistics.SuccessfulGenerations++;
579 Statistics.AddFrequencies(AssemblyMiner.SignatureFrequency);
580 Statistics.AddIncompatibleFrequencies(AssemblyMiner.IncompatibleSignatureFrequency);
581 var generationTime = endGenerationTime - startGenerationTime;
582 var unityPath = fullPath.Replace(Application.dataPath, "Assets");
583 AssetDatabase.ImportAsset(unityPath);
584
585 var configName = configuration.name;
586 var manifestName = Path.GetFileNameWithoutExtension(unityPath);
587 #if UNITY_2021_2_OR_NEWER
588 var configPath = AssetDatabase.GetAssetPath(configuration);
589 configName = $"<a href=\"{configPath}\">{configName}</a>";
590 manifestName = $"<a href=\"{unityPath}\">{manifestName}</a>";
591 #endif
592 VLog.D($"Conduit manifest generated\nConfiguration: {configName}\nManifest: {manifestName}\nGeneration Time: {generationTime.TotalMilliseconds} ms");
593
594 if (openManifest)
595 {
596 UnityEditorInternal.InternalEditorUtility.OpenFileAtLineExternal(fullPath, 1);
597 }
598 }

612 // Sync entities
613 private void SyncEntities(Action successCallback = null)
614 {
615 var instanceKey = Telemetry.StartEvent(Telemetry.TelemetryEventId.SyncEntities);
616
617 if (!EditorUtility.DisplayDialog("Synchronizing with Wit.Ai entities", "This will synchronize local enums with Wit.Ai entities. Part of this process involves generating code locally and may result in overwriting existing code. Please make sure to backup your work before proceeding.", "Proceed", "Cancel", DialogOptOutDecisionType.ForThisSession, ENTITY_SYNC_CONSENT_KEY))
618 {
619 Telemetry.EndEvent(instanceKey, Telemetry.ResultType.Cancel);
620 VLog.D("Entity Sync cancelled");
621 return;
622 }
623
624 // Fail without server token
625 var validServerToken = WitConfigurationUtility.IsServerTokenValid(_serverToken);
626 if (!validServerToken)
627 {
628 Telemetry.EndEventWithFailure(instanceKey, "Invalid server token");
629 VLog.E($"Conduit Sync Failed\nError: Invalid server token");
630 return;
631 }
632
633 // Generate
634 if (_enumSynchronizer == null)
635 {
636 _enumSynchronizer = new EnumSynchronizer(Configuration, AssemblyWalker, new FileIo(), VRequestFactory);
637 }
638
639 // Sync
640 _syncInProgress = true;
641 EditorUtility.DisplayProgressBar("Conduit Entity Sync", "Generating Manifest.", 0f );
642 GenerateManifest(Configuration, false);
643
644 var manifest = LoadManifest(Configuration.ManifestLocalPath);
645
646 const float initializationProgress = 0.1f;
647 EditorUtility.DisplayProgressBar("Conduit Entity Sync", "Synchronizing entities. Please wait...", initializationProgress);
648 VLog.D("Synchronizing enums with Wit.Ai entities");
649 CoroutineUtility.StartCoroutine(_enumSynchronizer.SyncWitEntities(manifest, (success, data) =>
650 {
651 _syncInProgress = false;
652 EditorUtility.ClearProgressBar();
653 if (!success)
654 {
655 Telemetry.EndEventWithFailure(instanceKey, data);
656 VLog.E($"Conduit failed to synchronize entities\nError: {data}");
657 }
658 else
659 {
660 Telemetry.EndEvent(instanceKey, Telemetry.ResultType.Success);
661 VLog.D("Conduit successfully synchronized entities");
662 successCallback?.Invoke();
663 }
664 },
665 (status, progress) =>
666 {
667 EditorUtility.DisplayProgressBar("Conduit Entity Sync", status,
668 initializationProgress + (1f - initializationProgress) * progress);
669 }));
670 }
