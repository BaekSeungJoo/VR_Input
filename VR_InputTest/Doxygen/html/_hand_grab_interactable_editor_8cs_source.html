66 private void Mirror()
67 {
68 HandGrabInteractable mirrorInteractable =
69 HandGrabUtils.CreateHandGrabInteractable(_target.RelativeTo,
70 $"{_target.gameObject.name}_mirror");
71
72 var data = HandGrabUtils.SaveData(_target);
73 data.poses = null;
74 HandGrabUtils.LoadData(mirrorInteractable, data);
75 foreach (HandGrabPose point in _target.HandGrabPoses)
76 {
77 HandGrabPose mirrorPose = HandGrabUtils.CreateHandGrabPose(mirrorInteractable.transform,
78 mirrorInteractable.RelativeTo);
79 HandGrabUtils.MirrorHandGrabPose(point, mirrorPose, _target.RelativeTo);
80 mirrorPose.transform.SetParent(mirrorInteractable.transform);
81 mirrorInteractable.HandGrabPoses.Add(mirrorPose);
82 }
