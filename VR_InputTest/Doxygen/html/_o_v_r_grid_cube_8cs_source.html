67 void UpdateCubeGrid()
68 {
69 //todo: enable for Unity Input System
70#if ENABLE_LEGACY_INPUT_MANAGER
71 // Toggle the grid cube display on 'G'
72 if (Input.GetKeyDown(GridKey))
73 {
74 if (CubeGridOn == false)
75 {
76 CubeGridOn = true;
77 Debug.LogWarning("CubeGrid ON");
78 if (CubeGrid != null)
79 CubeGrid.SetActive(true);
80 else
81 CreateCubeGrid();
82 }
83 else
84 {
85 CubeGridOn = false;
86 Debug.LogWarning("CubeGrid OFF");
87
88 if (CubeGrid != null)
89 CubeGrid.SetActive(false);
90 }
91 }
92#endif
93
94 if (CubeGrid != null)
95 {
96 // Set cube colors to let user know if camera is tracking
97 CubeSwitchColor = !OVRManager.tracker.isPositionTracked;
98
99 if (CubeSwitchColor != CubeSwitchColorOld)
100 CubeGridSwitchColor(CubeSwitchColor);
101 CubeSwitchColorOld = CubeSwitchColor;
102 }
103 }

105 void CreateCubeGrid()
106 {
107 Debug.LogWarning("Create CubeGrid");
108
109 // Create the visual cube grid
110 CubeGrid = new GameObject("CubeGrid");
111 // Set a layer to target a specific camera
112 CubeGrid.layer = CameraController.gameObject.layer;
113
114 for (int x = -gridSizeX; x <= gridSizeX; x++)
115 for (int y = -gridSizeY; y <= gridSizeY; y++)
116 for (int z = -gridSizeZ; z <= gridSizeZ; z++)
117 {
118 // Set the cube type:
119 // 0 = non-axis cube
120 // 1 = axis cube
121 // 2 = center cube
122 int CubeType = 0;
123 if ((x == 0 && y == 0) || (x == 0 && z == 0) || (y == 0 && z == 0))
124 {
125 if ((x == 0) && (y == 0) && (z == 0))
126 CubeType = 2;
127 else
128 CubeType = 1;
129 }
130
131 GameObject cube = GameObject.CreatePrimitive(PrimitiveType.Cube);
132
133 BoxCollider bc = cube.GetComponent<BoxCollider>();
134 bc.enabled = false;
135
136 cube.layer = CameraController.gameObject.layer;
137
138 // No shadows
139 Renderer r = cube.GetComponent<Renderer>();
140
141#if UNITY_4_0 || UNITY_4_1 || UNITY_4_2 || UNITY_4_3 || UNITY_4_5 || UNITY_4_6
142 // Renderer.castShadows was deprecated starting in Unity 5.0
143 r.castShadows = false;
144#else
145 r.shadowCastingMode = UnityEngine.Rendering.ShadowCastingMode.Off;
146#endif
147
148 r.receiveShadows = false;
149
150 // Cube line is white down the middle
151 if (CubeType == 0)
152 r.material.color = Color.red;
153 else if (CubeType == 1)
154 r.material.color = Color.white;
155 else
156 r.material.color = Color.yellow;
157
158 cube.transform.position =
159 new Vector3(((float)x * gridScale),
160 ((float)y * gridScale),
161 ((float)z * gridScale));
162
163 float s = 0.7f;
164
165 // Axis cubes are bigger
166 if (CubeType == 1)
167 s = 1.0f;
168 // Center cube is the largest
169 if (CubeType == 2)
170 s = 2.0f;
171
172 cube.transform.localScale =
173 new Vector3(cubeScale * s, cubeScale * s, cubeScale * s);
174
175 cube.transform.parent = CubeGrid.transform;
176 }
177 }

183 void CubeGridSwitchColor(bool CubeSwitchColor)
184 {
185 Color c = Color.red;
186 if (CubeSwitchColor == true)
187 c = Color.blue;
188
189 var cachedTransform = CubeGrid.transform;
190 for (int i = 0; i < cachedTransform.childCount; i++)
191 {
192 var child = cachedTransform.GetChild(i);
193 Material m = child.GetComponent<Renderer>().material;
194 // Cube line is white down the middle
195 if (m.color == Color.red || m.color == Color.blue)
196 m.color = c;
197 }
198 }
