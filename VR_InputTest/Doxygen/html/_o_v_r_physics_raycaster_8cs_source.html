90 public override void Raycast(PointerEventData eventData, List<RaycastResult> resultAppendList)
91 {
92 // This function is closely based on PhysicsRaycaster.Raycast
93
94 if (eventCamera == null)
95 return;
96
97 if (!eventData.IsVRPointer())
98 return;
99
100 var ray = eventData.GetRay();
101
102 float dist = eventCamera.farClipPlane - eventCamera.nearClipPlane;
103
104 var hits = Physics.RaycastAll(ray, dist, finalEventMask);
105
106 if (hits.Length > 1)
107 System.Array.Sort(hits, (r1, r2) => r1.distance.CompareTo(r2.distance));
108
109 if (hits.Length != 0)
110 {
111 for (int b = 0, bmax = hits.Length; b < bmax; ++b)
112 {
113 var result = new RaycastResult
114 {
115 gameObject = hits[b].collider.gameObject,
116 module = this,
117 distance = hits[b].distance,
118 index = resultAppendList.Count,
119 worldPosition = hits[0].point,
120 worldNormal = hits[0].normal,
121 };
122 resultAppendList.Add(result);
123 }
124 }
125 }
