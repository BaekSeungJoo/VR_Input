32 // Animated hand visuals for a user of a Touch controller.
33 [RequireComponent(typeof(OVRGrabber))]
34 public class Hand : MonoBehaviour
35 {
36 public const string ANIM_LAYER_NAME_POINT = "Point Layer";
37 public const string ANIM_LAYER_NAME_THUMB = "Thumb Layer";
38 public const string ANIM_PARAM_NAME_FLEX = "Flex";
39 public const string ANIM_PARAM_NAME_POSE = "Pose";
40 public const float THRESH_COLLISION_FLEX = 0.9f;
41
42 public const float INPUT_RATE_CHANGE = 20.0f;
43
44 public const float COLLIDER_SCALE_MIN = 0.01f;
45 public const float COLLIDER_SCALE_MAX = 1.0f;
46 public const float COLLIDER_SCALE_PER_SECOND = 1.0f;
47
48 public const float TRIGGER_DEBOUNCE_TIME = 0.05f;
49 public const float THUMB_DEBOUNCE_TIME = 0.15f;
50
51 [SerializeField]
52 private OVRInput.Controller m_controller = OVRInput.Controller.None;
53
54 [SerializeField]
55 private Animator m_animator = null;
56
57 [SerializeField]
58 private HandPose m_defaultGrabPose = null;

60 private Collider[] m_colliders = null;

64 List<Renderer> m_showAfterInputFocusAcquired;
65
66 private int m_animLayerIndexThumb = -1;
67 private int m_animLayerIndexPoint = -1;
68 private int m_animParamIndexFlex = -1;
69 private int m_animParamIndexPose = -1;
70
71 private bool m_isPointing = false;
72 private bool m_isGivingThumbsUp = false;
73 private float m_pointBlend = 0.0f;
74 private float m_thumbsUpBlend = 0.0f;
75

78 private void Awake()
79 {
80 m_grabber = GetComponent<OVRGrabber>();

83 private void Start()
84 {
85 m_showAfterInputFocusAcquired = new List<Renderer>();
86
87 // Collision starts disabled. We'll enable it for certain cases such as making a fist.
88 m_colliders = this.GetComponentsInChildren<Collider>().Where(childCollider => !childCollider.isTrigger)
89 .ToArray();
90 CollisionEnable(false);
91
92 // Get animator layer indices by name, for later use switching between hand visuals
93 m_animLayerIndexPoint = m_animator.GetLayerIndex(ANIM_LAYER_NAME_POINT);
94 m_animLayerIndexThumb = m_animator.GetLayerIndex(ANIM_LAYER_NAME_THUMB);
95 m_animParamIndexFlex = Animator.StringToHash(ANIM_PARAM_NAME_FLEX);
96 m_animParamIndexPose = Animator.StringToHash(ANIM_PARAM_NAME_POSE);
97
98 OVRManager.InputFocusAcquired += OnInputFocusAcquired;
99 OVRManager.InputFocusLost += OnInputFocusLost;
100#if UNITY_EDITOR
101 OVRPlugin.SendEvent("custom_hand", (SceneManager.GetActiveScene().name == "CustomHands").ToString(),
102 "sample_framework");
103#endif
104 }
105
106 private void OnDestroy()
107 {
108 OVRManager.InputFocusAcquired -= OnInputFocusAcquired;
109 OVRManager.InputFocusLost -= OnInputFocusLost;
110 }
111
112 private void Update()
113 {
114 UpdateCapTouchStates();
115
116 m_pointBlend = InputValueRateChange(m_isPointing, m_pointBlend);
117 m_thumbsUpBlend = InputValueRateChange(m_isGivingThumbsUp, m_thumbsUpBlend);
118
119 float flex = OVRInput.Get(OVRInput.Axis1D.PrimaryHandTrigger, m_controller);
120
121 bool collisionEnabled = m_grabber.grabbedObject == null && flex >= THRESH_COLLISION_FLEX;
122 CollisionEnable(collisionEnabled);
123
124 UpdateAnimStates();
125 }
126
127 // Just checking the state of the index and thumb cap touch sensors, but with a little bit of
128 // debouncing.
129 private void UpdateCapTouchStates()
130 {
131 m_isPointing = !OVRInput.Get(OVRInput.NearTouch.PrimaryIndexTrigger, m_controller);
132 m_isGivingThumbsUp = !OVRInput.Get(OVRInput.NearTouch.PrimaryThumbButtons, m_controller);
133 }
134
135 private void LateUpdate()
136 {
137 // Hand's collision grows over a short amount of time on enable, rather than snapping to on, to help somewhat with interpenetration issues.
138 if (m_collisionEnabled && m_collisionScaleCurrent + Mathf.Epsilon < COLLIDER_SCALE_MAX)
139 {
140 m_collisionScaleCurrent = Mathf.Min(COLLIDER_SCALE_MAX,
141 m_collisionScaleCurrent + Time.deltaTime * COLLIDER_SCALE_PER_SECOND);
142 for (int i = 0; i < m_colliders.Length; ++i)
143 {
144 Collider collider = m_colliders[i];
145 collider.transform.localScale = new Vector3(m_collisionScaleCurrent, m_collisionScaleCurrent,
146 m_collisionScaleCurrent);
147 }
148 }
149 }

151 // Simple Dash support. Just hide the hands.
152 private void OnInputFocusLost()
153 {
154 if (gameObject.activeInHierarchy)
155 {
156 m_showAfterInputFocusAcquired.Clear();
157 Renderer[] renderers = GetComponentsInChildren<Renderer>();
158 for (int i = 0; i < renderers.Length; ++i)
159 {
160 if (renderers[i].enabled)
161 {
162 renderers[i].enabled = false;
163 m_showAfterInputFocusAcquired.Add(renderers[i]);
164 }
165 }
166
167 CollisionEnable(false);
168
169 m_restoreOnInputAcquired = true;
170 }
171 }
172
173 private void OnInputFocusAcquired()
174 {
175 if (m_restoreOnInputAcquired)
176 {
177 for (int i = 0; i < m_showAfterInputFocusAcquired.Count; ++i)
178 {
179 if (m_showAfterInputFocusAcquired[i])
180 {
181 m_showAfterInputFocusAcquired[i].enabled = true;
182 }
183 }
184
185 m_showAfterInputFocusAcquired.Clear();
186
187 // Update function will update this flag appropriately. Do not set it to a potentially incorrect value here.
188 //CollisionEnable(true);
189
190 m_restoreOnInputAcquired = false;
191 }
192 }

194 private float InputValueRateChange(bool isDown, float value)
195 {
196 float rateDelta = Time.deltaTime * INPUT_RATE_CHANGE;
197 float sign = isDown ? 1.0f : -1.0f;
198 return Mathf.Clamp01(value + rateDelta * sign);
199 }
200
201 private void UpdateAnimStates()
202 {
203 bool grabbing = m_grabber.grabbedObject != null;
204 HandPose grabPose = m_defaultGrabPose;
205 if (grabbing)
206 {
207 HandPose customPose = m_grabber.grabbedObject.GetComponent<HandPose>();
208 if (customPose != null) grabPose = customPose;
209 }
210
211 // Pose
212 HandPoseId handPoseId = grabPose.PoseId;
213 m_animator.SetInteger(m_animParamIndexPose, (int)handPoseId);
214
215 // Flex
216 // blend between open hand and fully closed fist
217 float flex = OVRInput.Get(OVRInput.Axis1D.PrimaryHandTrigger, m_controller);
218 m_animator.SetFloat(m_animParamIndexFlex, flex);
219
220 // Point
221 bool canPoint = !grabbing || grabPose.AllowPointing;
222 float point = canPoint ? m_pointBlend : 0.0f;
223 m_animator.SetLayerWeight(m_animLayerIndexPoint, point);
224
225 // Thumbs up
226 bool canThumbsUp = !grabbing || grabPose.AllowThumbsUp;
227 float thumbsUp = canThumbsUp ? m_thumbsUpBlend : 0.0f;
228 m_animator.SetLayerWeight(m_animLayerIndexThumb, thumbsUp);
229
230 float pinch = OVRInput.Get(OVRInput.Axis1D.PrimaryIndexTrigger, m_controller);
231 m_animator.SetFloat("Pinch", pinch);
232 }
233
234 private float m_collisionScaleCurrent = 0.0f;
235
236 private void CollisionEnable(bool enabled)
237 {
238 if (m_collisionEnabled == enabled)
239 {
240 return;
241 }
242
243 m_collisionEnabled = enabled;
244
245 if (enabled)
246 {
247 m_collisionScaleCurrent = COLLIDER_SCALE_MIN;
248 for (int i = 0; i < m_colliders.Length; ++i)
249 {
250 Collider collider = m_colliders[i];
251 collider.transform.localScale =
252 new Vector3(COLLIDER_SCALE_MIN, COLLIDER_SCALE_MIN, COLLIDER_SCALE_MIN);
253 collider.enabled = true;
254 }
255 }
256 else
257 {
258 m_collisionScaleCurrent = COLLIDER_SCALE_MAX;
259 for (int i = 0; i < m_colliders.Length; ++i)
260 {
261 Collider collider = m_colliders[i];
262 collider.enabled = false;
263 collider.transform.localScale =
264 new Vector3(COLLIDER_SCALE_MIN, COLLIDER_SCALE_MIN, COLLIDER_SCALE_MIN);
265 }
266 }
267 }
