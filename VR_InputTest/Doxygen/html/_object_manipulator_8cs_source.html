153 void FindHoverObject(Vector3 controllerPos, Quaternion controllerRot)
154 {
155 RaycastHit[] objectsHit = Physics.RaycastAll(controllerPos, controllerRot * Vector3.forward);
156 float closestObject = Mathf.Infinity;
157 float rayDistance = 2.0f;
158 bool showLaser = true;
159 Vector3 labelPosition = Vector3.zero;
160 foreach (RaycastHit hit in objectsHit)
161 {
162 float thisHitDistance = Vector3.Distance(hit.point, controllerPos);
163 if (thisHitDistance < closestObject)
164 {
165 hoverObject = hit.collider.gameObject;
166 closestObject = thisHitDistance;
167 rayDistance = grabObject ? thisHitDistance : thisHitDistance - 0.1f;
168 labelPosition = hit.point;
169 }
170 }
171
172 if (objectsHit.Length == 0)
173 {
174 hoverObject = null;
175 }
176
177 // if intersecting with an object, grab it
178 Collider[] hitColliders = Physics.OverlapSphere(controllerPos, 0.05f);
179 foreach (var hitCollider in hitColliders)
180 {
181 // use the last object, if there are multiple hits.
182 // If objects overlap, this would require improvements.
183 hoverObject = hitCollider.gameObject;
184 showLaser = false;
185 labelPosition = hitCollider.ClosestPoint(controllerPos);
186 labelPosition += (Camera.main.transform.position - labelPosition).normalized * 0.05f;
187 }
188
189 if (objectInfo && objectInstructionsLabel)
190 {
191 bool showObjectInfo = hoverObject || grabObject;
192 objectInfo.gameObject.SetActive(showObjectInfo);
193 Vector3 toObj = labelPosition - Camera.main.transform.position;
194 if (hoverObject && !grabObject)
195 {
196 Vector3 targetPos = labelPosition - toObj.normalized * 0.05f;
197 objectInfo.position = Vector3.Lerp(targetPos, objectInfo.position, grabTime);
198 objectInfo.rotation = Quaternion.LookRotation(toObj);
199 //objectInstructionsLabel.gameObject.SetActive(false);
200 objectInfo.localScale = Vector3.one * toObj.magnitude * 2.0f;
201 if (hoverObject.GetComponent<GrabObject>())
202 {
203 AssignInstructions(hoverObject.GetComponent<GrabObject>());
204 }
205 }
206 else if (grabObject)
207 {
208 Vector3 targetPos = controllerPos + (Camera.main.transform.position - controllerPos).normalized * 0.1f;
209 objectInfo.position = Vector3.Lerp(objectInfo.position, targetPos, grabTime);
210 ;
211 objectInfo.rotation = Quaternion.LookRotation(objectInfo.position - Camera.main.transform.position);
212 //objectInstructionsLabel.gameObject.SetActive(true);
213 objectInfo.localScale = Vector3.one;
214 if (grabObject.GetComponent<GrabObject>())
215 showLaser = grabObject.GetComponent<GrabObject>().showLaserWhileGrabbed;
216 }
217 }
218
219 // show/hide laser pointer
220 if (laser)
221 {
222 laser.positionCount = 2;
223 Vector3 pos1 = controllerPos + controllerRot * (Vector3.forward * 0.05f);
224 cursorPosition = controllerPos + controllerRot * (Vector3.forward * rayDistance);
225 laser.SetPosition(0, pos1);
226 laser.SetPosition(1, cursorPosition);
227 laser.enabled = (showLaser);
228 if (grabObject && grabObject.GetComponent<GrabObject>())
229 {
230 grabObject.GetComponent<GrabObject>().CursorPos(cursorPosition);
231 }
232 }

235 // the heavy lifting code for moving objects
236 void ManipulateObject(GameObject obj, Vector3 controllerPos, Quaternion controllerRot)
237 {
238 bool useDefaultManipulation = true;
239 Vector2 thumbstick = OVRInput.Get(OVRInput.Axis2D.PrimaryThumbstick, controller);
240
241 if (obj.GetComponent<GrabObject>())
242 {
243 useDefaultManipulation = false;
244 switch (obj.GetComponent<GrabObject>().objectManipulationType)
245 {
246 case GrabObject.ManipulationType.Default:
247 useDefaultManipulation = true;
248 break;
249 case GrabObject.ManipulationType.ForcedHand:
250 obj.transform.position = Vector3.Lerp(obj.transform.position, controllerPos, grabTime);
251 obj.transform.rotation = Quaternion.Lerp(obj.transform.rotation, controllerRot, grabTime);
252 break;
253 case GrabObject.ManipulationType.DollyHand:
254 float targetDist = localGrabOffset.z + thumbstick.y * 0.01f;
255 targetDist = Mathf.Clamp(targetDist, 0.1f, 2.0f);
256 localGrabOffset = Vector3.forward * targetDist;
257 obj.transform.position = Vector3.Lerp(obj.transform.position,
258 controllerPos + controllerRot * localGrabOffset, grabTime);
259 obj.transform.rotation = Quaternion.Lerp(obj.transform.rotation, controllerRot, grabTime);
260 break;
261 case GrabObject.ManipulationType.DollyAttached:
262 obj.transform.position = controllerPos + controllerRot * localGrabOffset;
263 obj.transform.rotation = controllerRot * localGrabRotation;
264 ClampGrabOffset(ref localGrabOffset, thumbstick.y);
265 break;
266 case GrabObject.ManipulationType.HorizontalScaled:
267 obj.transform.position = controllerPos + controllerRot * localGrabOffset;
268 if (!OVRInput.Get(OVRInput.Button.PrimaryIndexTrigger, controller))
269 {
270 obj.transform.localScale = ClampScale(obj.transform.localScale, thumbstick);
271 }
272 else
273 {
274 rotationOffset -= thumbstick.x;
275 ClampGrabOffset(ref localGrabOffset, thumbstick.y);
276 }
277
278 Vector3 newFwd = obj.GetComponent<GrabObject>().grabbedRotation * Vector3.forward;
279 newFwd = new Vector3(newFwd.x, 0, newFwd.z);
280 obj.transform.rotation = Quaternion.Euler(0, rotationOffset, 0) * Quaternion.LookRotation(newFwd);
281 break;
282 case GrabObject.ManipulationType.VerticalScaled:
283 obj.transform.position = controllerPos + controllerRot * localGrabOffset;
284 if (!OVRInput.Get(OVRInput.Button.PrimaryIndexTrigger, controller))
285 {
286 obj.transform.localScale = ClampScale(obj.transform.localScale, thumbstick);
287 }
288 else
289 {
290 rotationOffset -= thumbstick.x;
291 ClampGrabOffset(ref localGrabOffset, thumbstick.y);
292 }
293
294 Vector3 newUp = obj.GetComponent<GrabObject>().grabbedRotation * Vector3.up;
295 newUp = new Vector3(newUp.x, 0, newUp.z);
296 obj.transform.rotation =
297 Quaternion.LookRotation(Vector3.up, Quaternion.Euler(0, rotationOffset, 0) * newUp);
298 break;
299 case GrabObject.ManipulationType.Menu:
300 Vector3 targetPos = handGrabPosition + (handGrabRotation * Vector3.forward * 0.4f);
301 Quaternion targetRotation = Quaternion.LookRotation(targetPos - camGrabPosition);
302 obj.transform.position = Vector3.Lerp(obj.transform.position, targetPos, grabTime);
303 obj.transform.rotation = Quaternion.Lerp(obj.transform.rotation, targetRotation, grabTime);
304 break;
305 default:
306 useDefaultManipulation = true;
307 break;
308 }
309 }
310
311 if (useDefaultManipulation)
312 {
313 obj.transform.position = controllerPos + controllerRot * localGrabOffset;
314 obj.transform.Rotate(Vector3.up * -thumbstick.x);
315 ClampGrabOffset(ref localGrabOffset, thumbstick.y);
316 }
317 }

319 void ClampGrabOffset(ref Vector3 localOffset, float thumbY)
320 {
321 Vector3 projectedGrabOffset = localOffset + Vector3.forward * thumbY * 0.01f;
322 if (projectedGrabOffset.z > 0.1f)
323 {
324 localOffset = projectedGrabOffset;
325 }
326 }

328 Vector3 ClampScale(Vector3 localScale, Vector2 thumb)
329 {
330 float newXscale = localScale.x + thumb.x * 0.01f;
331 if (newXscale <= 0.1f) newXscale = 0.1f;
332 float newZscale = localScale.z + thumb.y * 0.01f;
333 if (newZscale <= 0.1f) newZscale = 0.1f;
334 return new Vector3(newXscale, 0.0f, newZscale);
335 }
