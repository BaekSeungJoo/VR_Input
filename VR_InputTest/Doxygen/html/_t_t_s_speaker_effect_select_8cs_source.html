46 // Refresh dropdown using voice settings
47 private void RefreshDropdowns()
48 {
49 if (!Speaker)
50 {
51 VLog.W("No speaker found");
52 return;
53 }
54
55 // Refresh characters
56 RefreshDropdown("character", _characterDropdown, CHARACTER_IDS);
57
58 // Refresh environments
59 RefreshDropdown("environments", _environmentDropdown, ENVIRONMENT_IDS);
60 }
61 // Refresh a specific dropdown
62 private void RefreshDropdown(string id, SimpleDropdownList dropdown, string[] options)
63 {
64 if (!dropdown)
65 {
66 VLog.W($"No {id} dropdown found");
67 return;
68 }
69 if (options == null || options.Length == 0)
70 {
71 VLog.W($"No {id} options found");
72 return;
73 }
74
75 // Load dropdown & select first index
76 dropdown.LoadDropdown(options);
77 dropdown.SelectOption(0);
78 }
79
80 // Apply voice on option select
81 private void OnCharacterSelected(int newIndex)
82 {
83 RefreshSsml();
84 }
85
86 // Apply voice on option select
87 private void OnEnvironmentSelected(int newIndex)
88 {
89 RefreshSsml();
90 }
91
92 // Refresh speaker ssml
93 private void RefreshSsml()
94 {
95 if (!Speaker)
96 {
97 VLog.W("No speaker found");
98 return;
99 }
100
101 // Get SSMLs
102 StringBuilder prepend = new StringBuilder();
103 StringBuilder append = new StringBuilder();
104
105 // Add ssml tags
106 prepend.Append("<speak>");
107 append.Append("</speak>");
108
109 // Get character & environment ids
110 string characterId = _characterDropdown.SelectedOption;
111 if (string.Equals(characterId, NONE_ID))
112 {
113 characterId = null;
114 }
115 string environmentId = _environmentDropdown.SelectedOption;
116 if (string.Equals(environmentId, NONE_ID))
117 {
118 environmentId = null;
119 }
120
121 // Add sfx tag
122 bool hasCharacter = !string.IsNullOrEmpty(characterId);
123 bool hasEnvironment = !string.IsNullOrEmpty(environmentId);
124 if (hasCharacter || hasEnvironment)
125 {
126 // Add prefix & postfix
127 prepend.Append("<sfx");
128 append.Insert(0, "</sfx>");
129
130 // Add character
131 if (hasCharacter)
132 {
133 prepend.Append($" character=\"{characterId.ToLower()}\"");
134 }
135 // Add environment
136 if (hasEnvironment)
137 {
138 prepend.Append($" environment=\"{environmentId.ToLower()}\"");
139 }
140
141 // Finalize
142 prepend.Append(">");
143 }
144
145 // Set SSML
146 Speaker.PrependedText = prepend.ToString();
147 Speaker.AppendedText = append.ToString();
148 }
149 }
