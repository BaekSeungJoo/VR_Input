182 void UpdateSpeedAndRotationScaleMultiplier()

185 playerController.GetMoveScaleMultiplier(ref moveScaleMultiplier);
186
187 //todo: enable for Unity Input System
188#if ENABLE_LEGACY_INPUT_MANAGER
189 if (Input.GetKeyDown(KeyCode.Alpha7))
190 {
191 moveScaleMultiplier -= speedRotationIncrement;
192 }
193 else if (Input.GetKeyDown(KeyCode.Alpha8))
194 {
195 moveScaleMultiplier += speedRotationIncrement;
196 }
197#endif
198
199 playerController.SetMoveScaleMultiplier(moveScaleMultiplier);
200
201 float rotationScaleMultiplier = 0.0f;
202 playerController.GetRotationScaleMultiplier(ref rotationScaleMultiplier);
203
204 //todo: enable for Unity Input System

206 if (Input.GetKeyDown(KeyCode.Alpha9))
207 {
208 rotationScaleMultiplier -= speedRotationIncrement;

211 {

214#endif
