63 public void Update(BodyDataAsset data, int dataVersion,
64 Transform trackingSpace = null)
65 {
66 LocalDataVersion = dataVersion;
67 _mapping = data.SkeletonMapping;
68
69 for (int i = 0; i < DIRTY_ARRAY_SIZE; ++i)
70 {
71 _dirtyJointsFromRoot[i] = ulong.MaxValue;
72 _dirtyLocalJoints[i] = ulong.MaxValue;
73 _dirtyWorldJoints[i] = ulong.MaxValue;
74 }
75 if (!data.IsDataValid)
76 {
77 return;
78 }
79
80 _scale = Matrix4x4.Scale(Vector3.one * data.RootScale);
81 _rootPose = data.Root;
82 _worldRoot = _rootPose;
83
84 if (trackingSpace != null)
85 {
86 _scale *= Matrix4x4.Scale(trackingSpace.lossyScale);
87 _worldRoot.position = trackingSpace.TransformPoint(_rootPose.position);
88 _worldRoot.rotation = trackingSpace.rotation * _rootPose.rotation;
89 }
90
91 for (int i = 0; i < Constants.NUM_BODY_JOINTS; ++i)
92 {
93 _originalPoses[i] = data.JointPoses[i];
94 }
95 }

141 private void UpdateJointPoseFromRoot(BodyJointId jointId)
142 {
143 if (!CheckJointDirty(jointId, _dirtyJointsFromRoot))
144 {
145 return;
146 }
147
148 Pose originalPose = _originalPoses[(int)jointId];
149 Vector3 posFromRoot = Quaternion.Inverse(_rootPose.rotation) *
150 (originalPose.position - _rootPose.position);
151 Quaternion rotFromRoot = Quaternion.Inverse(_rootPose.rotation) *
152 originalPose.rotation;
153 _posesFromRoot[(int)jointId] = new Pose(posFromRoot, rotFromRoot);
154
155 SetJointClean(jointId, _dirtyJointsFromRoot);
156 }
157
158 private void UpdateLocalJointPose(BodyJointId jointId)
159 {
160 if (!CheckJointDirty(jointId, _dirtyLocalJoints))
161 {
162 return;
163 }
164
165 if (_mapping.TryGetParentJointId(jointId, out BodyJointId parentId) &&
166 parentId >= BodyJointId.Body_Root)
167 {
168 Pose originalPose = _originalPoses[(int)jointId];
169 Pose parentPose = _originalPoses[(int)parentId];
170
171 Vector3 localPos = Quaternion.Inverse(parentPose.rotation) *
172 (originalPose.position - parentPose.position);
173 Quaternion localRot = Quaternion.Inverse(parentPose.rotation) *
174 originalPose.rotation;
175 _localPoses[(int)jointId] = new Pose(localPos, localRot);
176 }
177 else
178 {
179 _localPoses[(int)jointId] = Pose.identity;
180 }
181 SetJointClean(jointId, _dirtyLocalJoints);
182 }
183
184 private void UpdateWorldJointPose(BodyJointId jointId)
185 {
186 if (!CheckJointDirty(jointId, _dirtyWorldJoints))
187 {
188 return;
189 }
190
191 Pose fromRoot = GetJointPoseFromRoot(jointId);
192 fromRoot.position = _scale * fromRoot.position;
193 fromRoot.Postmultiply(GetWorldRootPose());
194 _worldPoses[(int)jointId] = fromRoot;
195 SetJointClean(jointId, _dirtyWorldJoints);
196 }
197
198 private void UpdateAllWorldPoses()
199 {
200 foreach (BodyJointId joint in _mapping.Joints)
201 {
202 UpdateWorldJointPose(joint);
203 }
204 }
205

210 UpdateLocalJointPose(joint);
211 }
212 }

214 private void UpdateAllPosesFromRoot()

218 UpdateJointPoseFromRoot(joint);
219 }
220 }

222 private bool CheckJointDirty(BodyJointId jointId, ulong[] dirtyFlags)
223 {
224 int outerIdx = (int)jointId / 64;
225 int innerIdx = (int)jointId % 64;
226 return (dirtyFlags[outerIdx] & (1UL << innerIdx)) != 0;
227 }
228
229 private void SetJointClean(BodyJointId jointId, ulong[] dirtyFlags)
230 {
231 int outerIdx = (int)jointId / 64;
232 int innerIdx = (int)jointId % 64;
233 dirtyFlags[outerIdx] = dirtyFlags[outerIdx] & ~(1UL << innerIdx);
234 }
