131 private void ScaledHandPoseSlider()
132 {
133 float minScale = MIN_SCALE;
134 float maxScale = MAX_SCALE;
135 for (int i = 0; i < _handGrabPoses.Count; i++)
136 {
137 HandGrabPose grabPose = _handGrabPoses[i];
138 float scale = grabPose.transform.lossyScale.x;
139 if (scale < minScale)
140 {
141 minScale = scale;
142 }
143 if (scale > maxScale)
144 {
145 maxScale = scale;
146 }
147 }
148 _handScale = EditorGUILayout.Slider(_handScale, minScale, maxScale);
149 Rect backRect = GUILayoutUtility.GetLastRect();
150 backRect.x += LEFT_MARGIN;
151 backRect.width -= RIGHT_MARGIN;
152
153 for (int i = 0; i < _handGrabPoses.Count; i++)
154 {
155 HandGrabPose grabPose = _handGrabPoses[i];
156 if (grabPose == null)
157 {
158 continue;
159 }
160 float x = backRect.x + Mathf.InverseLerp(minScale, maxScale, grabPose.transform.lossyScale.x) * backRect.width;
161 Rect poseRect = new Rect(x - POSE_RECT_WIDTH * 0.5f, backRect.y, POSE_RECT_WIDTH, backRect.height);
162 EditorGUI.LabelField(poseRect, EditorGUIUtility.IconContent("curvekeyframeselected"));
163 }

166 private void CheckUniqueScales()
167 {
168 _scalesSet.Clear();
169 for (int i = 0; i < _handGrabPoses.Count; i++)
170 {
171 HandGrabPose grabPose = _handGrabPoses[i];
172 if (grabPose == null)
173 {
174 continue;
175 }
176
177 float scale = grabPose.transform.lossyScale.x /
178 _relativeToRef.RelativeTo.lossyScale.x;
179 if (_scalesSet.Contains(scale))
180 {
181 EditorGUILayout.HelpBox(
182 $"Duplicated {nameof(HandGrabPose)} of scale {scale} at index {i}.",
183 MessageType.Warning);
184 }
185 _scalesSet.Add(scale);
186 }
187 }
188
189 private void CheckUniqueHandedness()
190 {
191 bool handednessSet = false;
192 Handedness validHandedness = Handedness.Left;
193 for (int i = 0; i < _handGrabPoses.Count; i++)
194 {
195 HandGrabPose grabPose = _handGrabPoses[i];
196 if (grabPose == null || grabPose.HandPose == null)
197 {
198 continue;
199 }
200 Handedness grabPoseHandedness = grabPose.HandPose.Handedness;
201 if (!handednessSet)
202 {
203 handednessSet = true;
204 validHandedness = grabPoseHandedness;
205 }
206 else if (grabPoseHandedness != validHandedness)
207 {
208 EditorGUILayout.HelpBox($"Different Handedness at index {i}. " +
209 $"Ensure all HandGrabPoses have the same Handedness", MessageType.Warning);
210 }
211 }

227 private void AddHandGrabPose(float scale)
228 {
229 HandGrabPose handGrabPose = HandGrabUtils.CreateHandGrabPose(_target.transform,
230 _relativeToRef.RelativeTo);
231 float relativeScale = scale / _relativeToRef.RelativeTo.lossyScale.x;
232
233 bool rangeFound = GrabPoseFinder.FindInterpolationRange(relativeScale, _handGrabPoses,
234 out HandGrabPose from, out HandGrabPose to, out float t);
235
236 handGrabPose.transform.localScale = Vector3.one * relativeScale;
237 _handGrabPoses.Add(handGrabPose);
238 EditorUtility.SetDirty(_target);
239
240 if (!rangeFound)
241 {
242 return;
243 }
244
245 Pose relativePose = Pose.identity;
246 PoseUtils.Lerp(from.RelativePose, to.RelativePose, t, ref relativePose);
247 Pose rootPose = PoseUtils.GlobalPoseScaled(_relativeToRef.RelativeTo, relativePose);
248 HandPose resultHandPose = new HandPose(from.HandPose);
249 HandPose.Lerp(from.HandPose, to.HandPose, t, ref resultHandPose);
250 Grab.GrabSurfaces.IGrabSurface surface = from.SnapSurface?.CreateDuplicatedSurface(handGrabPose.gameObject);
251
252 handGrabPose.InjectAllHandGrabPose(_relativeToRef.RelativeTo);
253 handGrabPose.InjectOptionalHandPose(resultHandPose);
254 handGrabPose.InjectOptionalSurface(surface);
255 handGrabPose.transform.SetPose(rootPose);
256 }

262 _handGrabPoses.AddRange(handGrabPoses);
263 EditorUtility.SetDirty(_target);

268 private void UpdateGhost()
269 {
270 if (_handGrabPoses.Count == 0
271 || _handGrabPoses[0].HandPose == null)
272 {
273 DestroyGhost();
274 return;
275 }
276
277 Transform relativeTo = _relativeToRef.RelativeTo;
278 Pose rootPose = Pose.identity;
279 float relativeScale = _handScale / relativeTo.lossyScale.x;
280 bool rangeFound = GrabPoseFinder.FindInterpolationRange(relativeScale, _handGrabPoses,
281 out HandGrabPose from, out HandGrabPose to, out float t);
282 if (!rangeFound)
283 {
284 DestroyGhost();
285 return;
286 }
287
288 HandPose.Lerp(from.HandPose, to.HandPose, t, ref _ghostHandPose);
289 PoseUtils.Lerp(from.RelativePose, to.RelativePose, t, ref rootPose);
290 rootPose = PoseUtils.GlobalPoseScaled(relativeTo, rootPose);
291 DisplayGhost(_ghostHandPose, rootPose, relativeScale);
292 }
293
294 private void DisplayGhost(HandPose handPose, Pose rootPose, float scale)
295 {
296 if (_handGhost != null
297 && _lastHandedness != handPose.Handedness)
298 {
299 DestroyGhost();
300 }
301
302 _lastHandedness = handPose.Handedness;
303 if (_handGhost == null)
304 {
305 HandGhost ghostPrototype = _ghostVisualsProvider.GetHand(_lastHandedness);
306 _handGhost = GameObject.Instantiate(ghostPrototype, _target.transform);
307 _handGhost.gameObject.hideFlags = HideFlags.HideAndDontSave;
308 }
309 _handGhost.transform.localScale = Vector3.one * scale;
310 _handGhost.SetPose(handPose, rootPose);
311 }
312
313 private void DestroyGhost()
314 {
315 if (_handGhost == null)
316 {
317 return;
318 }
319 GameObject.DestroyImmediate(_handGhost.gameObject);
320 _handGhost = null;
321 }
