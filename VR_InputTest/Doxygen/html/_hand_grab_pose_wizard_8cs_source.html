170 public void RecordPose()
171 {
172 if (!Application.isPlaying)
173 {
174 Debug.LogError("Recording tracked hands only works in Play Mode!", this);
175 return;
176 }
177
178 if (Hand == null)
179 {
180 Debug.LogError("Missing Hand reference.", this);
181 return;
182 }
183
184 if (_item == null)
185 {
186 Debug.LogError("Missing recordable item", this);
187 return;
188 }
189
190 HandPose trackedHandPose = TrackedPose();
191 if (trackedHandPose == null)
192 {
193 Debug.LogError("Tracked Pose could not be retrieved", this);
194 return;
195 }
196
197 if (!Hand.GetRootPose(out Pose handRoot))
198 {
199 Debug.LogError("Hand Root pose could not be retrieved", this);
200 return;
201 }
202
203 Pose gripPoint = PoseUtils.DeltaScaled(_item.transform, handRoot);
204 HandGrabPose point = AddHandGrabPose(trackedHandPose, gripPoint);
205 AttachGhost(point);
206 }
207
208 private HandPose TrackedPose()
209 {
210 if (!Hand.GetJointPosesLocal(out ReadOnlyHandJointPoses localJoints))
211 {
212 return null;
213 }
214 HandPose result = new HandPose(Hand.Handedness);
215 for (int i = 0; i < FingersMetadata.HAND_JOINT_IDS.Length; ++i)
216 {
217 HandJointId jointID = FingersMetadata.HAND_JOINT_IDS[i];
218 result.JointRotations[i] = localJoints[jointID].rotation;
219 }
220 return result;
221 }

223 private void AttachGhost(HandGrabPose point)
224 {
225 if (_ghostProvider == null)
226 {
227 return;
228 }
229 HandGhost ghostPrefab = _ghostProvider.GetHand(Hand.Handedness);
230 HandGhost ghost = GameObject.Instantiate(ghostPrefab, point.transform);
231 ghost.SetPose(point);
232 }

241 private HandGrabPose AddHandGrabPose(HandPose rawPose, Pose snapPoint)
242 {
243 HandGrabInteractable interactable = HandGrabUtils.CreateHandGrabInteractable(_item.transform);
244 var pointData = new HandGrabUtils.HandGrabPoseData()
245 {
246 handPose = rawPose,
247 scale = Hand.Scale / interactable.RelativeTo.lossyScale.x,
248 gripPose = snapPoint,
249 };
250 return HandGrabUtils.LoadHandGrabPose(interactable, pointData);
251 }

272 private void SaveToAsset()
273 {
274 if (_posesCollection == null)
275 {
276 GenerateCollectionAsset();
277 }
278 var savedPoses = new List<HandGrabUtils.HandGrabInteractableData>();
279 foreach (HandGrabInteractable snap in _item.GetComponentsInChildren<HandGrabInteractable>(false))
280 {
281 savedPoses.Add(HandGrabUtils.SaveData(snap));
282 }
283 _posesCollection.StoreInteractables(savedPoses);
284 }
285
290 private void LoadFromAsset()
291 {
292 if (_posesCollection == null)
293 {
294 return;
295 }
296
297 foreach (var handPose in _posesCollection.InteractablesData)
298 {
299 LoadHandGrabInteractable(handPose);
300 }
301 }

303 public void GenerateCollectionAsset()
304 {
305 _posesCollection = ScriptableObject.CreateInstance<HandGrabInteractableDataCollection>();
306 string parentDir = Path.Combine("Assets", "HandGrabInteractableDataCollection");
307 if (!Directory.Exists(parentDir))
308 {
309 Directory.CreateDirectory(parentDir);
310 }
311 string name = _item != null ? _item.name : "Auto";
312 AssetDatabase.CreateAsset(_posesCollection, Path.Combine(parentDir, $"{name}_HandGrabCollection.asset"));
313 AssetDatabase.SaveAssets();
314 }
