82 public virtual TInteractable ComputeCandidate(
83 InteractableRegistry<TInteractor, TInteractable> registry,
84 TInteractor interactor,
85 out Vector3 bestHitPoint)
86 {
87 if (_detector == null)
88 {
89 _detector = new DistantPointDetector(DetectionFrustums);
90 }
91 if (_stableCandidate != null
92 && _detector.IsPointingWithoutAid(_stableCandidate.Colliders, out bestHitPoint))
93 {
94 return _stableCandidate;
95 }
96
97 if (_stableCandidate != null
98 && !_detector.ComputeIsPointing(_stableCandidate.Colliders, false,
99 out float score, out Vector3 hitPoint))
100 {
101 _stableCandidate = null;
102 }
103
104 TInteractable candidate = ComputeBestInteractable(
105 registry.List(interactor),
106 _stableCandidate == null,
107 out bestHitPoint);
108 if (candidate != _pointedCandidate)
109 {
110 _pointedCandidate = candidate;
111 if (candidate != null)
112 {
113 _hoverStartTime = Time.time;
114 }
115 }
116
117 if ((_stableCandidate == null
118 && candidate != null)
119 || (_stableCandidate != null
120 && candidate != null
121 && _stableCandidate != candidate
122 && Time.time - _hoverStartTime >= _detectionDelay))
123 {
124 _pointedCandidate = null;
125 _stableCandidate = candidate;
126 }
127
128 return _stableCandidate;
129 }

131 private TInteractable ComputeBestInteractable(IEnumerable<TInteractable> candidates,
132 bool narrowSearch, out Vector3 bestHitPoint)
133 {
134 TInteractable bestInteractable = null;
135 float bestScore = float.NegativeInfinity;
136 bestHitPoint = Vector3.zero;
137
138 foreach (TInteractable candidate in candidates)
139 {
140 if (_detector.ComputeIsPointing(candidate.Colliders, narrowSearch, out float score, out Vector3 hitPoint)
141 && score > bestScore)
142 {
143 bestScore = score;
144 bestInteractable = candidate;
145 bestHitPoint = hitPoint;
146 }
147 }
148 return bestInteractable;
149 }
