345 private void ScrollThroughDestinations()
346 {
347 if (PressUp())
348 {
349 if (!OnlyPushUpOnce)
350 {
351 DestinationIndex--;
352 if (DestinationIndex < 0)
353 {
354 DestinationIndex = DestinationAPINames.Count - 1;
355 }
356 OnlyPushUpOnce = true;
357 UpdateDestinationsConsole();
358 }
359 }
360 else if (PressDown())
361 {
362 if (!OnlyPushUpOnce)
363 {
364 DestinationIndex++;
365 if (DestinationIndex >= DestinationAPINames.Count)
366 {
367 DestinationIndex = 0;
368 }
369 OnlyPushUpOnce = true;
370 UpdateDestinationsConsole();
371 }
372 }
373 else
374 {
375 OnlyPushUpOnce = false;
376 }
