103 public void PopulateParametersFromNode(WitResponseNode responseNode)
104 {
105 _parametersOfType.Clear();
106
107 var parameters = new Dictionary<string, ConduitParameterValue>();
108 foreach (var entity in responseNode.AsObject["entities"].Childs)
109 {
110 var parameterName = entity[0]["role"].Value;
111 var parameterValue = entity[0]["value"].Value;
112 var parameterTypes = GetParameterTypes(entity[0]["name"].Value, parameterValue).ToList();
113
114 foreach (var parameterType in parameterTypes)
115 {
116 if (!_parametersOfType.ContainsKey(parameterType))
117 {
118 _parametersOfType.Add(parameterType, new List<string>());
119 }
120
121 _parametersOfType[parameterType].Add(parameterName);
122 }
123
124 parameters.Add(parameterName, new ConduitParameterValue(parameterValue, parameterTypes.First()));
125 }
126 parameters.Add(WitResponseNodeReservedName, new ConduitParameterValue(responseNode, typeof(WitResponseNode)));
127
128 PopulateParameters(parameters);
129 }

207 public object GetParameterValue(ParameterInfo formalParameter, Dictionary<string, string> parameterMap, bool relaxed)
208 {
209 if (SupportedSpecializedParameter(formalParameter))
210 {
211 return this.GetSpecializedParameter(formalParameter);
212 }
213
214 var actualParameterName = GetActualParameterName(formalParameter, parameterMap, relaxed);
215 if (string.IsNullOrEmpty(actualParameterName))
216 {
217 return null;
218 }
219
220 if (ActualParameters.TryGetValue(actualParameterName, out var parameterValue))
221 {
222 if (formalParameter.ParameterType == typeof(string))
223 {
224 return parameterValue.ToString();
225 }
226 else if (formalParameter.ParameterType.IsEnum)
227 {
228
229 try
230 {
231 return Enum.Parse(formalParameter.ParameterType, ConduitUtilities.SanitizeString(parameterValue.ToString()), true);
232 }
233 catch (Exception e)
234 {
235 VLog.E($"Parameter Provider - Parameter '{parameterValue}' could not be cast to enum\nEnum Type: {formalParameter.ParameterType.FullName}\n{e}");
236 throw;
237 }
238 }
239 else
240 {
241 try
242 {
243 return Convert.ChangeType(parameterValue, formalParameter.ParameterType);
244 }
245 catch (Exception e)
246 {
247 VLog.E($"Parameter Provider - Parameter '{parameterValue}' could not be cast\nType: {formalParameter.ParameterType.FullName}\n{e}");
248 return null;
249 }
250
251 }
252 }
253
254 return null;
255 }

299 protected virtual object GetSpecializedParameter(ParameterInfo formalParameter)
300 {
301 if (_specializedParameters.ContainsKey(formalParameter.ParameterType))

303 var parameterName = _specializedParameters[formalParameter.ParameterType];
304 if (ActualParameters.ContainsKey(parameterName))
305 {
306 return ActualParameters[parameterName];

310 // Log warning when not found
311 var error = new StringBuilder();
312 error.AppendLine("Specialized parameter not found");
313 error.AppendLine($"Parameter Type: {formalParameter.ParameterType}");
314 error.AppendLine($"Parameter Name: {formalParameter.Name}");
315 error.AppendLine($"Actual Parameters: {ActualParameters.Keys.Count}");
316 foreach (var key in ActualParameters.Keys)
317 {
318 var val = ActualParameters[key] == null ? "NULL" : ActualParameters[key].GetType().ToString();
319 error.AppendLine($"\t{key}: {val}");
320 }
321 VLog.W(error.ToString());
322 return null;
323 }

333 private IEnumerable<Type> GetParameterTypes(string typeString, string value)
334 {
335 if (_customTypes.ContainsKey(typeString))
336 {
337 return new List<Type>() { _customTypes[typeString] };
338 }
339
340 if (!BuiltInTypes.ContainsKey(typeString) || BuiltInTypes[typeString].Count == 0)
341 {
342 return new List<Type>() { typeof(string) };
343 }
344
345 return BuiltInTypes[typeString].Where(type => PerfectTypeMatch(type, value)).ToList();
346 }

348 private bool PerfectTypeMatch(Type targetType, string value)
349 {
350 try
351 {
352 var valueAsTarget = Convert.ChangeType(value, targetType);
353
354 if (valueAsTarget == null)
355 {
356 return false;
357 }
358
359 if (!targetType.IsPrimitive)
360 {
361 return true;
362 }
363
364 return value.Equals(valueAsTarget.ToString());
365 }
366 catch (Exception)
367 {
368 return false;
369 }
370 }

380 private string GetActualParameterName(ParameterInfo formalParameter, Dictionary<string, string> parameterMap,
381 bool relaxed)
382 {
383 var formalParameterName = formalParameter.Name;
384 string targetActualParameterName;
385
386 if (parameterMap.ContainsKey(formalParameterName))
387 {
388 targetActualParameterName = parameterMap[formalParameterName];
389 }
390 else
391 {
392 targetActualParameterName = formalParameterName;
393 }
394
395 if (ActualParameters.ContainsKey(targetActualParameterName))
396 {
397 return targetActualParameterName;
398 }
399
400 if (_parameterToRoleMap.ContainsKey(targetActualParameterName))
401 {
402 var roleName = _parameterToRoleMap[targetActualParameterName];
403 if (!string.IsNullOrEmpty(roleName) && this.ActualParameters.ContainsKey(roleName))
404 {
405 return roleName;
406 }
407 }
408
409 if (!relaxed)
410 {
411 VLog.E($"Parameter '{formalParameterName}' is missing");
412 return null;
413 }
414
415 var possibleNames = GetParameterNamesOfType(formalParameter.ParameterType);
416 if (possibleNames.Count != 1)
417 {
418 VLog.E(
419 $"Got multiple parameters of type {formalParameter.ParameterType} but none with the correct name");
420 return null;
421 }
422
423 targetActualParameterName = possibleNames[0];
424
425 return targetActualParameterName;
426 }
