90 protected override void Start()
91 {
92 this.BeginStart(ref _started, () => base.Start());
93 this.AssertField(CameraRigRef, nameof(CameraRigRef));
94 this.AssertField(TrackingToWorldTransformer, nameof(TrackingToWorldTransformer));

97 this.AssertField(CameraRigRef.LeftHand, nameof(CameraRigRef.LeftHand));
98 _ovrControllerAnchor = CameraRigRef.LeftController;

105 _ovrController = OVRInput.Controller.RTouch;
106 }

111 offset.position.x = -offset.position.x;

113 }
114 _poseOffset = offset;

116 UpdateSkeleton();
117 UpdateConfig();

186 protected override void UpdateData()
187 {
188 _handDataAsset.Config = Config;
189 _handDataAsset.IsDataValid = true;
190 _handDataAsset.IsConnected = (OVRInput.GetConnectedControllers() & _ovrController) > 0;
191 if (!_handDataAsset.IsConnected)
192 {
193 // revert state fields to their defaults
194 _handDataAsset.IsTracked = default;
195 _handDataAsset.RootPoseOrigin = default;
196 _handDataAsset.PointerPoseOrigin = default;
197 _handDataAsset.IsHighConfidence = default;
198 for (var fingerIdx = 0; fingerIdx < Constants.NUM_FINGERS; fingerIdx++)
199 {
200 _handDataAsset.IsFingerPinching[fingerIdx] = default;
201 _handDataAsset.IsFingerHighConfidence[fingerIdx] = default;
202 }
203 return;
204 }
205
206 _handDataAsset.IsTracked = true;
207 _handDataAsset.IsHighConfidence = true;
208 _handDataAsset.HandScale = 1f;
209
210 _handDataAsset.IsDominantHand =
211 OVRInput.GetDominantHand() == OVRInput.Handedness.LeftHanded
212 && _handedness == Handedness.Left
213 || (OVRInput.GetDominantHand() == OVRInput.Handedness.RightHanded
214 && _handedness == Handedness.Right);
215
216 float pinchAmount = OVRControllerUtility.GetPinchAmount(_ovrController);
217 float pinchStrength = _pinchCurve.Evaluate(pinchAmount);
218 float gripStrength = _pinchCurve.Evaluate(OVRInput.Get(OVRInput.Axis1D.PrimaryHandTrigger, _ovrController));
219
220 _handDataAsset.IsFingerHighConfidence[(int)HandFinger.Thumb] = true;
221 _handDataAsset.IsFingerPinching[(int)HandFinger.Thumb] = pinchStrength >= 1f || gripStrength >= 1f;
222 _handDataAsset.FingerPinchStrength[(int)HandFinger.Thumb] = Mathf.Max(pinchStrength, gripStrength);
223
224 _handDataAsset.IsFingerHighConfidence[(int)HandFinger.Index] = true;
225 _handDataAsset.IsFingerPinching[(int)HandFinger.Index] = pinchStrength >= 1f;
226 _handDataAsset.FingerPinchStrength[(int)HandFinger.Index] = pinchStrength;
227
228 _handDataAsset.IsFingerHighConfidence[(int)HandFinger.Middle] = true;
229 _handDataAsset.IsFingerPinching[(int)HandFinger.Middle] = gripStrength >= 1f;
230 _handDataAsset.FingerPinchStrength[(int)HandFinger.Middle] = gripStrength;
231
232 _handDataAsset.IsFingerHighConfidence[(int)HandFinger.Ring] = true;
233 _handDataAsset.IsFingerPinching[(int)HandFinger.Ring] = gripStrength >= 1f;
234 _handDataAsset.FingerPinchStrength[(int)HandFinger.Ring] = gripStrength;
235
236 _handDataAsset.IsFingerHighConfidence[(int)HandFinger.Pinky] = true;
237 _handDataAsset.IsFingerPinching[(int)HandFinger.Pinky] = gripStrength >= 1f;
238 _handDataAsset.FingerPinchStrength[(int)HandFinger.Pinky] = gripStrength;
239
240 _handDataAsset.PointerPoseOrigin = PoseOrigin.RawTrackedPose;
241 _handDataAsset.PointerPose = new Pose(
242 OVRInput.GetLocalControllerPosition(_ovrController),
243 OVRInput.GetLocalControllerRotation(_ovrController));
244
245 for (int i = 0; i < _bones.Length; i++)
246 {
247 _handDataAsset.Joints[i] = _bones[i].localRotation;
248 }
249
250 _handDataAsset.Joints[0] = WristFixupRotation;
251
252 // Convert controller pose from world to tracking space.
253 Pose pose = new Pose(_ovrControllerAnchor.position, _ovrControllerAnchor.rotation);
254 if (Config.TrackingToWorldTransformer != null)
255 {
256 pose = Config.TrackingToWorldTransformer.ToTrackingPose(pose);
257 }
258
259 PoseUtils.Multiply(pose, _poseOffset, ref _handDataAsset.Root);
260 _handDataAsset.RootPoseOrigin = PoseOrigin.RawTrackedPose;
261 }
