153 private void InitializeMidPoint(Pose pointer)
154 {
155 Vector3 direction = Vector3.ProjectOnPlane(pointer.position - _stabilizationPoint.position, Vector3.up).normalized;
156 Quaternion rotation = Quaternion.LookRotation(direction, Vector3.up);
157 Vector3 point = pointer.position;
158 _midPoint = Transformer.ToTrackingPose(new Pose(point, rotation));
159 }
160
161 private void UpdateMidPoint(Pose pointer, Pose midPoint)
162 {
163 float length = Vector3.ProjectOnPlane(pointer.position - _stabilizationPoint.position, Vector3.up).magnitude;
164 Vector3 position = _stabilizationPoint.position + midPoint.forward * length;
165 position.y = pointer.position.y;
166 Quaternion rotation = midPoint.rotation;
167 _midPoint = Transformer.ToTrackingPose(new Pose(position, rotation));
168 }
169
170 private void DragMidPoint(Pose worldMidPoint)
171 {
172 Vector3 midPointPos = worldMidPoint.position;
173 float distance = Mathf.Abs(_axisValue) - _dragThresold * this.transform.lossyScale.x;
174 if (distance <= 0)
175 {
176 return;
177 }
178
179 Vector3 right = worldMidPoint.right;
180 float direction = Math.Sign(_axisValue);
181 midPointPos += right * direction * distance;
182
183 Vector3 lookDirection = Vector3.ProjectOnPlane(midPointPos - _stabilizationPoint.position, Vector3.up).normalized;
184 Quaternion rotation = Quaternion.LookRotation(lookDirection, Vector3.up);
185
186 _midPoint = Transformer.ToTrackingPose(new Pose(midPointPos, rotation));
187 }
188
189 private void UpdateAxisValue(Pose pointer, Pose origin)
190 {
191 float prevSign = Mathf.Sign(_axisValue);
192 Vector3 diff = pointer.position - origin.position;
193 Vector3 deviation = Vector3.Project(pointer.position - origin.position, origin.right);
194 _axisValue = deviation.magnitude * Mathf.Sign(Vector3.Dot(origin.right, diff));
195 if (prevSign != Mathf.Sign(_axisValue))
196 {
197 _whenTurnDirectionChanged(prevSign);
198 }
199 }
