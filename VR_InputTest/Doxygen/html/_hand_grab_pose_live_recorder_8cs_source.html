175 private bool TakeSnapshot()
176 {
177 Rigidbody leftItem = FindNearestItem(_leftHand.Rigidbody, _leftDetector, out float leftDistance);
178 Rigidbody rightItem = FindNearestItem(_rightHand.Rigidbody, _rightDetector, out float rightDistance);
179
180 if (leftDistance < rightDistance
181 && leftItem != null)
182 {
183 return Record(_leftHand.Hand, leftItem);
184 }
185 else if (rightItem != null)
186 {
187 return Record(_rightHand.Hand, rightItem);
188 }
189
190 Debug.LogError("No rigidbody detected near any hand");
191 return false;
192 }
193
194 private Rigidbody FindNearestItem(Rigidbody handBody, RigidbodyDetector detector, out float bestDistance)
195 {
196 Vector3 referencePoint = handBody.worldCenterOfMass;
197 float minDistance = float.PositiveInfinity;
198 Rigidbody bestItem = null;
199 foreach (Rigidbody item in detector.IntersectingBodies)
200 {
201 Vector3 point = item.worldCenterOfMass;
202 float distance = Vector3.Distance(point, referencePoint);
203 if (distance < minDistance)
204 {
205 minDistance = distance;
206 bestItem = item;
207 }
208 }
209
210 bestDistance = minDistance;
211 return bestItem;
212 }

250 private bool Record(IHand hand, Rigidbody item)
251 {
252 HandPose trackedHandPose = TrackedPose(hand);
253 if (trackedHandPose == null)
254 {
255 Debug.LogError("Tracked Pose could not be retrieved", this);
256 return false;
257 }
258
259 if (!hand.GetRootPose(out Pose handRoot))
260 {
261 Debug.LogError("Hand Root pose could not be retrieved", this);
262 return false;
263 }
264
265 Pose gripPoint = PoseUtils.DeltaScaled(item.transform, handRoot);
266 RecorderStep recorderStep = new RecorderStep(trackedHandPose, gripPoint, hand.Scale, item);
267 AddHandGrabPose(recorderStep, out recorderStep.interactable, out HandGrabPose point);
268 AttachGhost(point, recorderStep.HandScale);
269
270 int nextStep = CurrentStepIndex + 1;
271 if (nextStep < _recorderSteps.Count)
272 {
273 _recorderSteps.RemoveRange(nextStep, _recorderSteps.Count - nextStep);
274 }
275 _recorderSteps.Add(recorderStep);
276 CurrentStepIndex = _recorderSteps.Count - 1;
277
278 return true;
279 }

281 private HandPose TrackedPose(IHand hand)
282 {
283 if (!hand.GetJointPosesLocal(out ReadOnlyHandJointPoses localJoints))
284 {
285 return null;
286 }
287 HandPose result = new HandPose(hand.Handedness);
288 for (int i = 0; i < FingersMetadata.HAND_JOINT_IDS.Length; ++i)
289 {
290 HandJointId jointID = FingersMetadata.HAND_JOINT_IDS[i];
291 result.JointRotations[i] = localJoints[jointID].rotation;
292 }
293 return result;
294 }

296 private void AddHandGrabPose(RecorderStep recorderStep,
297 out HandGrabInteractable interactable, out HandGrabPose handGrabPose)
298 {
299 interactable = HandGrabUtils.CreateHandGrabInteractable(recorderStep.Item.transform);
300 if (recorderStep.Item.TryGetComponent(out Grabbable grabbable))
301 {
302 interactable.InjectOptionalPointableElement(grabbable);
303 }
304
305 var pointData = new HandGrabUtils.HandGrabPoseData()
306 {
307 handPose = recorderStep.RawHandPose,
308 scale = (recorderStep.HandScale / interactable.RelativeTo.lossyScale.x),
309 gripPose = recorderStep.GrabPoint,
310 };
311 handGrabPose = HandGrabUtils.LoadHandGrabPose(interactable, pointData);
312 }

314 private void AttachGhost(HandGrabPose point, float referenceScale)
315 {
316 if (_ghostProvider == null)
317 {
318 return;
319 }
320 HandGhost ghostPrefab = _ghostProvider.GetHand(point.HandPose.Handedness);
321 HandGhost ghost = GameObject.Instantiate(ghostPrefab, point.transform);
322 ghost.transform.localScale = Vector3.one * (referenceScale / point.transform.lossyScale.x);
323 ghost.SetPose(point);
324 }

327 public class RigidbodyDetector : MonoBehaviour
328 {
329 private HashSet<Rigidbody> _ignoredBodies = new HashSet<Rigidbody>();
330
331 public List<Rigidbody> IntersectingBodies { get; private set; } = new List<Rigidbody>();
332
333 public void IgnoreBody(Rigidbody body)
334 {
335 if (!_ignoredBodies.Contains(body))
336 {
337 _ignoredBodies.Add(body);
338 }
339
340 if (IntersectingBodies.Contains(body))
341 {
342 IntersectingBodies.Remove(body);
343 }
344 }
345
346 public void UnIgnoreBody(Rigidbody body)
347 {
348 if (_ignoredBodies.Contains(body))
349 {
350 _ignoredBodies.Remove(body);
351 }
352 }
353
354 private void OnTriggerEnter(Collider collider)
355 {
356 Rigidbody rigidbody = collider.attachedRigidbody;
357 if (rigidbody == null || _ignoredBodies.Contains(rigidbody))
358 {
359 return;
360 }
361 if (!IntersectingBodies.Contains(rigidbody))
362 {
363 IntersectingBodies.Add(rigidbody);
364 }
365 }
366
367 private void OnTriggerExit(Collider collider)
368 {
369 Rigidbody rigidbody = collider.attachedRigidbody;
370 if (rigidbody == null)
371 {
372 return;
373 }
374 if (IntersectingBodies.Contains(rigidbody))
375 {
376 IntersectingBodies.Remove(rigidbody);
377 }
378 }
