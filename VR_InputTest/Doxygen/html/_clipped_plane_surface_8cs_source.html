69 public bool ClipBounds(in Bounds bounds, out Bounds clipped)
70 {
71 clipped = bounds;
72
73 foreach (var clipperMono in _clippers)
74 {
75 IBoundsClipper clipper = clipperMono as IBoundsClipper;
76 if (clipper == null ||
77 !clipper.GetLocalBounds(Transform, out Bounds clipTo))
78 {
79 continue;
80 }
81
82 if (!clipped.Clip(clipTo, out clipped))
83 {
84 return false;
85 }
86 }
87
88 return true;
89 }
90
91 private Vector3 ClampPoint(in Vector3 point, in Bounds bounds)
92 {
93 Vector3 min = bounds.min;
94 Vector3 max = bounds.max;
95 Vector3 localPoint = Transform.InverseTransformPoint(point);
96
97 Vector3 clamped = new Vector3(
98 Mathf.Clamp(localPoint.x, min.x, max.x),
99 Mathf.Clamp(localPoint.y, min.y, max.y),
100 Mathf.Clamp(localPoint.z, min.z, max.z));
101
102 return Transform.TransformPoint(clamped);
103 }
104
105 public bool ClosestSurfacePoint(in Vector3 point, out SurfaceHit hit, float maxDistance = 0)
106 {
107 if (_planeSurface.ClosestSurfacePoint(point, out hit, maxDistance) &&
108 ClipBounds(PlaneBounds, out Bounds clippedPlane))
109 {
110 hit.Point = ClampPoint(hit.Point, clippedPlane);
111 hit.Distance = Vector3.Distance(point, hit.Point);
112 return maxDistance <= 0 || hit.Distance <= maxDistance;
113 }
114 return false;
115 }
116
117 public bool Raycast(in Ray ray, out SurfaceHit hit, float maxDistance = 0)
118 {
119 return BackingSurface.Raycast(ray, out hit, maxDistance) &&
120 ClipBounds(InfiniteBounds, out Bounds clipBounds) &&
121 clipBounds.size != Vector3.zero &&
122 clipBounds.Contains(Transform.InverseTransformPoint(hit.Point));
123 }
