223 protected virtual void UpdateAnchors(bool updateEyeAnchors, bool updateHandAnchors)
224 {
225 if (!OVRManager.OVRManagerinitialized)
226 return;
227
228 EnsureGameObjectIntegrity();
229
230 if (!Application.isPlaying)
231 return;
232
233 if (_skipUpdate)
234 {
235 centerEyeAnchor.FromOVRPose(OVRPose.identity, true);
236 leftEyeAnchor.FromOVRPose(OVRPose.identity, true);
237 rightEyeAnchor.FromOVRPose(OVRPose.identity, true);
238
239 return;
240 }
241
242 bool monoscopic = OVRManager.instance.monoscopic;
243 bool hmdPresent = OVRNodeStateProperties.IsHmdPresent();
244
245 OVRPose tracker = OVRManager.tracker.GetPose();
246
247 trackerAnchor.localRotation = tracker.orientation;
248
249 Quaternion emulatedRotation = Quaternion.Euler(-OVRManager.instance.headPoseRelativeOffsetRotation.x,
250 -OVRManager.instance.headPoseRelativeOffsetRotation.y,
251 OVRManager.instance.headPoseRelativeOffsetRotation.z);
252
253 //Note: in the below code, when using UnityEngine's API, we only update anchor transforms if we have a new, fresh value this frame.
254 //If we don't, it could mean that tracking is lost, etc. so the pose should not change in the virtual world.
255 //This can be thought of as similar to calling InputTracking GetLocalPosition and Rotation, but only for doing so when the pose is valid.
256 //If false is returned for any of these calls, then a new pose is not valid and thus should not be updated.
257 if (updateEyeAnchors)
258 {
259 if (hmdPresent)
260 {
261 Vector3 centerEyePosition = Vector3.zero;
262 Quaternion centerEyeRotation = Quaternion.identity;
263
264 if (OVRNodeStateProperties.GetNodeStatePropertyVector3(Node.CenterEye, NodeStatePropertyType.Position,
265 OVRPlugin.Node.EyeCenter, OVRPlugin.Step.Render, out centerEyePosition))
266 centerEyeAnchor.localPosition = centerEyePosition;
267 if (OVRNodeStateProperties.GetNodeStatePropertyQuaternion(Node.CenterEye,
268 NodeStatePropertyType.Orientation, OVRPlugin.Node.EyeCenter, OVRPlugin.Step.Render,
269 out centerEyeRotation))
270 centerEyeAnchor.localRotation = centerEyeRotation;
271 }
272 else
273 {
274 centerEyeAnchor.localRotation = emulatedRotation;
275 centerEyeAnchor.localPosition = OVRManager.instance.headPoseRelativeOffsetTranslation;
276 }
277
278 if (!hmdPresent || monoscopic)
279 {
280 leftEyeAnchor.localPosition = centerEyeAnchor.localPosition;
281 rightEyeAnchor.localPosition = centerEyeAnchor.localPosition;
282 leftEyeAnchor.localRotation = centerEyeAnchor.localRotation;
283 rightEyeAnchor.localRotation = centerEyeAnchor.localRotation;
284 }
285 else
286 {
287 Vector3 leftEyePosition = Vector3.zero;
288 Vector3 rightEyePosition = Vector3.zero;
289 Quaternion leftEyeRotation = Quaternion.identity;
290 Quaternion rightEyeRotation = Quaternion.identity;
291
292 if (OVRNodeStateProperties.GetNodeStatePropertyVector3(Node.LeftEye, NodeStatePropertyType.Position,
293 OVRPlugin.Node.EyeLeft, OVRPlugin.Step.Render, out leftEyePosition))
294 leftEyeAnchor.localPosition = leftEyePosition;
295 if (OVRNodeStateProperties.GetNodeStatePropertyVector3(Node.RightEye, NodeStatePropertyType.Position,
296 OVRPlugin.Node.EyeRight, OVRPlugin.Step.Render, out rightEyePosition))
297 rightEyeAnchor.localPosition = rightEyePosition;
298 if (OVRNodeStateProperties.GetNodeStatePropertyQuaternion(Node.LeftEye,
299 NodeStatePropertyType.Orientation, OVRPlugin.Node.EyeLeft, OVRPlugin.Step.Render,
300 out leftEyeRotation))
301 leftEyeAnchor.localRotation = leftEyeRotation;
302 if (OVRNodeStateProperties.GetNodeStatePropertyQuaternion(Node.RightEye,
303 NodeStatePropertyType.Orientation, OVRPlugin.Node.EyeRight, OVRPlugin.Step.Render,
304 out rightEyeRotation))
305 rightEyeAnchor.localRotation = rightEyeRotation;
306 }
307 }
308
309 if (updateHandAnchors)
310 {
311 //Need this for controller offset because if we're on OpenVR, we want to set the local poses as specified by Unity, but if we're not, OVRInput local position is the right anchor
312 if (OVRManager.loadedXRDevice == OVRManager.XRDevice.OpenVR)
313 {
314 Vector3 leftPos = Vector3.zero;
315 Vector3 rightPos = Vector3.zero;
316 Quaternion leftQuat = Quaternion.identity;
317 Quaternion rightQuat = Quaternion.identity;
318
319 if (OVRNodeStateProperties.GetNodeStatePropertyVector3(Node.LeftHand, NodeStatePropertyType.Position,
320 OVRPlugin.Node.HandLeft, OVRPlugin.Step.Render, out leftPos))
321 leftHandAnchor.localPosition = leftPos;
322 if (OVRNodeStateProperties.GetNodeStatePropertyVector3(Node.RightHand, NodeStatePropertyType.Position,
323 OVRPlugin.Node.HandRight, OVRPlugin.Step.Render, out rightPos))
324 rightHandAnchor.localPosition = rightPos;
325 if (OVRNodeStateProperties.GetNodeStatePropertyQuaternion(Node.LeftHand,
326 NodeStatePropertyType.Orientation, OVRPlugin.Node.HandLeft, OVRPlugin.Step.Render,
327 out leftQuat))
328 leftHandAnchor.localRotation = leftQuat;
329 if (OVRNodeStateProperties.GetNodeStatePropertyQuaternion(Node.RightHand,
330 NodeStatePropertyType.Orientation, OVRPlugin.Node.HandRight, OVRPlugin.Step.Render,
331 out rightQuat))
332 rightHandAnchor.localRotation = rightQuat;
333 }
334 else
335 {
336 OVRInput.Controller leftActiveController =
337 OVRInput.GetActiveControllerForHand(OVRInput.Handedness.LeftHanded);
338 OVRInput.Controller rightActiveController =
339 OVRInput.GetActiveControllerForHand(OVRInput.Handedness.RightHanded);
340
341 // it's possible for niether hands or controllers to be active, in that case
342 // manually calculate which one we care about giving hands priority.
343 if (leftActiveController == OVRInput.Controller.None)
344 {
345 if (OVRInput.GetControllerPositionValid(OVRInput.Controller.LHand))
346 {
347 leftActiveController = OVRInput.Controller.LHand;
348 }
349 else if (OVRInput.GetControllerPositionValid(OVRInput.Controller.LTouch))
350 {
351 leftActiveController = OVRInput.Controller.LTouch;
352 }
353 }
354 if (rightActiveController == OVRInput.Controller.None)
355 {
356 if (OVRInput.GetControllerPositionValid(OVRInput.Controller.RHand))
357 {
358 rightActiveController = OVRInput.Controller.RHand;
359 }
360 else if (OVRInput.GetControllerPositionValid(OVRInput.Controller.RTouch))
361 {
362 rightActiveController = OVRInput.Controller.RTouch;
363 }
364 }
365
366 leftHandAnchor.localPosition = OVRInput.GetLocalControllerPosition(leftActiveController);
367 rightHandAnchor.localPosition = OVRInput.GetLocalControllerPosition(rightActiveController);
368 leftHandAnchor.localRotation = OVRInput.GetLocalControllerRotation(leftActiveController);
369 rightHandAnchor.localRotation = OVRInput.GetLocalControllerRotation(rightActiveController);
370
371 // left
372 OVRInput.ControllerInHandState controllerInHandState =
373 OVRInput.GetControllerIsInHandState(OVRInput.Hand.HandLeft);
374 if (controllerInHandState == OVRInput.ControllerInHandState.ControllerNotInHand)
375 {
376 leftHandAnchorDetached.localPosition =
377 OVRInput.GetLocalControllerPosition(OVRInput.Controller.LTouch);
378 leftHandAnchorDetached.localRotation =
379 OVRInput.GetLocalControllerRotation(OVRInput.Controller.LTouch);
380 leftHandOnControllerAnchor.localPosition = Vector3.zero;
381 leftHandOnControllerAnchor.localRotation = Quaternion.identity;
382 }
383 else if (controllerInHandState == OVRInput.ControllerInHandState.ControllerInHand)
384 {
385 Vector3 leftRelativeHandPosition =
386 trackingSpace.TransformPoint(OVRInput.GetLocalControllerPosition(OVRInput.Controller.LHand));
387 leftHandOnControllerAnchor.localPosition =
388 leftHandAnchor.InverseTransformPoint(leftRelativeHandPosition);
389 leftHandOnControllerAnchor.localRotation = Quaternion.Inverse(leftHandAnchor.localRotation) *
390 OVRInput.GetLocalControllerRotation(OVRInput.Controller
391 .LHand);
392 leftHandAnchorDetached.localPosition = Vector3.zero;
393 leftHandAnchorDetached.localRotation = Quaternion.identity;
394 }
395 else
396 {
397 leftHandAnchorDetached.localPosition = Vector3.zero;
398 leftHandAnchorDetached.localRotation = Quaternion.identity;
399 leftHandOnControllerAnchor.localPosition = Vector3.zero;
400 leftHandOnControllerAnchor.localRotation = Quaternion.identity;
401 }
402
403 // right
404 controllerInHandState = OVRInput.GetControllerIsInHandState(OVRInput.Hand.HandRight);
405 if (controllerInHandState == OVRInput.ControllerInHandState.ControllerNotInHand)
406 {
407 rightHandAnchorDetached.localPosition =
408 OVRInput.GetLocalControllerPosition(OVRInput.Controller.RTouch);
409 rightHandAnchorDetached.localRotation =
410 OVRInput.GetLocalControllerRotation(OVRInput.Controller.RTouch);
411 rightHandOnControllerAnchor.localPosition = Vector3.zero;
412 rightHandOnControllerAnchor.localRotation = Quaternion.identity;
413 }
414 else if (controllerInHandState == OVRInput.ControllerInHandState.ControllerInHand)
415 {
416 Vector3 rightRelativeHandPosition =
417 trackingSpace.TransformPoint(OVRInput.GetLocalControllerPosition(OVRInput.Controller.RHand));
418 rightHandOnControllerAnchor.localPosition =
419 rightHandAnchor.InverseTransformPoint(rightRelativeHandPosition);
420 rightHandOnControllerAnchor.localRotation = Quaternion.Inverse(rightHandAnchor.localRotation) *
421 OVRInput.GetLocalControllerRotation(OVRInput.Controller
422 .RHand);
423 rightHandAnchorDetached.localPosition = Vector3.zero;
424 rightHandAnchorDetached.localRotation = Quaternion.identity;
425 }
426 else
427 {
428 rightHandAnchorDetached.localPosition = Vector3.zero;
429 rightHandAnchorDetached.localRotation = Quaternion.identity;
430 rightHandOnControllerAnchor.localPosition = Vector3.zero;
431 rightHandOnControllerAnchor.localRotation = Quaternion.identity;
432 }
433 }
434
435 trackerAnchor.localPosition = tracker.position;
436
437 OVRPose leftOffsetPose = OVRPose.identity;
438 OVRPose rightOffsetPose = OVRPose.identity;
439 if (OVRManager.loadedXRDevice == OVRManager.XRDevice.OpenVR)
440 {
441 leftOffsetPose = OVRManager.GetOpenVRControllerOffset(Node.LeftHand);
442 rightOffsetPose = OVRManager.GetOpenVRControllerOffset(Node.RightHand);
443
444 //Sets poses of left and right nodes, local to the tracking space.
445 OVRManager.SetOpenVRLocalPose(trackingSpace.InverseTransformPoint(leftControllerAnchor.position),
446 trackingSpace.InverseTransformPoint(rightControllerAnchor.position),
447 Quaternion.Inverse(trackingSpace.rotation) * leftControllerAnchor.rotation,
448 Quaternion.Inverse(trackingSpace.rotation) * rightControllerAnchor.rotation);
449 }
450
451 rightControllerAnchor.localPosition = rightOffsetPose.position;
452 rightControllerAnchor.localRotation = rightOffsetPose.orientation;
453 leftControllerAnchor.localPosition = leftOffsetPose.position;
454 leftControllerAnchor.localRotation = leftOffsetPose.orientation;
455 }
456
457#if USING_XR_SDK
458#if UNITY_2020_3_OR_NEWER
459 if (OVRManager.instance.LateLatching)
460 {
461 XRDisplaySubsystem displaySubsystem = OVRManager.GetCurrentDisplaySubsystem();
462 if (displaySubsystem != null)
463 {
464 displaySubsystem.MarkTransformLateLatched(centerEyeAnchor.transform,
465 XRDisplaySubsystem.LateLatchNode.Head);
466 displaySubsystem.MarkTransformLateLatched(leftHandAnchor, XRDisplaySubsystem.LateLatchNode.LeftHand);
467 displaySubsystem.MarkTransformLateLatched(rightHandAnchor, XRDisplaySubsystem.LateLatchNode.RightHand);
468 }
469 }

471#endif
472 RaiseUpdatedAnchorsEvent();
473 CheckForTrackingSpaceChangesAndRaiseEvent();

668 public virtual Matrix4x4 ComputeTrackReferenceMatrix()
669 {
670 if (centerEyeAnchor == null)
671 {
672 Debug.LogError("centerEyeAnchor is required");
673 return Matrix4x4.identity;
674 }
675
676 // The ideal approach would be using UnityEngine.VR.VRNode.TrackingReference, then we would not have to depend on the OVRCameraRig. Unfortunately, it is not available in Unity 5.4.3
677
678 OVRPose headPose = OVRPose.identity;
679
680 Vector3 pos;
681 Quaternion rot;
682 if (OVRNodeStateProperties.GetNodeStatePropertyVector3(Node.Head, NodeStatePropertyType.Position,
683 OVRPlugin.Node.Head, OVRPlugin.Step.Render, out pos))
684 headPose.position = pos;
685 if (OVRNodeStateProperties.GetNodeStatePropertyQuaternion(Node.Head, NodeStatePropertyType.Orientation,
686 OVRPlugin.Node.Head, OVRPlugin.Step.Render, out rot))
687 headPose.orientation = rot;
688
689 OVRPose invHeadPose = headPose.Inverse();
690 Matrix4x4 invHeadMatrix = Matrix4x4.TRS(invHeadPose.position, invHeadPose.orientation, Vector3.one);
691
692 Matrix4x4 ret = centerEyeAnchor.localToWorldMatrix * invHeadMatrix;
693
694 return ret;
695 }

697 private void CheckForAnchorsInParent()
698 {
699 void Check<T>(Transform node) where T : MonoBehaviour
700 {
701 var anchor = node.GetComponent<T>();
702 if (anchor && anchor.enabled)
703 {
704 anchor.enabled = false;
705 Debug.LogError(
706 $"The {typeof(T).Name} '{anchor.name}' is a parent of the {nameof(OVRCameraRig)} '{name}', which is not allowed. An {typeof(T).Name} may not be the parent of an {nameof(OVRCameraRig)} because the {nameof(OVRCameraRig)} defines the tracking space for the anchor, and its transform is relative to the {nameof(OVRCameraRig)}.");
707 }
708 }
709
710 var parent = transform.parent;
711 while (parent)
712 {
713 Check<OVRSpatialAnchor>(parent);
714 Check<OVRSceneAnchor>(parent);
715 parent = parent.parent;
716 }
717 }
