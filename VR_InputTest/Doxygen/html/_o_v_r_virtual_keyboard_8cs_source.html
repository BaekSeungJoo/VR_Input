49 public class InteractorRootTransformOverride
50 {
51 private struct InteractorRootOverrideData
52 {
53 public Transform root;
54 public OVRPose originalPose;
55 public OVRPose targetPose;
56 }
57
58 private Queue<InteractorRootOverrideData> applyQueue = new Queue<InteractorRootOverrideData>();
59 private Queue<InteractorRootOverrideData> revertQueue = new Queue<InteractorRootOverrideData>();
60
61 public void Enqueue(Transform interactorRootTransform, OVRPlugin.Posef interactorRootPose)
62 {
63 if (interactorRootTransform == null)
64 {
65 throw new Exception("Transform is undefined");
66 }
67
68 applyQueue.Enqueue(new InteractorRootOverrideData()
69 {
70 root = interactorRootTransform,
71 originalPose = interactorRootTransform.ToOVRPose(),
72 targetPose = interactorRootPose.ToOVRPose()
73 });
74 }
75
76 public void LateApply(MonoBehaviour coroutineRunner)
77 {
78 while (applyQueue.Count > 0)
79 {
80 var queueItem = applyQueue.Dequeue();
81 var restoreToPose = queueItem.root.ToOVRPose();
82 if (!ApplyOverride(queueItem))
83 {
84 continue;
85 }
86
87 queueItem.originalPose = queueItem.root.ToOVRPose();
88 queueItem.targetPose = restoreToPose;
89 revertQueue.Enqueue(queueItem);
90 }
91
92 if (revertQueue.Count > 0 && coroutineRunner != null)
93 {
94 coroutineRunner.StartCoroutine(RevertInteractorOverrides());
95 }
96 }
97
98 public void Reset()
99 {
100 while (revertQueue.Count > 0)
101 {
102 ApplyOverride(revertQueue.Dequeue());
103 }
104 }
105
106 private IEnumerator RevertInteractorOverrides()
107 {
108 yield return new WaitForEndOfFrame();
109 Reset();
110 }
111
112 private static bool ApplyOverride(InteractorRootOverrideData interactorOverride)
113 {
114 if (interactorOverride.root.position != interactorOverride.originalPose.position ||
115 interactorOverride.root.rotation != interactorOverride.originalPose.orientation)
116 {
117 return false;
118 }
119
120 interactorOverride.root.position = interactorOverride.targetPose.position;
121 interactorOverride.root.rotation = interactorOverride.targetPose.orientation;
122 return true;
123 }
124 }

175 private class ControllerInputSource : BaseInputSource
176 {
177 private readonly Transform _rootTransform;
178 private readonly Transform _directTransform;
179 private readonly InputSource _inputSource;
180 private readonly OVRInput.Controller _controllerType;
181 private readonly OVRVirtualKeyboard _keyboard;
182 private int _lastFrameCount;
183 private readonly OVRInput.RawButton _triggerButton;
184
185 private bool TriggerIsPressed => OVRInput.Get(_triggerButton);
186
187 public ControllerInputSource(OVRVirtualKeyboard keyboard, InputSource inputSource,
188 OVRInput.Controller controllerType, Transform rootTransform, Transform directTransform) : base()
189 {
190 _keyboard = keyboard;
191 _inputSource = inputSource;
192 _controllerType = controllerType;
193 _triggerButton = _controllerType == OVRInput.Controller.LTouch
194 ? OVRInput.RawButton.LIndexTrigger
195 : OVRInput.RawButton.RIndexTrigger;
196 _rootTransform = rootTransform;
197 _directTransform = directTransform;
198 }
199
200 protected override void UpdateInput()
201 {
202 if (!_keyboard.InputEnabled || !OVRInput.GetControllerPositionValid(_controllerType) || !_rootTransform)
203 {
204 return;
205 }
206
207 if (Time.frameCount == _lastFrameCount)
208 {
209 // Input already sent for frame
210 return;
211 }
212 _lastFrameCount = Time.frameCount;
213
214 if (_keyboard.controllerRayInteraction)
215 {
216 _keyboard.SendVirtualKeyboardRayInput(
217 _directTransform, _inputSource,
218 TriggerIsPressed);
219 }
220
221 if (_keyboard.controllerDirectInteraction)
222 {
223 _keyboard.SendVirtualKeyboardDirectInput(
224 _directTransform.position,
225 _inputSource,
226 TriggerIsPressed,
227 _rootTransform);
228 }
229 }

233 {
234 private readonly OVRHand _hand;
235 private readonly InputSource _inputSource;
236 private readonly OVRVirtualKeyboard _keyboard;
237 private readonly OVRSkeleton _skeleton;
238 private int _lastFrameCount;

240 public HandInputSource(OVRVirtualKeyboard keyboard, InputSource inputSource, OVRHand hand) : base()
241 {
242 if (!keyboard)
243 {
244 throw new ArgumentNullException("keyboard");
245 }
246 _keyboard = keyboard;
247 if (!hand)
248 {
249 throw new ArgumentNullException("hand");
250 }
251 _hand = hand;
252 _skeleton = _hand.GetComponent<OVRSkeleton>();
253 if (!_skeleton && _keyboard.handDirectInteraction)
254 {
255 Debug.LogWarning("Hand Direct Interaction requires an OVRSkeleton on the OVRHand");
256 }
257 _inputSource = inputSource;
258 }
259
260 protected override void UpdateInput()
261 {
262 if (!_keyboard.InputEnabled || !_hand)
263 {
264 return;
265 }
266
267 if (Time.frameCount == _lastFrameCount)
268 {
269 // Input already sent for frame
270 return;
271 }
272 _lastFrameCount = Time.frameCount;
273
274 if (_keyboard.handRayInteraction && _hand.IsPointerPoseValid)
275 {
276 _keyboard.SendVirtualKeyboardRayInput(
277 _hand.PointerPose,
278 _inputSource, _hand.GetFingerIsPinching(OVRHand.HandFinger.Index));
279 }
280
281 if (_keyboard.handDirectInteraction && _skeleton && _skeleton.IsDataValid)
282 {
283 var indexTip = _skeleton.Bones.First(b => b.Id == OVRSkeleton.BoneId.Hand_IndexTip);
284 var interactorRoot = _skeleton.Bones.First(b => b.Id == OVRSkeleton.BoneId.Hand_WristRoot);
285 _keyboard.SendVirtualKeyboardDirectInput(
286 indexTip.Transform.position,
287 _inputSource, _hand.GetFingerIsPinching(OVRHand.HandFinger.Index), interactorRoot.Transform);
288 }
289 }

539 public void UseSuggestedLocation(KeyboardPosition position)
540 {
541 OVRPlugin.VirtualKeyboardLocationInfo locationInfo = new OVRPlugin.VirtualKeyboardLocationInfo();
542 switch (position)
543 {
544 case KeyboardPosition.Near:
545 locationInfo.locationType = OVRPlugin.VirtualKeyboardLocationType.Direct;
546 break;
547 case KeyboardPosition.Far:
548 locationInfo.locationType = OVRPlugin.VirtualKeyboardLocationType.Far;
549 break;
550 case KeyboardPosition.Custom:
551 locationInfo = ComputeLocation(transform);
552 break;
553 default:
554 Debug.LogError("Unknown KeyboardInputMode: " + position);
555 break;
556 }
557
558 var result = OVRPlugin.SuggestVirtualKeyboardLocation(locationInfo);
559 if (result != OVRPlugin.Result.Success)
560 {
561 Debug.LogError("SuggestVirtualKeyboardLocation failed: " + result);
562 return;
563 }
564 // Clear transform has changed state to avoid a custom location overriding the SuggestVirtualKeyboardLocation call
565 transform.hasChanged = false;
566 SyncKeyboardLocation();
567 }

569 public void SendVirtualKeyboardRayInput(Transform inputTransform,
570 InputSource source, bool isPressed, bool useRaycastMask = true)
571 {
572 var inputSource = source switch
573 {
574 InputSource.ControllerLeft => OVRPlugin.VirtualKeyboardInputSource.ControllerRayLeft,
575 InputSource.ControllerRight => OVRPlugin.VirtualKeyboardInputSource.ControllerRayRight,
576 InputSource.HandLeft => OVRPlugin.VirtualKeyboardInputSource.HandRayLeft,
577 InputSource.HandRight => OVRPlugin.VirtualKeyboardInputSource.HandRayRight,
578 _ => throw new Exception("Unknown input source: " + source)
579 };
580
581 var raycaster = (source == InputSource.ControllerLeft || source == InputSource.ControllerRight)
582 ? controllerRaycaster
583 : handRaycaster;
584
585 if (raycaster)
586 {
587 var pointerData = new OVRPointerEventData(EventSystem.current)
588 {
589 worldSpaceRay = new Ray(inputTransform.position, inputTransform.forward)
590 };
591 var results = new List<RaycastResult>();
592 raycaster.Raycast(pointerData, results);
593 if (results.Count <= 0 || results[0].gameObject != Collider.gameObject)
594 {
595 return;
596 }
597 }
598 SendVirtualKeyboardInput(inputSource, inputTransform.ToOVRPose(), isPressed);
599 }

601 public void SendVirtualKeyboardDirectInput(Vector3 position,
602 InputSource source, bool isPressed, Transform interactorRootTransform = null)
603 {
604 var inputSource = source switch
605 {
606 InputSource.ControllerLeft => OVRPlugin.VirtualKeyboardInputSource.ControllerDirectLeft,
607 InputSource.ControllerRight => OVRPlugin.VirtualKeyboardInputSource.ControllerDirectRight,
608 InputSource.HandLeft => OVRPlugin.VirtualKeyboardInputSource.HandDirectIndexTipLeft,
609 InputSource.HandRight => OVRPlugin.VirtualKeyboardInputSource.HandDirectIndexTipRight,
610 _ => throw new Exception("Unknown input source: " + source)
611 };
612 SendVirtualKeyboardInput(inputSource, new OVRPose()
613 {
614 position = position
615 }, isPressed, interactorRootTransform);
616 }

664 private bool LoadRuntimeVirtualKeyboardMesh()
665 {
666 modelAvailable_ = false;
667 Debug.Log("LoadRuntimeVirtualKeyboardMesh");
668 string[] modelPaths = OVRPlugin.GetRenderModelPaths();
669
670 var keyboardPath = modelPaths?.FirstOrDefault(p => p.Equals("/model_fb/virtual_keyboard")
671 || p.Equals("/model_meta/keyboard/virtual"));
672
673 if (String.IsNullOrEmpty(keyboardPath))
674 {
675 Debug.LogError("Failed to find keyboard model. Check Render Model support.");
676 return false;
677 }
678
679 OVRPlugin.RenderModelProperties modelProps = new OVRPlugin.RenderModelProperties();
680 if (OVRPlugin.GetRenderModelProperties(keyboardPath, ref modelProps))
681 {
682 if (modelProps.ModelKey != OVRPlugin.RENDER_MODEL_NULL_KEY)
683 {
684 virtualKeyboardModelKey_ = modelProps.ModelKey;
685 byte[] data = OVRPlugin.LoadRenderModel(modelProps.ModelKey);
686 if (data != null)
687 {
688 OVRGLTFLoader gltfLoader = new OVRGLTFLoader(data);
689 gltfLoader.textureUriHandler = (string rawUri, Material mat) =>
690 {
691 var uri = new Uri(rawUri);
692 // metaVirtualKeyboard://texture/{id}?w={width}&h={height}&ft=RGBA32
693 if (uri.Scheme != "metaVirtualKeyboard" && uri.Host != "texture")
694 {
695 return null;
696 }
697
698 var textureId = ulong.Parse(uri.LocalPath.Substring(1));
699 if (virtualKeyboardTextures_.ContainsKey(textureId) == false)
700 {
701 virtualKeyboardTextures_[textureId] = new List<Material>();
702 }
703
704 virtualKeyboardTextures_[textureId].Add(mat);
705 return null; // defer texture data loading
706 };
707 gltfLoader.SetModelShader(keyboardModelShader);
708 gltfLoader.SetModelAlphaBlendShader(keyboardModelAlphaBlendShader);
709 virtualKeyboardScene_ = gltfLoader.LoadGLB(supportAnimation: true, loadMips: true);
710
711 modelAvailable_ = virtualKeyboardScene_.root != null;
712 if (modelAvailable_)
713 {
714 virtualKeyboardScene_.root.transform.SetParent(transform, false);
715 virtualKeyboardScene_.root.gameObject.name = "OVRVirtualKeyboardModel";
716 // keyboard is not intended for modification
717 ApplyHideFlags(virtualKeyboardScene_.root.transform);
718 UseSuggestedLocation(InitialPosition);
719 PopulateCollision();
720 }
721 }
722 }
723 }
724
725 return modelAvailable_;
726 }

754 private void ShowKeyboard()
755 {
756 if (!isKeyboardCreated_)
757 {
758 var createInfo = new OVRPlugin.VirtualKeyboardCreateInfo();
759
760 var result = OVRPlugin.CreateVirtualKeyboard(createInfo);
761 if (result != OVRPlugin.Result.Success)
762 {
763#if UNITY_EDITOR
764 if (result == OVRPlugin.Result.Failure_Unsupported || result == OVRPlugin.Result.Failure_NotInitialized)
765 {
766 Debug.LogWarning("Virtual Keyboard Unity Editor support requires Quest Link.");
767 }
768 else
769#endif
770 {
771 Debug.LogError("Create failed: '" + result + "'. Check for Virtual Keyboard Support.");
772 }
773 return;
774 }
775
776 var createSpaceInfo = new OVRPlugin.VirtualKeyboardSpaceCreateInfo();
777 createSpaceInfo.pose = OVRPlugin.Posef.identity;
778 result = OVRPlugin.CreateVirtualKeyboardSpace(createSpaceInfo, out keyboardSpace_);
779 if (result != OVRPlugin.Result.Success)
780 {
781 Debug.LogError("Create failed to create keyboard space: " + result);
782 return;
783 }
784
785 UseSuggestedLocation(InitialPosition);
786
787 // Initialize the keyboard model
788 if (modelInitialized_ != true)
789 {
790 modelInitialized_ = true;
791 if (!LoadRuntimeVirtualKeyboardMesh())
792 {
793 DestroyKeyboard();
794 return;
795 }
796
797 UpdateVisibleState();
798 }
799
800 // Should call this whenever the keyboard is created or when the text focus changes
801 if (TextCommitField != null)
802 {
803 ChangeTextContextInternal(TextCommitField.text);
804 }
805 }
806
807 try
808 {
809 SetKeyboardVisibility(true);
810 isKeyboardCreated_ = true;
811 }
812 catch
813 {
814 DestroyKeyboard();
815 throw;
816 }
817 }

976 private void SyncKeyboardLocation()
977 {
978 // If unity transform has updated, sync with runtime
979 if (transform.hasChanged)
980 {
981 // ensure scale uniformity
982 var scale = MaxElement(transform.localScale);
983 var maxScale = Vector3.one * scale;
984 transform.localScale = maxScale;
985 UseSuggestedLocation(KeyboardPosition.Custom);
986 }
987
988 // query the runtime for the true position
989 if (!OVRPlugin.TryLocateSpace(keyboardSpace_, OVRPlugin.GetTrackingOriginType(), out var keyboardPose))
990 {
991 Debug.LogError("Failed to locate the virtual keyboard space.");
992 return;
993 }
994
995 var result = OVRPlugin.GetVirtualKeyboardScale(out var keyboardScale);
996 if (result != OVRPlugin.Result.Success)
997 {
998 Debug.LogError("Failed to get virtual keyboard scale.");
999 return;
1000 }
1001
1002 Transform keyboardTransform = transform;
1003 keyboardTransform.SetPositionAndRotation(
1004 keyboardPose.Position.FromFlippedZVector3f(),
1005 keyboardPose.Orientation.FromFlippedZQuatf());
1006 keyboardTransform.localScale = Vector3.one * keyboardScale;
1007 // Reset the change flag to prevent recursive updates
1008 keyboardTransform.hasChanged = false;
1009 }

1011 private void UpdateAnimationState()
1012 {
1013 if (!modelAvailable_)
1014 {
1015 return;
1016 }
1017
1018 OVRPlugin.GetVirtualKeyboardDirtyTextures(out var dirtyTextures);
1019 foreach (var textureId in dirtyTextures.TextureIds)
1020 {
1021 if (!virtualKeyboardTextures_.TryGetValue(textureId, out var textureMaterials))
1022 {
1023 continue;
1024 }
1025
1026 var textureData = new OVRPlugin.VirtualKeyboardTextureData();
1027 OVRPlugin.GetVirtualKeyboardTextureData(textureId, ref textureData);
1028 if (textureData.BufferCountOutput > 0)
1029 {
1030 try
1031 {
1032 textureData.Buffer = Marshal.AllocHGlobal((int)textureData.BufferCountOutput);
1033 textureData.BufferCapacityInput = textureData.BufferCountOutput;
1034 OVRPlugin.GetVirtualKeyboardTextureData(textureId, ref textureData);
1035
1036 var texBytes = new byte[textureData.BufferCountOutput];
1037 Marshal.Copy(textureData.Buffer, texBytes, 0, (int)textureData.BufferCountOutput);
1038
1039 var tex = new Texture2D((int)textureData.TextureWidth, (int)textureData.TextureHeight,
1040 TextureFormat.RGBA32, false);
1041 tex.filterMode = FilterMode.Trilinear;
1042 tex.SetPixelData(texBytes, 0);
1043 tex.Apply(true /*updateMipmaps*/, true /*makeNoLongerReadable*/);
1044 foreach (var material in textureMaterials)
1045 {
1046 material.mainTexture = tex;
1047 }
1048 }
1049 finally
1050 {
1051 Marshal.FreeHGlobal(textureData.Buffer);
1052 }
1053 }
1054 }
1055
1056 var result = OVRPlugin.GetVirtualKeyboardModelAnimationStates(out var animationStates);
1057 if (result == OVRPlugin.Result.Success)
1058 {
1059 for (var i = 0; i < animationStates.States.Length; i++)
1060 {
1061 if (!virtualKeyboardScene_.animationNodeLookup.ContainsKey(animationStates.States[i].AnimationIndex))
1062 {
1063 Debug.LogWarning($"Unknown Animation State Index {animationStates.States[i].AnimationIndex}");
1064 continue;
1065 }
1066
1067 var animationNodes =
1068 virtualKeyboardScene_.animationNodeLookup[animationStates.States[i].AnimationIndex];
1069 foreach (var animationNode in animationNodes)
1070 {
1071 animationNode.UpdatePose(animationStates.States[i].Fraction, false);
1072 }
1073 }
1074
1075 if (animationStates.States.Length > 0)
1076 {
1077 foreach (var morphTargets in virtualKeyboardScene_.morphTargetHandlers)
1078 {
1079 morphTargets.Update();
