115 void UpdateDefaultExternalCamera()
116 {
117#if UNITY_EDITOR_WIN || UNITY_STANDALONE_WIN || OVR_ANDROID_MRC
118 // always build a 1080p external camera
119 const int cameraPixelWidth = 1920;
120 const int cameraPixelHeight = 1080;
121 const float cameraAspect = (float)cameraPixelWidth / cameraPixelHeight;
122
123
124 string cameraName = "UnityExternalCamera";
125 OVRPlugin.CameraIntrinsics cameraIntrinsics = new OVRPlugin.CameraIntrinsics();
126 OVRPlugin.CameraExtrinsics cameraExtrinsics = new OVRPlugin.CameraExtrinsics();
127
128 // intrinsics
129
130 cameraIntrinsics.IsValid = OVRPlugin.Bool.True;
131 cameraIntrinsics.LastChangedTimeSeconds = Time.time;
132
133 float vFov = defaultExternalCamera.fieldOfView * Mathf.Deg2Rad;
134 float hFov = Mathf.Atan(Mathf.Tan(vFov * 0.5f) * cameraAspect) * 2.0f;
135 OVRPlugin.Fovf fov = new OVRPlugin.Fovf();
136 fov.UpTan = fov.DownTan = Mathf.Tan(vFov * 0.5f);
137 fov.LeftTan = fov.RightTan = Mathf.Tan(hFov * 0.5f);
138
139 cameraIntrinsics.FOVPort = fov;
140 cameraIntrinsics.VirtualNearPlaneDistanceMeters = defaultExternalCamera.nearClipPlane;
141 cameraIntrinsics.VirtualFarPlaneDistanceMeters = defaultExternalCamera.farClipPlane;
142 cameraIntrinsics.ImageSensorPixelResolution.w = cameraPixelWidth;
143 cameraIntrinsics.ImageSensorPixelResolution.h = cameraPixelHeight;
144
145 // extrinsics
146
147 cameraExtrinsics.IsValid = OVRPlugin.Bool.True;
148 cameraExtrinsics.LastChangedTimeSeconds = Time.time;
149 cameraExtrinsics.CameraStatusData = OVRPlugin.CameraStatus.CameraStatus_Calibrated;
150 cameraExtrinsics.AttachedToNode = OVRPlugin.Node.None;
151
152 Camera mainCamera = Camera.main;
153 OVRCameraRig cameraRig = mainCamera.GetComponentInParent<OVRCameraRig>();
154 if (cameraRig)
155 {
156 Transform trackingSpace = cameraRig.trackingSpace;
157 OVRPose trackingSpacePose = trackingSpace.ToOVRPose(false);
158 OVRPose cameraPose = defaultExternalCamera.transform.ToOVRPose(false);
159 OVRPose relativePose = trackingSpacePose.Inverse() * cameraPose;
160#if OVR_ANDROID_MRC
161 OVRPose stageToLocalPose =
162 OVRPlugin.GetTrackingTransformRelativePose(OVRPlugin.TrackingOrigin.Stage).ToOVRPose();
163 OVRPose localToStagePose = stageToLocalPose.Inverse();
164 relativePose = localToStagePose * relativePose;
165#endif
166 cameraExtrinsics.RelativePose = relativePose.ToPosef();
167 }
168 else
169 {
170 cameraExtrinsics.RelativePose = OVRPlugin.Posef.identity;
171 }
172
173 if (!OVRPlugin.SetDefaultExternalCamera(cameraName, ref cameraIntrinsics, ref cameraExtrinsics))
174 {
175 Debug.LogError("SetDefaultExternalCamera() failed");
176 }
177#endif
178 }

180 private void UpdateSpectatorCameraStatus()
181 {
182#if OVR_ANDROID_MRC
183 // Trigger to switch between 1st person and spectator mode during casting to phone
184 if (OVRInput.GetDown(OVRInput.Button.PrimaryIndexTrigger) || OVRInput.GetDown(OVRInput.Button.SecondaryIndexTrigger))
185 {
186 camMode = OVRPlugin.Media.GetPlatformCameraMode();
187
188 if (camMode == OVRPlugin.Media.PlatformCameraMode.Disabled && readyToSwitch)
189 {
190 OVRPlugin.Media.SetMrcFrameImageFlipped(false);
191 OVRPlugin.Media.SetPlatformCameraMode(OVRPlugin.Media.PlatformCameraMode.Initialized);
192 StartCoroutine(TimerCoroutine());
193 }
194
195 if (camMode == OVRPlugin.Media.PlatformCameraMode.Initialized && readyToSwitch)
196 {
197 OVRPlugin.Media.SetMrcFrameImageFlipped(true);
198 OVRPlugin.Media.SetPlatformCameraMode(OVRPlugin.Media.PlatformCameraMode.Disabled);
199 StartCoroutine(TimerCoroutine());
200 }
201 }
202
203 // Keep spectator camera on dome surface
204 Vector2 axis = OVRInput.Get(OVRInput.Axis2D.SecondaryThumbstick);
205 if (Mathf.Abs(axis.x) > 0.2f)
206 {
207 polar = polar - axis.x * 0.5f;
208 }
209
210 if (Mathf.Abs(axis.y) > 0.2f)
211 {
212 elevation = elevation + axis.y * 0.5f;
213 if (elevation < -90.0f + elevationLimit) elevation = -90.0f + elevationLimit;
214 if (elevation > 90.0f) elevation = 90.0f;
215 }
216
217 axis = OVRInput.Get(OVRInput.Axis2D.PrimaryThumbstick);
218 if (Mathf.Abs(axis.y) > 0.1f)
219 {
220 distance = axis.y * 0.05f + distance;
221 if (distance > distance_far) distance = distance_far;
222 if (distance < distance_near) distance = distance_near;
223 }
224
225 SpectatorCamera.position = SpectatorCameraDomePosition(SpectatorAnchor.position, distance, elevation, polar);
226 SpectatorCamera.rotation = Quaternion.LookRotation(SpectatorCamera.position - SpectatorAnchor.position);
227 Head.position = SpectatorAnchor.position;
228 Head.rotation = SpectatorAnchor.rotation;
229#endif
230 }

232 Vector3 SpectatorCameraDomePosition(Vector3 spectatorAnchorPosition, float d, float e, float p)
233 {
234 float x = d * Mathf.Cos(Mathf.Deg2Rad * e) * Mathf.Cos(Mathf.Deg2Rad * p);
235 float y = d * Mathf.Sin(Mathf.Deg2Rad * e);
236 float z = d * Mathf.Cos(Mathf.Deg2Rad * e) * Mathf.Sin(Mathf.Deg2Rad * p);
237
238 return new Vector3(x + spectatorAnchorPosition.x, y + spectatorAnchorPosition.y, z + spectatorAnchorPosition.z);
239 }
