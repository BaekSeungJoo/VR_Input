126 public void Update()
127 {
128 UpdateTextures();
129
130 if (recenterRequested && Time.frameCount > recenterRequestedFrameCount)
131 {
132 Debug.Log("Recenter event detected");
133 if (RecenteredPose != null)
134 {
135 RecenteredPose();
136 }
137
138 recenterRequested = false;
139 recenterRequestedFrameCount = int.MaxValue;
140 }
141
142 if (OVRPlugin.GetSystemHeadsetType() >= OVRPlugin.SystemHeadset.Oculus_Quest &&
143 OVRPlugin.GetSystemHeadsetType() < OVRPlugin.SystemHeadset.Rift_DK1) // all Oculus Standalone headsets
144 {
145 int recenterCount = OVRPlugin.GetLocalTrackingSpaceRecenterCount();
146 if (localTrackingSpaceRecenterCount != recenterCount)
147 {
148 Debug.Log("Recenter event detected");
149 if (RecenteredPose != null)
150 {
151 RecenteredPose();
152 }
153
154 localTrackingSpaceRecenterCount = recenterCount;
155 }
156 }

162 public event System.Action RecenteredPose;

167 public void RecenterPose()
168 {
169#if USING_XR_SDK
170 XRInputSubsystem currentInputSubsystem = OVRManager.GetCurrentInputSubsystem();
171 if (currentInputSubsystem != null)
172 {
173 currentInputSubsystem.TryRecenter();
174 }
175#elif !REQUIRES_XR_SDK
176#pragma warning disable 618
177 InputTracking.Recenter();
178#pragma warning restore 618
179#endif
180
181 // The current poses are cached for the current frame and won't be updated immediately
182 // after UnityEngine.VR.InputTracking.Recenter(). So we need to wait until next frame
183 // to trigger the RecenteredPose delegate. The application could expect the correct pose
184 // when the RecenteredPose delegate get called.
185 recenterRequested = true;
186 recenterRequestedFrameCount = Time.frameCount;
187
188#if UNITY_EDITOR_WIN || UNITY_STANDALONE_WIN
189 OVRMixedReality.RecenterPose();
190#endif
191 }

278 public LatencyData latency
279 {
280 get
281 {
282 if (!OVRManager.isHmdPresent)
283 return new LatencyData();
284
285 string latency = OVRPlugin.latency;
286
287 var r = new Regex(
288 "Render: ([0-9]+[.][0-9]+)ms, TimeWarp: ([0-9]+[.][0-9]+)ms, PostPresent: ([0-9]+[.][0-9]+)ms",
289 RegexOptions.None);
290
291 var ret = new LatencyData();
292
293 Match match = r.Match(latency);
294 if (match.Success)
295 {
296 ret.render = float.Parse(match.Groups[1].Value);
297 ret.timeWarp = float.Parse(match.Groups[2].Value);
298 ret.postPresent = float.Parse(match.Groups[3].Value);
299 }
300
301 return ret;
302 }

358 private void ConfigureEyeDesc(Node eye)
359 {
360 if (!OVRManager.isHmdPresent)
361 return;
362
363 int eyeTextureWidth = Settings.eyeTextureWidth;
364 int eyeTextureHeight = Settings.eyeTextureHeight;
365
366 eyeDescs[(int)eye] = new EyeRenderDesc();
367 eyeDescs[(int)eye].resolution = new Vector2(eyeTextureWidth, eyeTextureHeight);
368
369 OVRPlugin.Frustumf2 frust;
370 if (OVRPlugin.GetNodeFrustum2((OVRPlugin.Node)eye, out frust))
371 {
372 eyeDescs[(int)eye].fullFov.LeftFov = Mathf.Rad2Deg * Mathf.Atan(frust.Fov.LeftTan);
373 eyeDescs[(int)eye].fullFov.RightFov = Mathf.Rad2Deg * Mathf.Atan(frust.Fov.RightTan);
374 eyeDescs[(int)eye].fullFov.UpFov = Mathf.Rad2Deg * Mathf.Atan(frust.Fov.UpTan);
375 eyeDescs[(int)eye].fullFov.DownFov = Mathf.Rad2Deg * Mathf.Atan(frust.Fov.DownTan);
376 }
377 else
378 {
379 OVRPlugin.Frustumf frustOld = OVRPlugin.GetEyeFrustum((OVRPlugin.Eye)eye);
380 eyeDescs[(int)eye].fullFov.LeftFov = Mathf.Rad2Deg * frustOld.fovX * 0.5f;
381 eyeDescs[(int)eye].fullFov.RightFov = Mathf.Rad2Deg * frustOld.fovX * 0.5f;
382 eyeDescs[(int)eye].fullFov.UpFov = Mathf.Rad2Deg * frustOld.fovY * 0.5f;
383 eyeDescs[(int)eye].fullFov.DownFov = Mathf.Rad2Deg * frustOld.fovY * 0.5f;
384 }
385
386 // Symmetric Fov uses the maximum fov angle
387 float maxFovX = Mathf.Max(eyeDescs[(int)eye].fullFov.LeftFov, eyeDescs[(int)eye].fullFov.RightFov);
388 float maxFovY = Mathf.Max(eyeDescs[(int)eye].fullFov.UpFov, eyeDescs[(int)eye].fullFov.DownFov);
389 eyeDescs[(int)eye].fov.x = maxFovX * 2.0f;
390 eyeDescs[(int)eye].fov.y = maxFovY * 2.0f;
391
392 if (!OVRPlugin.AsymmetricFovEnabled)
393 {
394 eyeDescs[(int)eye].fullFov.LeftFov = maxFovX;
395 eyeDescs[(int)eye].fullFov.RightFov = maxFovX;
396
397 eyeDescs[(int)eye].fullFov.UpFov = maxFovY;
398 eyeDescs[(int)eye].fullFov.DownFov = maxFovY;
399 }
