43 // GUI
44 public override void OnInspectorGUI()
45 {
46 // Display default ui
47 base.OnInspectorGUI();
48
49 // Check voices
50 TTSService tts = _speaker.TTSService;
51 TTSVoiceSettings[] settings = tts?.GetAllPresetVoiceSettings();
52 if (_voicePresetIds == null
53 || (settings != null && _voicePresetIds.Length != settings.Length + 1)
54 || (_voiceIndex >= 0 && _voiceIndex < _voicePresetIds.Length - 1 && !string.Equals(_speaker.presetVoiceID, _voicePresetIds[_voiceIndex])))
55 {
56 RefreshVoices(settings);
57 }
58
59 // No preset voices found, assume custom
60 if (_voicePresetIds == null || _voicePresetIds.Length == 0)
61 {
62 GUI.enabled = false;
63 EditorGUILayout.TextField(UI_PRESET_VOICE_KEY, UI_CUSTOM_KEY);
64 GUI.enabled = true;
65 }
66 // Voice dropdown
67 else
68 {
69 bool updated = false;
70 WitEditorUI.LayoutPopup(UI_PRESET_VOICE_KEY, _voicePresetIds, ref _voiceIndex, ref updated);
71 if (updated)
72 {
73 if (_voiceIndex >= 0 && _voiceIndex < _voicePresetIds.Length - 1)
74 {
75 _presetVoiceProperty.stringValue = _voicePresetIds[_voiceIndex];
76 }
77 else
78 {
79 _presetVoiceProperty.stringValue = null;
80 }
81 }
82 }
83
84 // Add custom layout
85 if (_voicePresetIds == null || _voiceIndex < 0 || _voiceIndex >= _voicePresetIds.Length - 1)
86 {
87 EditorGUILayout.PropertyField(_customVoiceProperty, new GUIContent(UI_CUSTOM_VOICE_KEY));
88 }
89
90 // Apply all modified properties
91 serializedObject.ApplyModifiedProperties();
92
93 // Layout TTS clip queue
94 LayoutClipQueue();
95 }

97 // Refresh voices
98 private void RefreshVoices(TTSVoiceSettings[] settings)
99 {
100 // Reset voice data
101 _voiceIndex = -1;
102 _voicePresetIds = null;
103 if (settings == null)
104 {
105 return;
106 }
107
108 // Get all ids
109 List<string> presetIds = settings.Select(s => s.SettingsId).ToList();
110 // Get voice index
111 _voiceIndex = presetIds.IndexOf(_speaker.presetVoiceID);
112 if (_voiceIndex == -1)
113 {
114 _voiceIndex = presetIds.Count;
115 }
116 // Add custom key
117 presetIds.Add(UI_CUSTOM_KEY);
118 // Apply preset ids
119 _voicePresetIds = presetIds.ToArray();
120 }

122 // Layout clip queue
123 private const string UI_CLIP_HEADER_TEXT = "Clip Queue";
124 private const string UI_CLIP_SPEAKER_TEXT = "Speaker Clip:";
125 private const string UI_CLIP_QUEUE_TEXT = "Loading Clips:";
126 private bool _speakerFoldout = false;
127 private bool _queueFoldout = false;
128 private void LayoutClipQueue()
129 {
130 // Ignore unless playing
131 if (!Application.isPlaying)
132 {
133 return;
134 }
135
136 // Add header
137 EditorGUILayout.Space();
138 EditorGUILayout.LabelField(UI_CLIP_HEADER_TEXT, EditorStyles.boldLabel);
139
140 // Speaker Foldout
141 _speakerFoldout = EditorGUILayout.Foldout(_speakerFoldout, UI_CLIP_SPEAKER_TEXT);
142 if (_speakerFoldout)
143 {
144 EditorGUI.indentLevel++;
145 if (!_speaker.IsSpeaking)
146 {
147 EditorGUILayout.LabelField("None");
148 }
149 else
150 {
151 TTSServiceInspector.DrawClipGUI(_speaker.SpeakingClip);
152 }
153 EditorGUI.indentLevel--;
154 }
155
156 // Queue Foldout
157 List<TTSClipData> queuedClips = _speaker.QueuedClips;
158 _queueFoldout = EditorGUILayout.Foldout(_queueFoldout, $"{UI_CLIP_QUEUE_TEXT} {(queuedClips == null ? 0 : queuedClips.Count)}");
159 if (_queueFoldout)
160 {
161 EditorGUI.indentLevel++;
162 if (queuedClips == null || queuedClips.Count == 0)
163 {
164 EditorGUILayout.LabelField("None");
165 }
166 else
167 {
168 for (int i = 0; i < queuedClips.Count; i++)
169 {
170 TTSClipData clipData = queuedClips[i];
171 bool oldFoldout = WitEditorUI.GetFoldoutValue(clipData);
172 bool newFoldout = EditorGUILayout.Foldout(oldFoldout, $"Clip[{i}]");
173 if (oldFoldout != newFoldout)
174 {
175 WitEditorUI.SetFoldoutValue(clipData, newFoldout);
176 }
177 if (newFoldout)
178 {
179 EditorGUI.indentLevel++;
180 TTSServiceInspector.DrawClipGUI(clipData);
181 EditorGUI.indentLevel--;
182 }
183 }
184 }
185 EditorGUI.indentLevel--;
186 }
187 }
