174 private void InitVoiceSDK()
175 {
176 // Check voice sdk version
177 if (string.IsNullOrEmpty(PACKAGE_VERSION))
178 {
179 VLog.E("No SDK Version Set");
180 }
181
182 // Clean up if we're switching to native C# wit impl
183 if (!UsePlatformIntegrations)
184 {
185 if (voiceServiceImpl is VoiceSDKImpl)
186 {
187 ((VoiceSDKImpl) voiceServiceImpl).Disconnect();
188 }
189
190 if (voiceSDKLoggerImpl is VoiceSDKPlatformLoggerImpl)
191 {
192 try
193 {
194 ((VoiceSDKPlatformLoggerImpl)voiceSDKLoggerImpl).Disconnect();
195 }
196 catch (Exception e)
197 {
198 VLog.E($"Disconnection error: {e.Message}");
199 }
200 }
201 }
202#if UNITY_ANDROID && !UNITY_EDITOR
203 var loggerImpl = new VoiceSDKPlatformLoggerImpl();
204 loggerImpl.Connect(PACKAGE_VERSION);
205 voiceSDKLoggerImpl = loggerImpl;
206
207 if (UsePlatformIntegrations)
208 {
209 Debug.Log("Checking platform capabilities...");
210 var platformImpl = new VoiceSDKImpl(this);
211 platformImpl.OnServiceNotAvailableEvent += () => RevertToWitUnity();
212 platformImpl.Connect(PACKAGE_VERSION);
213 platformImpl.SetRuntimeConfiguration(RuntimeConfiguration);
214 if (platformImpl.PlatformSupportsWit)
215 {
216 voiceServiceImpl = platformImpl;
217
218 if (voiceServiceImpl is Wit wit)
219 {
220 wit.RuntimeConfiguration = witRuntimeConfiguration;
221 }
222
223 voiceServiceImpl.VoiceEvents = VoiceEvents;
224 voiceServiceImpl.TelemetryEvents = TelemetryEvents;
225 voiceSDKLoggerImpl.IsUsingPlatformIntegration = true;
226 }
227 else
228 {
229 Debug.Log("Platform registration indicated platform support is not currently available.");
230 RevertToWitUnity();
231 }
232 }
233 else
234 {
235 RevertToWitUnity();
236 }
237#else
238 voiceSDKLoggerImpl = new VoiceSDKConsoleLoggerImpl();
239 RevertToWitUnity();
240#endif
241 voiceSDKLoggerImpl.WitApplication = RuntimeConfiguration?.witConfiguration?.GetLoggerAppId();
242 voiceSDKLoggerImpl.ShouldLogToConsole = EnableConsoleLogging;
243
244 OnInitialized?.Invoke();
245 }
246
247 private void RevertToWitUnity()
248 {
249 Wit w = GetComponent<Wit>();
250 if (null == w)
251 {
252 w = gameObject.AddComponent<Wit>();
253 w.hideFlags = HideFlags.HideInInspector;
254 }
255 voiceServiceImpl = w;
256
257 if (voiceServiceImpl is Wit wit)
258 {
259 wit.RuntimeConfiguration = witRuntimeConfiguration;
260 }
261
262 voiceServiceImpl.VoiceEvents = VoiceEvents;
263 voiceServiceImpl.TelemetryEvents = TelemetryEvents;
264 voiceSDKLoggerImpl.IsUsingPlatformIntegration = false;
265 }

434 void OnRequestComplete(VoiceServiceRequest request)
435 {

438 VLog.E($"Request Failed\nError: {request.Results.Message}\n{request.Results.ResponseData}");
439 voiceSDKLoggerImpl.LogInteractionEndFailure(request.Results.Message);
440 }
441 else if (request.State == VoiceRequestState.Canceled)
442 {
443 VLog.D($"Request Canceled\nMessage: {request.Results.Message}");
444 voiceSDKLoggerImpl.LogInteractionEndFailure("aborted");
