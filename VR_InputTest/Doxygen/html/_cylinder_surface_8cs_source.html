105 public bool ClosestSurfacePoint(in Vector3 point, out SurfaceHit hit, float maxDistance)
106 {
107 hit = new SurfaceHit();
108
109 if (!IsValid)
110 {
111 return false;
112 }
113
114 Vector3 localPoint = _cylinder.transform.InverseTransformPoint(point);
115 Vector3 clampedOrigin = localPoint;
116
117 if (_height > 0)
118 {
119 clampedOrigin.y = Mathf.Clamp(clampedOrigin.y, -_height / 2f, _height / 2f);
120 }
121
122 Vector3 nearestPointOnCenterAxis = Vector3.Project(clampedOrigin, Vector3.up);
123
124 Vector3 direction = (clampedOrigin == nearestPointOnCenterAxis) ?
125 Vector3.forward : (clampedOrigin - nearestPointOnCenterAxis).normalized;
126
127 bool isOutside = (clampedOrigin - nearestPointOnCenterAxis).magnitude > Radius;
128 Vector3 hitPoint = nearestPointOnCenterAxis + direction * Radius;
129 float hitDistance = Vector3.Distance(localPoint, hitPoint);
130
131 if (maxDistance > 0 && TransformScale(hitDistance) > maxDistance)
132 {
133 return false;
134 }
135
136 Vector3 normal;
137 switch (_facing)
138 {
139 default:
140 case NormalFacing.Any:
141 normal = isOutside ? direction : -direction;
142 break;
143 case NormalFacing.In:
144 normal = -direction;
145 break;
146 case NormalFacing.Out:
147 normal = direction;
148 break;
149 }
150
151 hit.Point = _cylinder.transform.TransformPoint(nearestPointOnCenterAxis + direction * Radius);
152 hit.Normal = _cylinder.transform.TransformDirection(normal);
153 hit.Distance = TransformScale(hitDistance);
154
155 return true;
156 }

158 public bool Raycast(in Ray ray, out SurfaceHit hit, float maxDistance)
159 {
160 // Flatten to 2D and find intersection point with circle in local space,
161 // then convert back into 3D world space
162
163 hit = new SurfaceHit();
164
165 if (!IsValid)
166 {
167 return false;
168 }
169
170 Ray localRay3D = new Ray(_cylinder.transform.InverseTransformPoint(ray.origin),
171 _cylinder.transform.InverseTransformDirection(ray.direction).normalized);
172
173 Ray localRay2D = new Ray(CancelY(localRay3D.origin), CancelY(localRay3D.direction).normalized);
174
175 Vector3 originToCenter2D = -localRay2D.origin;
176 Vector3 projPoint2D = localRay2D.origin + Vector3.Project(originToCenter2D, localRay2D.direction);
177
178 float magDir2D = Vector3.Magnitude(CancelY(localRay3D.direction));
179 float magProj = Vector3.Magnitude(projPoint2D);
180
181 bool isOutside = originToCenter2D.magnitude > Radius;
182
183 NormalFacing effectiveFacing = _facing == NormalFacing.Any && !isOutside ? NormalFacing.In : _facing;
184
185 bool hasMissed = magProj > Radius || // Aiming toward but missing
186 Mathf.Approximately(magDir2D, 0f) || // Aiming up or down
187 (isOutside && Vector3.Dot(originToCenter2D, localRay2D.direction) < 0) || // Aiming away
188 (!isOutside && effectiveFacing == NormalFacing.Out); // Origin inside with normals out
189
190 if (hasMissed)
191 {
192 return false;
193 }
194
195 float projToWall = Mathf.Sqrt(Mathf.Pow(Radius, 2) - Mathf.Pow(magProj, 2));
196 float tOuter = Vector3.Distance(localRay2D.origin, projPoint2D - localRay2D.direction * projToWall) / magDir2D;
197 float tInner = Vector3.Distance(localRay2D.origin, projPoint2D + localRay2D.direction * projToWall) / magDir2D;
198
199 Vector3 localHitpointOuter = localRay3D.GetPoint(tOuter);
200 Vector3 localHitpointInner = localRay3D.GetPoint(tInner);
201
202 bool hasOuter = (maxDistance <= 0 || TransformScale(tOuter) <= maxDistance) &&
203 (_height <= 0 || Mathf.Abs(localHitpointOuter.y) <= _height / 2f);
204
205 bool hasInner = (maxDistance <= 0 || TransformScale(tInner) <= maxDistance) &&
206 (_height <= 0 || Mathf.Abs(localHitpointInner.y) <= _height / 2f);
207
208 if (effectiveFacing != NormalFacing.In && hasOuter)
209 {
210 hit.Point = _cylinder.transform.TransformPoint(localHitpointOuter);
211 hit.Normal = _cylinder.transform.TransformDirection(CancelY(localHitpointOuter).normalized);
212 hit.Distance = TransformScale(tOuter);
213 }
214 else if (hasInner)
215 {
216 hit.Point = _cylinder.transform.TransformPoint(localHitpointInner);
217 hit.Normal = _cylinder.transform.TransformDirection(CancelY(-localHitpointInner).normalized);
218 hit.Distance = TransformScale(tInner);
219 }
220 else
221 {
222 return false;
223 }
224
225 return true;
226 }

234 }
