66 public string GetDiskCachePath(TTSClipData clipData)
67 {
68 // Disabled
69 if (!ShouldCacheToDisk(clipData))
70 {
71 return string.Empty;
72 }
73
74 // Get directory path
75 TTSDiskCacheLocation location = clipData.diskCacheSettings.DiskCacheLocation;
76 string directory = string.Empty;
77 switch (location)
78 {
79 case TTSDiskCacheLocation.Persistent:
80 directory = Application.persistentDataPath;
81 break;
82 case TTSDiskCacheLocation.Temporary:
83 directory = Application.temporaryCachePath;
84 break;
85 case TTSDiskCacheLocation.Preload:
86 directory = Application.streamingAssetsPath;
87 break;
88 }
89 if (string.IsNullOrEmpty(directory))
90 {
91 return string.Empty;
92 }
93
94 // Add tts cache path & clean
95 directory = Path.Combine(directory, DiskPath);
96
97 // Generate tts directory if possible
98 if (location != TTSDiskCacheLocation.Preload || !Application.isPlaying)
99 {
100 if (!IOUtility.CreateDirectory(directory, true))
101 {
102 VLog.E($"Failed to create tts directory\nPath: {directory}\nLocation: {location}");
103 return string.Empty;
104 }
105 }
106
107 // Return clip path
108 return Path.Combine(directory, clipData.clipID + "." + WitTTSVRequest.GetAudioExtension(clipData.audioType));
109 }

116 public bool ShouldCacheToDisk(TTSClipData clipData)
117 {
118 return clipData != null && clipData.diskCacheSettings.DiskCacheLocation != TTSDiskCacheLocation.Stream && !string.IsNullOrEmpty(clipData.clipID);
119 }

126 public void CheckCachedToDisk(TTSClipData clipData, Action<TTSClipData, bool> onCheckComplete)
127 {
128 // Get path
129 string cachePath = GetDiskCachePath(clipData);
130 if (string.IsNullOrEmpty(cachePath))
131 {
132 onCheckComplete?.Invoke(clipData, false);
133 return;
134 }
135
136 // Check if file exists
137 VRequest request = new VRequest();
138 bool canPerform = request.RequestFileExists(cachePath, (success, error) =>
139 {
140 // Remove
141 if (_streamRequests.ContainsKey(clipData.clipID))
142 {
143 _streamRequests.Remove(clipData.clipID);
144 }
145 // Complete
146 onCheckComplete(clipData, success);
147 });
148 if (canPerform)
149 {
150 _streamRequests[clipData.clipID] = request;
151 }
152 }

157 public void StreamFromDiskCache(TTSClipData clipData)
158 {
159 // Invoke begin
160 DiskStreamEvents?.OnStreamBegin?.Invoke(clipData);
161
162 // Get file path
163 string filePath = GetDiskCachePath(clipData);
164
165 // Load clip async
166 VRequest request = new VRequest((progress) => clipData.loadProgress = progress);
167 bool canPerform = request.RequestAudioStream(clipData.clipStream, new Uri(request.CleanUrl(filePath)),
168 (clipStream, error) =>
169 {
170 clipData.clipStream = clipStream;
171 OnStreamComplete(clipData, error);
172 }, clipData.audioType, clipData.diskCacheSettings.StreamFromDisk);
173 if (canPerform)
174 {
175 _streamRequests[clipData.clipID] = request;
176 }
177 }
181 public void CancelDiskCacheStream(TTSClipData clipData)
182 {
183 // Ignore if not currently streaming
184 if (!_streamRequests.ContainsKey(clipData.clipID))
185 {
186 return;
187 }
188
189 // Get request
190 VRequest request = _streamRequests[clipData.clipID];
191 _streamRequests.Remove(clipData.clipID);
192
193 // Cancel immediately
194 request?.Cancel();
195 request = null;
196
197 // Call cancel
198 DiskStreamEvents?.OnStreamCancel?.Invoke(clipData);
199 }
200 // On stream completion
201 protected virtual void OnStreamComplete(TTSClipData clipData, string error)
202 {
203 // Ignore if not currently streaming
204 if (!_streamRequests.ContainsKey(clipData.clipID))
205 {
206 return;
207 }
208
209 // Remove from list
210 _streamRequests.Remove(clipData.clipID);
211
212 // Error
213 if (!string.IsNullOrEmpty(error))
214 {
215 DiskStreamEvents?.OnStreamError?.Invoke(clipData, error);
216 }
217 // Success
218 else
219 {
220 DiskStreamEvents?.OnStreamReady?.Invoke(clipData);
221 }
222 }
