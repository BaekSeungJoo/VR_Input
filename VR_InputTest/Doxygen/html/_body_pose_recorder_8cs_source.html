119 private void DrawUI()
120 {
121 GUI.enabled = !_willCapture;
122
123 SerializedProperty bodyProp =
124 _serializedEditor.FindProperty(nameof(_body));

143 _beepOnCapture = EditorGUILayout.Toggle("Play Sound on Capture", _beepOnCapture);
144 _serializedEditor.ApplyModifiedProperties();
145
146 GUILayout.Space(20);

149 string buttonLabel = _willCapture ?

151 $"seconds\n(Click To Cancel)" : "Capture Body Pose";
152

162 private void HandleCapture()
163 {
164 if (!_willCapture)
165 {
166 return;
167 }
168
169 if (TimeUntilCapture <= 0f)
170 {
171 _willCapture = false;
172 if (_beepOnCapture)
173 {
174 EditorApplication.Beep();
175 }
176
177 BodyPoseData assetToWrite = _targetAsset;
178 if (assetToWrite == null)
179 {
180 assetToWrite = GeneratePoseAsset();
181 }
182
183 assetToWrite.SetBodyPose(Body);
184 Debug.Log($"Captured Body Pose into " +
185 $"{AssetDatabase.GetAssetPath(assetToWrite)}");
186 EditorUtility.SetDirty(assetToWrite);
187 AssetDatabase.SaveAssetIfDirty(assetToWrite);
188 AssetDatabase.Refresh();
189 }
190 }

192 public BodyPoseData GeneratePoseAsset()
193 {
194 var poseDataAsset = ScriptableObject.CreateInstance<BodyPoseData>();
195 string parentDir = Path.Combine("Assets", POSE_DIRECTORY);
196 if (!Directory.Exists(parentDir))
197 {
198 Directory.CreateDirectory(parentDir);
199 }
200 string name = "BodyPose-" + $"{System.DateTime.Now.ToString("yyyyMMdd-HHmmss")}";
201 AssetDatabase.CreateAsset(poseDataAsset, Path.Combine(parentDir, $"{name}.asset"));
202 return poseDataAsset;
203 }
