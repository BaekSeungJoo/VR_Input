181 protected override TeleportInteractable ComputeCandidate()
182 {
183 Pose origin = ArcOrigin;
184 float bestScore = float.PositiveInfinity;
185 Vector3 arcEndPosition = TeleportArc.PointAtIndex(TeleportArc.PointsCount - 1);
186
187 TeleportInteractable bestCandidate = null;
188 TeleportHit bestHit = new TeleportHit(null, arcEndPosition, Vector3.up);
189 var interactables = TeleportInteractable.Registry.List(this);
190
191 Pose headPose = Pose.identity;
192 if (Hmd != null)
193 {
194 Hmd.TryGetRootPose(out headPose);
195 }
196
197 foreach (TeleportInteractable interactable in interactables)
198 {
199 if (interactable.AllowTeleport)
200 {
201 continue;
202 }
203
204 if (Hmd != null)
205 {
206 CheckViewToOriginBlockers(headPose.position, interactable);
207 }
208 CheckCandidate(interactable);
209 }
210
211 foreach (TeleportInteractable interactable in interactables)
212 {
213 if (interactable.AllowTeleport)
214 {
215 CheckCandidate(interactable);
216 }
217 }
218
219 _arcEnd = bestHit;
220 return bestCandidate;
221
222 void CheckViewToOriginBlockers(Vector3 viewPosition, TeleportInteractable candidate)
223 {
224 if (candidate.DetectHit(viewPosition, origin.position, out TeleportHit hit))
225 {
226 float score = -Vector3.Distance(origin.position, hit.Point);
227 bool isTie = Mathf.Abs(bestScore - score) < _equalDistanceThreshold;
228 if (bestCandidate == null
229 || (!isTie && score < bestScore)
230 || (isTie && ComputeCandidateTiebreaker(candidate, bestCandidate) > 0))
231 {
232 bestScore = score;
233 bestHit = hit;
234 bestCandidate = candidate;
235 }
236 }
237 }
238
239 void CheckCandidate(TeleportInteractable candidate)
240 {
241 Vector3 prevPoint = origin.position;
242 float accumulatedDistance = 0;
243 for (int i = 1; i < TeleportArc.PointsCount; i++)
244 {
245 if (accumulatedDistance > bestScore)
246 {
247 break;
248 }
249 Vector3 point = TeleportArc.PointAtIndex(i);
250 if (candidate.DetectHit(prevPoint, point, out TeleportHit hit))
251 {
252 float score = accumulatedDistance
253 + Vector3.Distance(prevPoint, hit.Point);
254
255 bool isTie = Mathf.Abs(bestScore - score) < _equalDistanceThreshold;
256 if (!isTie
257 && bestCandidate != null
258 && !bestCandidate.AllowTeleport
259 && candidate.AllowTeleport)
260 {
261 float snapRadius = hit.SnapRadius;
262 float pointToPoint = (bestHit.Point - hit.Point).sqrMagnitude;
263 if (pointToPoint < snapRadius * snapRadius)
264 {
265 isTie = true;
266 }
267 }
268
269 if (bestCandidate == null
270 || (!isTie && score < bestScore)
271 || (isTie && ComputeCandidateTiebreaker(candidate, bestCandidate) > 0))
272 {
273 bestScore = score;
274 bestHit = hit;
275 bestCandidate = candidate;
276 break;
277 }
278 }
279 accumulatedDistance += Vector3.Distance(prevPoint, point);
280 prevPoint = point;
281 }
282 }
283 }

285 protected override int ComputeCandidateTiebreaker(TeleportInteractable a, TeleportInteractable b)
286 {
287 int result = base.ComputeCandidateTiebreaker(a, b);
288 if (result != 0)

290 return result;
291 }
292
293 return a.TieBreakerScore.CompareTo(b.TieBreakerScore);
