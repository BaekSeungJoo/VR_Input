54 public void UpdateState(OVRHand hand, Interactable currFocusedInteractable)
55 {
56 float pinchStrength = hand.GetFingerPinchStrength(OVRHand.HandFinger.Index);
57 bool isPinching = Mathf.Abs(PINCH_STRENGTH_THRESHOLD - pinchStrength) < Mathf.Epsilon;
58 var oldPinchState = _currPinchState;
59
60 switch (oldPinchState)
61 {
62 case PinchState.PinchUp:
63 // can only be in pinch up for a single frame, so consider
64 // next frame carefully
65 if (isPinching)
66 {
67 _currPinchState = PinchState.PinchDown;
68 if (currFocusedInteractable != _firstFocusedInteractable)
69 {
70 _firstFocusedInteractable = null;
71 }
72 }
73 else
74 {
75 _currPinchState = PinchState.None;
76 _firstFocusedInteractable = null;
77 }
78
79 break;
80 case PinchState.PinchStay:
81 if (!isPinching)
82 {
83 _currPinchState = PinchState.PinchUp;
84 }
85
86 // if object is not focused anymore, then forget it
87 if (currFocusedInteractable != _firstFocusedInteractable)
88 {
89 _firstFocusedInteractable = null;
90 }
91
92 break;
93 // pinch down lasts for a max of 1 frame. either go to pinch stay or up
94 case PinchState.PinchDown:
95 _currPinchState = isPinching ? PinchState.PinchStay : PinchState.PinchUp;
96 // if the focused interactable changes, then the original one is now invalid
97 if (_firstFocusedInteractable != currFocusedInteractable)
98 {
99 _firstFocusedInteractable = null;
100 }
101
102 break;
103 default:
104 if (isPinching)
105 {
106 _currPinchState = PinchState.PinchDown;
107 // this is the interactable that must be focused through out the pinch up and down
108 // gesture.
109 _firstFocusedInteractable = currFocusedInteractable;
110 }
111
112 break;
113 }
114 }
115 }
