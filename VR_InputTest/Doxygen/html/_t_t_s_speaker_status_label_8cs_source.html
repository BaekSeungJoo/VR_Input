61 private void RefreshLabel()
62 {
63 StringBuilder status = new StringBuilder();
64 int count = 1;
65 if (Speaker.IsSpeaking)
66 {
67 status.Append($"Speaking[{count}]: '{Speaker.SpeakingClip.textToSpeak}'");
68 status.AppendLine("\n");
69 count++;
70 }
71 foreach (var clip in Speaker.QueuedClips)
72 {
73 status.Append($"{clip.loadState.ToString()}[{count}]: '{clip.textToSpeak}'");
74 status.AppendLine("\n");
75 count++;
76 }
77 _label.text = status.ToString();
78 _label.rectTransform.SetSizeWithCurrentAnchors(RectTransform.Axis.Vertical, _label.preferredHeight);
