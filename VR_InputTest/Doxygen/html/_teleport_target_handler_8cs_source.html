59 private IEnumerator TargetAimCoroutine()
60 {
61 // While the teleport system is in the aim state, perform the aim logic and consider teleporting.
62 while (LocomotionTeleport.CurrentState == LocomotionTeleport.States.Aim)
63 {
64 // With each targeting test, we need to reset the AimData to clear the point list and reset flags.
65 ResetAimData();
66
67 // Start the testing with the character's current position to the aiming origin to ensure they
68 // haven't just stuck their hand through something that should have prevented movement.
69 //
70 // The first test won't be added to the aim data results because the visual effects should be from
71 // the aiming origin.
72
73 var current = LocomotionTeleport.transform.position;
74
75 // Enumerate through all the line segments provided by the aim handler, checking for a valid target on each segment,
76 // stopping at the first valid target or when the enumerable runs out of line segments.
77 _aimPoints.Clear();
78 LocomotionTeleport.AimHandler.GetPoints(_aimPoints);
79
80 for (int i = 0; i < _aimPoints.Count; i++)
81 {
82 var adjustedPoint = _aimPoints[i];
83 AimData.TargetValid = ConsiderTeleport(current, ref adjustedPoint);
84 AimData.Points.Add(adjustedPoint);
85 if (AimData.TargetValid)
86 {
87 AimData.Destination = ConsiderDestination(adjustedPoint);
88 AimData.TargetValid = AimData.Destination.HasValue;
89 break;
90 }
91
92 current = _aimPoints[i];
93 }
94
95 LocomotionTeleport.OnUpdateAimData(AimData);
96 yield return null;
97 }
98 }

126 public virtual Vector3? ConsiderDestination(Vector3 location)
127 {
128 var character = LocomotionTeleport.LocomotionController.CharacterController;
129 var radius = character.radius - ERROR_MARGIN;
130
131 var start = location;
132 start.y += radius + ERROR_MARGIN;
133 var end = start;
134 end.y += character.height - ERROR_MARGIN;
135
136 var result = Physics.CheckCapsule(start, end, radius,
137 AimCollisionLayerMask, QueryTriggerInteraction.Ignore);
138 if (result)
139 {
140 return null;
141 }
142
143 return location;
144 }
