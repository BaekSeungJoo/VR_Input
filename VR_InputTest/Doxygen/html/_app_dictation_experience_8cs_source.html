106 private void InitDictation()
107 {
108 // Check voice sdk version
109 if (string.IsNullOrEmpty(PACKAGE_VERSION))
110 {
111 VLog.E("No SDK Version Set");
112 }
113 // Clean up if we're switching to native C# wit impl
114 if (!UsePlatformIntegrations)
115 {
116 if (_dictationServiceImpl is PlatformDictationImpl)
117 {
118 ((PlatformDictationImpl) _dictationServiceImpl).Disconnect();
119 }
120
121 if (_voiceSDKLogger is VoiceSDKPlatformLoggerImpl)
122 {
123 try
124 {
125 ((VoiceSDKPlatformLoggerImpl)_voiceSDKLogger).Disconnect();
126 }
127 catch (Exception e)
128 {
129 VLog.E($"Disconnection error: {e.Message}");
130 }
131 }
132 }
133#if UNITY_ANDROID && !UNITY_EDITOR
134 var loggerImpl = new VoiceSDKPlatformLoggerImpl();
135 loggerImpl.Connect(PACKAGE_VERSION);
136 _voiceSDKLogger = loggerImpl;
137
138 if (UsePlatformIntegrations)
139 {
140 VLog.D("Checking platform dictation capabilities...");
141 var platformDictationImpl = new PlatformDictationImpl(this);
142 platformDictationImpl.OnServiceNotAvailableEvent += OnPlatformServiceNotAvailable;
143 platformDictationImpl.Connect(PACKAGE_VERSION);
144 if (platformDictationImpl.PlatformSupportsDictation)
145 {
146 _dictationServiceImpl = platformDictationImpl;
147 _dictationServiceImpl.DictationEvents = DictationEvents;
148 _dictationServiceImpl.TelemetryEvents = TelemetryEvents;
149 platformDictationImpl.SetDictationRuntimeConfiguration(RuntimeDictationConfiguration);
150 VLog.D("Dictation platform init complete");
151 _voiceSDKLogger.IsUsingPlatformIntegration = true;
152 }
153 else
154 {
155 OnPlatformServiceNotAvailable();
156 }
157 }
158 else
159 {
160 RevertToWitDictation();
161 }
162#else
163 _voiceSDKLogger = new VoiceSDKConsoleLoggerImpl();
164 RevertToWitDictation();
165#endif
166 _voiceSDKLogger.WitApplication = RuntimeDictationConfiguration?.witConfiguration?.GetLoggerAppId();
167 _voiceSDKLogger.ShouldLogToConsole = enableConsoleLogging;
168
169 OnInitialized?.Invoke();
170 }
171
172 private void OnPlatformServiceNotAvailable()
173 {
174#if !UNITY_EDITOR
175 if (DoNotFallbackToWit)
176 {
177 VLog.D("Platform dictation service unavailable. Falling back to WitDictation is disabled");
178 DictationEvents.OnError?.Invoke("Platform dictation unavailable", "Platform dictation service is not available");
179 return;
180 }
181#endif
182
183 VLog.D("Platform dictation service unavailable. Falling back to WitDictation");
184 RevertToWitDictation();
185 }
186
187 private void OnDictationServiceNotAvailable()

189 VLog.D("Dictation service unavailable");
190 DictationEvents.OnError?.Invoke("Dictation unavailable", "Dictation service is not available");
191 }

193 private void RevertToWitDictation()
194 {
195 WitDictation witDictation = GetComponent<WitDictation>();
196 if (null == witDictation)
197 {
198 witDictation = gameObject.AddComponent<WitDictation>();
199 witDictation.hideFlags = HideFlags.HideInInspector;
200 }
201
202 witDictation.RuntimeConfiguration = RuntimeDictationConfiguration;
203 witDictation.DictationEvents = DictationEvents;
204 witDictation.TelemetryEvents = TelemetryEvents;
205 _dictationServiceImpl = witDictation;
206 VLog.D("WitDictation init complete");
207 _voiceSDKLogger.IsUsingPlatformIntegration = false;
208 }

415 void OnComplete(VoiceServiceRequest request)
416 {
417 if (request.State == VoiceRequestState.Failed)
418 {
419 VLog.E($"Dictation Request Failed\nError: {request.Results.Message}");
420 _voiceSDKLogger.LogInteractionEndFailure(request.Results.Message);
421 }
422 else if (request.State == VoiceRequestState.Canceled)
423 {
424 VLog.W($"Dictation Request Canceled\nMessage: {request.Results.Message}");
425 _voiceSDKLogger.LogInteractionEndFailure("aborted");
426 }
427 else
428 {
429 VLog.D($"Dictation Request Success");
430 var tokens = request.ResponseData?["speech"]?["tokens"];
431 if (tokens != null)
432 {
433 int speechTokensLength = tokens.Count;
434 string speechLength = request.ResponseData["speech"]["tokens"][speechTokensLength - 1]?["end"]?.Value;
435 _voiceSDKLogger.LogAnnotation("audioLength", speechLength);
436 }
437 _voiceSDKLogger.LogInteractionEndSuccess();
438 }
439 if (!_isActive)
440 {
441 DictationEvents.OnDictationSessionStopped?.Invoke(_activeSession);
442 CleanupSession();
443 }
444 }
