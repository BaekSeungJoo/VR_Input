73 public override void AddSamples(float[] newSamples)
74 {
75 // Cannot add samples to non-streamable clip
76 if (!_streamable)
77 {
78 VLog.E(GetType().ToString(), "Cannot add samples to a non-streamable AudioClip");
79 return;
80 }

82 // Generate initial clip
83 if (Clip == null)
84 {
85 int newMaxSamples = Mathf.Max(_chunkSize,
86 AddedSamples + newSamples.Length);
87 UpdateClip(newMaxSamples);
88 }
89 // Generate larger clip if needed
90 else if (AddedSamples + newSamples.Length > TotalSamples)

92 int newMaxSamples = Mathf.Max(TotalSamples + _chunkSize,

141 // Generate audio clip for a specific sample count
142 private void UpdateClip(int samples)
143 {
144 // Cannot update a non-streamable clip
145 if (!_streamable)
146 {
147 return;
148 }
149 // Already generated
150 if (Clip != null && TotalSamples == samples)
151 {
152 return;
153 }
154
155 // Get old clip if applicable
156 AudioClip oldClip = Clip;
157 int oldClipSamples = TotalSamples;
158
159 // Generate new clip
160 TotalSamples = samples;
161 Clip = GetCachedClip(TotalSamples, Channels, SampleRate);
162
163 // If previous clip existed, get previous data
164 if (oldClip != null)
165 {
166 // Apply existing data
167 int copySamples = Mathf.Min(oldClipSamples, samples);
168 float[] oldSamples = new float[copySamples];
169 oldClip.GetData(oldSamples, 0);
170 Clip.SetData(oldSamples, 0);
171
172 // Invoke clip updated callback
173 VLog.D($"Clip Stream - Clip Updated\nNew Samples: {samples}\nOld Samples: {oldClipSamples}");
174
175 // Requeue previous clip
176 ReuseCachedClip(oldClip);
177 }
178 else
179 {
180 VLog.D($"Clip Stream - Clip Generated\nSamples: {samples}");
181 }
182
183 // Handle update
184 HandleStreamUpdated();
185 }

204 // Preload a single clip
205 private static void GenerateCacheClip(int lengthSamples, int channels, int frequency)
206 {
207 ClipsGenerated++;
208 AudioClip clip = AudioClip.Create($"AudioClip_{ClipsGenerated:000}", lengthSamples, channels, frequency, false);
209 Clips.Add(clip);
210 }
211 // Preload a single clip
212 private static AudioClip GetCachedClip(int lengthSamples, int channels, int frequency)
213 {
214 // Find a matching clip
215 int clipIndex = Clips.FindIndex((clip) => DoesClipMatch(clip, lengthSamples, channels, frequency));
216
217 // Generate a clip with the specified size
218 if (clipIndex == -1)
219 {
220 clipIndex = Clips.Count;
221 GenerateCacheClip(lengthSamples, channels, frequency);
222 }
223
224 // Get clip, remove from preload list & return
225 AudioClip result = Clips[clipIndex];
226 Clips.RemoveAt(clipIndex);
227 return result;
228 }
229 // Check if clip matches
230 private static bool DoesClipMatch(AudioClip clip, int lengthSamples, int channels, int frequency)
231 {
232 return clip.samples == lengthSamples && clip.channels == channels && clip.frequency == frequency;
233 }
234 // Reuse clip
235 private static void ReuseCachedClip(AudioClip clip)
236 {
237 Clips.Add(clip);
238 }
242 public static void DestroyCachedClips()
243 {
244 foreach (var clip in Clips)
245 {
246 clip.DestroySafely();
247 }
248 Clips.Clear();
249 }
