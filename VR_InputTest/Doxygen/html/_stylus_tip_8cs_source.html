3using UnityEngine;

5public class StylusTip : MonoBehaviour
6{
7 private const int MaxBreadCrumbs = 60;
8 private const float BreadCrumbMinSize = 0.005f;
9 private const float BreadCrumbMaxSize = 0.02f;
10
11 [Header("External")]
12 [SerializeField] private Transform m_trackingSpace;
13
14 [Header("Settings")]
15 [SerializeField] private OVRInput.Handedness m_handedness = OVRInput.Handedness.LeftHanded;
16
17 [SerializeField] private GameObject m_breadCrumbPf;
18
19 private GameObject m_breadCrumbContainer;
20 private GameObject[] m_breadCrumbs;
21
22 private int m_breadCrumbIndexPrev = -1;
23 private int m_breadCrumbIndexCurr = 0;
24
25 private OVRInput.Controller m_controller;
26
27 private void Awake()
28 {
29 m_controller = m_handedness == OVRInput.Handedness.LeftHanded
30 ? OVRInput.Controller.LTouch
31 : OVRInput.Controller.RTouch;
32
33 // Create the bread crumbs
34 m_breadCrumbContainer = new GameObject($"BreadCrumbContainer ({m_handedness})");
35 m_breadCrumbs = new GameObject[MaxBreadCrumbs];
36 for (int i = 0; i < m_breadCrumbs.Length; ++i)
37 {
38 // Create bread crumb
39 GameObject breadCrumb = GameObject.Instantiate(m_breadCrumbPf, m_breadCrumbContainer.transform);
40 breadCrumb.name = $"BreadCrumb ({i})";
41 breadCrumb.SetActive(false);
42
43 // Store bread crumb
44 m_breadCrumbs[i] = breadCrumb;
45 }
46 }
47
48 private void Update()
49 {
50 // Update stylus tip position
51 Pose T_device = new Pose(OVRInput.GetLocalControllerPosition(m_controller),
52 OVRInput.GetLocalControllerRotation(m_controller));
53 Pose T_world_device = T_device.GetTransformedBy(m_trackingSpace);
54 Pose T_world_stylusTip = GetT_Device_StylusTip(m_controller).GetTransformedBy(T_world_device);
55 this.transform.SetPositionAndRotation(T_world_stylusTip.position, T_world_stylusTip.rotation);
56
57 // Get stylus tip data
58 float stylusTipForce = OVRInput.Get(OVRInput.Axis1D.PrimaryStylusForce, m_controller);
59 bool isStylusTipTouching = stylusTipForce > 0;
60
61 // Set the next crumb position
62 GameObject nextCrumb = m_breadCrumbs[m_breadCrumbIndexCurr];
63 nextCrumb.transform.position = this.transform.position;
64
65 // Set next crumb visuals based on stylus tip force
66 float nextCrumbSize = Mathf.Lerp(BreadCrumbMinSize, BreadCrumbMaxSize, stylusTipForce);
67 nextCrumb.transform.localScale = new Vector3(nextCrumbSize, nextCrumbSize, nextCrumbSize);
68 nextCrumb.GetComponent<MeshRenderer>().material.color = Color.Lerp(Color.white, Color.red, stylusTipForce);
69 nextCrumb.SetActive(isStylusTipTouching);
70
71 float crumbSeparation = 0;
72 float distanceToPrevCrumb = Mathf.Infinity;
73 if (m_breadCrumbIndexPrev >= 0)
74 {
75 // Compute next crumb distance to stylus tip
76 distanceToPrevCrumb = (this.transform.position - m_breadCrumbs[m_breadCrumbIndexPrev].transform.position)
77 .magnitude;
78
79 // Compute next crumb separation by averaging the previous and next crumb sizes
80 crumbSeparation = (nextCrumbSize + m_breadCrumbs[m_breadCrumbIndexPrev].transform.localScale.x) * 0.5f;
81 }
82
83 // Determine if a new crumb should drop
84 if (isStylusTipTouching && (distanceToPrevCrumb >= crumbSeparation))
85 {
86 // Drop the crumb
87 m_breadCrumbIndexPrev = m_breadCrumbIndexCurr;
88 m_breadCrumbIndexCurr = (m_breadCrumbIndexCurr + 1) % m_breadCrumbs.Length;
89 }
90 }
91
92 private static Pose GetT_Device_StylusTip(OVRInput.Controller controller)
93 {
94 // @Note: Only the next controller supports the stylus tip, but we compute the
95 // transforms for all controllers so we can draw the tip at the correct location.
96 Pose T_device_stylusTip = Pose.identity;
97
98 if (controller == OVRInput.Controller.LTouch || controller == OVRInput.Controller.RTouch)
99 {
100 T_device_stylusTip = new Pose(
101 new Vector3(0.0094f, -0.07145f, -0.07565f),
102 Quaternion.Euler(35.305f, 50.988f, 37.901f)
103 );
104 }
105
106 if (controller == OVRInput.Controller.LTouch)
107 {
108 T_device_stylusTip.position.x *= -1;
109 T_device_stylusTip.rotation.y *= -1;
110 T_device_stylusTip.rotation.z *= -1;
111 }
112
113 return T_device_stylusTip;
114 }
