43 // Use to determine if built in configuration
44 public static bool IsBuiltInConfiguration(WitConfiguration witConfiguration)
45 {
46 if (witConfiguration == null)
47 {
48 return false;

52 public static bool IsBuiltInConfiguration(WitAppInfo appInfo)

54 return IsBuiltInConfiguration(appInfo.id);
55 }
56 public static bool IsBuiltInConfiguration(string applicationID)
57 {
58 return !string.IsNullOrEmpty(applicationID) && applicationID.StartsWith("voice");
59 }
60
61 // Get safe app url
62 public static string GetSafeAppUrl(WitConfiguration witConfiguration, WitTexts.WitAppEndpointType endpointType)
63 {
64 // Use built in app url
65 if (IsBuiltInConfiguration(witConfiguration))
66 {
67 return VoiceSDKStyles.Texts.BuiltInAppUrl;
68 }
69 // Return wit app id
70 return WitTexts.GetAppURL(witConfiguration?.GetApplicationId(), endpointType);
