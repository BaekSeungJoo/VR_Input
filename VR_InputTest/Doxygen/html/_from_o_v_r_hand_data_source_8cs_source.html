159 protected override void UpdateData()
160 {
161 _handDataAsset.Config = Config;
162 _handDataAsset.IsDataValid = true;
163 _handDataAsset.IsConnected =
164 (OVRInput.GetConnectedControllers() & _ovrController) > 0;
165
166 if (_ovrHand != null)
167 {
168 IOVRSkeletonDataProvider skeletonProvider = _ovrHand;
169 SkeletonPoseData poseData = skeletonProvider.GetSkeletonPoseData();
170 if (poseData.IsDataValid && poseData.RootScale <= 0.0f)
171 {
172 if (_lastHandScale <= 0.0f)
173 {
174 poseData.IsDataValid = false;
175 }
176 else
177 {
178 poseData.RootScale = _lastHandScale;
179 }
180 }
181 else
182 {
183 _lastHandScale = poseData.RootScale;
184 }
185
186 if (poseData.IsDataValid && _handDataAsset.IsConnected)
187 {
188 UpdateDataPoses(poseData);
189 return;
190 }
191 }
192
193 // revert state fields to their defaults
194 _handDataAsset.IsConnected = default;
195 _handDataAsset.IsTracked = default;
196 _handDataAsset.RootPoseOrigin = default;
197 _handDataAsset.PointerPoseOrigin = default;
198 _handDataAsset.IsHighConfidence = default;
199 for (var fingerIdx = 0; fingerIdx < Constants.NUM_FINGERS; fingerIdx++)
200 {
201 _handDataAsset.IsFingerPinching[fingerIdx] = default;
202 _handDataAsset.IsFingerHighConfidence[fingerIdx] = default;
203 }
204 }

206 private void UpdateDataPoses(SkeletonPoseData poseData)
207 {
208 _handDataAsset.HandScale = poseData.RootScale;
209 _handDataAsset.IsTracked = _ovrHand.IsTracked;
210 _handDataAsset.IsHighConfidence = poseData.IsDataHighConfidence;
211 _handDataAsset.IsDominantHand = _ovrHand.IsDominantHand;
212 _handDataAsset.RootPoseOrigin = _handDataAsset.IsTracked
213 ? PoseOrigin.RawTrackedPose
214 : PoseOrigin.None;
215
216 for (var fingerIdx = 0; fingerIdx < Constants.NUM_FINGERS; fingerIdx++)
217 {
218 var ovrFingerIdx = (OVRHand.HandFinger)fingerIdx;
219 bool isPinching = _ovrHand.GetFingerIsPinching(ovrFingerIdx);
220 _handDataAsset.IsFingerPinching[fingerIdx] = isPinching;
221
222 bool isHighConfidence =
223 _ovrHand.GetFingerConfidence(ovrFingerIdx) == OVRHand.TrackingConfidence.High;
224 _handDataAsset.IsFingerHighConfidence[fingerIdx] = isHighConfidence;
225
226 float fingerPinchStrength = _ovrHand.GetFingerPinchStrength(ovrFingerIdx);
227 _handDataAsset.FingerPinchStrength[fingerIdx] = fingerPinchStrength;
228 }
229
230 // Read the poses directly from the poseData, so it isn't in conflict with
231 // any modifications that the application makes to OVRSkeleton
232 _handDataAsset.Root = new Pose()
233 {
234 position = poseData.RootPose.Position.FromFlippedZVector3f(),
235 rotation = poseData.RootPose.Orientation.FromFlippedZQuatf()
236 };
237
238 if (_ovrHand.IsPointerPoseValid)
239 {
240 _handDataAsset.PointerPoseOrigin = PoseOrigin.RawTrackedPose;
241 _handDataAsset.PointerPose = new Pose(_ovrHand.PointerPose.localPosition,
242 _ovrHand.PointerPose.localRotation);
243 }
244 else
245 {
246 _handDataAsset.PointerPoseOrigin = PoseOrigin.None;
247 }
248
249 // Hand joint rotations X axis needs flipping to get to Unity's coordinate system.
250 var bones = poseData.BoneRotations;
251 for (int i = 0; i < bones.Length; i++)
252 {
253 // When using Link in the Unity Editor, the first frame of hand data
254 // sometimes contains bad joint data.
255 _handDataAsset.Joints[i] = float.IsNaN(bones[i].w)
256 ? Config.HandSkeleton.joints[i].pose.rotation
257 : bones[i].FromFlippedXQuatf();
258 }
259
260 _handDataAsset.Joints[0] = WristFixupRotation;
261 }
