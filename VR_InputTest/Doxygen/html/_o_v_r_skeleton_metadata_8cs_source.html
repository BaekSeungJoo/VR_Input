167 private void AssembleSkeleton(OVRSkeleton skeleton, bool useBindPose,
168 Dictionary<OVRSkeleton.BoneId, HumanBodyBones> customBoneIdToHumanBodyBone
169 )
170 {
171 if (BodyToBoneData.Count != 0)
172 {
173 BodyToBoneData.Clear();
174 }
175
176 var allBones = useBindPose ? skeleton.BindPoses : skeleton.Bones;
177 for (var i = 0; i < allBones.Count; i++)
178 {
179 var bone = allBones[i];
180
181 if (!customBoneIdToHumanBodyBone.ContainsKey(bone.Id))
182 {
183 continue;
184 }
185
186 var humanBodyBone = customBoneIdToHumanBodyBone[bone.Id];
187
188 var boneData = new BoneData();
189 boneData.OriginalJoint = bone.Transform;
190
191 if (!OVRHumanBodyBonesMappings.BoneIdToJointPair.ContainsKey(bone.Id))
192 {
193 Debug.LogError($"Can't find {bone.Id} in bone Id to joint pair map!");
194 continue;
195 }
196
197 var jointPair =
198 OVRHumanBodyBonesMappings.BoneIdToJointPair[bone.Id];
199 var startOfPair = jointPair.Item1;
200 var endofPair = jointPair.Item2;
201
202 // for some tip transforms, the start of pair starts with one joint before
203 boneData.JointPairStart = (startOfPair == bone.Id)
204 ? bone.Transform
205 : FindBoneWithBoneId(allBones, startOfPair).Transform;
206 boneData.JointPairEnd = endofPair != OVRSkeleton.BoneId.Invalid
207 ? FindBoneWithBoneId(allBones, endofPair).Transform
208 : boneData.JointPairStart;
209 boneData.ParentTransform = allBones[bone.ParentBoneIndex].Transform;
210
211 if (boneData.JointPairStart == null)
212 {
213 Debug.LogWarning($"{bone.Id} has invalid start joint.");
214 }
215
216 if (boneData.JointPairEnd == null)
217 {
218 Debug.LogWarning($"{bone.Id} has invalid end joint.");
219 }
220
221 BodyToBoneData.Add(humanBodyBone, boneData);
222 }
223 }

242 private void BuildBoneData(Animator animator)
243 {
244 if (BodyToBoneData.Count != 0)
245 {
246 BodyToBoneData.Clear();
247 }
248
249 foreach (var humanBodyBone in _boneEnumValues)
250 {
251 if (humanBodyBone == HumanBodyBones.LastBone)
252 {
253 continue;
254 }
255
256 var currTransform = animator.GetBoneTransform(humanBodyBone);
257 if (currTransform == null)
258 {
259 continue;
260 }
261
262 var boneData = new BoneData();
263 boneData.OriginalJoint = currTransform;
264 BodyToBoneData.Add(humanBodyBone, boneData);
265 }
266
267 // Find paired joints after all transforms have been tracked.
268 // A joint start starts from a joints, ends at its child joint, and serves
269 // as the "axis" of the joint.
270 foreach (var key in BodyToBoneData.Keys)
271 {
272 var boneData = BodyToBoneData[key];
273 var jointPair = OVRHumanBodyBonesMappings.BoneToJointPair[key];
274
275 boneData.JointPairStart = jointPair.Item1 != HumanBodyBones.LastBone
276 ? animator.GetBoneTransform(jointPair.Item1)
277 : boneData.OriginalJoint;
278
279 boneData.JointPairEnd = jointPair.Item2 != HumanBodyBones.LastBone
280 ? animator.GetBoneTransform(jointPair.Item2)
281 : FindFirstChild(boneData.OriginalJoint, boneData.OriginalJoint);
282
283 boneData.ParentTransform = boneData.OriginalJoint.parent;
284
285 if (boneData.JointPairStart == null)
286 {
287 Debug.LogWarning($"{key} has invalid start joint.");
288 }
289
290 if (boneData.JointPairEnd == null)
291 {
292 Debug.LogWarning($"{key} has invalid end joint.");
293 }
294 }
295 }

300 public void BuildCoordinateAxesForAllBones()
301 {
302 foreach (var key in BodyToBoneData.Keys)
303 {
304 var boneData = BodyToBoneData[key];
305 var jointPairStartPosition = boneData.JointPairStart.position;
306 Vector3 jointPairEndPosition;
307
308 // Edge case: joint pair end is null or same node. If that's the case,
309 // make joint pair end follow the axis from previous node.
310 if (boneData.JointPairEnd == null ||
311 boneData.JointPairEnd == boneData.JointPairStart ||
312 (boneData.JointPairEnd.position - boneData.JointPairStart.position).magnitude <
313 Mathf.Epsilon)
314 {
315 var node1 = boneData.ParentTransform;
316 var node2 = boneData.JointPairStart;
317 jointPairStartPosition = node1.position;
318 jointPairEndPosition = node2.position;
319 boneData.DegenerateJoint = true;
320 }
321 else
322 {
323 jointPairEndPosition = boneData.JointPairEnd.position;
324 boneData.DegenerateJoint = false;
325 }
326
327 // with some joints like hands, fix the right vector. that's because the hand is nice and
328 // flat, and the right vector should point to a thumb bone.
329 if (key == HumanBodyBones.LeftHand || key == HumanBodyBones.RightHand)
330 {
331 var jointToCreateRightVecWith = key == HumanBodyBones.LeftHand
332 ? HumanBodyBones.LeftThumbIntermediate
333 : HumanBodyBones.RightThumbIntermediate;
334 var rightVec = BodyToBoneData[jointToCreateRightVecWith].OriginalJoint.position -
335 jointPairStartPosition;
336 boneData.JointPairOrientation =
337 CreateQuaternionForBoneDataWithRightVec(
338 jointPairStartPosition,
339 jointPairEndPosition,
340 rightVec);
341 }
342 else
343 {
344 boneData.JointPairOrientation =
345 CreateQuaternionForBoneData(
346 jointPairStartPosition,
347 jointPairEndPosition);
348 }
349
350 var position = boneData.OriginalJoint.position;
351 boneData.FromPosition = position;
352 boneData.ToPosition = position +
353 (jointPairEndPosition - jointPairStartPosition);
354 }
355 }

357 private static Transform FindFirstChild(
358 Transform startTransform,
359 Transform currTransform)
360 {
361 if (startTransform != currTransform)
362 {
363 return currTransform;
364 }
365
366 // Dead end.
367 if (currTransform.childCount == 0)
368 {
369 return null;
370 }
371
372 Transform foundChild = null;
373 for (var i = 0; i < currTransform.childCount; i++)
374 {
375 var currChild = FindFirstChild(startTransform,
376 currTransform.GetChild(i));
377 if (currChild != null)
378 {
379 foundChild = currChild;
380 break;
381 }
382 }
383
384 return foundChild;
385 }

387 private static Quaternion CreateQuaternionForBoneDataWithRightVec(
388 Vector3 fromPosition,
389 Vector3 toPosition,
390 Vector3 rightVector)
391 {
392 var forwardVec = (toPosition - fromPosition).normalized;
393 if (forwardVec.sqrMagnitude < Mathf.Epsilon)
394 {
395 forwardVec = Vector3.forward;
396 }
397
398 var upVector = Vector3.Cross(forwardVec, rightVector);
399 return Quaternion.LookRotation(forwardVec, upVector);
400 }
401
402 private static Quaternion CreateQuaternionForBoneData(
403 Vector3 fromPosition,
404 Vector3 toPosition)
405 {
406 var forwardVec = (toPosition - fromPosition).normalized;
407
408 if (forwardVec.sqrMagnitude < Mathf.Epsilon)
409 {
410 forwardVec = Vector3.forward;
411 }
412
413 return Quaternion.LookRotation(forwardVec);
414 }
