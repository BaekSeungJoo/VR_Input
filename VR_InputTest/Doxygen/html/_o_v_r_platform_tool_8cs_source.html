529 private void OnUpload(TargetPlatform targetPlatform)
530 {
531 OVRPlatformTool.log = string.Empty;
532 SetDirtyOnGUIChange();
533 var lintCount = 0;
534 if (OVRPlatformToolSettings.RunOvrLint)
535 {
536 lintCount = OVRLint.RunCheck();
537 }
538
539 if (lintCount != 0)
540 {
541 OVRPlatformTool.log += lintCount.ToString() + " lint suggestions are found. \n" +
542 "Please run Oculus\\Tools\\OVR Performance Lint Tool to review and fix lint errors. \n" +
543 "You can uncheck Run OVR Lint to bypass lint errors. \n";
544 OVRPlugin.SendEvent("oculus_platform_tool_lint", lintCount.ToString());
545 }
546 else
547 {
548 // Continue uploading process
549 ExecuteCommand(targetPlatform);
550 }
551 }

553 static void ExecuteCommand(TargetPlatform targetPlatform)
554 {
555 string dataPath = Application.dataPath;
556
557 // If we already have a copy of the platform util, check if it needs to be updated
558 if (!ranSelfUpdate && File.Exists(dataPath + "/Oculus/VR/Editor/Tools/ovr-platform-util.exe"))
559 {
560 ranSelfUpdate = true;
561 activeProcess = true;
562 var updateThread = new Thread(delegate ()
563 {
564 retryCount = 0;
565 CheckForUpdate(dataPath);
566 });
567 updateThread.Start();
568 }
569
570 string uploadCommand;
571 if (genUploadCommand(targetPlatform, out uploadCommand))
572 {
573 var thread = new Thread(delegate ()
574 {
575 // Wait for update process to finish before starting upload process
576 while (activeProcess)
577 {
578 Thread.Sleep(100);
579 }
580
581 retryCount = 0;
582 Command(targetPlatform, dataPath, uploadCommand);
583 });
584 thread.Start();
585 }
586 else
587 {
588 UnityEngine.Debug.LogError("Failed to generated upload command.");
589 }
590 }

592 private static string CheckForPlatformUtil(string dataPath)
593 {
594 string toolDataPath = dataPath + "/Oculus/VR/Editor/Tools";
595 if (!Directory.Exists(toolDataPath))
596 {
597 Directory.CreateDirectory(toolDataPath);
598 }
599
600 string platformUtil = toolDataPath + "/ovr-platform-util.exe";
601 if (!System.IO.File.Exists(platformUtil))
602 {
603 OVRPlugin.SendEvent("oculus_platform_tool", "provision_util");
604 EditorCoroutine downloadCoroutine = EditorCoroutine.Start(ProvisionPlatformUtil(platformUtil));
605 while (!downloadCoroutine.GetCompleted())
606 {
607 }
608 }
609
610 return platformUtil;
611 }

613 private static void InitializePlatformUtilProcess(string path, string args)
614 {
615 ovrPlatUtilProcess = new Process();
616 var processInfo = new ProcessStartInfo(path, args);
617
618 processInfo.CreateNoWindow = true;
619 processInfo.UseShellExecute = false;
620 processInfo.RedirectStandardError = true;
621 processInfo.RedirectStandardOutput = true;
622
623 ovrPlatUtilProcess.StartInfo = processInfo;
624 ovrPlatUtilProcess.EnableRaisingEvents = true;
625 }
626
627 static void CheckForUpdate(string dataPath)
628 {
629 string platformUtilPath = CheckForPlatformUtil(dataPath);
630 InitializePlatformUtilProcess(platformUtilPath, "self-update");
631
632 OVRPlatformTool.log += "Checking for update...\n";
633
634 ovrPlatUtilProcess.Exited += new EventHandler(
635 (s, e) =>
636 {
637 if (File.Exists(dataPath + ".ovr-platform-util.exe"))
638 {
639 OVRPlatformTool.log += "Cleaning up...\n";
640 while (File.Exists(dataPath + ".ovr-platform-util.exe"))
641 {
642 }
643
644 OVRPlatformTool.log += "Finished updating platform utility.\n";
645 }
646
647 activeProcess = false;
648 }
649 );
650
651 ovrPlatUtilProcess.OutputDataReceived += new DataReceivedEventHandler(
652 (s, e) =>
653 {
654 if (e.Data != null && e.Data.Length != 0 && !e.Data.Contains("\u001b"))
655 {
656 OVRPlatformTool.log += e.Data + "\n";
657 }
658 }
659 );
660
661 try
662 {
663 ovrPlatUtilProcess.Start();
664 ovrPlatUtilProcess.BeginOutputReadLine();
665 }
666 catch
667 {
668 if (ThrowPlatformUtilStartupError(platformUtilPath))
669 {
670 CheckForUpdate(dataPath);
671 }
672 }
673 }

675 static void LoadRedistPackages(string dataPath)
676 {
677 // Check / Download the platform util and call list-redists on it
678 activeProcess = true;
679 string platformUtilPath = CheckForPlatformUtil(dataPath);
680 InitializePlatformUtilProcess(platformUtilPath, "list-redists");
681
682 OVRPlatformTool.log += "Loading redistributable packages...\n";
683
684 List<RedistPackage> redistPacks = new List<RedistPackage>();
685
686 ovrPlatUtilProcess.Exited += new EventHandler(
687 (s, e) => { activeProcess = false; }
688 );
689
690 ovrPlatUtilProcess.OutputDataReceived += new DataReceivedEventHandler(
691 (s, e) =>
692 {
693 if (e.Data != null && e.Data.Length != 0 && !e.Data.Contains("\u001b") && !e.Data.Contains("ID"))
694 {
695 // Get the name / ID pair from the CLI and create a redist package instance
696 string[] terms = e.Data.Split('|');
697 if (terms.Length == 2)
698 {
699 RedistPackage redistPack = new RedistPackage(terms[1], terms[0]);
700 redistPacks.Add(redistPack);
701 }
702 }
703 }
704 );
705
706 try
707 {
708 ovrPlatUtilProcess.Start();
709 ovrPlatUtilProcess.BeginOutputReadLine();
710
711 ovrPlatUtilProcess.WaitForExit();
712
713 if (redistPacks.Count != OVRPlatformToolSettings.RiftRedistPackages.Count)
714 {
715 OVRPlatformTool.log += "Successfully updated redistributable packages.\n";
