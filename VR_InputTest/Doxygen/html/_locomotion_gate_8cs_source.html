114 public class GateSection
115 {
116 public float minAngle = _wristLimit;
117 public float maxAngle = 360f + _wristLimit;
118 public bool canEnterDirectly = true;
119 public LocomotionMode locomotionMode = LocomotionMode.None;
120
121 public float ScoreToAngle(float angle)
122 {
123 float dif = Mathf.Repeat(angle - minAngle, 360f);
124 float range = Mathf.Repeat(maxAngle - minAngle, 360f);
125 if (dif > range)
126 {
127 return float.PositiveInfinity;
128 }
129
130 float average = (minAngle + maxAngle) / 2f;
131 float score = Mathf.Repeat(Mathf.DeltaAngle(angle, average), 360f);
132 return score;
133 }
134 }

201 private void HandleHandupdated()
202 {
203 if (!Hand.GetRootPose(out Pose handPose))
204 {
205 Disable();
206 return;
207 }
208
209 bool isRightHand = Hand.Handedness == Handedness.Right;
210 Vector3 trackingUp = Vector3.up;
211 Vector3 shoulderToHand = (handPose.position - _shoulder.position).normalized;
212 Vector3 trackingRight = Vector3.Cross(trackingUp, shoulderToHand).normalized;
213 trackingRight = isRightHand ? trackingRight : -trackingRight;
214 Vector3 wristDir = isRightHand ? -handPose.forward : handPose.forward;
215 Vector3 fingersDir = isRightHand ? -handPose.right : handPose.right;
216 bool fingersAway = (Vector3.Dot(fingersDir, Vector3.ProjectOnPlane(shoulderToHand, trackingUp).normalized) * 0.5 + 0.5f) > _enterPoseThreshold;
217
218 wristDir = Vector3.ProjectOnPlane(wristDir, shoulderToHand).normalized;
219 float angle = Vector3.SignedAngle(wristDir, trackingRight, shoulderToHand);
220
221 angle = Hand.Handedness == Handedness.Right ? -angle : angle;
222 if (angle < _wristLimit)
223 {
224 angle += 360f;
225 }
226
227 CurrentAngle = angle;
228 StabilizationPose = new Pose(_shoulder.position, Quaternion.LookRotation(shoulderToHand));
229 WristDirection = wristDir;
230
231 bool shapeGateEnabled = false;
232 if (EnableShape.Active && !_previousShapeEnabled)
233 {
234 shapeGateEnabled = true;
235 }
236 _previousShapeEnabled = EnableShape.Active;
237
238 if (_currentGateIndex < 0
239 && shapeGateEnabled
240 && fingersAway)
241 {
242 GateSection gateSection = GetBestGateSection(CurrentAngle, out _currentGateIndex);
243 if (gateSection.canEnterDirectly)
244 {
245 ActiveMode = gateSection.locomotionMode;
246 }
247 else
248 {
249 _currentGateIndex = -1;
250 }
251 return;
252 }
253
254 if (_currentGateIndex < 0)
255 {
256 return;
257 }
258
259 if (_currentGateIndex >= 0
260 && DisableShape.Active)
261 {
262 _currentGateIndex = -1;
263 ActiveMode = LocomotionMode.None;
264 return;
265 }
266
267
268 GateSection currentGate = _gateSections[_currentGateIndex];
269
270 if (CurrentAngle < currentGate.minAngle)
271 {
272 _currentGateIndex = Mathf.Max(0, _currentGateIndex - 1);
273 ActiveMode = _gateSections[_currentGateIndex].locomotionMode;
274 }
275 else if (CurrentAngle > currentGate.maxAngle)
276 {
277 _currentGateIndex = Mathf.Min(_gateSections.Length - 1, _currentGateIndex + 1);
278 ActiveMode = _gateSections[_currentGateIndex].locomotionMode;
279 }
280
281 }

283 private GateSection GetBestGateSection(float angle, out int index)
284 {
285 float bestScore = float.PositiveInfinity;
286 index = -1;
287
288 for (int i = 0; i < _gateSections.Length; i++)
289 {
290 float score = _gateSections[i].ScoreToAngle(angle);
291 if (score < bestScore)
292 {
293 bestScore = score;
294 index = i;
295 }
296 }
297
298 if (index == -1)
299 {
300 return DefaultSection;
301 }
302 return _gateSections[index];
303 }
