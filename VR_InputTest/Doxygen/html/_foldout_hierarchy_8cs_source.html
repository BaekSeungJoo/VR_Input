16 public void Add(string path, FoldoutHierarchyItem<T> item)
17 {
18 string[] parts = path.Split('/');

20
21
22 for (int i = 0; i < parts.Length; i++)
23 {
24 string key = string.Join("/", parts, 0, i + 1);
25
26 if (!_groups.ContainsKey(key))
27 {
28 FoldoutGroup<T> newGroup = new FoldoutGroup<T>(parts[i]);
29 _groups.Add(key, newGroup);

35 }
36 }
37
38 currentGroup = _groups[key];
39 }
