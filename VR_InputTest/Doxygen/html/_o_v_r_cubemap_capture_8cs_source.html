95 public static void TriggerCubemapCapture(Vector3 capturePos, int cubemapSize = 2048, string pathName = null)
96 {
97 GameObject ownerObj = new GameObject("CubemapCamera", typeof(Camera));
98 ownerObj.hideFlags = HideFlags.HideAndDontSave;
99 ownerObj.transform.position = capturePos;
100 ownerObj.transform.rotation = Quaternion.identity;
101 Camera camComponent = ownerObj.GetComponent<Camera>();
102 camComponent.farClipPlane = 10000.0f;
103 camComponent.enabled = false;
104
105 Cubemap cubemap = new Cubemap(cubemapSize, TextureFormat.RGB24, false);
106 RenderIntoCubemap(camComponent, cubemap);
107 SaveCubemapCapture(cubemap, pathName);
108 DestroyImmediate(cubemap);
109 DestroyImmediate(ownerObj);
110 }

113 public static void RenderIntoCubemap(Camera ownerCamera, Cubemap outCubemap)
114 {
115 int width = (int)outCubemap.width;
116 int height = (int)outCubemap.height;
117
118 CubemapFace[] faces = new CubemapFace[]
119 {
120 CubemapFace.PositiveX, CubemapFace.NegativeX, CubemapFace.PositiveY, CubemapFace.NegativeY,
121 CubemapFace.PositiveZ, CubemapFace.NegativeZ
122 };
123 Vector3[] faceAngles = new Vector3[]
124 {
125 new Vector3(0.0f, 90.0f, 0.0f), new Vector3(0.0f, -90.0f, 0.0f), new Vector3(-90.0f, 0.0f, 0.0f),
126 new Vector3(90.0f, 0.0f, 0.0f), new Vector3(0.0f, 0.0f, 0.0f), new Vector3(0.0f, 180.0f, 0.0f)
127 };
128
129 // Backup states
130 RenderTexture backupRenderTex = RenderTexture.active;
131 float backupFieldOfView = ownerCamera.fieldOfView;
132 float backupAspect = ownerCamera.aspect;
133 Quaternion backupRot = ownerCamera.transform.rotation;
134 //RenderTexture backupRT = ownerCamera.targetTexture;
135
136 // Enable 8X MSAA
137 RenderTexture faceTexture = new RenderTexture(width, height, 24);
138 faceTexture.antiAliasing = 8;
139 faceTexture.dimension = UnityEngine.Rendering.TextureDimension.Tex2D;
140 faceTexture.hideFlags = HideFlags.HideAndDontSave;
141
142 // For intermediate saving
143 Texture2D swapTex = new Texture2D(width, height, TextureFormat.RGB24, false);
144 swapTex.hideFlags = HideFlags.HideAndDontSave;
145
146 // Capture 6 Directions
147 ownerCamera.targetTexture = faceTexture;
148 ownerCamera.fieldOfView = 90;
149 ownerCamera.aspect = 1.0f;
150
151 Color[] mirroredPixels = new Color[swapTex.height * swapTex.width];
152 for (int i = 0; i < faces.Length; i++)
153 {
154 ownerCamera.transform.eulerAngles = faceAngles[i];
155 ownerCamera.Render();
156 RenderTexture.active = faceTexture;
157 swapTex.ReadPixels(new Rect(0, 0, width, height), 0, 0);
158
159 // Mirror vertically to meet the standard of unity cubemap
160 Color[] OrignalPixels = swapTex.GetPixels();
161 for (int y1 = 0; y1 < height; y1++)
162 {
163 for (int x1 = 0; x1 < width; x1++)
164 {
165 mirroredPixels[y1 * width + x1] = OrignalPixels[((height - 1 - y1) * width) + x1];
166 }
167 }
168
169 outCubemap.SetPixels(mirroredPixels, faces[i]);
170 }
171
172 outCubemap.SmoothEdges();
173
174 // Restore states
175 RenderTexture.active = backupRenderTex;
176 ownerCamera.fieldOfView = backupFieldOfView;
177 ownerCamera.aspect = backupAspect;
178 ownerCamera.transform.rotation = backupRot;
179 ownerCamera.targetTexture = backupRenderTex;
180
181 DestroyImmediate(swapTex);
182 DestroyImmediate(faceTexture);
183 }

193 public static bool SaveCubemapCapture(Cubemap cubemap, string pathName = null)
194 {
195 string fileName;
196 string dirName;
197 int width = cubemap.width;
198 int height = cubemap.height;
199 int x = 0;
200 int y = 0;
201 bool saveToPNG = true;
202
203 if (string.IsNullOrEmpty(pathName))
204 {
205 dirName = Application.persistentDataPath + "/OVR_ScreenShot360/";
206 fileName = null;
207 }
208 else
209 {
210 dirName = Path.GetDirectoryName(pathName);
211 fileName = Path.GetFileName(pathName);
212
213 if (dirName[dirName.Length - 1] != '/' || dirName[dirName.Length - 1] != '\\')
214 dirName += "/";
215 }
216
217 if (string.IsNullOrEmpty(fileName))
218 fileName = "OVR_" + System.DateTime.Now.ToString("hh_mm_ss") + ".png";
219
220 string extName = Path.GetExtension(fileName);
221 if (extName == ".png")
222 {
223 saveToPNG = true;
224 }
225 else if (extName == ".jpg")
226 {
227 saveToPNG = false;
228 }
229 else
230 {
231 Debug.LogError("Unsupported file format" + extName);
232 return false;
233 }
234
235 // Validate path
236 try
237 {
238 System.IO.Directory.CreateDirectory(dirName);
239 }
240 catch (System.Exception e)
241 {
242 Debug.LogError("Failed to create path " + dirName + " since " + e.ToString());
243 return false;
244 }
245
246
247 // Create the new texture
248 Texture2D tex = new Texture2D(width * 6, height, TextureFormat.RGB24, false);
249 if (tex == null)
250 {
251 Debug.LogError("[OVRScreenshotWizard] Failed creating the texture!");
252 return false;
253 }
254
255 // Merge all the cubemap faces into the texture
256 // Reference cubemap format: http://docs.unity3d.com/Manual/class-Cubemap.html
257 CubemapFace[] faces = new CubemapFace[]
258 {
259 CubemapFace.PositiveX, CubemapFace.NegativeX, CubemapFace.PositiveY, CubemapFace.NegativeY,
260 CubemapFace.PositiveZ, CubemapFace.NegativeZ
261 };
262 for (int i = 0; i < faces.Length; i++)
263 {
264 // get the pixels from the cubemap
265 Color[] srcPixels = null;
266 Color[] pixels = cubemap.GetPixels(faces[i]);
267 // if desired, flip them as they are ordered left to right, bottom to top
268 srcPixels = new Color[pixels.Length];
269 for (int y1 = 0; y1 < height; y1++)
270 {
271 for (int x1 = 0; x1 < width; x1++)
272 {
273 srcPixels[y1 * width + x1] = pixels[((height - 1 - y1) * width) + x1];
274 }
275 }
276
277 // Copy them to the dest texture
278 tex.SetPixels(x, y, width, height, srcPixels);
279 x += width;
280 }
281
282 try
283 {
284 // Encode the texture and save it to disk
285 byte[] bytes = saveToPNG ? tex.EncodeToPNG() : tex.EncodeToJPG();
286
287 System.IO.File.WriteAllBytes(dirName + fileName, bytes);
288 Debug.Log("Cubemap file created " + dirName + fileName);
289 }
290 catch (System.Exception e)
291 {
292 Debug.LogError("Failed to save cubemap file since " + e.ToString());
293 return false;
294 }
295
296 DestroyImmediate(tex);
297 return true;
298 }
