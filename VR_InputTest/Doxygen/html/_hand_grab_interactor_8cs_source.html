338 protected override HandGrabInteractable ComputeCandidate()
339 {
340 var interactables = HandGrabInteractable.Registry.List(this);
341 float bestFingerScore = float.NegativeInfinity;
342 GrabPoseScore bestPoseScore = GrabPoseScore.Max;
343 HandGrabInteractable bestInteractable = null;
344
345 foreach (HandGrabInteractable interactable in interactables)
346 {
347 GrabTypeFlags selectingGrabTypes = SelectingGrabTypes(interactable, bestFingerScore,
348 out float fingerScore);
349
350 if (selectingGrabTypes == GrabTypeFlags.None)
351 {
352 continue;
353 }
354
355 GrabPoseScore poseScore = this.GetPoseScore(interactable, selectingGrabTypes, ref _cachedResult);
356 if (fingerScore > bestFingerScore
357 || poseScore.IsBetterThan(bestPoseScore))
358 {
359 bestFingerScore = fingerScore;
360 bestPoseScore = poseScore;
361 bestInteractable = interactable;
362 }
363 }
364
365 return bestInteractable;
366 }

368 private GrabTypeFlags SelectingGrabTypes(HandGrabInteractable interactable,
369 float minFingerScoreRequired, out float fingerScore)
370 {
371 fingerScore = 1.0f;
372 GrabTypeFlags selectingGrabTypes;
373 if (State == InteractorState.Select
374 || !this.ComputeShouldSelect(interactable, out selectingGrabTypes))
375 {
376 fingerScore = HandGrabInteraction.ComputeHandGrabScore(this, interactable, out selectingGrabTypes);
377 }
378
379 if (fingerScore < minFingerScoreRequired)
380 {
381 return GrabTypeFlags.None;
382 }
383
384 if (selectingGrabTypes == GrabTypeFlags.None)
385 {
386 if (_hoverOnZeroStrength)
387 {
388 selectingGrabTypes = interactable.SupportedGrabTypes & this.SupportedGrabTypes;
389 }
390 else
391 {
392 return GrabTypeFlags.None;
393 }
394 }
395
396 if (_gripCollider != null
397 && (selectingGrabTypes & GrabTypeFlags.Palm) != 0
398 && !OverlapsSphere(interactable, _gripCollider))
399 {
400 selectingGrabTypes &= ~GrabTypeFlags.Palm;
401 }
402
403 if (_pinchCollider != null
404 && (selectingGrabTypes & GrabTypeFlags.Pinch) != 0
405 && !OverlapsSphere(interactable, _pinchCollider))
406 {
407 selectingGrabTypes &= ~GrabTypeFlags.Pinch;
408 }
409
410 return selectingGrabTypes;
411 }

470 private bool OverlapsSphere(HandGrabInteractable interactable, SphereCollider sphere)
471 {
472 Vector3 point = sphere.transform.position;
473 float radius = sphere.bounds.extents.x;
474
475 foreach (Collider collider in interactable.Colliders)
476 {
477 if (collider.enabled &&
478 Collisions.IsSphereWithinCollider(point, radius, collider))
479 {
480 return true;
481 }
482 }
483
484 return false;
485 }
