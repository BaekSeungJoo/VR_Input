88 public static string SanitizeName(string input)
89 {
90 // Ensure no empty/null names
91 if (string.IsNullOrEmpty(input))
92 {
93 return string.Empty;
94 }
95 // Standard sanitize
96 var result = SanitizeString(input);
97 // Capitalize first letter

99 }

107 public static string SanitizeString(string input)
108 {
109 // Ensure no empty/null strings
110 if (string.IsNullOrEmpty(input))
111 {
112 return string.Empty;
113 }
114 // Remove all non word characters, underscore & hyphen
115 var result = Regex.Replace(input, @"[^\w_-]", "");
116 // Starts with number, append N
117 if (Regex.IsMatch(result[0].ToString(), @"^\d$"))
118 {
119 result = $"N{result}";
120 }
121 // Sanitized string
122 return result;
123 }
