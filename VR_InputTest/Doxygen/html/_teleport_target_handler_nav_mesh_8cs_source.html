41 protected override bool ConsiderTeleport(Vector3 start, ref Vector3 end)
42 {
43 // If the ray hits the world, consider it valid and update the aimRay to the end point.
44 if (LocomotionTeleport.AimCollisionTest(start, end, AimCollisionLayerMask, out AimData.TargetHitInfo))
45 {
46 var d = (end - start).normalized;
47
48 end = start + d * AimData.TargetHitInfo.distance;
49 return true;
50 }
51
52 return false;
53 }

61 public override Vector3? ConsiderDestination(Vector3 location)
62 {
63 var result = base.ConsiderDestination(location);
64 if (result.HasValue)
65 {
66 Vector3 start = LocomotionTeleport.GetCharacterPosition();
67 Vector3 dest = result.GetValueOrDefault();
68 UnityEngine.AI.NavMesh.CalculatePath(start, dest, NavMeshAreaMask, _path);
69
70 if (_path.status == UnityEngine.AI.NavMeshPathStatus.PathComplete)
71 {
72 return result;
73 }
74 }
75
76 return null;
77 }
