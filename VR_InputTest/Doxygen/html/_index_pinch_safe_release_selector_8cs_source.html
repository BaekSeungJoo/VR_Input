88 private void HandleHandUpdated()
89 {
90 bool isPinchDetected = Hand.GetIndexFingerIsPinching();
91 bool isReleaseDetected = !isPinchDetected && IndexReleaseSafeguard.Active;

103 private void ProcessSelectOnPinch(bool isPinchDetected, bool isReleaseDetected)
104 {
105 if (!_isIndexFingerPinching && isPinchDetected && !_cancelled)
106 {
107 _isIndexFingerPinching = true;
108 WhenSelected.Invoke();
109 }
110 else if (_isIndexFingerPinching && isReleaseDetected)
111 {
112 _isIndexFingerPinching = false;
113 WhenUnselected.Invoke();
114 }
115
116 if (!_isIndexFingerPinching && _cancelled)
117 {
118 _cancelled = false;
119 }
120 }

122 private void ProcessSelectOnRelease(bool isPinchDetected, bool isReleaseDetected)
123 {
124 if (!_isIndexFingerPinching && isPinchDetected)
125 {
126 _isIndexFingerPinching = true;
127 }
128
129 if (!_isIndexFingerPinching && _pendingUnselect)
130 {
131 _pendingUnselect = false;
132 WhenUnselected.Invoke();
133 }
134
135 if (_isIndexFingerPinching && isReleaseDetected)
136 {
137 _isIndexFingerPinching = false;
138 if (!_cancelled)
139 {
140 WhenSelected.Invoke();
141 _pendingUnselect = true;
142 }
143 _cancelled = false;
144 }
145 }
